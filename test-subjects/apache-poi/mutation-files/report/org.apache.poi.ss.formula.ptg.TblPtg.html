 23: /**
 24:  * This ptg indicates a data table.
 25:  * It only occurs in a FORMULA record, never in an
 26:  *  ARRAY or NAME record.  When ptgTbl occurs in a
 27:  *  formula, it is the only token in the formula.
 28:  *
 29:  * This indicates that the cell containing the
 30:  *  formula is an interior cell in a data table;
 31:  *  the table description is found in a TABLE
 32:  *  record. Rows and columns which contain input
 33:  *  values to be substituted in the table do
 34:  *  not contain ptgTbl.
 35:  * See page 811 of the june 08 binary docs.
 36:  */
 37: public final class TblPtg extends ControlPtg {
 38:     private final static int  SIZE = 5;
 39:     public final static short sid  = 0x02;
 40:     /** The row number of the upper left corner */
 41:     private final int field_1_first_row;
 42:     /** The column number of the upper left corner */
 43:     private final int field_2_first_col;
 44: 
 45:     public TblPtg(LittleEndianInput in)  {
 46:       field_1_first_row = in.readUShort();
 47:       field_2_first_col = in.readUShort();
 48:     }
 49: 
 50:     public void write(LittleEndianOutput out) {
 51:         out.writeByte(sid + getPtgClass());
 52:         out.writeShort(field_1_first_row);
 53:         out.writeShort(field_2_first_col);
 54:     }
 55: 
 56:     public int getSize() {
 57:         return SIZE;
 58:     }
 59: 
 60:     public int getRow() {
 61:       return field_1_first_row;
 62:     }
 63: 
 64:     public int getColumn() {
 65:       return field_2_first_col;
 66:     }
 67: 
 68:     public String toFormulaString()
 69:     {
 70:         // table(....)[][]
 71:         throw new RuntimeException("Table and Arrays are not yet supported");
 72:     }
 73: 
 74:     public String toString() {
 75:         StringBuffer buffer = new StringBuffer("[Data Table - Parent cell is an interior cell in a data table]\n");
 76:         buffer.append("top left row = ").append(getRow()).append("\n");
 77:         buffer.append("top left col = ").append(getColumn()).append("\n");
 78:         return buffer.toString();
 79:     }
