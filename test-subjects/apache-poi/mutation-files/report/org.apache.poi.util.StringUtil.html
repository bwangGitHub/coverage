 45: 	 *  Given a byte array of 16-bit unicode characters in Little Endian
 46: 	 *  format (most important byte last), return a Java String representation
 47: 	 *  of it.
 48: 	 *
 49: 	 * { 0x16, 0x00 } -0x16
 50: 	 *
 51: 	 * @param  string  the byte array to be converted
 52: 	 * @param  offset  the initial offset into the
 53: 	 *                 byte array. it is assumed that string[ offset ] and string[ offset +
 54: 	 *                 1 ] contain the first 16-bit unicode character
 55:      * @param len the length of the final string
 56: 	 * @return the converted string, never <code>null</code>.
 57: 	 * @exception  ArrayIndexOutOfBoundsException  if offset is out of bounds for
 58: 	 *      the byte array (i.e., is negative or is greater than or equal to
 59: 	 *      string.length)
 60: 	 * @exception  IllegalArgumentException        if len is too large (i.e.,
 61: 	 *      there is not enough data in string to create a String of that
 62: 	 *      length)
 63: 	 */
 64: 	public static String getFromUnicodeLE(
 65: 		final byte[] string,
 66: 		final int offset,
 67: 		final int len)
 68: 		throws ArrayIndexOutOfBoundsException, IllegalArgumentException {
 69: 		if ((offset < 0) || (offset >= string.length)) {
 70: 			throw new ArrayIndexOutOfBoundsException("Illegal offset " + offset + " (String data is of length " + string.length + ")");
 71: 		}
 72: 		if ((len < 0) || (((string.length - offset) / 2) < len)) {
 73: 			throw new IllegalArgumentException("Illegal length " + len);
 74: 		}
 75: 
 76: 		try {
 77: 			return new String(string, offset, len * 2, "UTF-16LE");
 78: 		} catch (UnsupportedEncodingException e) {
 79: 			throw new RuntimeException(e);
 80: 		}
 81: 	}

283: 	 *  Apply printf() like formatting to a string.
284: 	 *  Primarily used for logging.
285: 	 * @param  message  the string with embedded formatting info
286: 	 *                 eg. "This is a test %2.2"
287: 	 * @param  params   array of values to format into the string
288: 	 * @return          The formatted string
289: 	 */
290: 	public static String format(String message, Object[] params) {
291: 		int currentParamNumber = 0;
292: 		StringBuffer formattedMessage = new StringBuffer();
293: 		for (int i = 0; i < message.length(); i++) {
294: 			if (message.charAt(i) == '%') {
295: 				if (currentParamNumber >= params.length) {
296: 					formattedMessage.append("?missing data?");
297: 				} else if (
298: 					(params[currentParamNumber] instanceof Number)
299: 						&& (i + 1 < message.length())) {
300: 					i
301: 						+= matchOptionalFormatting(
302: 							(Number) params[currentParamNumber++],
303: 							message.substring(i + 1),
304: 							formattedMessage);
305: 				} else {
306: 					formattedMessage.append(
307: 						params[currentParamNumber++].toString());
308: 				}
309: 			} else {
310: 				if ((message.charAt(i) == '\\')
311: 					&& (i + 1 < message.length())
312: 					&& (message.charAt(i + 1) == '%')) {
313: 					formattedMessage.append('%');
314: 					i++;
315: 				} else {
316: 					formattedMessage.append(message.charAt(i));
317: 				}
318: 			}
319: 		}
320: 		return formattedMessage.toString();
321: 	}

324: 	private static int matchOptionalFormatting(
325: 		Number number,
326: 		String formatting,
327: 		StringBuffer outputTo) {
328: 		NumberFormat numberFormat = NumberFormat.getInstance();
329: 		if ((0 < formatting.length())
330: 			&& Character.isDigit(formatting.charAt(0))) {
331: 			numberFormat.setMinimumIntegerDigits(
332: 				Integer.parseInt(formatting.charAt(0) + ""));
333: 			if ((2 < formatting.length())
334: 				&& (formatting.charAt(1) == '.')
335: 				&& Character.isDigit(formatting.charAt(2))) {
336: 				numberFormat.setMaximumFractionDigits(
337: 					Integer.parseInt(formatting.charAt(2) + ""));
338: 				numberFormat.format(number, outputTo, new FieldPosition(0));
339: 				return 3;
340: 			}
341: 			numberFormat.format(number, outputTo, new FieldPosition(0));
342: 			return 1;
343: 		} else if (
344: 			(0 < formatting.length()) && (formatting.charAt(0) == '.')) {
345: 			if ((1 < formatting.length())
346: 				&& Character.isDigit(formatting.charAt(1))) {
347: 				numberFormat.setMaximumFractionDigits(
348: 					Integer.parseInt(formatting.charAt(1) + ""));
349: 				numberFormat.format(number, outputTo, new FieldPosition(0));
350: 				return 2;
351: 			}
352: 		}
353: 		numberFormat.format(number, outputTo, new FieldPosition(0));
354: 		return 1;
355: 	}
