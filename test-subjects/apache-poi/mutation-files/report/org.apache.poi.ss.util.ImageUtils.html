 42:     /**
 43:      * Return the dimension of this image
 44:      *
 45:      * @param is the stream containing the image data
 46:      * @param type type of the picture: {@link org.apache.poi.ss.usermodel.Workbook#PICTURE_TYPE_JPEG},
 47:      * {@link org.apache.poi.ss.usermodel.Workbook#PICTURE_TYPE_PNG} or {@link org.apache.poi.ss.usermodel.Workbook#PICTURE_TYPE_DIB}
 48:      *
 49:      * @return image dimension in pixels
 50:      */
 51:     public static Dimension getImageDimension(InputStream is, int type){
 52:         Dimension size = new Dimension();
 53: 
 54:         switch (type){
 55:             //we can calculate the preferred size only for JPEG, PNG and BMP
 56:             //other formats like WMF, EMF and PICT are not supported in Java
 57:             case Workbook.PICTURE_TYPE_JPEG:
 58:             case Workbook.PICTURE_TYPE_PNG:
 59:             case Workbook.PICTURE_TYPE_DIB:
 60:                 try {
 61:                     //read the image using javax.imageio.*
 62:                     ImageInputStream iis = ImageIO.createImageInputStream( is );
 63:                     Iterator i = ImageIO.getImageReaders( iis );
 64:                     ImageReader r = (ImageReader) i.next();
 65:                     r.setInput( iis );
 66:                     BufferedImage img = r.read(0);
 67: 
 68:                     int[] dpi = getResolution(r);
 69: 
 70:                     //if DPI is zero then assume standard 96 DPI
 71:                     //since cannot divide by zero
 72:                     if (dpi[0] == 0) dpi[0] = PIXEL_DPI;
 73:                     if (dpi[1] == 0) dpi[1] = PIXEL_DPI;
 74: 
 75:                     size.width = img.getWidth()*PIXEL_DPI/dpi[0];
 76:                     size.height = img.getHeight()*PIXEL_DPI/dpi[1];
 77: 
 78:                     r.dispose();
 79:                     iis.close();
 80: 
 81:                 } catch (IOException e){
 82:                     //silently return if ImageIO failed to read the image
 83:                     logger.log(POILogger.WARN, e);
 84:                 }
 85: 
 86:                 break;
 87:             default:
 88:                 logger.log(POILogger.WARN, "Only JPEG, PNG and DIB pictures can be automatically sized");
 89:         }
 90:         return size;
 91:     }

 93:     /**
 94:      * The metadata of PNG and JPEG can contain the width of a pixel in millimeters.
 95:      * Return the the "effective" dpi calculated as <code>25.4/HorizontalPixelSize</code>
 96:      * and <code>25.4/VerticalPixelSize</code>.  Where 25.4 is the number of mm in inch.
 97:      *
 98:      * @return array of two elements: <code>{horisontalPdi, verticalDpi}</code>.
 99:      * {96, 96} is the default.
100:      */
101:     public static int[] getResolution(ImageReader r) throws IOException {
102:         int hdpi=96, vdpi=96;
103:         double mm2inch = 25.4;
104: 
105:         NodeList lst;
106:         Element node = (Element)r.getImageMetadata(0).getAsTree("javax_imageio_1.0");
107:         lst = node.getElementsByTagName("HorizontalPixelSize");
108:         if(lst != null && lst.getLength() == 1) hdpi = (int)(mm2inch/Float.parseFloat(((Element)lst.item(0)).getAttribute("value")));
109: 
110:         lst = node.getElementsByTagName("VerticalPixelSize");
111:         if(lst != null && lst.getLength() == 1) vdpi = (int)(mm2inch/Float.parseFloat(((Element)lst.item(0)).getAttribute("value")));
112: 
113:         return new int[]{hdpi, vdpi};
114:     }
