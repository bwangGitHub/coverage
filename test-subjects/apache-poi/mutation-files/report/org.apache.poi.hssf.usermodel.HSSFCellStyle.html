 69:      *  cases there'll be a fully defined parent.
 70:      */
 71:     public HSSFCellStyle getParentStyle() {
 72:         short parentIndex = _format.getParentIndex();
 73:         // parentIndex equal 0xFFF indicates no inheritance from a cell style XF (See 2.4.353 XF)
 74:     	if(parentIndex == 0 || parentIndex == 0xFFF) {
 75:     		return null;
 76:     	}
 77:     	return new HSSFCellStyle(
 78:     			parentIndex,
 79:     			_workbook.getExFormatAt(parentIndex),
 80:     			_workbook
 81:     	);
 82:     }

286:     /**
287:      * set the degree of rotation for the text in the cell
288:      * @param rotation degrees (between -90 and 90 degrees, of 0xff for vertical)
289:      */
290:     public void setRotation(short rotation)
291:     {
292:       if (rotation == 0xff) {
293:           // Special cases for vertically aligned text
294:       } 
295:       else if ((rotation < 0)&&(rotation >= -90)) {
296:         //Take care of the funny 4th quadrant issue
297:         //The 4th quadrant (-1 to -90) is stored as (91 to 180)
298:         rotation = (short)(90 - rotation);
299:       }
300:       else if ((rotation < -90)  ||(rotation > 90)) {
301:         //Do not allow an incorrect rotation to be set
302:         throw new IllegalArgumentException("The rotation must be between -90 and 90 degrees, or 0xff");
303:       }
304:       _format.setRotation(rotation);
305:     }
306: 
307:     /**
308:      * get the degree of rotation for the text in the cell
309:      * @return rotation degrees (between -90 and 90 degrees, or 0xff for vertical)
310:      */
311:     public short getRotation()
312:     {
313:       short rotation = _format.getRotation();
314:       if (rotation == 0xff) {
315:          // Vertical aligned special case
316:          return rotation;
317:       }
318:       if (rotation > 90) {
319:         //This is actually the 4th quadrant
320:         rotation = (short)(90-rotation);
321:       }
322:       return rotation;
323:     }
