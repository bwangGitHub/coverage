 32: /**
 33:  * Converts the text meta-data file into a <tt>FunctionMetadataRegistry</tt>
 34:  * 
 35:  * @author Josh Micich
 36:  */
 37: final class FunctionMetadataReader {
 38: 
 39: 	private static final String METADATA_FILE_NAME = "functionMetadata.txt";
 40: 	
 41: 	/** plain ASCII text metadata file uses three dots for ellipsis */
 42: 	private static final String ELLIPSIS = "...";
 43: 
 44: 	private static final Pattern TAB_DELIM_PATTERN = Pattern.compile("\t");
 45: 	private static final Pattern SPACE_DELIM_PATTERN = Pattern.compile(" ");
 46: 	private static final byte[] EMPTY_BYTE_ARRAY = { };
 47: 
 48: 	private static final String[] DIGIT_ENDING_FUNCTION_NAMES = {
 49: 		// Digits at the end of a function might be due to a left-over footnote marker.
 50: 		// except in these cases
 51: 		"LOG10", "ATAN2", "DAYS360", "SUMXMY2", "SUMX2MY2", "SUMX2PY2",
 52: 	};
 53: 	private static final Set DIGIT_ENDING_FUNCTION_NAMES_SET = new HashSet(Arrays.asList(DIGIT_ENDING_FUNCTION_NAMES));
 54: 
 55: 	public static FunctionMetadataRegistry createRegistry() {
 56: 		InputStream is = FunctionMetadataReader.class.getResourceAsStream(METADATA_FILE_NAME);
 57: 		if (is == null) {
 58: 			throw new RuntimeException("resource '" + METADATA_FILE_NAME + "' not found");
 59: 		}
 60: 
 61: 		BufferedReader br;
 62: 		try {
 63: 			br = new BufferedReader(new InputStreamReader(is,"UTF-8"));
 64: 		} catch(UnsupportedEncodingException e) {
 65: 			throw new RuntimeException(e);
 66: 		}
 67: 		FunctionDataBuilder fdb = new FunctionDataBuilder(400);
 68: 
 69: 		try {
 70: 			while (true) {
 71: 				String line = br.readLine();
 72: 				if (line == null) {
 73: 					break;
 74: 				}
 75: 				if (line.length() < 1 || line.charAt(0) == '#') {
 76: 					continue;
 77: 				}
 78: 				String trimLine = line.trim();
 79: 				if (trimLine.length() < 1) {
 80: 					continue;
 81: 				}
 82: 				processLine(fdb, line);
 83: 			}
 84: 			br.close();
 85: 		} catch (IOException e) {
 86: 			throw new RuntimeException(e);
 87: 		}
 88: 
 89: 		return fdb.build();
 90: 	}
 91: 
 92: 	private static void processLine(FunctionDataBuilder fdb, String line) {
 93: 
 94: 		String[] parts = TAB_DELIM_PATTERN.split(line, -2);
 95: 		if(parts.length != 8) {
 96: 			throw new RuntimeException("Bad line format '" + line + "' - expected 8 data fields");
 97: 		}
 98: 		int functionIndex = parseInt(parts[0]);
 99: 		String functionName = parts[1];
100: 		int minParams = parseInt(parts[2]);
101: 		int maxParams = parseInt(parts[3]);
102: 		byte returnClassCode = parseReturnTypeCode(parts[4]);
103: 		byte[] parameterClassCodes = parseOperandTypeCodes(parts[5]);
104: 		// 6 isVolatile
105: 		boolean hasNote = parts[7].length() > 0;
106: 
107: 		validateFunctionName(functionName);
108: 		// TODO - make POI use isVolatile
109: 		fdb.add(functionIndex, functionName, minParams, maxParams, 
110: 				returnClassCode, parameterClassCodes, hasNote);
111: 	}
112: 	
113: 
114: 	private static byte parseReturnTypeCode(String code) {
115: 		if(code.length() == 0) {
116: 			return Ptg.CLASS_REF; // happens for GETPIVOTDATA
117: 		}
118: 		return parseOperandTypeCode(code);
119: 	}
120: 
121: 	private static byte[] parseOperandTypeCodes(String codes) {
122: 		if(codes.length() < 1) {
123: 			return EMPTY_BYTE_ARRAY; // happens for GETPIVOTDATA
124: 		}
125: 		if(isDash(codes)) {
126: 			// '-' means empty:
127: 			return EMPTY_BYTE_ARRAY;
128: 		}
129: 		String[] array = SPACE_DELIM_PATTERN.split(codes);
130: 		int nItems = array.length;
131: 		if(ELLIPSIS.equals(array[nItems-1])) {
132: 			// final ellipsis is optional, and ignored
133: 			// (all unspecified params are assumed to be the same as the last)
134: 			nItems --;
135: 		}
136: 		byte[] result = new byte[nItems];
137: 		for (int i = 0; i < nItems; i++) {
138: 			result[i] = parseOperandTypeCode(array[i]);
139: 		}
140: 		return result;
141: 	}
142: 
143: 	private static boolean isDash(String codes) {
144: 		if(codes.length() == 1) {
145: 			switch (codes.charAt(0)) {
146: 				case '-':
147: 					return true;
148: 			}
149: 		}
150: 		return false;
151: 	}
152: 
153: 	private static byte parseOperandTypeCode(String code) {
154: 		if(code.length() != 1) {
155: 			throw new RuntimeException("Bad operand type code format '" + code  + "' expected single char");
156: 		}
157: 		switch(code.charAt(0)) {
158: 			case 'V': return Ptg.CLASS_VALUE;
159: 			case 'R': return Ptg.CLASS_REF;
160: 			case 'A': return Ptg.CLASS_ARRAY;
161: 		}
162: 		throw new IllegalArgumentException("Unexpected operand type code '" + code + "' (" + (int)code.charAt(0) + ")");
163: 	}
164: 
165: 	/**
166: 	 * Makes sure that footnote digits from the original OOO document have not been accidentally 
167: 	 * left behind
168: 	 */
169: 	private static void validateFunctionName(String functionName) {
170: 		int len = functionName.length();
171: 		int ix = len - 1;
172: 		if (!Character.isDigit(functionName.charAt(ix))) {
173: 			return;
174: 		}
175: 		while(ix >= 0) {
176: 			if (!Character.isDigit(functionName.charAt(ix))) {
177: 				break;
178: 			}
179: 			ix--;
180: 		}
181: 		if(DIGIT_ENDING_FUNCTION_NAMES_SET.contains(functionName)) {
182: 			return;
183: 		}
184: 		throw new RuntimeException("Invalid function name '" + functionName 
185: 				+ "' (is footnote number incorrectly appended)");
186: 	}
187: 
188: 	private static int parseInt(String valStr) {
189: 		try {
190: 			return Integer.parseInt(valStr);
191: 		} catch (NumberFormatException e) {
192: 			throw new RuntimeException("Value '" + valStr + "' could not be parsed as an integer");
193: 		}
194: 	}
