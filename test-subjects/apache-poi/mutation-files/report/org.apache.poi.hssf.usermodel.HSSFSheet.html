222:      * Remove a row from this sheet.  All cells contained in the row are removed as well
223:      *
224:      * @param row   representing a row to remove.
225:      */
226:     public void removeRow(Row row) {
227:         HSSFRow hrow = (HSSFRow) row;
228:         if (row.getSheet() != this) {
229:             throw new IllegalArgumentException("Specified row does not belong to this sheet");
230:         }
231:         for(Cell cell : row) {
232:             HSSFCell xcell = (HSSFCell)cell;
233:             if(xcell.isPartOfArrayFormulaGroup()){
234:                 String msg = "Row[rownum="+row.getRowNum()+"] contains cell(s) included in a multi-cell array formula. You cannot change part of an array.";
235:                 xcell.notifyArrayFormulaChanging(msg);
236:             }
237:         }
238: 
239:         if (_rows.size() > 0) {
240:             Integer key = Integer.valueOf(row.getRowNum());
241:             HSSFRow removedRow = _rows.remove(key);
242:             if (removedRow != row) {
243:                 //should not happen if the input argument is valid
244:                 throw new IllegalArgumentException("Specified row does not belong to this sheet");
245:             }
246:             if (hrow.getRowNum() == getLastRowNum())
247:             {
248:                 _lastrow = findLastRow(_lastrow);
249:             }
250:             if (hrow.getRowNum() == getFirstRowNum())
251:             {
252:                 _firstrow = findFirstRow(_firstrow);
253:             }
254:             _sheet.removeRow(hrow.getRowRecord());

258:     /**
259:      * used internally to refresh the "last row" when the last row is removed.
260:      */
261:     private int findLastRow(int lastrow) {
262:         if (lastrow < 1) {
263:             return 0;
264:         }
265:         int rownum = lastrow - 1;
266:         HSSFRow r = getRow(rownum);
267: 
268:         while (r == null && rownum > 0) {
269:             r = getRow(--rownum);
270:         }
271:         if (r == null) {
272:             return 0;
273:         }
274:         return rownum;

1275:      * @param startRow the row to start shifting
1276:      * @param endRow the row to end shifting
1277:      * @param n the number of rows to shift
1278:      * @param copyRowHeight whether to copy the row height during the shift
1279:      * @param resetOriginalRowHeight whether to set the original row's height to the default
1280:      * @param moveComments whether to move comments at the same time as the cells they are attached to
1281:      */
1282:     public void shiftRows(int startRow, int endRow, int n,
1283:             boolean copyRowHeight, boolean resetOriginalRowHeight, boolean moveComments) {
1284:         int s, inc;
1285:         if (n < 0) {
1286:             s = startRow;
1287:             inc = 1;
1288:         } else if (n > 0) {
1289:             s = endRow;
1290:             inc = -1;
1291:         } else {
1292:            // Nothing to do
1293:            return;
1294:         }
1295:         
1296:         NoteRecord[] noteRecs;
1297:         if (moveComments) {
1298:             noteRecs = _sheet.getNoteRecords();
1299:         } else {
1300:             noteRecs = NoteRecord.EMPTY_ARRAY;
1301:         }
1302: 
1303:         shiftMerged(startRow, endRow, n, true);
1304:         _sheet.getPageSettings().shiftRowBreaks(startRow, endRow, n);
1305: 
1306:         for ( int rowNum = s; rowNum >= startRow && rowNum <= endRow && rowNum >= 0 && rowNum < 65536; rowNum += inc ) {
1307:             HSSFRow row = getRow( rowNum );
1308:             // notify all cells in this row that we are going to shift them,
1309:             // it can throw IllegalStateException if the operation is not allowed, for example,
1310:             // if the row contains cells included in a multi-cell array formula
1311:             if(row != null) notifyRowShifting(row);
1312: 
1313:             HSSFRow row2Replace = getRow( rowNum + n );
1314:             if ( row2Replace == null )
1315:                 row2Replace = createRow( rowNum + n );
1316: 
1317: 
1318:             // Remove all the old cells from the row we'll
1319:             //  be writing too, before we start overwriting
1320:             //  any cells. This avoids issues with cells
1321:             //  changing type, and records not being correctly
1322:             //  overwritten
1323:             row2Replace.removeAllCells();
1324: 
1325:             // If this row doesn't exist, nothing needs to
1326:             //  be done for the now empty destination row
1327:             if (row == null) continue; // Nothing to do for this row
1328: 
1329:             // Fix up row heights if required
1330:             if (copyRowHeight) {
1331:                 row2Replace.setHeight(row.getHeight());
1332:             }
1333:             if (resetOriginalRowHeight) {
1334:                 row.setHeight((short)0xff);
1335:             }
1336: 
1337:             // Copy each cell from the source row to
1338:             //  the destination row
1339:             for(Iterator<Cell> cells = row.cellIterator(); cells.hasNext(); ) {
1340:                 HSSFCell cell = (HSSFCell)cells.next();
1341:                 row.removeCell( cell );
1342:                 CellValueRecordInterface cellRecord = cell.getCellValueRecord();
1343:                 cellRecord.setRow( rowNum + n );
1344:                 row2Replace.createCellFromRecord( cellRecord );
1345:                 _sheet.addValueRecord( rowNum + n, cellRecord );
1346: 
1347:                 HSSFHyperlink link = cell.getHyperlink();

1349:                     link.setFirstRow(link.getFirstRow() + n);
1350:                     link.setLastRow(link.getLastRow() + n);
1351:                 }
1352:             }
1353:             // Now zap all the cells in the source row
1354:             row.removeAllCells();
1355: 
1356:             // Move comments from the source row to the
1357:             //  destination row. Note that comments can
1358:             //  exist for cells which are null
1359:             if(moveComments) {
1360:                 // This code would get simpler if NoteRecords could be organised by HSSFRow.
1361:                 for(int i=noteRecs.length-1; i>=0; i--) {
1362:                     NoteRecord nr = noteRecs[i];

1375:         if(n > 0) {

1377:            if ( startRow == _firstrow ) {

1379:               _firstrow = Math.max( startRow + n, 0 );
1380:               for( int i=startRow+1; i < startRow+n; i++ ) {
1381:                  if (getRow(i) != null) {
1382:                     _firstrow = i;
1383:                     break;
1384:                  }
1385:               }
1386:            }
1387:            if ( endRow + n > _lastrow ) {
1388:               _lastrow = Math.min( endRow + n, SpreadsheetVersion.EXCEL97.getLastRowIndex() );
1389:            }
1390:         } else {
1391:            // Rows are moving up
1392:            if ( startRow + n < _firstrow ) {
1393:               _firstrow = Math.max( startRow + n, 0 );
1394:            }
1395:            if ( endRow == _lastrow  ) {
1396:               // Need to walk backward to find the last non-blank row
1397:               _lastrow = Math.min( endRow + n, SpreadsheetVersion.EXCEL97.getLastRowIndex() );
1398:               for (int i=endRow-1; i > endRow+n; i++) {
1399:                  if (getRow(i) != null) {
1400:                     _lastrow = i;
1401:                     break;
1402:                  }
1403:               }
1404:            }
1405:         }
1406: 
1407:         // Update any formulas on this sheet that point to
1408:         //  rows which have been moved
1409:         int sheetIndex = _workbook.getSheetIndex(this);
1410:         short externSheetIndex = _book.checkExternSheet(sheetIndex);
1411:         FormulaShifter shifter = FormulaShifter.createForRowShift(externSheetIndex, startRow, endRow, n);
1412:         _sheet.updateFormulasAfterCellShift(shifter, externSheetIndex);
