 26:  * Temporarily collects <tt>FunctionMetadata</tt> instances for creation of a
 27:  * <tt>FunctionMetadataRegistry</tt>.

 29:  * @author Josh Micich

 31: final class FunctionDataBuilder {
 32: 	private int _maxFunctionIndex;
 33: 	private final Map _functionDataByName;
 34: 	private final Map _functionDataByIndex;
 35: 	/** stores indexes of all functions with footnotes (i.e. whose definitions might change) */
 36: 	private final Set _mutatingFunctionIndexes;
 37: 
 38: 	public FunctionDataBuilder(int sizeEstimate) {
 39: 		_maxFunctionIndex = -1;
 40: 		_functionDataByName = new HashMap(sizeEstimate * 3 / 2);
 41: 		_functionDataByIndex = new HashMap(sizeEstimate * 3 / 2);
 42: 		_mutatingFunctionIndexes = new HashSet();
 43: 	}
 44: 
 45: 	public void add(int functionIndex, String functionName, int minParams, int maxParams,
 46: 			byte returnClassCode, byte[] parameterClassCodes, boolean hasFootnote) {
 47: 		FunctionMetadata fm = new FunctionMetadata(functionIndex, functionName, minParams, maxParams,
 48: 				returnClassCode, parameterClassCodes);
 49: 
 50: 		Integer indexKey = Integer.valueOf(functionIndex);
 51: 
 52: 
 53: 		if(functionIndex > _maxFunctionIndex) {
 54: 			_maxFunctionIndex = functionIndex;
 55: 		}
 56: 		// allow function definitions to change only if both previous and the new items have footnotes
 57: 		FunctionMetadata prevFM;
 58: 		prevFM = (FunctionMetadata) _functionDataByName.get(functionName);
 59: 		if(prevFM != null) {
 60: 			if(!hasFootnote || !_mutatingFunctionIndexes.contains(indexKey)) {
 61: 				throw new RuntimeException("Multiple entries for function name '" + functionName + "'");
 62: 			}
 63: 			_functionDataByIndex.remove(Integer.valueOf(prevFM.getIndex()));
 64: 		}
 65: 		prevFM = (FunctionMetadata) _functionDataByIndex.get(indexKey);
 66: 		if(prevFM != null) {
 67: 			if(!hasFootnote || !_mutatingFunctionIndexes.contains(indexKey)) {
 68: 				throw new RuntimeException("Multiple entries for function index (" + functionIndex + ")");
 69: 			}
 70: 			_functionDataByName.remove(prevFM.getName());
 71: 		}
 72: 		if(hasFootnote) {
 73: 			_mutatingFunctionIndexes.add(indexKey);
 74: 		}
 75: 		_functionDataByIndex.put(indexKey, fm);
 76: 		_functionDataByName.put(functionName, fm);
 77: 	}
 78: 
 79: 	public FunctionMetadataRegistry build() {
 80: 
 81: 		FunctionMetadata[] jumbledArray =  new FunctionMetadata[_functionDataByName.size()];
 82: 		_functionDataByName.values().toArray(jumbledArray);
 83: 		FunctionMetadata[] fdIndexArray = new FunctionMetadata[_maxFunctionIndex+1];
 84: 		for (int i = 0; i < jumbledArray.length; i++) {
 85: 			FunctionMetadata fd = jumbledArray[i];
 86: 			fdIndexArray[fd.getIndex()] = fd;
 87: 		}
 88: 
 89: 		return new FunctionMetadataRegistry(fdIndexArray, _functionDataByName);
 90: 	}
