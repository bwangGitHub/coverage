 32: /**
 33:  * Implementation for the Excel function SUMPRODUCT<p>
 34:  *
 35:  * Syntax : <br/>
 36:  *  SUMPRODUCT ( array1[, array2[, array3[, ...]]])
 37:  *    <table border="0" cellpadding="1" cellspacing="0" summary="Parameter descriptions">
 38:  *      <tr><th>array1, ... arrayN&nbsp;&nbsp;</th><td>typically area references,
 39:  *      possibly cell references or scalar values</td></tr>
 40:  *    </table><br/>
 41:  *
 42:  * Let A<b>n</b><sub>(<b>i</b>,<b>j</b>)</sub> represent the element in the <b>i</b>th row <b>j</b>th column
 43:  * of the <b>n</b>th array<br/>
 44:  * Assuming each array has the same dimensions (W, H), the result is defined as:<br/>
 45:  * SUMPRODUCT = &Sigma;<sub><b>i</b>: 1..H</sub> &nbsp;
 46:  * 	(&nbsp; &Sigma;<sub><b>j</b>: 1..W</sub> &nbsp;
 47:  * 	  (&nbsp; &Pi;<sub><b>n</b>: 1..N</sub>
 48:  * 			A<b>n</b><sub>(<b>i</b>,<b>j</b>)</sub>&nbsp;
 49:  *    )&nbsp;
 50:  *  )
 51:  * </p>
 52:  * @author Josh Micich
 53:  */
 54: public final class Sumproduct implements Function {

 57: 	public ValueEval evaluate(ValueEval[] args, int srcCellRow, int srcCellCol) {
 58: 
 59: 		int maxN = args.length;
 60: 
 61: 		if(maxN < 1) {
 62: 			return ErrorEval.VALUE_INVALID;
 63: 		}
 64: 		ValueEval firstArg = args[0];
 65: 		try {
 66: 			if(firstArg instanceof NumericValueEval) {
 67: 				return evaluateSingleProduct(args);
 68: 			}
 69: 			if(firstArg instanceof RefEval) {
 70: 				return evaluateSingleProduct(args);
 71: 			}
 72: 			if (firstArg instanceof TwoDEval) {
 73: 				TwoDEval ae = (TwoDEval) firstArg;
 74: 				if(ae.isRow() && ae.isColumn()) {
 75: 					return evaluateSingleProduct(args);
 76: 				}
 77: 				return evaluateAreaSumProduct(args);
 78: 			}
 79: 		} catch (EvaluationException e) {
 80: 			return e.getErrorEval();
 81: 		}
 82: 		throw new RuntimeException("Invalid arg type for SUMPRODUCT: ("
 83: 				+ firstArg.getClass().getName() + ")");
 84: 	}

 86: 	private static ValueEval evaluateSingleProduct(ValueEval[] evalArgs) throws EvaluationException {
 87: 		int maxN = evalArgs.length;
 88: 
 89: 		double term = 1D;
 90: 		for(int n=0; n<maxN; n++) {
 91: 			double val = getScalarValue(evalArgs[n]);
 92: 			term *= val;
 93: 		}
 94: 		return new NumberEval(term);
 95: 	}
 96: 
 97: 	private static double getScalarValue(ValueEval arg) throws EvaluationException {
 98: 
 99: 		ValueEval eval;
100: 		if (arg instanceof RefEval) {
101: 			RefEval re = (RefEval) arg;
102: 			eval = re.getInnerValueEval();
103: 		} else {
104: 			eval = arg;
105: 		}
106: 
107: 		if (eval == null) {
108: 			throw new RuntimeException("parameter may not be null");
109: 		}
110: 		if (eval instanceof AreaEval) {
111: 			AreaEval ae = (AreaEval) eval;
112: 			// an area ref can work as a scalar value if it is 1x1
113: 			if(!ae.isColumn() || !ae.isRow()) {
114: 				throw new EvaluationException(ErrorEval.VALUE_INVALID);
115: 			}
116: 			eval = ae.getRelativeValue(0, 0);
117: 		}
118: 
119: 		return getProductTerm(eval, true);
120: 	}
121: 
122: 	private static ValueEval evaluateAreaSumProduct(ValueEval[] evalArgs) throws EvaluationException {
123: 		int maxN = evalArgs.length;
124: 		TwoDEval[] args = new TwoDEval[maxN];
125: 		try {
126: 			System.arraycopy(evalArgs, 0, args, 0, maxN);
127: 		} catch (ArrayStoreException e) {
128: 			// one of the other args was not an AreaRef
129: 			return ErrorEval.VALUE_INVALID;
130: 		}
131: 
132: 
133: 		TwoDEval firstArg = args[0];
134: 
135: 		int height = firstArg.getHeight();
136: 		int width = firstArg.getWidth(); // TODO - junit
137: 
138: 		// first check dimensions
139: 		if (!areasAllSameSize(args, height, width)) {
140: 			// normally this results in #VALUE!,
141: 			// but errors in individual cells take precedence
142: 			for (int i = 1; i < args.length; i++) {
143: 				throwFirstError(args[i]);
144: 			}
145: 			return ErrorEval.VALUE_INVALID;
146: 		}
147: 
148: 		double acc = 0;
149: 
150: 		for (int rrIx=0; rrIx<height; rrIx++) {
151: 			for (int rcIx=0; rcIx<width; rcIx++) {
152: 				double term = 1D;
153: 				for(int n=0; n<maxN; n++) {
154: 					double val = getProductTerm(args[n].getValue(rrIx, rcIx), false);
155: 					term *= val;
156: 				}
157: 				acc += term;
158: 			}
159: 		}
160: 
161: 		return new NumberEval(acc);
162: 	}
163: 
164: 	private static void throwFirstError(TwoDEval areaEval) throws EvaluationException {
165: 		int height = areaEval.getHeight();
166: 		int width = areaEval.getWidth();
167: 		for (int rrIx=0; rrIx<height; rrIx++) {
168: 			for (int rcIx=0; rcIx<width; rcIx++) {
169: 				ValueEval ve = areaEval.getValue(rrIx, rcIx);
170: 				if (ve instanceof ErrorEval) {
171: 					throw new EvaluationException((ErrorEval) ve);
172: 				}
173: 			}
174: 		}
175: 	}
176: 
177: 	private static boolean areasAllSameSize(TwoDEval[] args, int height, int width) {
178: 		for (int i = 0; i < args.length; i++) {
179: 			TwoDEval areaEval = args[i];
180: 			// check that height and width match
181: 			if(areaEval.getHeight() != height) {
182: 				return false;
183: 			}
184: 			if(areaEval.getWidth() != width) {
185: 				return false;
186: 			}
187: 		}
188: 		return true;
189: 	}

192: 	/**
193: 	 * Determines a <code>double</code> value for the specified <code>ValueEval</code>.
194: 	 * @param isScalarProduct <code>false</code> for SUMPRODUCTs over area refs.
195: 	 * @throws EvaluationException if <code>ve</code> represents an error value.
196: 	 * <p/>
197: 	 * Note - string values and empty cells are interpreted differently depending on
198: 	 * <code>isScalarProduct</code>.  For scalar products, if any term is blank or a string, the
199: 	 * error (#VALUE!) is raised.  For area (sum)products, if any term is blank or a string, the
200: 	 * result is zero.
201: 	 */
202: 	private static double getProductTerm(ValueEval ve, boolean isScalarProduct) throws EvaluationException {
203: 
204: 		if(ve instanceof BlankEval || ve == null) {
205: 			// TODO - shouldn't BlankEval.INSTANCE be used always instead of null?
206: 			// null seems to occur when the blank cell is part of an area ref (but not reliably)
207: 			if(isScalarProduct) {
208: 				throw new EvaluationException(ErrorEval.VALUE_INVALID);
209: 			}
210: 			return 0;
211: 		}
212: 
213: 		if(ve instanceof ErrorEval) {
214: 			throw new EvaluationException((ErrorEval)ve);
215: 		}
216: 		if(ve instanceof StringEval) {
217: 			if(isScalarProduct) {
218: 				throw new EvaluationException(ErrorEval.VALUE_INVALID);
219: 			}
220: 			// Note for area SUMPRODUCTs, string values are interpreted as zero
221: 			// even if they would parse as valid numeric values
222: 			return 0;
223: 		}
224: 		if(ve instanceof NumericValueEval) {
225: 			NumericValueEval nve = (NumericValueEval) ve;
226: 			return nve.getNumberValue();
227: 		}
228: 		throw new RuntimeException("Unexpected value eval class ("
229: 				+ ve.getClass().getName() + ")");
230: 	}
