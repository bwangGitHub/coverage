267: 	private ParseNode parseRangeExpression() {
268: 		ParseNode result = parseRangeable();
269: 		boolean hasRange = false;
270: 		while (look == ':') {
271: 			int pos = _pointer;
272: 			GetChar();
273: 			ParseNode nextPart = parseRangeable();
274: 			// Note - no range simplification here. An expr like "A1:B2:C3:D4:E5" should be
275: 			// grouped into area ref pairs like: "(A1:B2):(C3:D4):E5"
276: 			// Furthermore, Excel doesn't seem to simplify
277: 			// expressions like "Sheet1!A1:Sheet1:B2" into "Sheet1!A1:B2"
278: 
279: 			checkValidRangeOperand("LHS", pos, result);
280: 			checkValidRangeOperand("RHS", pos, nextPart);

283: 			result = new ParseNode(RangePtg.instance, children);

304: 	 *  a tMemFunc token is used"

308: 		Ptg token = root.getToken();
309: 		if (token instanceof AbstractFunctionPtg) {
310: 			return true;
311: 		}

313: 			return true;

315: 		if (token instanceof NamePtg || token instanceof NameXPtg) { // 3D refs
316: 			return true;
317: 		}

320: 			// expect RangePtg, but perhaps also UnionPtg, IntersectionPtg etc
321: 			for(ParseNode child : root.getChildren()) {

349: 	/**
350: 	 * @return <code>false</code> if sub-expression represented the specified ParseNode definitely
351: 	 * cannot appear on either side of the range (':') operator
352: 	 */
353: 	private static boolean isValidRangeOperand(ParseNode a) {
354: 		Ptg tkn = a.getToken();
355: 		// Note - order is important for these instance-of checks
356: 		if (tkn instanceof OperandPtg) {
357: 			// notably cell refs and area refs
358: 			return true;
359: 		}
360: 
361: 		// next 2 are special cases of OperationPtg
362: 		if (tkn instanceof AbstractFunctionPtg) {
363: 			AbstractFunctionPtg afp = (AbstractFunctionPtg) tkn;
364: 			byte returnClass = afp.getDefaultOperandClass();
365: 			return Ptg.CLASS_REF == returnClass;
366: 		}
367: 		if (tkn instanceof ValueOperatorPtg) {
368: 			return false;
369: 		}
370: 		if (tkn instanceof OperationPtg) {
371: 			return true;
372: 		}
373: 
374: 		// one special case of ControlPtg
375: 		if (tkn instanceof ParenthesisPtg) {
376: 			// parenthesis Ptg should have only one child
377: 			return isValidRangeOperand(a.getChildren()[0]);
378: 		}
379: 
380: 		// one special case of ScalarConstantPtg
381: 		if (tkn == ErrPtg.REF_INVALID) {
382: 			return true;
383: 		}
384: 
385: 		// All other ControlPtgs and ScalarConstantPtgs cannot be used with ':'
386: 		return false;
387: 	}

524: 	 * i.e. '!', ':'(and equiv '...') do not appear
525: 	 * Examples
526: 	 * <pre>
527: 	 *   my.named...range.

535: 		resetPointer(savePointer);

537: 		if (Character.isDigit(look)) {
538: 			return new ParseNode(parseNumber());
539: 		}
540: 		if (look == '"') {
541: 			return new ParseNode(new StringPtg(parseStringLiteral()));
542: 		}
543: 		// from now on we can only be dealing with non-quoted identifiers
544: 		// which will either be named ranges or functions
545: 		StringBuilder sb = new StringBuilder();

547: 		// defined names may begin with a letter or underscore
548: 		if (!Character.isLetter(look) && look != '_') {
549: 			throw expected("number, string, or defined name");
550: 		}

552: 			sb.append(look);
553: 			GetChar();
554: 		}

557: 		if (look == '(') {
558: 			return function(name);
559: 		}
560: 		if (name.equalsIgnoreCase("TRUE") || name.equalsIgnoreCase("FALSE")) {
561: 			return  new ParseNode(BoolPtg.valueOf(name.equalsIgnoreCase("TRUE")));
562: 		}
563: 		if (_book == null) {
564: 			// Only test cases omit the book (expecting it not to be needed)
565: 			throw new IllegalStateException("Need book to evaluate name '" + name + "'");
566: 		}

568: 		if (evalName == null) {

574: 		}
575: 		// TODO - what about NameX ?
