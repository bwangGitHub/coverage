 26: /**
 27:  * Title: Write Access Record (0x005C)<p/>
 28:  * 
 29:  * Description: Stores the username of that who owns the spreadsheet generator (on unix the user's 
 30:  * login, on Windoze its the name you typed when you installed the thing)
 31:  * <p/>
 32:  * REFERENCE: PG 424 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)
 33:  * <p/>
 34:  * 
 35:  * @author Andrew C. Oliver (acoliver at apache dot org)
 36:  */
 37: public final class WriteAccessRecord extends StandardRecord {
 38: 	public final static short sid = 0x005C;
 39: 
 40: 	private static final byte PAD_CHAR = (byte) ' ';
 41: 	private static final int DATA_SIZE = 112;
 42: 	private String field_1_username;
 43: 	/** this record is always padded to a constant length */
 44: 	private static final byte[] PADDING = new byte[DATA_SIZE];
 45: 	static {
 46: 		Arrays.fill(PADDING, PAD_CHAR);
 47: 	}
 48: 
 49: 	public WriteAccessRecord() {
 50: 		setUsername("");
 51: 	}
 52: 
 53: 	public WriteAccessRecord(RecordInputStream in) {
 54: 		if (in.remaining() > DATA_SIZE) {
 55: 			throw new RecordFormatException("Expected data size (" + DATA_SIZE + ") but got ("
 56: 					+ in.remaining() + ")");
 57: 		}
 58: 		// The string is always 112 characters (padded with spaces), therefore
 59: 		// this record can not be continued.
 60: 
 61: 		int nChars = in.readUShort();
 62: 		int is16BitFlag = in.readUByte();
 63: 		if (nChars > DATA_SIZE || (is16BitFlag & 0xFE) != 0) {
 64: 			// String header looks wrong (probably missing)
 65: 			// OOO doc says this is optional anyway.
 66: 			// reconstruct data
 67: 			byte[] data = new byte[3 + in.remaining()];
 68: 			LittleEndian.putUShort(data, 0, nChars);
 69: 			LittleEndian.putByte(data, 2, is16BitFlag);
 70: 			in.readFully(data, 3, data.length-3);
 71: 			String rawValue = new String(data);
 72: 			setUsername(rawValue.trim());
 73: 			return;
 74: 		}
 75: 
 76: 		String rawText;
 77: 		if ((is16BitFlag & 0x01) == 0x00) {
 78: 			rawText = StringUtil.readCompressedUnicode(in, nChars);
 79: 		} else {
 80: 			rawText = StringUtil.readUnicodeLE(in, nChars);
 81: 		}
 82: 		field_1_username = rawText.trim();
 83: 
 84: 		// consume padding
 85: 		int padSize = in.remaining();
 86: 		while (padSize > 0) {
 87: 			// in some cases this seems to be garbage (non spaces)
 88: 			in.readUByte();
 89: 			padSize--;
 90: 		}
 91: 	}
 92: 
 93: 	/**
 94: 	 * set the username for the user that created the report. HSSF uses the
 95: 	 * logged in user.
 96: 	 * 
 97: 	 * @param username of the user who is logged in (probably "tomcat" or "apache")
 98: 	 */
 99: 	public void setUsername(String username) {
100: 		boolean is16bit = StringUtil.hasMultibyte(username);
101: 		int encodedByteCount = 3 + username.length() * (is16bit ? 2 : 1);
102: 		int paddingSize = DATA_SIZE - encodedByteCount;
103: 		if (paddingSize < 0) {
104: 			throw new IllegalArgumentException("Name is too long: " + username);
105: 		}
106: 
107: 		field_1_username = username;
108: 	}
109: 
110: 	/**
111: 	 * get the username for the user that created the report. HSSF uses the
112: 	 * logged in user. On natively created M$ Excel sheet this would be the name
113: 	 * you typed in when you installed it in most cases.
114: 	 * 
115: 	 * @return username of the user who is logged in (probably "tomcat" or "apache")
116: 	 */
117: 	public String getUsername() {
118: 		return field_1_username;
119: 	}
120: 
121: 	public String toString() {
122: 		StringBuffer buffer = new StringBuffer();
123: 
124: 		buffer.append("[WRITEACCESS]\n");
125: 		buffer.append("    .name = ").append(field_1_username.toString()).append("\n");
126: 		buffer.append("[/WRITEACCESS]\n");
127: 		return buffer.toString();
128: 	}
129: 
130: 	public void serialize(LittleEndianOutput out) {
131: 		String username = getUsername();
132: 		boolean is16bit = StringUtil.hasMultibyte(username);
133: 
134: 		out.writeShort(username.length());
135: 		out.writeByte(is16bit ? 0x01 : 0x00);
136: 		if (is16bit) {
137: 			StringUtil.putUnicodeLE(username, out);
138: 		} else {
139: 			StringUtil.putCompressedUnicode(username, out);
140: 		}
141: 		int encodedByteCount = 3 + username.length() * (is16bit ? 2 : 1);
142: 		int paddingSize = DATA_SIZE - encodedByteCount;
143: 		out.write(PADDING, 0, paddingSize);
144: 	}
145: 
146: 	protected int getDataSize() {
147: 		return DATA_SIZE;
148: 	}
149: 
150: 	public short getSid() {
151: 		return sid;
152: 	}
153: }
