 28:  * @author Werner Froidevaux

 32: public final class StringPtg extends ScalarConstantPtg {
 33:      public final static byte sid = 0x17;
 34:     /** the character (") used in formulas to delimit string literals */
 35:     private static final char FORMULA_DELIMITER = '"';
 36: 
 37:     private final boolean _is16bitUnicode;
 38:     /**
 39:      * NOTE: OO doc says 16bit length, but BiffViewer says 8 Book says something
 40:      * totally different, so don't look there!
 41:      */
 42:     private final String field_3_string;
 43: 
 44:     /** Create a StringPtg from a stream */
 45:     public StringPtg(LittleEndianInput in)  {
 46:     	int nChars = in.readUByte(); // Note - nChars is 8-bit
 47:     	_is16bitUnicode = (in.readByte() & 0x01) != 0;
 48:     	if (_is16bitUnicode) {
 49:     		field_3_string = StringUtil.readUnicodeLE(in, nChars);
 50:     	} else {
 51:     		field_3_string = StringUtil.readCompressedUnicode(in, nChars);
 52:     	}
 53:     }
 54: 
 55:     /**
 56:      * Create a StringPtg from a string representation of the number Number
 57:      * format is not checked, it is expected to be validated in the parser that
 58:      * calls this method.
 59:      * 
 60:      * @param value :
 61:      *            String representation of a floating point number
 62:      */
 63:     public StringPtg(String value) {
 64:         if (value.length() > 255) {
 65:             throw new IllegalArgumentException(
 66:                     "String literals in formulas can't be bigger than 255 characters ASCII");
 67:         }
 68:         _is16bitUnicode = StringUtil.hasMultibyte(value);
 69:         field_3_string = value;
 70:     }
 71: 
 72:     public String getValue() {
 73:         return field_3_string;
 74:     }
 75: 
 76:     public void write(LittleEndianOutput out) {
 77:         out.writeByte(sid + getPtgClass());
 78:         out.writeByte(field_3_string.length()); // Note - nChars is 8-bit
 79:         out.writeByte(_is16bitUnicode ? 0x01 : 0x00);
 80:         if (_is16bitUnicode) {
 81:         	StringUtil.putUnicodeLE(field_3_string, out);
 82:         } else {
 83:         	StringUtil.putCompressedUnicode(field_3_string, out);
 84:         }
 85:     }
 86: 
 87:     public int getSize() {
 88:     	return 3 +  field_3_string.length() * (_is16bitUnicode ? 2 : 1);
 89:     }
 90: 
 91:     public String toFormulaString() {
 92:         String value = field_3_string;
 93:         int len = value.length();
 94:         StringBuffer sb = new StringBuffer(len + 4);
 95:         sb.append(FORMULA_DELIMITER);
 96: 
 97:         for (int i = 0; i < len; i++) {
 98:             char c = value.charAt(i);
 99:             if (c == FORMULA_DELIMITER) {
100:                 sb.append(FORMULA_DELIMITER);
101:             }
102:             sb.append(c);
103:         }
104: 
105:         sb.append(FORMULA_DELIMITER);
106:         return sb.toString();
107:     }
