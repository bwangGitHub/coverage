 36: /**
 37:  * Link Table (OOO pdf reference: 4.10.3 ) <p/>
 38:  *
 39:  * The main data of all types of references is stored in the Link Table inside the Workbook Globals
 40:  * Substream (4.2.5). The Link Table itself is optional and occurs only, if  there are any
 41:  * references in the document.
 42:  *  <p/>
 43:  *
 44:  *  In BIFF8 the Link Table consists of
 45:  *  <ul>
 46:  *  <li>zero or more EXTERNALBOOK Blocks<p/>
 47:  *  	each consisting of
 48:  *  	<ul>
 49:  *  	<li>exactly one EXTERNALBOOK (0x01AE) record</li>
 50:  *  	<li>zero or more EXTERNALNAME (0x0023) records</li>
 51:  *  	<li>zero or more CRN Blocks<p/>
 52:  *			each consisting of
 53:  *  		<ul>
 54:  *  		<li>exactly one XCT (0x0059)record</li>
 55:  *  		<li>zero or more CRN (0x005A) records (documentation says one or more)</li>
 56:  *  		</ul>
 57:  *  	</li>
 58:  *  	</ul>
 59:  *  </li>
 60:  *  <li>zero or one EXTERNSHEET (0x0017) record</li>
 61:  *  <li>zero or more DEFINEDNAME (0x0018) records</li>
 62:  *  </ul>
 63:  *
 64:  *
 65:  * @author Josh Micich
 66:  */

180: 	public LinkTable(List inputList, int startIndex, WorkbookRecordList workbookRecordList, Map<String, NameCommentRecord> commentRecords) {
181: 
182: 		_workbookRecordList = workbookRecordList;
183: 		RecordStream rs = new RecordStream(inputList, startIndex);
184: 
185: 		List<ExternalBookBlock> temp = new ArrayList<ExternalBookBlock>();
186: 		while(rs.peekNextClass() == SupBookRecord.class) {
187: 		   temp.add(new ExternalBookBlock(rs));
188: 		}
189: 
190: 		_externalBookBlocks = new ExternalBookBlock[temp.size()];
191: 		temp.toArray(_externalBookBlocks);
192: 		temp.clear();
193: 
194: 		if (_externalBookBlocks.length > 0) {
195: 			// If any ExternalBookBlock present, there is always 1 of ExternSheetRecord
196: 			if (rs.peekNextClass() != ExternSheetRecord.class) {
197: 				// not quite - if written by google docs
198: 				_externSheetRecord = null;
199: 			} else {
200: 				_externSheetRecord = readExtSheetRecord(rs);
201: 			}
202: 		} else {
203: 			_externSheetRecord = null;
204: 		}
205: 
206: 		_definedNames = new ArrayList<NameRecord>();
207: 		// collect zero or more DEFINEDNAMEs id=0x18,
208: 		//  with their comments if present
209: 		while(true) {
210: 		  Class nextClass = rs.peekNextClass();
211: 		  if (nextClass == NameRecord.class) {
212: 		    NameRecord nr = (NameRecord)rs.getNext();
213: 		    _definedNames.add(nr);
214: 		  }
215: 		  else if (nextClass == NameCommentRecord.class) {
216: 		    NameCommentRecord ncr = (NameCommentRecord)rs.getNext();
217: 		    commentRecords.put(ncr.getNameText(), ncr);
218: 		  }
219: 		  else {
220: 		    break;
221: 		  }
222: 		}
223: 
224: 		_recordCount = rs.getCountRead();
225: 		_workbookRecordList.getRecords().addAll(inputList.subList(startIndex, startIndex + _recordCount));
226: 	}

228: 	private static ExternSheetRecord readExtSheetRecord(RecordStream rs) {
229: 		List<ExternSheetRecord> temp = new ArrayList<ExternSheetRecord>(2);
230: 		while(rs.peekNextClass() == ExternSheetRecord.class) {
231: 			temp.add((ExternSheetRecord) rs.getNext());
232: 		}
233: 
234: 		int nItems = temp.size();
235: 		if (nItems < 1) {
236: 			throw new RuntimeException("Expected an EXTERNSHEET record but got ("
237: 					+ rs.peekNextClass().getName() + ")");
238: 		}
239: 		if (nItems == 1) {
240: 			// this is the normal case. There should be just one ExternSheetRecord
241: 			return temp.get(0);
242: 		}
243: 		// Some apps generate multiple ExternSheetRecords (see bug 45698).
244: 		// It seems like the best thing to do might be to combine these into one
245: 		ExternSheetRecord[] esrs = new ExternSheetRecord[nItems];
246: 		temp.toArray(esrs);
247: 		return ExternSheetRecord.combine(esrs);
248: 	}

490:     /**
491:      * Register an external name in this workbook
492:      *
493:      * @param name  the name to register
494:      * @return a NameXPtg describing this name 
495:      */
496:     public NameXPtg addNameXPtg(String name) {
497:         int extBlockIndex = -1;
498:         ExternalBookBlock extBlock = null;
499: 
500:         // find ExternalBlock for Add-In functions and remember its index
501:         for (int i = 0; i < _externalBookBlocks.length; i++) {
502:             SupBookRecord ebr = _externalBookBlocks[i].getExternalBookRecord();
503:             if (ebr.isAddInFunctions()) {
504:                 extBlock = _externalBookBlocks[i];
505:                 extBlockIndex = i;
506:                 break;
507:             }
508:         }
509:         // An ExternalBlock for Add-In functions was not found. Create a new one.
510:         if (extBlock == null) {
511:             extBlock = new ExternalBookBlock();
512: 
513:             ExternalBookBlock[] tmp = new ExternalBookBlock[_externalBookBlocks.length + 1];
514:             System.arraycopy(_externalBookBlocks, 0, tmp, 0, _externalBookBlocks.length);
515:             tmp[tmp.length - 1] = extBlock;
516:             _externalBookBlocks = tmp;
517: 
518:             extBlockIndex = _externalBookBlocks.length - 1;
519: 
520:             // add the created SupBookRecord before ExternSheetRecord
521:             int idx = findFirstRecordLocBySid(ExternSheetRecord.sid);
522:             _workbookRecordList.add(idx, extBlock.getExternalBookRecord());
523: 
524:             // register the SupBookRecord in the ExternSheetRecord
525:             // -2 means that the scope of this name is Workbook and the reference applies to the entire workbook.
526:             _externSheetRecord.addRef(_externalBookBlocks.length - 1, -2, -2);
527:         }
528: 
529:         // create a ExternalNameRecord that will describe this name
530:         ExternalNameRecord extNameRecord = new ExternalNameRecord();
531:         extNameRecord.setText(name);
532:         // The docs don't explain why Excel set the formula to #REF!
533:         extNameRecord.setParsedExpression(new Ptg[]{ErrPtg.REF_INVALID});
534: 
535:         int nameIndex = extBlock.addExternalName(extNameRecord);
536:         int supLinkIndex = 0;
537:         // find the posistion of the Add-In SupBookRecord in the workbook stream,
538:         // the created ExternalNameRecord will be appended to it
539:         for (Iterator iterator = _workbookRecordList.iterator(); iterator.hasNext(); supLinkIndex++) {
540:             Record record = (Record) iterator.next();
541:             if (record instanceof SupBookRecord) {
542:                 if (((SupBookRecord) record).isAddInFunctions()) break;
543:             }
544:         }
545:         int numberOfNames = extBlock.getNumberOfNames();
546:         // a new name is inserted in the end of the SupBookRecord, after the last name
547:         _workbookRecordList.add(supLinkIndex + numberOfNames, extNameRecord);
548:         int ix = _externSheetRecord.getRefIxForSheet(extBlockIndex, -2 /* the scope is workbook*/);
549:         return new NameXPtg(ix, nameIndex);
550:     }

557: 	 * Changes an external referenced file to another file.
558: 	 * A formular in Excel which refers a cell in another file is saved in two parts: 
559: 	 * The referenced file is stored in an reference table. the row/cell information is saved separate.
560: 	 * This method invokation will only change the reference in the lookup-table itself.
561: 	 * @param oldUrl The old URL to search for and which is to be replaced
562: 	 * @param newUrl The URL replacement
563: 	 * @return true if the oldUrl was found and replaced with newUrl. Otherwise false
564: 	 */
565: 	public boolean changeExternalReference(String oldUrl, String newUrl) {
566: 		for(ExternalBookBlock ex : _externalBookBlocks) {
567: 			SupBookRecord externalRecord = ex.getExternalBookRecord();
568: 			if (externalRecord.isExternalReferences() 
569: 				&& externalRecord.getURL().equals(oldUrl)) {
570: 				
571: 				externalRecord.setURL(newUrl);
572: 				return true;
573: 			}
574: 		}
575: 		return false;
576: 	}
