254: 	/**
255: 	 * @return never <code>null</code>, never {@link BlankEval}
256: 	 */
257: 	private ValueEval evaluateAny(EvaluationCell srcCell, int sheetIndex,
258: 				int rowIndex, int columnIndex, EvaluationTracker tracker) {
259: 
260: 		// avoid tracking dependencies to cells that have constant definition
261: 		boolean shouldCellDependencyBeRecorded = _stabilityClassifier == null ? true
262: 					: !_stabilityClassifier.isCellFinal(sheetIndex, rowIndex, columnIndex);
263: 		if (srcCell == null || srcCell.getCellType() != Cell.CELL_TYPE_FORMULA) {
264: 			ValueEval result = getValueFromNonFormulaCell(srcCell);
265: 			if (shouldCellDependencyBeRecorded) {
266: 				tracker.acceptPlainValueDependency(_workbookIx, sheetIndex, rowIndex, columnIndex, result);
267: 			}
268: 			return result;
269: 		}
270: 
271: 		FormulaCellCacheEntry cce = _cache.getOrCreateFormulaCellEntry(srcCell);
272: 		if (shouldCellDependencyBeRecorded || cce.isInputSensitive()) {
273: 			tracker.acceptFormulaDependency(cce);
274: 		}
275: 		IEvaluationListener evalListener = _evaluationListener;
276: 		ValueEval result;
277: 		if (cce.getValue() == null) {
278: 			if (!tracker.startEvaluate(cce)) {
279: 				return ErrorEval.CIRCULAR_REF_ERROR;
280: 			}
281: 			OperationEvaluationContext ec = new OperationEvaluationContext(this, _workbook, sheetIndex, rowIndex, columnIndex, tracker);
282: 
283: 			try {
284: 
285: 				Ptg[] ptgs = _workbook.getFormulaTokens(srcCell);
286: 				if (evalListener == null) {
287: 					result = evaluateFormula(ec, ptgs);
288: 				} else {
289: 					evalListener.onStartEvaluate(srcCell, cce);
290: 					result = evaluateFormula(ec, ptgs);
291: 					evalListener.onEndEvaluate(cce, result);
292: 				}
293: 
294: 				tracker.updateCacheResult(result);
295: 			}
296: 			 catch (NotImplementedException e) {
297: 				throw addExceptionInfo(e, sheetIndex, rowIndex, columnIndex);
298: 			 } catch (RuntimeException re) {
299: 				 if (re.getCause() instanceof WorkbookNotFoundException && _ignoreMissingWorkbooks) {
300:  					logInfo(re.getCause().getMessage() + " - Continuing with cached value!");
301:  					switch(srcCell.getCachedFormulaResultType()) {
302: 	 					case Cell.CELL_TYPE_NUMERIC:
303: 	 						result = new NumberEval(srcCell.getNumericCellValue());
304: 	 						break;
305: 	 					case Cell.CELL_TYPE_STRING:
306: 	 						result =  new StringEval(srcCell.getStringCellValue());
307: 	 						break;
308: 	 					case Cell.CELL_TYPE_BLANK:
309: 	 						result = BlankEval.instance;
310: 	 						break;
311: 	 					case Cell.CELL_TYPE_BOOLEAN:
312: 	 						result =  BoolEval.valueOf(srcCell.getBooleanCellValue());
313: 	 						break;
314: 	 					case Cell.CELL_TYPE_ERROR:
315: 							result =  ErrorEval.valueOf(srcCell.getErrorCellValue());
316: 							break;
317: 	 					case Cell.CELL_TYPE_FORMULA:
318: 						default:
319: 							throw new RuntimeException("Unexpected cell type '" + srcCell.getCellType()+"' found!");
320:  					}
321: 				 } else {
322: 					 throw re;
323: 				 }
324: 			 } finally {
325: 				tracker.endEvaluate(cce);
326: 			}
327: 		} else {
328: 			if(evalListener != null) {
329: 				evalListener.onCacheHit(sheetIndex, rowIndex, columnIndex, cce.getValue());
330: 			}
331: 			return cce.getValue();
332: 		}
333: 		if (isDebugLogEnabled()) {
334: 			String sheetName = getSheetName(sheetIndex);
335: 			CellReference cr = new CellReference(rowIndex, columnIndex);
336: 			logDebug("Evaluated " + sheetName + "!" + cr.formatAsString() + " to " + result.toString());
337: 		}
338: 		// Usually (result === cce.getValue())
339: 		// But sometimes: (result==ErrorEval.CIRCULAR_REF_ERROR, cce.getValue()==null)
340: 		// When circular references are detected, the cache entry is only updated for
341: 		// the top evaluation frame
342: 		return result;
343: 	}

388: 	/* package */ ValueEval evaluateFormula(OperationEvaluationContext ec, Ptg[] ptgs) {
389: 
390: 		Stack<ValueEval> stack = new Stack<ValueEval>();
391: 		for (int i = 0, iSize = ptgs.length; i < iSize; i++) {
392: 
393: 			// since we don't know how to handle these yet :(
394: 			Ptg ptg = ptgs[i];
395: 			if (ptg instanceof AttrPtg) {
396: 				AttrPtg attrPtg = (AttrPtg) ptg;
397: 				if (attrPtg.isSum()) {
398: 					// Excel prefers to encode 'SUM()' as a tAttr token, but this evaluator
399: 					// expects the equivalent function token
400: 					ptg = FuncVarPtg.SUM;
401: 				}
402: 				if (attrPtg.isOptimizedChoose()) {
403: 					ValueEval arg0 = stack.pop();
404: 					int[] jumpTable = attrPtg.getJumpTable();
405: 					int dist;
406: 					int nChoices = jumpTable.length;
407: 					try {
408: 						int switchIndex = Choose.evaluateFirstArg(arg0, ec.getRowIndex(), ec.getColumnIndex());
409: 						if (switchIndex<1 || switchIndex > nChoices) {
410: 							stack.push(ErrorEval.VALUE_INVALID);
411: 							dist = attrPtg.getChooseFuncOffset() + 4; // +4 for tFuncFar(CHOOSE)
412: 						} else {
413: 							dist = jumpTable[switchIndex-1];
414: 						}
415: 					} catch (EvaluationException e) {
416: 						stack.push(e.getErrorEval());
417: 						dist = attrPtg.getChooseFuncOffset() + 4; // +4 for tFuncFar(CHOOSE)
418: 					}
419: 					// Encoded dist for tAttrChoose includes size of jump table, but
420: 					// countTokensToBeSkipped() does not (it counts whole tokens).
421: 					dist -= nChoices*2+2; // subtract jump table size
422: 					i+= countTokensToBeSkipped(ptgs, i, dist);
423: 					continue;
424: 				}
425: 				if (attrPtg.isOptimizedIf()) {
426: 					ValueEval arg0 = stack.pop();
427: 					boolean evaluatedPredicate;
428: 					try {
429: 						evaluatedPredicate = IfFunc.evaluateFirstArg(arg0, ec.getRowIndex(), ec.getColumnIndex());
430: 					} catch (EvaluationException e) {
431: 						stack.push(e.getErrorEval());
432: 						int dist = attrPtg.getData();
433: 						i+= countTokensToBeSkipped(ptgs, i, dist);
434: 						attrPtg = (AttrPtg) ptgs[i];
435: 						dist = attrPtg.getData()+1;
436: 						i+= countTokensToBeSkipped(ptgs, i, dist);
437: 						continue;
438: 					}
439: 					if (evaluatedPredicate) {
440: 						// nothing to skip - true param folows
441: 					} else {
442: 						int dist = attrPtg.getData();
443: 						i+= countTokensToBeSkipped(ptgs, i, dist);
444: 						Ptg nextPtg = ptgs[i+1];
445: 						if (ptgs[i] instanceof AttrPtg && nextPtg instanceof FuncVarPtg) {
446: 							// this is an if statement without a false param (as opposed to MissingArgPtg as the false param)
447: 							i++;
448: 							stack.push(BoolEval.FALSE);
449: 						}
450: 					}
451: 					continue;
452: 				}
453: 				if (attrPtg.isSkip()) {
454: 					int dist = attrPtg.getData()+1;
455: 					i+= countTokensToBeSkipped(ptgs, i, dist);
456: 					if (stack.peek() == MissingArgEval.instance) {
457: 						stack.pop();
458: 						stack.push(BlankEval.instance);
459: 					}
460: 					continue;
461: 				}
462: 			}
463: 			if (ptg instanceof ControlPtg) {
464: 				// skip Parentheses, Attr, etc
465: 				continue;
466: 			}
467: 			if (ptg instanceof MemFuncPtg || ptg instanceof MemAreaPtg) {
468: 				// can ignore, rest of tokens for this expression are in OK RPN order
469: 				continue;
470: 			}
471: 			if (ptg instanceof MemErrPtg) {
472: 				continue;
473: 			}
474: 
475: 			ValueEval opResult;
476: 			if (ptg instanceof OperationPtg) {
477: 				OperationPtg optg = (OperationPtg) ptg;
478: 
479: 				if (optg instanceof UnionPtg) { continue; }
480: 
481: 
482: 				int numops = optg.getNumberOfOperands();
483: 				ValueEval[] ops = new ValueEval[numops];
484: 
485: 				// storing the ops in reverse order since they are popping
486: 				for (int j = numops - 1; j >= 0; j--) {
487: 					ValueEval p = stack.pop();
488: 					ops[j] = p;
489: 				}
490: //				logDebug("invoke " + operation + " (nAgs=" + numops + ")");
491: 				opResult = OperationEvaluatorFactory.evaluate(optg, ops, ec);
492: 			} else {
493: 				opResult = getEvalForPtg(ptg, ec);
494: 			}
495: 			if (opResult == null) {
496: 				throw new RuntimeException("Evaluation result must not be null");
497: 			}
498: //			logDebug("push " + opResult);
499: 			stack.push(opResult);
500: 		}
501: 
502: 		ValueEval value = stack.pop();
503: 		if (!stack.isEmpty()) {
504: 			throw new IllegalStateException("evaluation stack not empty");
505: 		}
506: 		return dereferenceResult(value, ec.getRowIndex(), ec.getColumnIndex());
507: 	}

509: 	/**
510: 	 * Calculates the number of tokens that the evaluator should skip upon reaching a tAttrSkip.
511: 	 *
512: 	 * @return the number of tokens (starting from <tt>startIndex+1</tt>) that need to be skipped
513: 	 * to achieve the specified <tt>distInBytes</tt> skip distance.
514: 	 */
515: 	private static int countTokensToBeSkipped(Ptg[] ptgs, int startIndex, int distInBytes) {
516: 		int remBytes = distInBytes;
517: 		int index = startIndex;
518: 		while (remBytes != 0) {
519: 			index++;
520: 			remBytes -= ptgs[index].getSize();
521: 			if (remBytes < 0) {
522: 				throw new RuntimeException("Bad skip distance (wrong token size calculation).");
523: 			}
524: 			if (index >= ptgs.length) {
525: 				throw new RuntimeException("Skip distance too far (ran out of formula tokens).");
526: 			}
527: 		}
528: 		return index-startIndex;
529: 	}

540: 	public static ValueEval dereferenceResult(ValueEval evaluationResult, int srcRowNum, int srcColNum) {
541: 		ValueEval value;

558: 	 * returns an appropriate Eval impl instance for the Ptg. The Ptg must be
559: 	 * one of: Area3DPtg, AreaPtg, ReferencePtg, Ref3DPtg, IntPtg, NumberPtg,
560: 	 * StringPtg, BoolPtg <br/>special Note: OperationPtg subtypes cannot be
561: 	 * passed here!
562: 	 */
563: 	private ValueEval getEvalForPtg(Ptg ptg, OperationEvaluationContext ec) {
564: 		//  consider converting all these (ptg instanceof XxxPtg) expressions to (ptg.getClass() == XxxPtg.class)
565: 
566: 		if (ptg instanceof NamePtg) {
567: 			// named ranges, macro functions
568: 			NamePtg namePtg = (NamePtg) ptg;
569: 			EvaluationName nameRecord = _workbook.getName(namePtg);
570: 			if (nameRecord.isFunctionName()) {
571: 				return new NameEval(nameRecord.getNameText());
572: 			}
573: 			if (nameRecord.hasFormula()) {
574: 				return evaluateNameFormula(nameRecord.getNameDefinition(), ec);
575: 			}
576: 
577: 			throw new RuntimeException("Don't now how to evalate name '" + nameRecord.getNameText() + "'");
578: 		}
579: 		if (ptg instanceof NameXPtg) {
580: 		   return ec.getNameXEval(((NameXPtg) ptg));
581: 		}
582: 
583: 		if (ptg instanceof IntPtg) {
584: 			return new NumberEval(((IntPtg)ptg).getValue());
585: 		}
586: 		if (ptg instanceof NumberPtg) {
587: 			return new NumberEval(((NumberPtg)ptg).getValue());
588: 		}
589: 		if (ptg instanceof StringPtg) {
590: 			return new StringEval(((StringPtg) ptg).getValue());
591: 		}
592: 		if (ptg instanceof BoolPtg) {
593: 			return BoolEval.valueOf(((BoolPtg) ptg).getValue());
594: 		}
595: 		if (ptg instanceof ErrPtg) {
596: 			return ErrorEval.valueOf(((ErrPtg) ptg).getErrorCode());
597: 		}
598: 		if (ptg instanceof MissingArgPtg) {
599: 			return MissingArgEval.instance;
600: 		}
601: 		if (ptg instanceof AreaErrPtg ||ptg instanceof RefErrorPtg
602: 				|| ptg instanceof DeletedArea3DPtg || ptg instanceof DeletedRef3DPtg) {
603: 				return ErrorEval.REF_INVALID;
604: 		}
605: 		if (ptg instanceof Ref3DPtg) {
606: 			Ref3DPtg rptg = (Ref3DPtg) ptg;
607: 			return ec.getRef3DEval(rptg.getRow(), rptg.getColumn(), rptg.getExternSheetIndex());
608: 		}
609: 		if (ptg instanceof Area3DPtg) {
610: 			Area3DPtg aptg = (Area3DPtg) ptg;
611: 			return ec.getArea3DEval(aptg.getFirstRow(), aptg.getFirstColumn(), aptg.getLastRow(), aptg.getLastColumn(), aptg.getExternSheetIndex());
612: 		}
613: 		if (ptg instanceof RefPtg) {
614: 			RefPtg rptg = (RefPtg) ptg;
615: 			return ec.getRefEval(rptg.getRow(), rptg.getColumn());
616: 		}
617: 		if (ptg instanceof AreaPtg) {
618: 			AreaPtg aptg = (AreaPtg) ptg;
619: 			return ec.getAreaEval(aptg.getFirstRow(), aptg.getFirstColumn(), aptg.getLastRow(), aptg.getLastColumn());
620: 		}
621: 
622: 		if (ptg instanceof UnknownPtg) {
623: 			// POI uses UnknownPtg when the encoded Ptg array seems to be corrupted.
624: 			// This seems to occur in very rare cases (e.g. unused name formulas in bug 44774, attachment 21790)
625: 			// In any case, formulas are re-parsed before execution, so UnknownPtg should not get here
626: 			throw new RuntimeException("UnknownPtg not allowed");
627: 		}
628: 		if (ptg instanceof ExpPtg) {
629: 			// ExpPtg is used for array formulas and shared formulas.
630: 			// it is currently unsupported, and may not even get implemented here
631: 			throw new RuntimeException("ExpPtg currently not supported");
632: 		}
633: 
634: 		throw new RuntimeException("Unexpected ptg class (" + ptg.getClass().getName() + ")");
635: 	}
