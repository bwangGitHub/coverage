143:     /**
144:      * <p>Creates a {@link Section} instance from a byte array.</p>
145:      *
146:      * @param src Contains the complete property set stream.
147:      * @param offset The position in the stream that points to the
148:      * section's format ID.
149:      *
150:      * @exception UnsupportedEncodingException if the section's codepage is not
151:      * supported.
152:      */
153:     public Section(final byte[] src, final int offset)
154:     throws UnsupportedEncodingException
155:     {
156:         int o1 = offset;
157: 
158:         /*
159:          * Read the format ID.
160:          */
161:         formatID = new ClassID(src, o1);
162:         o1 += ClassID.LENGTH;
163: 
164:         /*
165:          * Read the offset from the stream's start and positions to
166:          * the section header.
167:          */
168:         this.offset = LittleEndian.getUInt(src, o1);
169:         o1 = (int) this.offset;
170: 
171:         /*
172:          * Read the section length.
173:          */
174:         size = (int) LittleEndian.getUInt(src, o1);
175:         o1 += LittleEndian.INT_SIZE;
176: 
177:         /*
178:          * Read the number of properties.
179:          */
180:         final int propertyCount = (int) LittleEndian.getUInt(src, o1);
181:         o1 += LittleEndian.INT_SIZE;
182: 
183:         /*
184:          * Read the properties. The offset is positioned at the first
185:          * entry of the property list. There are two problems:
186:          *
187:          * 1. For each property we have to find out its length. In the
188:          *    property list we find each property's ID and its offset relative
189:          *    to the section's beginning. Unfortunately the properties in the
190:          *    property list need not to be in ascending order, so it is not
191:          *    possible to calculate the length as
192:          *    (offset of property(i+1) - offset of property(i)). Before we can
193:          *    that we first have to sort the property list by ascending offsets.
194:          *
195:          * 2. We have to read the property with ID 1 before we read other
196:          *    properties, at least before other properties containing strings.
197:          *    The reason is that property 1 specifies the codepage. If it is
198:          *    1200, all strings are in Unicode. In other words: Before we can
199:          *    read any strings we have to know whether they are in Unicode or
200:          *    not. Unfortunately property 1 is not guaranteed to be the first in
201:          *    a section.
202:          *
203:          *    The algorithm below reads the properties in two passes: The first
204:          *    one looks for property ID 1 and extracts the codepage number. The
205:          *    seconds pass reads the other properties.
206:          */
207:         properties = new Property[propertyCount];
208: 
209:         /* Pass 1: Read the property list. */
210:         int pass1Offset = o1;
211:         final List<PropertyListEntry> propertyList = new ArrayList<PropertyListEntry>(propertyCount);
212:         PropertyListEntry ple;
213:         for (int i = 0; i < properties.length; i++)
214:         {
215:             ple = new PropertyListEntry();
216: 
217:             /* Read the property ID. */
218:             ple.id = (int) LittleEndian.getUInt(src, pass1Offset);
219:             pass1Offset += LittleEndian.INT_SIZE;
220: 
221:             /* Offset from the section's start. */
222:             ple.offset = (int) LittleEndian.getUInt(src, pass1Offset);
223:             pass1Offset += LittleEndian.INT_SIZE;
224: 
225:             /* Add the entry to the property list. */
226:             propertyList.add(ple);
227:         }
228: 
229:         /* Sort the property list by ascending offsets: */
230:         Collections.sort(propertyList);
231: 
232:         /* Calculate the properties' lengths. */
233:         for (int i = 0; i < propertyCount - 1; i++)
234:         {
235:             PropertyListEntry ple1 = propertyList.get(i);
236:             PropertyListEntry ple2 = propertyList.get(i + 1);
237:             ple1.length = ple2.offset - ple1.offset;
238:         }
239:         if (propertyCount > 0)
240:         {
241:             ple = propertyList.get(propertyCount - 1);
242:             ple.length = size - ple.offset;
243:         }
244: 
245:         /* Look for the codepage. */
246:         int codepage = -1;
247:         for (final Iterator<PropertyListEntry> i = propertyList.iterator();
248:              codepage == -1 && i.hasNext();)
249:         {
250:             ple = i.next();
251: 
252:             /* Read the codepage if the property ID is 1. */
253:             if (ple.id == PropertyIDMap.PID_CODEPAGE)
254:             {
255:                 /* Read the property's value type. It must be
256:                  * VT_I2. */
257:                 int o = (int) (this.offset + ple.offset);
258:                 final long type = LittleEndian.getUInt(src, o);
259:                 o += LittleEndian.INT_SIZE;
260: 
261:                 if (type != Variant.VT_I2)
262:                     throw new HPSFRuntimeException
263:                         ("Value type of property ID 1 is not VT_I2 but " +
264:                          type + ".");
265: 
266:                 /* Read the codepage number. */
267:                 codepage = LittleEndian.getUShort(src, o);
268:             }
269:         }
270: 
271:         /* Pass 2: Read all properties - including the codepage property,
272:          * if available. */
273:         int i1 = 0;
274:         for (final Iterator<PropertyListEntry> i = propertyList.iterator(); i.hasNext();)
275:         {
276:             ple = i.next();
277:             Property p = new Property(ple.id, src,
278:                     this.offset + ple.offset,
279:                     ple.length, codepage);
280:             if (p.getID() == PropertyIDMap.PID_CODEPAGE)
281:                 p = new Property(p.getID(), p.getType(), Integer.valueOf(codepage));
282:             properties[i1++] = p;
283:         }
284: 
285:         /*
286:          * Extract the dictionary (if available).
287:          */
288:         dictionary = (Map) getProperty(0);
289:     }

462:      * <p>Checks whether this section is equal to another object. The result is
463:      * <code>false</code> if one of the the following conditions holds:</p>
464:      *
465:      * <ul>
466:      *
467:      * <li><p>The other object is not a {@link Section}.</p></li>
468:      *
469:      * <li><p>The format IDs of the two sections are not equal.</p></li>
470:      *
471:      * <li><p>The sections have a different number of properties. However,
472:      * properties with ID 1 (codepage) are not counted.</p></li>
473:      *
474:      * <li><p>The other object is not a {@link Section}.</p></li>
475:      *
476:      * <li><p>The properties have different values. The order of the properties
477:      * is irrelevant.</p></li>
478:      *
479:      * </ul>
480:      *
481:      * @param o The object to compare this section with
482:      * @return <code>true</code> if the objects are equal, <code>false</code> if
483:      * not
484:      */
485:     public boolean equals(final Object o)
486:     {
487:         if (o == null || !(o instanceof Section))
488:             return false;
489:         final Section s = (Section) o;
490:         if (!s.getFormatID().equals(getFormatID()))
491:             return false;
492: 
493:         /* Compare all properties except 0 and 1 as they must be handled
494:          * specially. */
495:         Property[] pa1 = new Property[getProperties().length];
496:         Property[] pa2 = new Property[s.getProperties().length];
497:         System.arraycopy(getProperties(), 0, pa1, 0, pa1.length);
498:         System.arraycopy(s.getProperties(), 0, pa2, 0, pa2.length);
499: 
500:         /* Extract properties 0 and 1 and remove them from the copy of the
501:          * arrays. */
502:         Property p10 = null;
503:         Property p20 = null;
504:         for (int i = 0; i < pa1.length; i++)
505:         {
506:             final long id = pa1[i].getID();
507:             if (id == 0)
508:             {
509:                 p10 = pa1[i];
510:                 pa1 = remove(pa1, i);
511:                 i--;
512:             }
513:             if (id == 1)
514:             {
515:                 // p11 = pa1[i];
516:                 pa1 = remove(pa1, i);
517:                 i--;
518:             }
519:         }
520:         for (int i = 0; i < pa2.length; i++)
521:         {
522:             final long id = pa2[i].getID();
523:             if (id == 0)
524:             {
525:                 p20 = pa2[i];
526:                 pa2 = remove(pa2, i);
527:                 i--;
528:             }
529:             if (id == 1)
530:             {
531:                 // p21 = pa2[i];
532:                 pa2 = remove(pa2, i);
533:                 i--;
534:             }
535:         }
536: 
537:         /* If the number of properties (not counting property 1) is unequal the
538:          * sections are unequal. */
539:         if (pa1.length != pa2.length)
540:             return false;
541: 
542:         /* If the dictionaries are unequal the sections are unequal. */
543:         boolean dictionaryEqual = true;
544:         if (p10 != null && p20 != null)
545:             dictionaryEqual = p10.getValue().equals(p20.getValue());
546:         else if (p10 != null || p20 != null)
547:             dictionaryEqual = false;
548:         if (dictionaryEqual) {
549:             return Util.equals(pa1, pa2);
550:         }
551:         return false;
552:     }

561:      * @param i The index of the field to be removed.
562:      * @return the compactified array.

564:     private Property[] remove(final Property[] pa, final int i)
565:     {
566:         final Property[] h = new Property[pa.length - 1];
567:         if (i > 0)

569:         System.arraycopy(pa, i + 1, h, i, h.length - i);
