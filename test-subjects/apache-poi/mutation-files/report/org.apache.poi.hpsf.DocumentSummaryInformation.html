568:     /**
569:      * <p>Gets the custom properties.</p>
570:      *
571:      * @return The custom properties.
572:      */
573:     public CustomProperties getCustomProperties()
574:     {
575:         CustomProperties cps = null;
576:         if (getSectionCount() >= 2)
577:         {
578:             cps = new CustomProperties();
579:             final Section section = (Section) getSections().get(1);
580:             final Map<Long,String> dictionary = section.getDictionary();
581:             final Property[] properties = section.getProperties();
582:             int propertyCount = 0;
583:             for (int i = 0; i < properties.length; i++)
584:             {
585:                 final Property p = properties[i];
586:                 final long id = p.getID();
587:                 if (id != 0 && id != 1)
588:                 {
589:                     propertyCount++;
590:                     final CustomProperty cp = new CustomProperty(p,
591:                             dictionary.get(Long.valueOf(id)));
592:                     cps.put(cp.getName(), cp);
593:                 }
594:             }
595:             if (cps.size() != propertyCount)
596:                 cps.setPure(false);
597:         }
598:         return cps;
599:     }
600: 
601:     /**
602:      * <p>Sets the custom properties.</p>
603:      *
604:      * @param customProperties The custom properties
605:      */
606:     public void setCustomProperties(final CustomProperties customProperties)
607:     {
608:         ensureSection2();
609:         final MutableSection section = (MutableSection) getSections().get(1);
610:         final Map<Long,String> dictionary = customProperties.getDictionary();
611:         section.clear();
612: 
613:         /* Set the codepage. If both custom properties and section have a
614:          * codepage, the codepage from the custom properties wins, else take the
615:          * one that is defined. If none is defined, take Unicode. */
616:         int cpCodepage = customProperties.getCodepage();
617:         if (cpCodepage < 0)
618:             cpCodepage = section.getCodepage();
619:         if (cpCodepage < 0)
620:             cpCodepage = Constants.CP_UNICODE;
621:         customProperties.setCodepage(cpCodepage);
622:         section.setCodepage(cpCodepage);
623:         section.setDictionary(dictionary);
624:         for (final Iterator<CustomProperty> i = customProperties.values().iterator(); i.hasNext();)
625:         {
626:             final Property p = i.next();
627:             section.setProperty(p);
628:         }
629:     }
