266:      * Register a new function in runtime.
267:      *
268:      * @param name  the function name
269:      * @param func  the functoin to register
270:      * @throws IllegalArgumentException if the function is unknown or already  registered.
271:      * @since 3.8 beta6
272:      */
273:     public static void registerFunction(String name, Function func){
274:         FunctionMetadata metaData = FunctionMetadataRegistry.getFunctionByName(name);
275:         if(metaData == null) {
276:             if(AnalysisToolPak.isATPFunction(name)) {
277:                 throw new IllegalArgumentException(name + " is a function from the Excel Analysis Toolpack. " +
278:                         "Use AnalysisToolpack.registerFunction(String name, FreeRefFunction func) instead.");
279:             } else {
280:                 throw new IllegalArgumentException("Unknown function: " + name);
281:             }
282:         }
283: 
284:         int idx = metaData.getIndex();
285:         if(functions[idx] instanceof NotImplementedFunction) {
286:             functions[idx] = func;
287:         } else {
288:             throw new IllegalArgumentException("POI already implememts " + name +
289:                     ". You cannot override POI's implementations of Excel functions");
290:         }
291:     }
