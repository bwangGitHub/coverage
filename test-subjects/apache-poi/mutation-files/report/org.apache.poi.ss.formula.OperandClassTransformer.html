 31: /**
 32:  * This class performs 'operand class' transformation. Non-base tokens are classified into three
 33:  * operand classes:
 34:  * <ul>
 35:  * <li>reference</li>
 36:  * <li>value</li>
 37:  * <li>array</li>
 38:  * </ul>
 39:  * <p/>
 40:  *
 41:  * The final operand class chosen for each token depends on the formula type and the token's place
 42:  * in the formula. If POI gets the operand class wrong, Excel <em>may</em> interpret the formula
 43:  * incorrectly.  This condition is typically manifested as a formula cell that displays as '#VALUE!',
 44:  * but resolves correctly when the user presses F2, enter.<p/>
 45:  *
 46:  * The logic implemented here was partially inspired by the description in
 47:  * "OpenOffice.org's Documentation of the Microsoft Excel File Format".  The model presented there
 48:  * seems to be inconsistent with observed Excel behaviour (These differences have not been fully
 49:  * investigated). The implementation in this class has been heavily modified in order to satisfy
 50:  * concrete examples of how Excel performs the same logic (see TestRVA).<p/>
 51:  *
 52:  * Hopefully, as additional important test cases are identified and added to the test suite,
 53:  * patterns might become more obvious in this code and allow for simplification.
 54:  *
 55:  * @author Josh Micich
 56:  */

 65: 	/**
 66: 	 * Traverses the supplied formula parse tree, calling <tt>Ptg.setClass()</tt> for each non-base
 67: 	 * token to set its operand class.
 68: 	 */
 69: 	public void transformFormula(ParseNode rootNode) {
 70: 		byte rootNodeOperandClass;
 71: 		switch (_formulaType) {
 72: 			case FormulaType.CELL:
 73: 				rootNodeOperandClass = Ptg.CLASS_VALUE;
 74: 				break;
 75: 			case FormulaType.ARRAY:
 76: 				rootNodeOperandClass = Ptg.CLASS_ARRAY;
 77: 				break;
 78: 			case FormulaType.NAMEDRANGE:
 79: 			case FormulaType.DATAVALIDATION_LIST:
 80: 				rootNodeOperandClass = Ptg.CLASS_REF;
 81: 				break;
 82: 			default:
 83: 				throw new RuntimeException("Incomplete code - formula type ("
 84: 						+ _formulaType + ") not supported yet");
 85: 
 86: 		}
 87: 		transformNode(rootNode, rootNodeOperandClass, false);
 88: 	}
 89: 
 90: 	/**
 91: 	 * @param callerForceArrayFlag <code>true</code> if one of the current node's parents is a
 92: 	 * function Ptg which has been changed from default 'V' to 'A' type (due to requirements on
 93: 	 * the function return value).
 94: 	 */
 95: 	private void transformNode(ParseNode node, byte desiredOperandClass,
 96: 			boolean callerForceArrayFlag) {
 97: 		Ptg token = node.getToken();
 98: 		ParseNode[] children = node.getChildren();
 99: 		boolean isSimpleValueFunc = isSimpleValueFunction(token);
100: 
101: 		if (isSimpleValueFunc) {
102: 			boolean localForceArray = desiredOperandClass == Ptg.CLASS_ARRAY;
103: 			for (int i = 0; i < children.length; i++) {
104: 				transformNode(children[i], desiredOperandClass, localForceArray);
105: 			}
106: 			setSimpleValueFuncClass((AbstractFunctionPtg) token, desiredOperandClass, callerForceArrayFlag);
107: 			return;
108: 		}
109: 
110: 		if (isSingleArgSum(token)) {
111: 			// Need to process the argument of SUM with transformFunctionNode below
112: 			// so make a dummy FuncVarPtg for that call.
113: 			token = FuncVarPtg.SUM;
114: 			// Note - the tAttrSum token (node.getToken()) is a base
115: 			// token so does not need to have its operand class set
116: 		}
117: 		if (token instanceof ValueOperatorPtg || token instanceof ControlPtg
118: 				|| token instanceof MemFuncPtg
119: 				|| token instanceof MemAreaPtg
120: 				|| token instanceof UnionPtg) {
121: 			// Value Operator Ptgs and Control are base tokens, so token will be unchanged
122: 			// but any child nodes are processed according to desiredOperandClass and callerForceArrayFlag
123: 
124: 			// As per OOO documentation Sec 3.2.4 "Token Class Transformation", "Step 1"
125: 			// All direct operands of value operators that are initially 'R' type will
126: 			// be converted to 'V' type.
127: 			byte localDesiredOperandClass = desiredOperandClass == Ptg.CLASS_REF ? Ptg.CLASS_VALUE : desiredOperandClass;
128: 			for (int i = 0; i < children.length; i++) {
129: 				transformNode(children[i], localDesiredOperandClass, callerForceArrayFlag);
130: 			}
131: 			return;
132: 		}
133: 		if (token instanceof AbstractFunctionPtg) {
134: 			transformFunctionNode((AbstractFunctionPtg) token, children, desiredOperandClass, callerForceArrayFlag);
135: 			return;
136: 		}
137: 		if (children.length > 0) {
138: 			if (token == RangePtg.instance) {
139: 				// TODO is any token transformation required under the various ref operators?
140: 				return;
141: 			}
142: 			throw new IllegalStateException("Node should not have any children");
143: 		}
144: 
145: 		if (token.isBaseToken()) {
146: 			// nothing to do
147: 			return;
148: 		}
149: 		token.setClass(transformClass(token.getPtgClass(), desiredOperandClass, callerForceArrayFlag));
150: 	}

152: 	private static boolean isSingleArgSum(Ptg token) {
153: 		if (token instanceof AttrPtg) {
154: 			AttrPtg attrPtg = (AttrPtg) token;
155: 			return attrPtg.isSum();
156: 		}
157: 		return false;
158: 	}
159: 
160: 	private static boolean isSimpleValueFunction(Ptg token) {
161: 		if (token instanceof AbstractFunctionPtg) {
162: 			AbstractFunctionPtg aptg = (AbstractFunctionPtg) token;
163: 			if (aptg.getDefaultOperandClass() != Ptg.CLASS_VALUE) {
164: 				return false;
165: 			}
166: 			int numberOfOperands = aptg.getNumberOfOperands();
167: 			for (int i=numberOfOperands-1; i>=0; i--) {
168: 				if (aptg.getParameterClass(i) != Ptg.CLASS_VALUE) {
169: 					return false;
170: 				}
171: 			}
172: 			return true;
173: 		}
174: 		return false;
175: 	}

177: 	private byte transformClass(byte currentOperandClass, byte desiredOperandClass,
178: 			boolean callerForceArrayFlag) {
179: 		switch (desiredOperandClass) {
180: 			case Ptg.CLASS_VALUE:
181: 				if (!callerForceArrayFlag) {
182: 					return Ptg.CLASS_VALUE;
183: 				}
184: 				// else fall through
185: 			case Ptg.CLASS_ARRAY:
186: 				return Ptg.CLASS_ARRAY;
187: 			case Ptg.CLASS_REF:
188: 				if (!callerForceArrayFlag) {
189: 					return currentOperandClass;
190: 				}
191: 				return Ptg.CLASS_REF;
192: 		}
193: 		throw new IllegalStateException("Unexpected operand class (" + desiredOperandClass + ")");
194: 	}

196: 	private void transformFunctionNode(AbstractFunctionPtg afp, ParseNode[] children,
197: 			byte desiredOperandClass, boolean callerForceArrayFlag) {
198: 
199: 		boolean localForceArrayFlag;
200: 		byte defaultReturnOperandClass = afp.getDefaultOperandClass();
201: 
202: 		if (callerForceArrayFlag) {
203: 			switch (defaultReturnOperandClass) {
204: 				case Ptg.CLASS_REF:
205: 					if (desiredOperandClass == Ptg.CLASS_REF) {
206: 						afp.setClass(Ptg.CLASS_REF);
207: 					} else {
208: 						afp.setClass(Ptg.CLASS_ARRAY);
209: 					}
210: 					localForceArrayFlag = false;
211: 					break;
212: 				case Ptg.CLASS_ARRAY:
213: 					afp.setClass(Ptg.CLASS_ARRAY);
214: 					localForceArrayFlag = false;
215: 					break;
216: 				case Ptg.CLASS_VALUE:
217: 					afp.setClass(Ptg.CLASS_ARRAY);
218: 					localForceArrayFlag = true;
219: 					break;
220: 				default:
221: 					throw new IllegalStateException("Unexpected operand class ("
222: 							+ defaultReturnOperandClass + ")");
223: 			}
224: 		} else {
225: 			if (defaultReturnOperandClass == desiredOperandClass) {
226: 				localForceArrayFlag = false;
227: 				// an alternative would have been to for non-base Ptgs to set their operand class
228: 				// from their default, but this would require the call in many subclasses because
229: 				// the default OC is not known until the end of the constructor
230: 				afp.setClass(defaultReturnOperandClass);
231: 			} else {
232: 				switch (desiredOperandClass) {
233: 					case Ptg.CLASS_VALUE:
234: 						// always OK to set functions to return 'value'
235: 						afp.setClass(Ptg.CLASS_VALUE);
236: 						localForceArrayFlag = false;
237: 						break;
238: 					case Ptg.CLASS_ARRAY:
239: 						switch (defaultReturnOperandClass) {
240: 							case Ptg.CLASS_REF:
241: 								afp.setClass(Ptg.CLASS_REF);
242: //								afp.setClass(Ptg.CLASS_ARRAY);
243: 								break;
244: 							case Ptg.CLASS_VALUE:
245: 								afp.setClass(Ptg.CLASS_ARRAY);
246: 								break;
247: 							default:
248: 								throw new IllegalStateException("Unexpected operand class ("
249: 										+ defaultReturnOperandClass + ")");
250: 						}
251: 						localForceArrayFlag = (defaultReturnOperandClass == Ptg.CLASS_VALUE);
252: 						break;
253: 					case Ptg.CLASS_REF:
254: 						switch (defaultReturnOperandClass) {
255: 							case Ptg.CLASS_ARRAY:
256: 								afp.setClass(Ptg.CLASS_ARRAY);
257: 								break;
258: 							case Ptg.CLASS_VALUE:
259: 								afp.setClass(Ptg.CLASS_VALUE);
260: 								break;
261: 							default:
262: 								throw new IllegalStateException("Unexpected operand class ("
263: 										+ defaultReturnOperandClass + ")");
264: 						}
265: 						localForceArrayFlag = false;
266: 						break;
267: 					default:
268: 						throw new IllegalStateException("Unexpected operand class ("
269: 								+ desiredOperandClass + ")");
270: 				}
271: 
272: 			}
273: 		}
274: 
275: 		for (int i = 0; i < children.length; i++) {
276: 			ParseNode child = children[i];
277: 			byte paramOperandClass = afp.getParameterClass(i);
278: 			transformNode(child, paramOperandClass, localForceArrayFlag);
279: 		}
280: 	}
281: 
282: 	private void setSimpleValueFuncClass(AbstractFunctionPtg afp,
283: 			byte desiredOperandClass, boolean callerForceArrayFlag) {
284: 
285: 		if (callerForceArrayFlag  || desiredOperandClass == Ptg.CLASS_ARRAY) {
286: 			afp.setClass(Ptg.CLASS_ARRAY);
287: 		} else {
288: 			afp.setClass(Ptg.CLASS_VALUE);
289: 		}
290: 	}
