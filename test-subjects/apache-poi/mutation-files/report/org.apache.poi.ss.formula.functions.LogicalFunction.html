 38: 		ValueEval ve;
 39: 		try {
 40: 			ve = OperandResolver.getSingleValue(arg0, srcRowIndex, srcColumnIndex);
 41: 		} catch (EvaluationException e) {
 42: 			if (false) {
 43: 				// Note - it is more usual to propagate error codes straight to the result like this:

 46: 			}
 47: 			// this will usually cause a 'FALSE' result except for ISNONTEXT()
 48: 			ve = e.getErrorEval();
 49: 		}
