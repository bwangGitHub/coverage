 29: /**
 30:  * Factory for turning an array of RawDataBlock instances containing
 31:  * Property data into an array of proper Property objects.
 32:  *
 33:  * The array produced may be sparse, in that any portion of data that
 34:  * should correspond to a Property, but which does not map to a proper
 35:  * Property (i.e., a DirectoryProperty, DocumentProperty, or
 36:  * RootProperty) will get mapped to a null Property in the array.
 37:  *
 38:  * @author Marc Johnson (mjohnson at apache dot org)
 39:  */
 40: 
 41: class PropertyFactory
 42: {
 43:     // no need for an accessible constructor
 44:     private PropertyFactory()
 45:     {
 46:     }
 47: 
 48:     /**
 49:      * Convert raw data blocks to an array of Property's
 50:      *
 51:      * @param blocks to be converted
 52:      *
 53:      * @return the converted List of Property objects. May contain
 54:      *         nulls, but will not be null
 55:      *
 56:      * @exception IOException if any of the blocks are empty
 57:      */
 58:     static List<Property> convertToProperties(ListManagedBlock [] blocks)
 59:         throws IOException
 60:     {
 61:         List<Property> properties = new ArrayList<Property>();
 62: 
 63:         for (int j = 0; j < blocks.length; j++) {
 64:             byte[] data = blocks[ j ].getData();
 65:             convertToProperties(data, properties);
 66:         }
 67:         return properties;
 68:     }
 69:     
 70:     static void convertToProperties(byte[] data, List<Property> properties)
 71:         throws IOException
 72:     {
 73:        int property_count = data.length / POIFSConstants.PROPERTY_SIZE;
 74:        int offset         = 0;
 75: 
 76:        for (int k = 0; k < property_count; k++) {
 77:           switch (data[ offset + PropertyConstants.PROPERTY_TYPE_OFFSET ]) {
 78:           case PropertyConstants.DIRECTORY_TYPE :
 79:              properties.add(
 80:                    new DirectoryProperty(properties.size(), data, offset)
 81:              );
 82:              break;
 83: 
 84:           case PropertyConstants.DOCUMENT_TYPE :
 85:              properties.add(
 86:                    new DocumentProperty(properties.size(), data, offset)
 87:              );
 88:              break;
 89: 
 90:           case PropertyConstants.ROOT_TYPE :
 91:              properties.add(
 92:                    new RootProperty(properties.size(), data, offset)
 93:              );
 94:              break;
 95: 
 96:           default :
 97:              properties.add(null);
 98:              break;
 99:           }
100:           
101:           offset += POIFSConstants.PROPERTY_SIZE;
102:        }
103:     }
104:     
105: }   // end package scope class PropertyFactory
