 24: import org.apache.poi.poifs.filesystem.DocumentDescriptor;
 25: import org.apache.poi.poifs.filesystem.POIFSDocumentPath;

 27: /**
 28:  * A registry for POIFSReaderListeners and the DocumentDescriptors of
 29:  * the documents those listeners are interested in
 30:  *
 31:  * @author Marc Johnson (mjohnson at apache dot org)
 32:  * @version %I%, %G%
 33:  */
 34: 
 35: class POIFSReaderRegistry
 36: {
 37: 
 38:     // the POIFSReaderListeners who listen to all POIFSReaderEvents
 39:     private Set omnivorousListeners;
 40: 
 41:     // Each mapping in this Map has a key consisting of a
 42:     // POIFSReaderListener and a value cosisting of a Set of
 43:     // DocumentDescriptors for the documents that POIFSReaderListener
 44:     // is interested in; used to efficiently manage the registry
 45:     private Map selectiveListeners;
 46: 
 47:     // Each mapping in this Map has a key consisting of a
 48:     // DocumentDescriptor and a value consisting of a Set of
 49:     // POIFSReaderListeners for the document matching that
 50:     // DocumentDescriptor; used when a document is found, to quickly
 51:     // get the listeners interested in that document
 52:     private Map chosenDocumentDescriptors;
 53: 
 54:     /**
 55:      * Construct the registry
 56:      */
 57: 
 58:     POIFSReaderRegistry()
 59:     {
 60:         omnivorousListeners       = new HashSet();
 61:         selectiveListeners        = new HashMap();
 62:         chosenDocumentDescriptors = new HashMap();
 63:     }
 64: 
 65:     /**
 66:      * register a POIFSReaderListener for a particular document
 67:      *
 68:      * @param listener the listener
 69:      * @param path the path of the document of interest
 70:      * @param documentName the name of the document of interest
 71:      */
 72: 
 73:     void registerListener(final POIFSReaderListener listener,
 74:                           final POIFSDocumentPath path,
 75:                           final String documentName)
 76:     {
 77:         if (!omnivorousListeners.contains(listener))
 78:         {
 79: 
 80:             // not an omnivorous listener (if it was, this method is a
 81:             // no-op)
 82:             Set descriptors = ( Set ) selectiveListeners.get(listener);
 83: 
 84:             if (descriptors == null)
 85:             {
 86: 
 87:                 // this listener has not registered before
 88:                 descriptors = new HashSet();
 89:                 selectiveListeners.put(listener, descriptors);
 90:             }
 91:             DocumentDescriptor descriptor = new DocumentDescriptor(path,
 92:                                                 documentName);
 93: 
 94:             if (descriptors.add(descriptor))
 95:             {
 96: 
 97:                 // this listener wasn't already listening for this
 98:                 // document -- add the listener to the set of
 99:                 // listeners for this document
100:                 Set listeners =
101:                     ( Set ) chosenDocumentDescriptors.get(descriptor);
102: 
103:                 if (listeners == null)
104:                 {
105: 
106:                     // nobody was listening for this document before
107:                     listeners = new HashSet();
108:                     chosenDocumentDescriptors.put(descriptor, listeners);
109:                 }
110:                 listeners.add(listener);
111:             }
112:         }
113:     }
114: 
115:     /**
116:      * register for all documents
117:      *
118:      * @param listener the listener who wants to get all documents
119:      */
120: 
121:     void registerListener(final POIFSReaderListener listener)
122:     {
123:         if (!omnivorousListeners.contains(listener))
124:         {
125: 
126:             // wasn't already listening for everything, so drop
127:             // anything listener might have been listening for and
128:             // then add the listener to the set of omnivorous
129:             // listeners
130:             removeSelectiveListener(listener);
131:             omnivorousListeners.add(listener);
132:         }
133:     }
134: 
135:     /**
136:      * get am iterator of listeners for a particular document
137:      *
138:      * @param path the document path
139:      * @param name the name of the document
140:      *
141:      * @return an Iterator POIFSReaderListeners; may be empty
142:      */
143: 
144:     Iterator getListeners(final POIFSDocumentPath path, final String name)
145:     {
146:         Set rval               = new HashSet(omnivorousListeners);
147:         Set selectiveListeners =
148:             ( Set ) chosenDocumentDescriptors.get(new DocumentDescriptor(path,
149:                 name));
150: 
151:         if (selectiveListeners != null)
152:         {
153:             rval.addAll(selectiveListeners);
154:         }
155:         return rval.iterator();
156:     }
157: 
158:     private void removeSelectiveListener(final POIFSReaderListener listener)
159:     {
160:         Set selectedDescriptors = ( Set ) selectiveListeners.remove(listener);
161: 
162:         if (selectedDescriptors != null)
163:         {
164:             Iterator iter = selectedDescriptors.iterator();
165: 
166:             while (iter.hasNext())
167:             {
168:                 dropDocument(listener, ( DocumentDescriptor ) iter.next());
169:             }
170:         }
171:     }
172: 
173:     private void dropDocument(final POIFSReaderListener listener,
174:                               final DocumentDescriptor descriptor)
175:     {
176:         Set listeners = ( Set ) chosenDocumentDescriptors.get(descriptor);
177: 
178:         listeners.remove(listener);
179:         if (listeners.size() == 0)
180:         {
181:             chosenDocumentDescriptors.remove(descriptor);
182:         }
183:     }
184: }   // end package scope class POIFSReaderRegistry
