128: 	 * An implementation of the CLEAN function:
129: 	 * In Excel, the Clean function removes all non-printable characters from a string.
130:      *
131: 	 * Author: Aniket Banerjee(banerjee@google.com)
132: 	 */
133:     public static final Function CLEAN = new SingleArgTextFunc() {
134:         protected ValueEval evaluate(String arg) {
135:             StringBuilder result = new StringBuilder();
136:             for (int i = 0; i < arg.length(); i++) {
137:                 char c = arg.charAt(i);
138:                 if (isPrintable(c)) {
139:                     result.append(c);
140:                 }
141:             }
142:             return new StringEval(result.toString());
143:         }
144: 
145:         /**
146:          * From Excel docs: The CLEAN function was designed to remove the first 32 nonprinting characters
147:          * in the 7-bit ASCII code (values 0 through 31) from text. In the Unicode character set,
148:          * there are additional nonprinting characters (values 127, 129, 141, 143, 144, and 157). By itself,
149:          * the CLEAN function does not remove these additional  nonprinting characters. To do this task,
150:          * use the SUBSTITUTE function to replace the higher value Unicode characters with the 7-bit ASCII
151:          * characters for which the TRIM and CLEAN functions were designed.
152:          *
153:          * @param c the character to test
154:          * @return  whether the character is printable
155:          */
156:         private boolean isPrintable(char c){
157:             int charCode = (int)c ;
158:             return charCode >= 32;
159:         }

163: 	 * An implementation of the MID function<br/>
164: 	 * MID returns a specific number of
165: 	 * characters from a text string, starting at the specified position.<p/>
166: 	 *
167: 	 * <b>Syntax<b>:<br/> <b>MID</b>(<b>text</b>, <b>start_num</b>,
168: 	 * <b>num_chars</b>)<br/>
169: 	 *
170: 	 * Author: Manda Wilson &lt; wilson at c bio dot msk cc dot org &gt;
171: 	 */
172: 	public static final Function MID = new Fixed3ArgFunction() {
173: 
174: 		public ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0,
175: 				ValueEval arg1, ValueEval arg2) {
176: 			String text;
177: 			int startCharNum;
178: 			int numChars;
179: 			try {
180: 				text = evaluateStringArg(arg0, srcRowIndex, srcColumnIndex);
181: 				startCharNum = evaluateIntArg(arg1, srcRowIndex, srcColumnIndex);
182: 				numChars = evaluateIntArg(arg2, srcRowIndex, srcColumnIndex);
183: 			} catch (EvaluationException e) {
184: 				return e.getErrorEval();
185: 			}
186: 			int startIx = startCharNum - 1; // convert to zero-based
187: 
188: 			// Note - for start_num arg, blank/zero causes error(#VALUE!),
189: 			// but for num_chars causes empty string to be returned.
190: 			if (startIx < 0) {
191: 				return ErrorEval.VALUE_INVALID;
192: 			}
193: 			if (numChars < 0) {
194: 				return ErrorEval.VALUE_INVALID;
195: 			}
196: 			int len = text.length();
197: 			if (numChars < 0 || startIx > len) {
198: 				return new StringEval("");
199: 			}
200: 			int endIx = Math.min(startIx + numChars, len);
201: 			String result = text.substring(startIx, endIx);
202: 			return new StringEval(result);
203: 		}

323: 				ValueEval arg2) {
324: 			try {
325: 				String needle = TextFunction.evaluateStringArg(arg0, srcRowIndex, srcColumnIndex);
326: 				String haystack = TextFunction.evaluateStringArg(arg1, srcRowIndex, srcColumnIndex);
327: 				// evaluate third arg and convert from 1-based to 0-based index

329: 				if (startpos < 0) {
330: 					return ErrorEval.VALUE_INVALID;

333: 			} catch (EvaluationException e) {
334: 				return e.getErrorEval();
335: 			}
336: 		}
337: 		private ValueEval eval(String haystack, String needle, int startIndex) {
338: 			int result;
339: 			if (_isCaseSensitive) {
340: 				result = haystack.indexOf(needle, startIndex);
341: 			} else {
342: 				result = haystack.toUpperCase().indexOf(needle.toUpperCase(), startIndex);
343: 			}
344: 			if (result == -1) {
345: 				return ErrorEval.VALUE_INVALID;
346: 			}
347: 			return new NumberEval(result + 1);
348: 		}
