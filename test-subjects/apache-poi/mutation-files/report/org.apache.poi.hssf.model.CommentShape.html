 35: /**
 36:  * Represents a cell comment.
 37:  * This class converts highlevel model data from <code>HSSFComment</code>
 38:  * to low-level records.
 39:  *
 40:  * @author Yegor Kozlov
 41:  */
 42: public final class CommentShape extends TextboxShape {
 43: 
 44:     private NoteRecord _note;
 45: 
 46:     /**
 47:      * Creates the low-level records for a comment.
 48:      *
 49:      * @param hssfShape  The highlevel shape.
 50:      * @param shapeId    The shape id to use for this shape.
 51:      */
 52:      public CommentShape( HSSFComment hssfShape, int shapeId )
 53:     {
 54:         super(hssfShape, shapeId);
 55: 
 56:         _note = createNoteRecord(hssfShape, shapeId);
 57: 
 58:         ObjRecord obj = getObjRecord();
 59:         List<SubRecord> records = obj.getSubRecords();
 60:         int cmoIdx = 0;
 61:         for (int i = 0; i < records.size(); i++) {
 62:             Object r = records.get(i);
 63: 
 64:             if (r instanceof CommonObjectDataSubRecord){
 65:                 //modify autofill attribute inherited from <code>TextObjectRecord</code>
 66:                 CommonObjectDataSubRecord cmo = (CommonObjectDataSubRecord)r;
 67:                 cmo.setAutofill(false);
 68:                 cmoIdx = i;
 69:             }
 70:         }
 71:         //add NoteStructure sub record
 72:         //we don't know it's format, for now the record data is empty
 73:         NoteStructureSubRecord u = new NoteStructureSubRecord();
 74:         obj.addSubRecord(cmoIdx+1, u);
 75:     }
 76: 
 77:     /**
 78:      * Creates the low level <code>NoteRecord</code>
 79:      *  which holds the comment attributes.
 80:      */
 81:      private NoteRecord createNoteRecord( HSSFComment shape, int shapeId )
 82:     {
 83:         NoteRecord note = new NoteRecord();
 84:         note.setColumn(shape.getColumn());
 85:         note.setRow(shape.getRow());
 86:         note.setFlags(shape.isVisible() ? NoteRecord.NOTE_VISIBLE : NoteRecord.NOTE_HIDDEN);
 87:         note.setShapeId(shapeId);
 88:         note.setAuthor(shape.getAuthor() == null ? "" : shape.getAuthor());
 89:         return note;
 90:     }
 91: 
 92:     /**
 93:      * Sets standard escher options for a comment.
 94:      * This method is responsible for setting default background,
 95:      * shading and other comment properties.
 96:      *
 97:      * @param shape   The highlevel shape.
 98:      * @param opt     The escher records holding the proerties
 99:      * @return number of escher options added
100:      */
101:     protected int addStandardOptions( HSSFShape shape, EscherOptRecord opt )
102:     {
103:         super.addStandardOptions(shape, opt);
104: 
105:         //remove unnecessary properties inherited from TextboxShape
106:         List<EscherProperty> props = opt.getEscherProperties();
107:         for (Iterator<EscherProperty> iterator = props.iterator(); iterator.hasNext(); ) {
108:             EscherProperty prop = iterator.next();
109:             switch (prop.getId()){
110:                 case EscherProperties.TEXT__TEXTLEFT:
111:                 case EscherProperties.TEXT__TEXTRIGHT:
112:                 case EscherProperties.TEXT__TEXTTOP:
113:                 case EscherProperties.TEXT__TEXTBOTTOM:
114:                 case EscherProperties.GROUPSHAPE__PRINT:
115:                 case EscherProperties.FILL__FILLBACKCOLOR:
116:                 case EscherProperties.LINESTYLE__COLOR:
117:                     iterator.remove();
118:                     break;
119:             }
120:         }
121: 
122:         HSSFComment comment = (HSSFComment)shape;
123:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.GROUPSHAPE__PRINT, comment.isVisible() ? 0x000A0000 : 0x000A0002) );
124:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.SHADOWSTYLE__SHADOWOBSURED, 0x00030003 ) );
125:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.SHADOWSTYLE__COLOR, 0x00000000 ) );
126:         opt.sortProperties();
127:         return opt.getEscherProperties().size();   // # options added
128:     }
129: 
130:     /**
131:      * Return the <code>NoteRecord</code> holding the comment attributes
132:      *
133:      * @return <code>NoteRecord</code> holding the comment attributes
134:      */
135:     public NoteRecord getNoteRecord()
136:     {
137:         return _note;
138:     }
