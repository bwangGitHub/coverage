 56: 	public String getDocumentSummaryInformationText() {
 57: 		DocumentSummaryInformation dsi = document.getDocumentSummaryInformation();
 58: 		StringBuffer text = new StringBuffer();
 59: 
 60: 		// Normal properties
 61: 		text.append( getPropertiesText(dsi) );
 62: 
 63: 		// Now custom ones
 64: 		CustomProperties cps = dsi == null ? null : dsi.getCustomProperties();
 65: 		if(cps != null) {
 66: 			Iterator<String> keys = cps.nameSet().iterator();
 67: 			while(keys.hasNext()) {
 68: 				String key = keys.next();
 69: 				String val = getPropertyValueText( cps.get(key) );
 70: 				text.append(key + " = " + val + "\n");
 71: 			}
 72: 		}
 73: 
 74: 		// All done
 75: 		return text.toString();
 76: 	}
 77: 	public String getSummaryInformationText() {
 78: 		SummaryInformation si = document.getSummaryInformation();
 79: 
 80: 		// Just normal properties
 81: 		return getPropertiesText(si);
 82: 	}

 84: 	private static String getPropertiesText(SpecialPropertySet ps) {
 85: 		if(ps == null) {
 86: 			// Not defined, oh well
 87: 			return "";
 88: 		}
 89: 
 90: 		StringBuffer text = new StringBuffer();
 91: 
 92: 		PropertyIDMap idMap = ps.getPropertySetIDMap();
 93: 		Property[] props = ps.getProperties();
 94: 		for(int i=0; i<props.length; i++) {
 95: 			String type = Long.toString( props[i].getID() );
 96: 			Object typeObj = idMap.get(props[i].getID());
 97: 			if(typeObj != null) {
 98: 				type = typeObj.toString();
 99: 			}
100: 
101: 			String val = getPropertyValueText( props[i].getValue() );
102: 			text.append(type + " = " + val + "\n");
103: 		}
104: 
105: 		return text.toString();
106: 	}
107: 	private static String getPropertyValueText(Object val) {
108: 		if(val == null) {
109: 			return "(not set)";
110: 		}
111: 		if(val instanceof byte[]) {
112: 			byte[] b = (byte[])val;
113: 			if(b.length == 0) {
114: 				return "";
115: 			}
116: 			if(b.length == 1) {
117: 				return Byte.toString(b[0]);
118: 			}
119: 			if(b.length == 2) {
120: 				return Integer.toString( LittleEndian.getUShort(b) );
121: 			}
122: 			if(b.length == 4) {
123: 				return Long.toString( LittleEndian.getUInt(b) );
124: 			}
125: 			// Maybe it's a string? who knows!
126: 			return new String(b);
127: 		}
128: 		return val.toString();
129: 	}
