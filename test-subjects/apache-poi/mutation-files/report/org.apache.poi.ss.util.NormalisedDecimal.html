 20: import java.math.BigDecimal;

 23: /**
 24:  * Represents a transformation of a 64 bit IEEE double quantity having a decimal exponent and a
 25:  * fixed point (15 decimal digit) significand.  Some quirks of Excel's calculation behaviour are
 26:  * simpler to reproduce with numeric quantities in this format.  This class is currently used to
 27:  * help:
 28:  * <ol>
 29:  * <li>Comparison operations</li>
 30:  * <li>Conversions to text</li>
 31:  * </ol>
 32:  *
 33:  * <p/>
 34:  * This class does not handle negative numbers or zero.
 35:  * <p/>
 36:  * The value of a {@link NormalisedDecimal} is given by<br/>
 37:  * <tt> significand &times; 10<sup>decimalExponent</sup></tt>
 38:  * <br/>
 39:  * where:<br/>
 40:  *
 41:  * <tt>significand</tt> = wholePart + fractionalPart / 2<sup>24</sup><br/>
 42:  *
 43:  * @author Josh Micich
 44:  */
 45: final class NormalisedDecimal {
 46: 	/**
 47: 	 * Number of powers of ten contained in the significand
 48: 	 */
 49: 	private static final int EXPONENT_OFFSET = 14;
 50: 
 51: 	private static final BigDecimal BD_2_POW_24 = new BigDecimal(BigInteger.ONE.shiftLeft(24));
 52: 
 53: 	/**
 54: 	 * log<sub>10</sub>(2)&times;2<sup>20</sup>
 55: 	 */
 56: 	private static final int LOG_BASE_10_OF_2_TIMES_2_POW_20 = 315653; // 315652.8287
 57: 
 58: 	/**
 59: 	 * 2<sup>19</sup>
 60: 	 */
 61: 	private static final int C_2_POW_19 = 1 << 19;
 62: 
 63: 
 64: 	/**
 65: 	 * the value of {@link #_fractionalPart} that represents 0.5
 66: 	 */
 67: 	private static final int FRAC_HALF = 0x800000;
 68: 
 69: 	/**
 70: 	 * 10<sup>15</sup>
 71: 	 */
 72: 	private static final long MAX_REP_WHOLE_PART = 0x38D7EA4C68000L;

 76: 	public static NormalisedDecimal create(BigInteger frac, int binaryExponent) {
 77: 		// estimate pow2&pow10 first, perform optional mulShift, then normalize
 78: 		int pow10;
 79: 		if (binaryExponent > 49 || binaryExponent < 46) {
 80: 
 81: 			// working with ints (left shifted 20) instead of doubles
 82: 			// x = 14.5 - binaryExponent * log10(2);
 83: 			int x = (29 << 19) - binaryExponent * LOG_BASE_10_OF_2_TIMES_2_POW_20;
 84: 			x += C_2_POW_19; // round
 85: 			pow10 = -(x >> 20);
 86: 		} else {
 87: 			pow10 = 0;
 88: 		}
 89: 		MutableFPNumber cc = new MutableFPNumber(frac, binaryExponent);
 90: 		if (pow10 != 0) {
 91: 			cc.multiplyByPowerOfTen(-pow10);
 92: 		}
 93: 
 94: 		switch (cc.get64BitNormalisedExponent()) {
 95: 			case 46:
 96: 				if (cc.isAboveMinRep()) {
 97: 					break;
 98: 				}
 99: 			case 44:
100: 			case 45:
101: 				cc.multiplyByPowerOfTen(1);
102: 				pow10--;
103: 				break;
104: 			case 47:
105: 			case 48:
106: 				break;
107: 			case 49:
108: 				if (cc.isBelowMaxRep()) {
109: 					break;
110: 				}
111: 			case 50:
112: 				cc.multiplyByPowerOfTen(-1);
113: 				pow10++;
114: 				break;
115: 
116: 			default:
117: 				throw new IllegalStateException("Bad binary exp " + cc.get64BitNormalisedExponent() + ".");
118: 		}
119: 		cc.normalise64bit();
120: 
121: 		return cc.createNormalisedDecimal(pow10);
122: 	}

124: 	/**
125: 	 * Rounds at the digit with value 10<sup>decimalExponent</sup>
126: 	 */
127: 	public NormalisedDecimal roundUnits() {
128: 		long wholePart = _wholePart;
129: 		if (_fractionalPart >= FRAC_HALF) {
130: 			wholePart++;
131: 		}
132: 
133: 		int de = _relativeDecimalExponent;
134: 
135: 		if (wholePart < MAX_REP_WHOLE_PART) {
136: 			return new NormalisedDecimal(wholePart, 0, de);
137: 		}
138: 		return new NormalisedDecimal(wholePart/10, 0, de+1);
139: 	}

147: 	 *

149: 	 * LSB is units bit.

166: 	/**
167: 	 * Convert to an equivalent {@link ExpandedDouble} representation (binary frac and exponent).
168: 	 * The resulting transformed object is easily converted to a 64 bit IEEE double:
169: 	 * <ul>
170: 	 * <li>bits 2-53 of the {@link #getSignificand()} become the 52 bit 'fraction'.</li>
171: 	 * <li>{@link #getBinaryExponent()} is biased by 1023 to give the 'exponent'.</li>
172: 	 * </ul>
173: 	 * The sign bit must be obtained from somewhere else.
174: 	 * @return a new {@link NormalisedDecimal} normalised to base 2 representation.
175: 	 */
176: 	public ExpandedDouble normaliseBaseTwo() {
177: 		MutableFPNumber cc = new MutableFPNumber(composeFrac(), 39);
178: 		cc.multiplyByPowerOfTen(_relativeDecimalExponent);
179: 		cc.normalise64bit();
180: 		return cc.createExpandedDouble();
181: 	}

183: 	/**
184: 	 * @return the significand as a fixed point number (with 24 fraction bits and 47-50 whole bits)
185: 	 */
186: 	BigInteger composeFrac() {
187: 		long wp = _wholePart;
188: 		int fp = _fractionalPart;
189: 		return new BigInteger(new byte[] {
190: 				(byte) (wp >> 56), // N.B. assuming sign bit is zero
191: 				(byte) (wp >> 48),
192: 				(byte) (wp >> 40),
193: 				(byte) (wp >> 32),
194: 				(byte) (wp >> 24),
195: 				(byte) (wp >> 16),
196: 				(byte) (wp >>  8),
197: 				(byte) (wp >>  0),
198: 				(byte) (fp >> 16),
199: 				(byte) (fp >> 8),
200: 				(byte) (fp >> 0),
201: 		});
202: 	}

222: 	 */

224: 		return _relativeDecimalExponent+EXPONENT_OFFSET;
225: 	}

227: 	/**
228: 	 * assumes both this and other are normalised
229: 	 */
230: 	public int compareNormalised(NormalisedDecimal other) {
231: 		int cmp = _relativeDecimalExponent - other._relativeDecimalExponent;
232: 		if (cmp != 0) {
233: 			return cmp;
234: 		}
235: 		if (_wholePart > other._wholePart) {
236: 			return 1;
237: 		}
238: 		if (_wholePart < other._wholePart) {
239: 			return -1;
240: 		}
241: 		return _fractionalPart - other._fractionalPart;
242: 	}
243: 	public BigDecimal getFractionalPart() {
244: 		return new BigDecimal(_fractionalPart).divide(BD_2_POW_24);
245: 	}

248: 		if (_fractionalPart == 0) {
249: 			return "0";
250: 		}
251: 		return getFractionalPart().toString().substring(2);
252: 	}
