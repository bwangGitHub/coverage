 31: /**
 32:  * Returns the rank of a number in a list of numbers. The rank of a number is its size relative to other values in a list.
 33: 
 34:  * Syntax:
 35:  *    RANK(number,ref,order)
 36:  *       Number   is the number whose rank you want to find.
 37:  *       Ref     is an array of, or a reference to, a list of numbers. Nonnumeric values in ref are ignored.
 38:  *       Order   is a number specifying how to rank number.
 39: 
 40:  * If order is 0 (zero) or omitted, Microsoft Excel ranks number as if ref were a list sorted in descending order.
 41:  * If order is any nonzero value, Microsoft Excel ranks number as if ref were a list sorted in ascending order.
 42:  * 
 43:  * @author Rubin Wang
 44:  */

 52: 			ValueEval ve = OperandResolver.getSingleValue(arg0, srcRowIndex, srcColumnIndex);

 91: 	private static ValueEval eval(int srcRowIndex, int srcColumnIndex, double arg0, AreaEval aeRange, boolean descending_order) {
 92: 		
 93: 		int rank = 1;
 94: 		int height=aeRange.getHeight();
 95: 		int width= aeRange.getWidth();
 96: 		for (int r=0; r<height; r++) {
 97: 			for (int c=0; c<width; c++) {
 98: 				
 99: 				Double value = getValue(aeRange, r, c);
100: 				if(value==null)continue;
101: 				if(descending_order && value>arg0 || !descending_order && value<arg0){
102: 					rank++;
103: 				}
104: 			}
105: 		}
106: 		return new NumberEval(rank);
107: 	}

109: 	private static Double getValue(AreaEval aeRange, int relRowIndex, int relColIndex) {
110:

112: 		if (addend instanceof NumberEval) {
113: 			return ((NumberEval)addend).getNumberValue();

115: 		// everything else (including string and boolean values) counts as zero
116: 		return null;
