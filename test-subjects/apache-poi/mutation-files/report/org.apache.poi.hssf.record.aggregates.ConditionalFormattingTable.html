 45: 	public ConditionalFormattingTable(RecordStream rs) {
 46: 
 47: 		List temp = new ArrayList();
 48: 		while (rs.peekNextClass() == CFHeaderRecord.class) {
 49: 			temp.add(CFRecordsAggregate.createCFAggregate(rs));
 50: 		}
 51: 		_cfHeaders = temp;
 52: 	}

 55: 		for (int i = 0; i < _cfHeaders.size(); i++) {
 56: 			CFRecordsAggregate subAgg = (CFRecordsAggregate) _cfHeaders.get(i);
 57: 			subAgg.visitContainedRecords(rv);
 58: 		}
 59: 	}

 62: 	 * @return index of the newly added CF header aggregate
 63: 	 */
 64: 	public int add(CFRecordsAggregate cfAggregate) {
 65: 		_cfHeaders.add(cfAggregate);
 66: 		return _cfHeaders.size() - 1;
 67: 	}
 68:

 71: 	}
 72:

 76: 	}

 84: 		if (index < 0 || index >= _cfHeaders.size()) {

 86: 					+ " is outside the allowable range (0.." + (_cfHeaders.size() - 1) + ")");
 87: 		}
 88: 	}

 91: 		for (int i = 0; i < _cfHeaders.size(); i++) {
 92: 			CFRecordsAggregate subAgg = (CFRecordsAggregate) _cfHeaders.get(i);
