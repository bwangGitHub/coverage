 25: /**
 26:  * Title:        Iteration Record (0x0011) <p/>
 27:  * Description:  Tells whether to iterate over forumla calculations or not
 28:  *               (if a formula is dependant upon another formula's result)
 29:  *               (odd feature for something that can only have 32 elements in
 30:  *                a formula!)<P>
 31:  * REFERENCE:  PG 325 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<p/>
 32:  * @author Andrew C. Oliver (acoliver at apache dot org)
 33:  * @author Jason Height (jheight at chariot dot net dot au)
 34:  */
 35: public final class IterationRecord extends StandardRecord {
 36:     public final static short sid = 0x0011;
 37: 
 38:     private static final BitField iterationOn = BitFieldFactory.getInstance(0x0001);
 39: 
 40:     private int _flags;
 41: 
 42:     public IterationRecord(boolean iterateOn) {
 43:         _flags = iterationOn.setBoolean(0, iterateOn);
 44:     }
 45: 
 46:     public IterationRecord(RecordInputStream in)
 47:     {
 48:         _flags = in.readShort();
 49:     }
 50: 
 51:     /**
 52:      * set whether or not to iterate for calculations
 53:      * @param iterate or not
 54:      */
 55:     public void setIteration(boolean iterate) {
 56:         _flags = iterationOn.setBoolean(_flags, iterate);
 57:     }
 58: 
 59:     /**
 60:      * get whether or not to iterate for calculations
 61:      *
 62:      * @return whether iterative calculations are turned off or on
 63:      */
 64:     public boolean getIteration() {
 65:         return iterationOn.isSet(_flags);
 66:     }
 67: 
 68:     public String toString() {
 69:         StringBuffer buffer = new StringBuffer();
 70: 
 71:         buffer.append("[ITERATION]\n");
 72:         buffer.append("    .flags      = ").append(HexDump.shortToHex(_flags)).append("\n");
 73:         buffer.append("[/ITERATION]\n");
 74:         return buffer.toString();
 75:     }
 76: 
 77:     public void serialize(LittleEndianOutput out) {
 78:         out.writeShort(_flags);
 79:     }
 80: 
 81:     protected int getDataSize() {
 82:         return 2;
 83:     }
 84: 
 85:     public short getSid() {
 86:         return sid;
 87:     }
 88: 
 89:     public Object clone() {
 90:         return new IterationRecord(getIteration());
 91:     }
 92: }
