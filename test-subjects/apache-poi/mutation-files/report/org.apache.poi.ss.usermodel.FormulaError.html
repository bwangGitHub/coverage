 22: /**
 23:  * Enumerates error values in SpreadsheetML formula calculations.
 24:  *
 25:  * @author Yegor Kozlov
 26:  */
 27: public enum FormulaError {
 28:     /**
 29:      * Intended to indicate when two areas are required to intersect, but do not.
 30:      * <p>Example:
 31:      * In the case of SUM(B1 C1), the space between B1 and C1 is treated as the binary
 32:      * intersection operator, when a comma was intended. end example]
 33:      * </p>
 34:      */
 35:     NULL(0x00, "#NULL!"),
 36: 
 37:     /**
 38:      * Intended to indicate when any number, including zero, is divided by zero.
 39:      * Note: However, any error code divided by zero results in that error code.
 40:      */
 41:     DIV0(0x07, "#DIV/0!"),
 42: 
 43:     /**
 44:      * Intended to indicate when an incompatible type argument is passed to a function, or
 45:      * an incompatible type operand is used with an operator.
 46:      * <p>Example:
 47:      * In the case of a function argument, text was expected, but a number was provided
 48:      * </p>
 49:      */
 50:     VALUE(0x0F, "#VALUE!"),
 51: 
 52:     /**
 53:      * Intended to indicate when a cell reference is invalid.
 54:      * <p>Example:
 55:      * If a formula contains a reference to a cell, and then the row or column containing that cell is deleted,
 56:      * a #REF! error results. If a worksheet does not support 20,001 columns,
 57:      * OFFSET(A1,0,20000) will result in a #REF! error.
 58:      * </p>
 59:      */
 60:     REF(0x17, "#REF!"),
 61: 
 62:     /**
 63:      * Intended to indicate when what looks like a name is used, but no such name has been defined.
 64:      * <p>Example:
 65:      * XYZ/3, where XYZ is not a defined name. Total is & A10,
 66:      * where neither Total nor is is a defined name. Presumably, "Total is " & A10
 67:      * was intended. SUM(A1C10), where the range A1:C10 was intended.
 68:      * </p>
 69:      */
 70:     NAME(0x1D, "#NAME?"),
 71: 
 72:     /**
 73:      * Intended to indicate when an argument to a function has a compatible type, but has a
 74:      * value that is outside the domain over which that function is defined. (This is known as
 75:      * a domain error.)
 76:      * <p>Example:
 77:      * Certain calls to ASIN, ATANH, FACT, and SQRT might result in domain errors.
 78:      * </p>
 79:      * Intended to indicate that the result of a function cannot be represented in a value of
 80:      * the specified type, typically due to extreme magnitude. (This is known as a range
 81:      * error.)
 82:      * <p>Example: FACT(1000) might result in a range error. </p>
 83:      */
 84:     NUM(0x24, "#NUM!"),
 85: 
 86:     /**
 87:      * Intended to indicate when a designated value is not available.
 88:      * <p>Example:
 89:      * Some functions, such as SUMX2MY2, perform a series of operations on corresponding
 90:      * elements in two arrays. If those arrays do not have the same number of elements, then
 91:      * for some elements in the longer array, there are no corresponding elements in the
 92:      * shorter one; that is, one or more values in the shorter array are not available.
 93:      * </p>
 94:      * This error value can be produced by calling the function NA
 95:      */
 96:     NA(0x2A, "#N/A");
 97: 
 98:     private byte type;
 99:     private String repr;
100: 
101:     private FormulaError(int type, String repr) {
102:         this.type = (byte) type;
103:         this.repr = repr;
104:     }
105: 
106:     /**
107:      * @return numeric code of the error
108:      */
109:     public byte getCode() {
110:         return type;
111:     }
112: 
113:     /**
114:      * @return string representation of the error
115:      */
116:     public String getString() {
117:         return repr;
118:     }
119: 
120:     private static Map<String, FormulaError> smap = new HashMap<String, FormulaError>();
121:     private static Map<Byte, FormulaError> imap = new HashMap<Byte, FormulaError>();
122:     static{
123:         for (FormulaError error : values()) {
124:             imap.put(error.getCode(), error);
125:             smap.put(error.getString(), error);
126:         }
127:     }
128: 
129:     public static FormulaError forInt(byte type){
130:         FormulaError err = imap.get(type);
131:         if(err == null) throw new IllegalArgumentException("Unknown error type: " + type);
132:         return err;
133:     }
134: 
135:     public static FormulaError forString(String code){
136:         FormulaError err = smap.get(code);
137:         if(err == null) throw new IllegalArgumentException("Unknown error code: " + code);
138:         return err;
139:     }
