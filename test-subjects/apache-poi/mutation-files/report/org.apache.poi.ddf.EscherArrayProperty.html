 23: /**
 24:  * Escher array properties are the most wierd construction ever invented
 25:  * with all sorts of special cases.  I'm hopeful I've got them all.
 26:  *
 27:  * @author Glen Stampoultzis (glens at superlinksoftware.com)
 28:  */
 29: public final class EscherArrayProperty extends EscherComplexProperty {
 30:     /**
 31:      * The size of the header that goes at the
 32:      *  start of the array, before the data
 33:      */
 34:     private static final int FIXED_SIZE = 3 * 2;
 35:     /**
 36:      * Normally, the size recorded in the simple data (for the complex
 37:      *  data) includes the size of the header.
 38:      * There are a few cases when it doesn't though...
 39:      */
 40:     private boolean sizeIncludesHeaderSize = true;
 41: 
 42:     /**
 43:      * When reading a property from data stream remeber if the complex part is empty and set this flag.
 44:      */
 45:     private boolean emptyComplexPart = false;
 46: 
 47:     public EscherArrayProperty(short id, byte[] complexData) {
 48:         super(id, checkComplexData(complexData));
 49:         emptyComplexPart = complexData.length == 0;
 50:     }
 51: 
 52:     public EscherArrayProperty(short propertyNumber, boolean isBlipId, byte[] complexData) {
 53:         super(propertyNumber, isBlipId, checkComplexData(complexData));
 54:     }
 55: 
 56:     private static byte[] checkComplexData(byte[] complexData) {
 57:         if (complexData == null || complexData.length == 0) {
 58:             return new byte[6];
 59:         }
 60: 
 61:         return complexData;
 62:     }
 63: 
 64:     public int getNumberOfElementsInArray() {
 65:         return LittleEndian.getUShort(_complexData, 0);
 66:     }
 67: 
 68:     public void setNumberOfElementsInArray(int numberOfElements) {
 69:         int expectedArraySize = numberOfElements * getActualSizeOfElements(getSizeOfElements()) + FIXED_SIZE;
 70:         if (expectedArraySize != _complexData.length) {
 71:             byte[] newArray = new byte[expectedArraySize];
 72:             System.arraycopy(_complexData, 0, newArray, 0, _complexData.length);
 73:             _complexData = newArray;
 74:         }
 75:         LittleEndian.putShort(_complexData, 0, (short) numberOfElements);
 76:     }
 77: 
 78:     public int getNumberOfElementsInMemory() {
 79:         return LittleEndian.getUShort(_complexData, 2);
 80:     }
 81: 
 82:     public void setNumberOfElementsInMemory(int numberOfElements) {
 83:         int expectedArraySize = numberOfElements * getActualSizeOfElements(getSizeOfElements()) + FIXED_SIZE;
 84:         if (expectedArraySize != _complexData.length) {
 85:             byte[] newArray = new byte[expectedArraySize];
 86:             System.arraycopy(_complexData, 0, newArray, 0, expectedArraySize);
 87:             _complexData = newArray;
 88:         }
 89:         LittleEndian.putShort(_complexData, 2, (short) numberOfElements);
 90:     }
 91: 
 92:     public short getSizeOfElements() {
 93:         return LittleEndian.getShort( _complexData, 4 );
 94:     }
 95: 
 96:     public void setSizeOfElements(int sizeOfElements) {
 97:         LittleEndian.putShort( _complexData, 4, (short) sizeOfElements );
 98: 
 99:         int expectedArraySize = getNumberOfElementsInArray() * getActualSizeOfElements(getSizeOfElements()) + FIXED_SIZE;
100:         if (expectedArraySize != _complexData.length) {
101:             // Keep just the first 6 bytes.  The rest is no good to us anyway.
102:             byte[] newArray = new byte[expectedArraySize];
103:             System.arraycopy( _complexData, 0, newArray, 0, 6 );
104:             _complexData = newArray;
105:         }
106:     }

108:     public byte[] getElement(int index) {
109:         int actualSize = getActualSizeOfElements(getSizeOfElements());
110:         byte[] result = new byte[actualSize];
111:         System.arraycopy(_complexData, FIXED_SIZE + index * actualSize, result, 0, result.length );
112:         return result;
113:     }
114: 
115:     public void setElement(int index, byte[] element) {
116:         int actualSize = getActualSizeOfElements(getSizeOfElements());
117:         System.arraycopy( element, 0, _complexData, FIXED_SIZE + index * actualSize, actualSize);
118:     }

120:     public String toString() {
121:         StringBuffer results = new StringBuffer();
122:         results.append("    {EscherArrayProperty:" + '\n');
123:         results.append("     Num Elements: " + getNumberOfElementsInArray() + '\n');
124:         results.append("     Num Elements In Memory: " + getNumberOfElementsInMemory() + '\n');
125:         results.append("     Size of elements: " + getSizeOfElements() + '\n');
126:         for (int i = 0; i < getNumberOfElementsInArray(); i++) {
127:             results.append("     Element " + i + ": " + HexDump.toHex(getElement(i)) + '\n');
128:         }
129:         results.append("}" + '\n');
130: 
131:         return "propNum: " + getPropertyNumber()
132:                 + ", propName: " + EscherProperties.getPropertyName( getPropertyNumber() )
133:                 + ", complex: " + isComplex()
134:                 + ", blipId: " + isBlipId()
135:                 + ", data: " + '\n' + results.toString();
136:     }

138:     /**
139:      * We have this method because the way in which arrays in escher works
140:      * is screwed for seemly arbitary reasons.  While most properties are
141:      * fairly consistent and have a predictable array size, escher arrays
142:      * have special cases.
143:      *
144:      * @param data      The data array containing the escher array information
145:      * @param offset    The offset into the array to start reading from.
146:      * @return  the number of bytes used by this complex property.
147:      */
148:     public int setArrayData(byte[] data, int offset) {
149:         if (emptyComplexPart){
150:             _complexData = new byte[0];
151:         } else {
152:             short numElements = LittleEndian.getShort(data, offset);
153:             LittleEndian.getShort(data, offset + 2); // numReserved
154:             short sizeOfElements = LittleEndian.getShort(data, offset + 4);
155: 
156:             int arraySize = getActualSizeOfElements(sizeOfElements) * numElements;
157:             if (arraySize == _complexData.length) {
158:                 // The stored data size in the simple block excludes the header size
159:                 _complexData = new byte[arraySize + 6];
160:                 sizeIncludesHeaderSize = false;
161:             }
162:             System.arraycopy(data, offset, _complexData, 0, _complexData.length );
163:         }
164:         return _complexData.length;
165:     }
166: 
167:     /**
168:      * Serializes the simple part of this property.  ie the first 6 bytes.
169:      *
170:      * Needs special code to handle the case when the size doesn't
171:      *  include the size of the header block
172:      */
173:     public int serializeSimplePart(byte[] data, int pos) {
174:         LittleEndian.putShort(data, pos, getId());
175:         int recordSize = _complexData.length;
176:         if(!sizeIncludesHeaderSize) {
177:             recordSize -= 6;
178:         }
179:         LittleEndian.putInt(data, pos + 2, recordSize);
180:         return 6;
181:     }
182: 
183:     /**
184:      * Sometimes the element size is stored as a negative number.  We
185:      * negate it and shift it to get the real value.
186:      */
187:     public static int getActualSizeOfElements(short sizeOfElements) {
188:         if (sizeOfElements < 0) {
189:             return (short) ( ( -sizeOfElements ) >> 2 );
190:         }
191:         return sizeOfElements;
192:     }
