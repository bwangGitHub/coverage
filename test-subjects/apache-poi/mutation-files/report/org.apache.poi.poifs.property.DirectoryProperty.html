132:         /**
133:          * compare method. Assumes both parameters are non-null
134:          * instances of Property. One property is less than another if
135:          * its name is shorter than the other property's name. If the
136:          * names are the same length, the property whose name comes
137:          * before the other property's name, alphabetically, is less
138:          * than the other property.
139:          *
140:          * @param o1 first object to compare, better be a Property
141:          * @param o2 second object to compare, better be a Property
142:          *
143:          * @return negative value if o1 <  o2,
144:          *         zero           if o1 == o2,
145:          *         positive value if o1 >  o2.
146:          */
147:         public int compare(Property o1, Property o2)
148:         {
149:             String VBA_PROJECT = "_VBA_PROJECT";
150:             String name1  = o1.getName();
151:             String name2  = o2.getName();
152:             int  result = name1.length() - name2.length();
153: 
154:             if (result == 0)
155:             {
156:               // _VBA_PROJECT, it seems, will always come last
157:               if (name1.compareTo(VBA_PROJECT) == 0)
158:                 result = 1;
159:               else if (name2.compareTo(VBA_PROJECT) == 0)
160:                 result = -1;
161:               else
162:               {
163:                 if (name1.startsWith("__") && name2.startsWith("__"))
164:                 {
165:                   // Betweeen __SRP_0 and __SRP_1 just sort as normal
166:                   result = name1.compareToIgnoreCase(name2);
167:                 }
168:                 else if (name1.startsWith("__"))
169:                 {
170:                   // If only name1 is __XXX then this will be placed after name2
171:                   result = 1;
172:                 }
173:                 else if (name2.startsWith("__"))
174:                 {
175:                   // If only name2 is __XXX then this will be placed after name1
176:                   result = -1;
177:                 }
178:                 else
179:                   // result = name1.compareTo(name2);
180:                   // The default case is to sort names ignoring case
181:                   result = name1.compareToIgnoreCase(name2);
182:               }
183:             }
184:             return result;
185:         }
