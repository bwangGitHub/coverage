 30: /**
 31:  * Allows the user to lookup the font metrics for a particular font without
 32:  * actually having the font on the system. The font details are loaded as a
 33:  * resource from the POI jar file (or classpath) and should be contained in path
 34:  * "/font_metrics.properties". The font widths are for a 10 point version of the
 35:  * font. Use a multiplier for other sizes.
 36:  *
 37:  * @author Glen Stampoultzis (glens at apache.org)
 38:  */
 39: final class StaticFontMetrics {
 40: 	/** The font metrics property file we're using */
 41: 	private static Properties fontMetricsProps;
 42: 	/** Our cache of font details we've already looked up */
 43: 	private static Map<String, FontDetails> fontDetailsMap = new HashMap<String, FontDetails>();
 44: 
 45: 	/**
 46: 	 * Retrieves the fake font details for a given font.
 47: 	 *
 48: 	 * @param font
 49: 	 *            the font to lookup.
 50: 	 * @return the fake font.
 51: 	 */
 52: 	public static FontDetails getFontDetails(Font font) {
 53: 		// If we haven't already identified out font metrics file,
 54: 		// figure out which one to use and load it
 55: 		if (fontMetricsProps == null) {
 56: 			InputStream metricsIn = null;
 57: 			try {
 58: 				fontMetricsProps = new Properties();
 59: 
 60: 				// Check to see if the font metric file was specified
 61: 				// as a system property
 62: 				String propFileName = null;
 63: 				try {
 64: 					propFileName = System.getProperty("font.metrics.filename");
 65: 				} catch (SecurityException e) {
 66: 				}
 67: 
 68: 				if (propFileName != null) {
 69: 					File file = new File(propFileName);
 70: 					if (!file.exists())
 71: 						throw new FileNotFoundException(
 72: 								"font_metrics.properties not found at path "
 73: 										+ file.getAbsolutePath());
 74: 					metricsIn = new FileInputStream(file);
 75: 				} else {
 76: 					// Use the built-in font metrics file off the classpath
 77: 					metricsIn = FontDetails.class.getResourceAsStream("/font_metrics.properties");
 78: 					if (metricsIn == null)
 79: 						throw new FileNotFoundException(
 80: 								"font_metrics.properties not found in classpath");
 81: 				}
 82: 				fontMetricsProps.load(metricsIn);
 83: 			} catch (IOException e) {
 84: 				throw new RuntimeException("Could not load font metrics: " + e.getMessage());
 85: 			} finally {
 86: 				if (metricsIn != null) {
 87: 					try {
 88: 						metricsIn.close();
 89: 					} catch (IOException ignore) {
 90: 					}
 91: 				}
 92: 			}
 93: 		}
 94: 
 95: 		// Grab the base name of the font they've asked about
 96: 		String fontName = font.getName();
 97: 
 98: 		// Some fonts support plain/bold/italic/bolditalic variants
 99: 		// Others have different font instances for bold etc
100: 		// (eg font.dialog.plain.* vs font.Californian FB Bold.*)
101: 		String fontStyle = "";
102: 		if (font.isPlain())
103: 			fontStyle += "plain";
104: 		if (font.isBold())
105: 			fontStyle += "bold";
106: 		if (font.isItalic())
107: 			fontStyle += "italic";
108: 
109: 		// Do we have a definition for this font with just the name?
110: 		// If not, check with the font style added
111: 		if (fontMetricsProps.get(FontDetails.buildFontHeightProperty(fontName)) == null
112: 				&& fontMetricsProps.get(FontDetails.buildFontHeightProperty(fontName + "."
113: 						+ fontStyle)) != null) {
114: 			// Need to add on the style to the font name
115: 			fontName += "." + fontStyle;
116: 		}
117: 
118: 		// Get the details on this font
119: 		if (fontDetailsMap.get(fontName) == null) {
120: 			FontDetails fontDetails = FontDetails.create(fontName, fontMetricsProps);
121: 			fontDetailsMap.put(fontName, fontDetails);
122: 			return fontDetails;
123: 		}
124: 		return fontDetailsMap.get(fontName);
125: 	}
126: }
