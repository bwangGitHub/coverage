 35:  * The formula record aggregate is used to join together the formula record and it's
 36:  * (optional) string record and (optional) Shared Formula Record (template reads, excel optimization).

 38:  * @author Glen Stampoultzis (glens at apache.org)
 39:  * @author Vladimirs Abramovs(Vladimirs.Abramovs at exigenservices.com) - Array Formula support

 49: 	/**
 50: 	 * @param stringRec may be <code>null</code> if this formula does not have a cached text
 51: 	 * value.
 52: 	 * @param svm the {@link SharedValueManager} for the current sheet
 53: 	 */
 54: 	public FormulaRecordAggregate(FormulaRecord formulaRec, StringRecord stringRec, SharedValueManager svm) {
 55: 		if (svm == null) {
 56: 			throw new IllegalArgumentException("sfm must not be null");
 57: 		}
 58: 		if (formulaRec.hasCachedResultString()) {
 59: 			if (stringRec == null) {
 60: 				throw new RecordFormatException("Formula record flag is set but String record was not found");
 61: 			}
 62: 			_stringRecord = stringRec;
 63: 		} else {
 64: 			// Usually stringRec is null here (in agreement with what the formula rec says).
 65: 			// In the case where an extra StringRecord is erroneously present, Excel (2007)
 66: 			// ignores it (see bug 46213).
 67: 			_stringRecord = null;
 68: 		}
 69: 
 70: 		_formulaRecord = formulaRec;
 71: 		_sharedValueManager = svm;
 72: 		if (formulaRec.isSharedFormula()) {
 73: 			CellReference firstCell = formulaRec.getFormula().getExpReference();
 74: 			if (firstCell == null) {
 75: 				handleMissingSharedFormulaRecord(formulaRec);
 76: 			} else {
 77: 				_sharedFormulaRecord = svm.linkSharedFormulaRecord(firstCell, this);
 78: 			}
 79: 		}
 80: 	}
 81: 	/**
 82: 	 * Sometimes the shared formula flag "seems" to be erroneously set (because the corresponding
 83: 	 * {@link SharedFormulaRecord} does not exist). Normally this would leave no way of determining
 84: 	 * the {@link Ptg} tokens for the formula.  However as it turns out in these
 85: 	 * cases, Excel encodes the unshared {@link Ptg} tokens in the right place (inside the {@link
 86: 	 * FormulaRecord}).  So the the only thing that needs to be done is to ignore the erroneous
 87: 	 * shared formula flag.<br/>
 88: 	 *
 89: 	 * This method may also be used for setting breakpoints to help diagnose issues regarding the
 90: 	 * abnormally-set 'shared formula' flags.
 91: 	 * (see TestValueRecordsAggregate.testSpuriousSharedFormulaFlag()).<p/>
 92: 	 */
 93: 	private static void handleMissingSharedFormulaRecord(FormulaRecord formula) {
 94: 		// make sure 'unshared' formula is actually available
 95: 		Ptg firstToken = formula.getParsedExpression()[0];
 96: 		if (firstToken instanceof ExpPtg) {
 97: 			throw new RecordFormatException(
 98: 					"SharedFormulaRecord not found for FormulaRecord with (isSharedFormula=true)");
 99: 		}
100: 		// could log an info message here since this is a fairly unusual occurrence.
101: 		formula.setSharedFormula(false); // no point leaving the flag erroneously set
102: 	}

189: 		if (_sharedFormulaRecord != null) {
190: 			return _sharedFormulaRecord.getFormulaTokens(_formulaRecord);
191: 		}
192: 		CellReference expRef = _formulaRecord.getFormula().getExpReference();
193: 		if (expRef != null) {
194: 			ArrayRecord arec = _sharedValueManager.getArrayRecord(expRef.getRow(), expRef.getCol());
195: 			return arec.getFormulaTokens();
196: 		}
197: 		return _formulaRecord.getParsedExpression();

208: 	public void unlinkSharedFormula() {
209: 		SharedFormulaRecord sfr = _sharedFormulaRecord;
210: 		if (sfr == null) {
211: 			throw new IllegalStateException("Formula not linked to shared formula");
212: 		}
213: 		Ptg[] ptgs = sfr.getFormulaTokens(_formulaRecord);
214: 		_formulaRecord.setParsedExpression(ptgs);
215: 		//Now its not shared!
216: 		_formulaRecord.setSharedFormula(false);
217: 		_sharedFormulaRecord = null;
218: 	}
219: 	/**
220: 	 * Should be called by any code which is either deleting this formula cell, or changing
221: 	 * its type.  This method gives the aggregate a chance to unlink any shared formula
222: 	 * that may be involved with this cell formula.
223: 	 */
224: 	public void notifyFormulaChanging() {
225: 		if (_sharedFormulaRecord != null) {
226: 			_sharedValueManager.unlink(_sharedFormulaRecord);
227: 		}

230: 		if (_sharedFormulaRecord != null) {
231: 			return false;
232: 		}
233:         CellReference expRef = _formulaRecord.getFormula().getExpReference();
234:         ArrayRecord arec = expRef == null ? null : _sharedValueManager.getArrayRecord(expRef.getRow(), expRef.getCol());
235: 		return arec != null;

238: 	public CellRangeAddress getArrayFormulaRange() {
239: 		if (_sharedFormulaRecord != null) {
240: 			throw new IllegalStateException("not an array formula cell.");
241: 		}
242: 		CellReference expRef = _formulaRecord.getFormula().getExpReference();
243: 		if (expRef == null) {
244: 			throw new IllegalStateException("not an array formula cell.");
245: 		}
246: 		ArrayRecord arec = _sharedValueManager.getArrayRecord(expRef.getRow(), expRef.getCol());
247: 		if (arec == null) {
248: 			throw new IllegalStateException("ArrayRecord was not found for the locator " + expRef.formatAsString());
249: 		}
250: 		CellRangeAddress8Bit a = arec.getRange();
251: 		return new CellRangeAddress(a.getFirstRow(), a.getLastRow(), a.getFirstColumn(),a.getLastColumn());
252: 	}

255: 
256: 		ArrayRecord arr = new ArrayRecord(Formula.create(ptgs), new CellRangeAddress8Bit(r.getFirstRow(), r.getLastRow(), r.getFirstColumn(), r.getLastColumn()));
257: 		_sharedValueManager.addArrayRecord(arr);
258: 	}
259: 	/**
260: 	 * Removes an array formula
261: 	 * @return the range of the array formula containing the specified cell. Never <code>null</code>
262: 	 */
263: 	public CellRangeAddress removeArrayFormula(int rowIndex, int columnIndex) {
264: 		CellRangeAddress8Bit a = _sharedValueManager.removeArrayFormula(rowIndex, columnIndex);
265:         // at this point FormulaRecordAggregate#isPartOfArrayFormula() should return false
266:         _formulaRecord.setParsedExpression(null);
267:         return new CellRangeAddress(a.getFirstRow(), a.getLastRow(), a.getFirstColumn(), a.getLastColumn());
268: 	}
