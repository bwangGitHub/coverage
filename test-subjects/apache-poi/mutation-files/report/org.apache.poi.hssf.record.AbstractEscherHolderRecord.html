 33:  * The escher container record is used to hold escher records.  It is abstract and
 34:  * must be subclassed for maximum benefit.

 36:  * @author Glen Stampoultzis (glens at apache.org)

 39: public abstract class AbstractEscherHolderRecord extends Record {
 40:     private static boolean DESERIALISE;
 41:     static {
 42:     try {
 43:             DESERIALISE = (System.getProperty("poi.deserialize.escher") != null);
 44:         } catch (SecurityException e) {
 45:             DESERIALISE = false;
 46:         }
 47:     }

 49:     private List<EscherRecord> escherRecords;
 50:     private LazilyConcatenatedByteArray rawDataContainer = new LazilyConcatenatedByteArray();
 51: 
 52:     public AbstractEscherHolderRecord()
 53:     {
 54:         escherRecords = new ArrayList<EscherRecord>();
 55:     }
 56: 
 57:     public AbstractEscherHolderRecord(RecordInputStream in)
 58:     {
 59:         escherRecords = new ArrayList<EscherRecord>();
 60:         if (! DESERIALISE )
 61:         {
 62:             rawDataContainer.concatenate(in.readRemainder());
 63:         }
 64:         else
 65:         {
 66:             byte[] data = in.readAllContinuedRemainder();
 67:             convertToEscherRecords( 0, data.length, data );
 68:         }
 69:     }
 70: 
 71:     protected void convertRawBytesToEscherRecords() {
 72:         byte[] rawData = getRawData();
 73:     	convertToEscherRecords(0, rawData.length, rawData);
 74:     }
 75:     private void convertToEscherRecords( int offset, int size, byte[] data )
 76:     {
 77:          escherRecords.clear();
 78:         EscherRecordFactory recordFactory = new DefaultEscherRecordFactory();
 79:         int pos = offset;
 80:         while ( pos < offset + size )
 81:         {
 82:             EscherRecord r = recordFactory.createRecord(data, pos);
 83:             int bytesRead = r.fillFields(data, pos, recordFactory );
 84:             escherRecords.add(r);
 85:             pos += bytesRead;
 86:         }
 87:     }

 89:     public String toString()
 90:     {
 91:         StringBuffer buffer = new StringBuffer();
 92: 
 93:         final String nl = System.getProperty("line.separator");
 94:         buffer.append('[' + getRecordName() + ']' + nl);
 95:         if (escherRecords.size() == 0)
 96:             buffer.append("No Escher Records Decoded" + nl);
 97:         for ( Iterator<EscherRecord> iterator = escherRecords.iterator(); iterator.hasNext(); )
 98:         {
 99:             EscherRecord r = iterator.next();
100:             buffer.append(r.toString());
101:         }
102:         buffer.append("[/" + getRecordName() + ']' + nl);
103: 
104:         return buffer.toString();
105:     }

109:     public int serialize(int offset, byte[] data)
110:     {
111:         LittleEndian.putShort( data, 0 + offset, getSid() );
112:         LittleEndian.putShort( data, 2 + offset, (short) ( getRecordSize() - 4 ) );
113:         byte[] rawData = getRawData();
114:         if ( escherRecords.size() == 0 && rawData != null )
115:         {
116:             LittleEndian.putShort(data, 0 + offset, getSid());
117:             LittleEndian.putShort(data, 2 + offset, (short)(getRecordSize() - 4));
118:             System.arraycopy( rawData, 0, data, 4 + offset, rawData.length);
119:             return rawData.length + 4;
120:         }
121:         LittleEndian.putShort(data, 0 + offset, getSid());
122:         LittleEndian.putShort(data, 2 + offset, (short)(getRecordSize() - 4));
123: 
124:         int pos = offset + 4;
125:         for ( Iterator<EscherRecord> iterator = escherRecords.iterator(); iterator.hasNext(); )
126:         {
127:             EscherRecord r = iterator.next();
128:             pos += r.serialize( pos, data, new NullEscherSerializationListener() );
129:         }
130:         return getRecordSize();
131:     }

133:     public int getRecordSize() {
134:         byte[] rawData = getRawData();
135:         if (escherRecords.size() == 0 && rawData != null) {
136:             // XXX: It should be possible to derive this without concatenating the array, too.
137:             return rawData.length;
138:         }
139:         int size = 0;
140:         for ( Iterator<EscherRecord> iterator = escherRecords.iterator(); iterator.hasNext(); )
141:         {
142:             EscherRecord r = iterator.next();
143:             size += r.getRecordSize();
144:         }
145:         return size;
146:     }

192:     /**
193:      * Descends into all our children, returning the
194:      *  first EscherRecord with the given id, or null
195:      *  if none found
196:      */
197:     public EscherRecord findFirstWithId(short id) {
198:     	return findFirstWithId(id, getEscherRecords());
199:     }
200:     private EscherRecord findFirstWithId(short id, List<EscherRecord> records) {
201:     	// Check at our level
202:     	for(Iterator<EscherRecord> it = records.iterator(); it.hasNext();) {
203:     		EscherRecord r = it.next();
204:     		if(r.getRecordId() == id) {
205:     			return r;
206:     		}
207:     	}
208: 
209:     	// Then check our children in turn
210:     	for(Iterator<EscherRecord> it = records.iterator(); it.hasNext();) {
211:     		EscherRecord r = it.next();
212:     		if(r.isContainerRecord()) {
213:     			EscherRecord found = findFirstWithId(id, r.getChildRecords());
214:     			if(found != null) {
215:     				return found;
216:     			}
217:     		}
218:     	}
219: 
220:     	// Not found in this lot
221:     	return null;
222:     }
