 36:     public static final int LINESTYLE_LONGDASHGEL = 7;        // long dash style
 37:     public static final int LINESTYLE_DASHDOTGEL = 8;         // dash short dash

 55:     HSSFShape( HSSFShape parent, HSSFAnchor anchor )

 57:         this.parent = parent;
 58:         this.anchor = anchor;
 59:     }
 60:

 62:      * Gets the parent shape.
 63:      */
 64:     public HSSFShape getParent()
 65:     {
 66:         return parent;
 67:     }
 68: 
 69:     /**
 70:      * @return  the anchor that is used by this shape.
 71:      */
 72:     public HSSFAnchor getAnchor()
 73:     {
 74:         return anchor;
 75:     }
 76: 
 77:     /**
 78:      * Sets a particular anchor.  A top-level shape must have an anchor of
 79:      * HSSFClientAnchor.  A child anchor must have an anchor of HSSFChildAnchor
 80:      *
 81:      * @param anchor    the anchor to use.
 82:      * @throws IllegalArgumentException     when the wrong anchor is used for
 83:      *                                      this particular shape.
 84:      *
 85:      * @see HSSFChildAnchor
 86:      * @see HSSFClientAnchor
 87:      */
 88:     public void setAnchor( HSSFAnchor anchor )
 89:     {
 90:         if ( parent == null )
 91:         {
 92:             if ( anchor instanceof HSSFChildAnchor )
 93:                 throw new IllegalArgumentException( "Must use client anchors for shapes directly attached to sheet." );
 94:         }
 95:         else
 96:         {
 97:             if ( anchor instanceof HSSFClientAnchor )
 98:                 throw new IllegalArgumentException( "Must use child anchors for shapes attached to groups." );
 99:         }
100: 
101:         this.anchor = anchor;
102:     }
103:

105:      * The color applied to the lines of this shape.
106:      */
107:     public int getLineStyleColor() {
108:         return _lineStyleColor;
109:     }
110: 
111:     /**
112:      * The color applied to the lines of this shape.
113:      */
114:     public void setLineStyleColor(int lineStyleColor) {
115:         _lineStyleColor = lineStyleColor;

143:     public void setFillColor( int red, int green, int blue )
144:     {
145:         this._fillColor = ((blue) << 16) | ((green) << 8) | red;

163:         _lineWidth = lineWidth;

170:         return _lineStyle;

178:     public void setLineStyle(int lineStyle) {

180:     }
