 50:     /**
 51:      * @param in the RecordInputstream to read the record from

 53:     public LabelRecord(RecordInputStream in)
 54:     {
 55:         field_1_row          = in.readUShort();
 56:         field_2_column       = in.readShort();
 57:         field_3_xf_index     = in.readShort();
 58:         field_4_string_len   = in.readShort();
 59:         field_5_unicode_flag = in.readByte();
 60:         if (field_4_string_len > 0) {
 61:             if (isUnCompressedUnicode()) {
 62:                 field_6_value = in.readUnicodeLEString(field_4_string_len);
 63:             } else {
 64:                 field_6_value = in.readCompressedUnicode(field_4_string_len);
 65:             }
 66:         } else {
 67:             field_6_value = "";
 68:         }
 69: 
 70:         if (in.remaining() > 0) {
 71:            logger.log(POILogger.INFO,
 72:                    "LabelRecord data remains: " + in.remaining() +
 73:                            " : " + HexDump.toHex(in.readRemainder())
 74:            );
 75:         }
 76:     }
