 31:  *
 32:  * <p>The property's <strong>ID</strong> gives the property a meaning
 33:  * in the context of its {@link Section}. Each {@link Section} spans
 34:  * its own name space of property IDs.</p>
 35:  *
 36:  * <p>The property's <strong>type</strong> determines how its
 37:  * <strong>value </strong> is interpreted. For example, if the type is

 41:  * string. The type {@link Variant#VT_I4} denotes a four-byte integer
 42:  * value, {@link Variant#VT_FILETIME} some date and time (of a

 44:  *
 45:  * <p>Please note that not all {@link Variant} types yet. This might change
 46:  * over time but largely depends on your feedback so that the POI team knows
 47:  * which variant types are really needed. So please feel free to submit error
 48:  * reports or patches for the types you need.</p>
 49:  *

129:      * <p>Creates a {@link Property} instance by reading its bytes
130:      * from the property set stream.</p>
131:      *
132:      * @param id The property's ID.
133:      * @param src The bytes the property set stream consists of.
134:      * @param offset The property's type/value pair's offset in the
135:      * section.
136:      * @param length The property's type/value pair's length in bytes.
137:      * @param codepage The section's and thus the property's
138:      * codepage. It is needed only when reading string values.
139:      * @exception UnsupportedEncodingException if the specified codepage is not
140:      * supported.
141:      */
142:     public Property(final long id, final byte[] src, final long offset,
143:                     final int length, final int codepage)
144:     throws UnsupportedEncodingException
145:     {
146:         this.id = id;
147: 
148:         /*
149:          * ID 0 is a special case since it specifies a dictionary of
150:          * property IDs and property names.
151:          */
152:         if (id == 0)
153:         {
154:             value = readDictionary(src, offset, length, codepage);
155:             return;
156:         }
157: 
158:         int o = (int) offset;
159:         type = LittleEndian.getUInt(src, o);
160:         o += LittleEndian.INT_SIZE;
161: 
162:         try
163:         {
164:             value = VariantSupport.read(src, o, length, (int) type, codepage);
165:         }
166:         catch (UnsupportedVariantTypeException ex)
167:         {
168:             VariantSupport.writeUnsupportedTypeMessage(ex);
169:             value = ex.getValue();
170:         }
171:     }

184:     /**
185:      * <p>Reads a dictionary.</p>
186:      *
187:      * @param src The byte array containing the bytes making out the dictionary.
188:      * @param offset At this offset within <var>src </var> the dictionary
189:      *        starts.
190:      * @param length The dictionary contains at most this many bytes.
191:      * @param codepage The codepage of the string values.
192:      * @return The dictonary
193:      * @throws UnsupportedEncodingException if the dictionary's codepage is not
194:      *         (yet) supported.
195:      */
196:     protected Map<?, ?> readDictionary(final byte[] src, final long offset,
197:                                  final int length, final int codepage)
198:     throws UnsupportedEncodingException
199:     {
200:         /* Check whether "offset" points into the "src" array". */
201:         if (offset < 0 || offset > src.length)
202:             throw new HPSFRuntimeException
203:                 ("Illegal offset " + offset + " while HPSF stream contains " +
204:                  length + " bytes.");
205:         int o = (int) offset;
206: 
207:         /*
208:          * Read the number of dictionary entries.
209:          */
210:         final long nrEntries = LittleEndian.getUInt(src, o);
211:         o += LittleEndian.INT_SIZE;
212: 
213:         final Map<Object, Object> m = new LinkedHashMap<Object, Object>(
214:                 (int) nrEntries, (float) 1.0 );
215: 
216:         try
217:         {
218:             for (int i = 0; i < nrEntries; i++)
219:             {
220:                 /* The key. */
221:                 final Long id = Long.valueOf(LittleEndian.getUInt(src, o));
222:                 o += LittleEndian.INT_SIZE;
223: 
224:                 /* The value (a string). The length is the either the
225:                  * number of (two-byte) characters if the character set is Unicode
226:                  * or the number of bytes if the character set is not Unicode.
227:                  * The length includes terminating 0x00 bytes which we have to strip
228:                  * off to create a Java string. */
229:                 long sLength = LittleEndian.getUInt(src, o);
230:                 o += LittleEndian.INT_SIZE;
231: 
232:                 /* Read the string. */
233:                 final StringBuffer b = new StringBuffer();
234:                 switch (codepage)
235:                 {
236:                     case -1:
237:                     {
238:                         /* Without a codepage the length is equal to the number of
239:                          * bytes. */
240:                         b.append(new String(src, o, (int) sLength));
241:                         break;
242:                     }
243:                     case Constants.CP_UNICODE:
244:                     {
245:                         /* The length is the number of characters, i.e. the number
246:                          * of bytes is twice the number of the characters. */
247:                         final int nrBytes = (int) (sLength * 2);
248:                         final byte[] h = new byte[nrBytes];
249:                         for (int i2 = 0; i2 < nrBytes; i2 += 2)
250:                         {
251:                             h[i2] = src[o + i2 + 1];
252:                             h[i2 + 1] = src[o + i2];
253:                         }
254:                         b.append(new String(h, 0, nrBytes,
255:                                 VariantSupport.codepageToEncoding(codepage)));
256:                         break;
257:                     }
258:                     default:
259:                     {
260:                         /* For encodings other than Unicode the length is the number
261:                          * of bytes. */
262:                         b.append(new String(src, o, (int) sLength,
263:                                  VariantSupport.codepageToEncoding(codepage)));
264:                         break;
265:                     }
266:                 }
267: 
268:                 /* Strip 0x00 characters from the end of the string: */
269:                 while (b.length() > 0 && b.charAt(b.length() - 1) == 0x00)
270:                     b.setLength(b.length() - 1);
271:                 if (codepage == Constants.CP_UNICODE)
272:                 {
273:                     if (sLength % 2 == 1)
274:                         sLength++;
275:                     o += (sLength + sLength);
276:                 }
277:                 else
278:                     o += sLength;
279:                 m.put(id, b.toString());
280:             }
281:         }
282:         catch (RuntimeException ex)
283:         {
284:             final POILogger l = POILogFactory.getLogger(getClass());
285:             l.log(POILogger.WARN,
286:                     "The property set's dictionary contains bogus data. "
287:                     + "All dictionary entries starting with the one with ID "
288:                     + id + " will be ignored.", ex);
289:         }
290:         return m;
291:     }

295:     /**
296:      * <p>Returns the property's size in bytes. This is always a multiple of
297:      * 4.</p>
298:      *
299:      * @return the property's size in bytes
300:      *
301:      * @exception WritingNotSupportedException if HPSF does not yet support the
302:      * property's variant type.
303:      */
304:     protected int getSize() throws WritingNotSupportedException
305:     {
306:         int length = VariantSupport.getVariantLength(type);
307:         if (length >= 0)
308:             return length; /* Fixed length */
309:         if (length == -2)
310:             /* Unknown length */
311:             throw new WritingNotSupportedException(type, null);
312: 
313:         /* Variable length: */
314:         final int PADDING = 4; /* Pad to multiples of 4. */
315:         switch ((int) type)
316:         {
317:             case Variant.VT_LPSTR:
318:             {
319:                 int l = ((String) value).length() + 1;
320:                 int r = l % PADDING;
321:                 if (r > 0)
322:                     l += PADDING - r;
323:                 length += l;
324:                 break;
325:             }
326:             case Variant.VT_EMPTY:
327:                 break;
328:             default:
329:                 throw new WritingNotSupportedException(type, value);
330:         }
331:         return length;
332:     }

337:      * <p>Compares two properties.</p> <p>Please beware that a property with
338:      * ID == 0 is a special case: It does not have a type, and its value is the
339:      * section's dictionary. Another special case are strings: Two properties
340:      * may have the different types Variant.VT_LPSTR and Variant.VT_LPWSTR;</p>
341:      *
342:      * @see Object#equals(java.lang.Object)
343:      */
344:     public boolean equals(final Object o)
345:     {
346:         if (!(o instanceof Property)) {
347:             return false;
348:         }
349:         final Property p = (Property) o;
350:         final Object pValue = p.getValue();
351:         final long pId = p.getID();
352:         if (id != pId || (id != 0 && !typesAreEqual(type, p.getType())))
353:             return false;
354:         if (value == null && pValue == null)
355:             return true;
356:         if (value == null || pValue == null)
357:             return false;
358: 
359:         /* It's clear now that both values are non-null. */
360:         final Class<?> valueClass = value.getClass();
361:         final Class<?> pValueClass = pValue.getClass();
362:         if (!(valueClass.isAssignableFrom(pValueClass)) &&
363:             !(pValueClass.isAssignableFrom(valueClass)))
364:             return false;
365: 
366:         if (value instanceof byte[])
367:             return Util.equal((byte[]) value, (byte[]) pValue);
368: 
369:         return value.equals(pValue);
370:     }
