 22: /**
 23:  *  <p>Represents a class ID (16 bytes). Unlike other little-endian
 24:  *  type the {@link ClassID} is not just 16 bytes stored in the wrong

 26:  *  words (2 bytes each) followed by 8 bytes.</p>

 28:  * @author Rainer Klute <a

110:     /**

112:      * little-endian into big-endian.</p>
113:      *
114:      * @param src The byte array to read from
115:      *
116:      * @param offset The offset within the <var>src</var> byte array
117:      *
118:      * @return A byte array containing the class ID.
119:      */
120:     public byte[] read(final byte[] src, final int offset)
121:     {
122:         bytes = new byte[16];
123: 
124:         /* Read double word. */
125:         bytes[0] = src[3 + offset];
126:         bytes[1] = src[2 + offset];
127:         bytes[2] = src[1 + offset];
128:         bytes[3] = src[0 + offset];
129: 
130:         /* Read first word. */
131:         bytes[4] = src[5 + offset];
132:         bytes[5] = src[4 + offset];
133: 
134:         /* Read second word. */
135:         bytes[6] = src[7 + offset];
136:         bytes[7] = src[6 + offset];
137: 
138:         /* Read 8 bytes. */
139:         for (int i = 8; i < 16; i++)
140:             bytes[i] = src[i + offset];
141: 
142:         return bytes;
143:     }

147:     /**
148:      * <p>Writes the class ID to a byte array in the
149:      * little-endian format.</p>
150:      *
151:      * @param dst The byte array to write to.
152:      *
153:      * @param offset The offset within the <var>dst</var> byte array.
154:      *
155:      * @exception ArrayStoreException if there is not enough room for the class
156:      * ID 16 bytes in the byte array after the <var>offset</var> position.
157:      */
158:     public void write(final byte[] dst, final int offset)
159:     throws ArrayStoreException
160:     {
161:         /* Check array size: */
162:         if (dst.length < 16)
163:             throw new ArrayStoreException
164:                 ("Destination byte[] must have room for at least 16 bytes, " +
165:                  "but has a length of only " + dst.length + ".");
166:         /* Write double word. */
167:         dst[0 + offset] = bytes[3];
168:         dst[1 + offset] = bytes[2];
169:         dst[2 + offset] = bytes[1];
170:         dst[3 + offset] = bytes[0];
171: 
172:         /* Write first word. */
173:         dst[4 + offset] = bytes[5];
174:         dst[5 + offset] = bytes[4];
175: 
176:         /* Write second word. */
177:         dst[6 + offset] = bytes[7];
178:         dst[7 + offset] = bytes[6];
179: 
180:         /* Write 8 bytes. */
181:         for (int i = 8; i < 16; i++)
182:             dst[i + offset] = bytes[i];
183:     }
