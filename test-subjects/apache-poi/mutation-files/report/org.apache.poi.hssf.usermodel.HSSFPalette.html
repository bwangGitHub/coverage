 23: /**
 24:  * Represents a workbook color palette.
 25:  * Internally, the XLS format refers to colors using an offset into the palette
 26:  * record.  Thus, the first color in the palette has the index 0x8, the second
 27:  * has the index 0x9, etc. through 0x40
 28:  *
 29:  * @author Brian Sanders (bsanders at risklabs dot com)
 30:  */
 31: public final class HSSFPalette {
 32:     private PaletteRecord _palette;
 33: 
 34:     protected HSSFPalette(PaletteRecord palette)
 35:     {
 36:         _palette = palette;
 37:     }
 38: 
 39:     /**
 40:      * Retrieves the color at a given index
 41:      *
 42:      * @param index the palette index, between 0x8 to 0x40 inclusive
 43:      * @return the color, or null if the index is not populated
 44:      */
 45:     public HSSFColor getColor(short index)
 46:     {
 47:         //Handle the special AUTOMATIC case
 48:         if (index == HSSFColor.AUTOMATIC.index) {
 49:             return HSSFColor.AUTOMATIC.getInstance();
 50:         }
 51:         byte[] b = _palette.getColor(index);
 52:           if (b != null)
 53:           {
 54:              return new CustomColor(index, b);
 55:           }
 56:         return null;
 57:     }
 58:     /**
 59:      * Retrieves the color at a given index
 60:      *
 61:      * @param index the palette index, between 0x8 to 0x40 inclusive
 62:      * @return the color, or null if the index is not populated
 63:      */
 64:     public HSSFColor getColor(int index) {
 65:     	return getColor((short)index);
 66:     }
 67: 
 68:     /**
 69:      * Finds the first occurance of a given color
 70:      *
 71:      * @param red the RGB red component, between 0 and 255 inclusive
 72:      * @param green the RGB green component, between 0 and 255 inclusive
 73:      * @param blue the RGB blue component, between 0 and 255 inclusive
 74:      * @return the color, or null if the color does not exist in this palette
 75:      */
 76:     public HSSFColor findColor(byte red, byte green, byte blue)
 77:     {
 78:         byte[] b = _palette.getColor(PaletteRecord.FIRST_COLOR_INDEX);
 79:         for (short i = PaletteRecord.FIRST_COLOR_INDEX; b != null;
 80:             b = _palette.getColor(++i))
 81:         {
 82:             if (b[0] == red && b[1] == green && b[2] == blue)
 83:             {
 84:                 return new CustomColor(i, b);
 85:             }
 86:         }
 87:         return null;
 88:     }

 90:     /**
 91:      * Finds the closest matching color in the custom palette.  The
 92:      * method for finding the distance between the colors is fairly
 93:      * primative.
 94:      *
 95:      * @param red   The red component of the color to match.
 96:      * @param green The green component of the color to match.
 97:      * @param blue  The blue component of the color to match.
 98:      * @return  The closest color or null if there are no custom
 99:      *          colors currently defined.
100:      */
101:     public HSSFColor findSimilarColor(byte red, byte green, byte blue) {
102:     	return findSimilarColor(unsignedInt(red), unsignedInt(green), unsignedInt(blue));
103:     }
104:     /**
105:      * Finds the closest matching color in the custom palette.  The
106:      * method for finding the distance between the colors is fairly
107:      * primative.
108:      *
109:      * @param red   The red component of the color to match.
110:      * @param green The green component of the color to match.
111:      * @param blue  The blue component of the color to match.
112:      * @return  The closest color or null if there are no custom
113:      *          colors currently defined.
114:      */
115:     public HSSFColor findSimilarColor(int red, int green, int blue) {
116:         HSSFColor result = null;
117:         int minColorDistance = Integer.MAX_VALUE;
118:         byte[] b = _palette.getColor(PaletteRecord.FIRST_COLOR_INDEX);
119:         for (short i = PaletteRecord.FIRST_COLOR_INDEX; b != null;
120:             b = _palette.getColor(++i))
121:         {
122:             int colorDistance = Math.abs(red - unsignedInt(b[0])) +
123:             	Math.abs(green - unsignedInt(b[1])) +
124:             	Math.abs(blue - unsignedInt(b[2]));
125:             if (colorDistance < minColorDistance)
126:             {
127:                 minColorDistance = colorDistance;
128:                 result = getColor(i);
129:             }
130:         }
131:         return result;
132:     }

134:     /**
135:      * Turn a byte of between -127 and 127 into something between
136:      *  0 and 255, so distance calculations work as expected.
137:      */
138:     private int unsignedInt(byte b) {
139:     	return 0xFF & ((int) b);
140:     }

143:      * Sets the color at the given offset
144:      *
145:      * @param index the palette index, between 0x8 to 0x40 inclusive
146:      * @param red the RGB red component, between 0 and 255 inclusive
147:      * @param green the RGB green component, between 0 and 255 inclusive
148:      * @param blue the RGB blue component, between 0 and 255 inclusive
149:      */
150:     public void setColorAtIndex(short index, byte red, byte green, byte blue)
151:     {
152:         _palette.setColor(index, red, green, blue);

155:     /**
156:      * Adds a new color into an empty color slot.
157:      * @param red       The red component
158:      * @param green     The green component
159:      * @param blue      The blue component
160:      *
161:      * @return  The new custom color.
162:      *
163:      * @throws RuntimeException if there are more more free color indexes.
164:      */
165:     public HSSFColor addColor( byte red, byte green, byte blue )
166:     {
167:         byte[] b = _palette.getColor(PaletteRecord.FIRST_COLOR_INDEX);
168:         short i;
169:         for (i = PaletteRecord.FIRST_COLOR_INDEX; i < PaletteRecord.STANDARD_PALETTE_SIZE + PaletteRecord.FIRST_COLOR_INDEX; b = _palette.getColor(++i))
170:         {
171:             if (b == null)
172:             {
173:                 setColorAtIndex( i, red, green, blue );
174:                 return getColor(i);
175:             }
176:         }
177:         throw new RuntimeException("Could not find free color index");
178:     }

180:     private static final class CustomColor extends HSSFColor {
181:         private short _byteOffset;
182:         private byte _red;
183:         private byte _green;
184:         private byte _blue;
185: 
186:         public CustomColor(short byteOffset, byte[] colors)
187:         {
188:             this(byteOffset, colors[0], colors[1], colors[2]);
189:         }
190: 
191:         private CustomColor(short byteOffset, byte red, byte green, byte blue)
192:         {
193:             _byteOffset = byteOffset;
194:             _red = red;
195:             _green = green;
196:             _blue = blue;
197:         }
198: 
199:         public short getIndex()
200:         {
201:             return _byteOffset;
202:         }
203: 
204:         public short[] getTriplet()
205:         {
206:             return new short[]
207:             {
208:                 (short) (_red   & 0xff),
209:                 (short) (_green & 0xff),
210:                 (short) (_blue  & 0xff)
211:             };
212:         }
213: 
214:         public String getHexString()
215:         {
216:             StringBuffer sb = new StringBuffer();
217:             sb.append(getGnumericPart(_red));
218:             sb.append(':');
219:             sb.append(getGnumericPart(_green));
220:             sb.append(':');
221:             sb.append(getGnumericPart(_blue));
222:             return sb.toString();
223:         }
224: 
225:         private String getGnumericPart(byte color)
226:         {
227:             String s;
228:             if (color == 0)
229:             {
230:                 s = "0";
231:             }
232:             else
233:             {
234:                 int c = color & 0xff; //as unsigned
235:                 c = (c << 8) | c; //pad to 16-bit
236:                 s = Integer.toHexString(c).toUpperCase();
237:                 while (s.length() < 4)
238:                 {
239:                     s = "0" + s;
240:                 }
241:             }
242:             return s;
243:         }
244:     }
