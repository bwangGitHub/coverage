108:     private int writeData( int offset, byte[] data, byte[] rawData )
109:     {
110:         int writtenActualData = 0;
111:         int writtenRawData = 0;
112:         while (writtenRawData < rawData.length)
113:         {
114:             int segmentLength = Math.min( rawData.length - writtenRawData, MAX_DATA_SIZE);
115:             if (writtenRawData / MAX_DATA_SIZE >= 2)
