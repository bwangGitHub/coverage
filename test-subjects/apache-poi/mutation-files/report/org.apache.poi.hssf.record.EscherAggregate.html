 61: /**
 62:  * This class is used to aggregate the MSODRAWING and OBJ record
 63:  * combinations.  This is necessary due to the bizare way in which
 64:  * these records are serialized.  What happens is that you get a

 67:  * records.
 68:  * <p>
 69:  * It gets even worse when you start looking at TXO records.
 70:  * <p>
 71:  * So what we do with this class is aggregate lazily.  That is

 73:  * need to modify them.

336: 	public static EscherAggregate createAggregate( List records, int locFirstDrawingRecord, DrawingManager2 drawingManager )
337: 	{
338: 		// Keep track of any shape records created so we can match them back to the object id's.
339: 		// Textbox objects are also treated as shape objects.
340: 		final List<EscherRecord> shapeRecords = new ArrayList<EscherRecord>();
341: 		EscherRecordFactory recordFactory = new DefaultEscherRecordFactory()
342: 		{
343: 			public EscherRecord createRecord( byte[] data, int offset )
344: 			{
345: 				EscherRecord r = super.createRecord( data, offset );
346: 				if ( r.getRecordId() == EscherClientDataRecord.RECORD_ID || r.getRecordId() == EscherTextboxRecord.RECORD_ID )
347: 				{
348: 					shapeRecords.add( r );
349: 				}
350: 				return r;
351: 			}
352: 		};
353: 
354: 		// Calculate the size of the buffer
355: 		EscherAggregate agg = new EscherAggregate(drawingManager);
356: 		int loc = locFirstDrawingRecord;
357: 		int dataSize = 0;
358: 		while ( loc + 1 < records.size()
359: 				&& sid( records, loc ) == DrawingRecord.sid
360: 				&& isObjectRecord( records, loc + 1 ) )
361: 		{
362: 			dataSize += ( (DrawingRecord) records.get( loc ) ).getData().length;
363: 			loc += 2;
364: 		}
365: 
366: 		// Create one big buffer
367: 		byte buffer[] = new byte[dataSize];
368: 		int offset = 0;
369: 		loc = locFirstDrawingRecord;
370: 		while ( loc + 1 < records.size()
371: 				&& sid( records, loc ) == DrawingRecord.sid
372: 				&& isObjectRecord( records, loc + 1 ) )
373: 		{
374: 			DrawingRecord drawingRecord = (DrawingRecord) records.get( loc );
375: 			System.arraycopy( drawingRecord.getData(), 0, buffer, offset, drawingRecord.getData().length );
376: 			offset += drawingRecord.getData().length;
377: 			loc += 2;
378: 		}
379: 
380: 		// Decode the shapes
381: 		//		agg.escherRecords = new ArrayList();
382: 		int pos = 0;
383: 		while ( pos < dataSize )
384: 		{
385: 			EscherRecord r = recordFactory.createRecord( buffer, pos );
386: 			int bytesRead = r.fillFields( buffer, pos, recordFactory );
387: 			agg.addEscherRecord( r );
388: 			pos += bytesRead;
389: 		}
390: 
391: 		// Associate the object records with the shapes
392: 		loc = locFirstDrawingRecord;
393: 		int shapeIndex = 0;
394: 		agg.shapeToObj = new HashMap<EscherRecord, Record>();
395: 		while ( loc + 1 < records.size()
396: 				&& sid( records, loc ) == DrawingRecord.sid
397: 				&& isObjectRecord( records, loc + 1 ) )
398: 		{
399: 			Record objRecord = (Record) records.get( loc + 1 );
400: 			agg.shapeToObj.put( shapeRecords.get( shapeIndex++ ), objRecord );
401: 			loc += 2;
402: 		}
403: 
404: 		return agg;
