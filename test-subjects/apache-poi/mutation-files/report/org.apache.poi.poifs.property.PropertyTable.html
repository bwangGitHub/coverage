 30: /**
 31:  * This class embodies the Property Table for the {@link POIFSFileSystem}; 
 32:  *  this is basically the directory for all of the documents in the
 33:  * filesystem.
 34:  *
 35:  * @author Marc Johnson (mjohnson at apache dot org)
 36:  */
 37: public final class PropertyTable extends PropertyTableBase implements BlockWritable {
 38:     private POIFSBigBlockSize _bigBigBlockSize;
 39:     private BlockWritable[]   _blocks;
 40: 
 41:     public PropertyTable(HeaderBlock headerBlock)
 42:     {
 43:         super(headerBlock);
 44:         _bigBigBlockSize = headerBlock.getBigBlockSize();
 45:         _blocks = null;
 46:     }
 47: 
 48:     /**
 49:      * reading constructor (used when we've read in a file and we want
 50:      * to extract the property table from it). Populates the
 51:      * properties thoroughly
 52:      *
 53:      * @param headerBlock the header block of the file
 54:      * @param blockList the list of blocks
 55:      *
 56:      * @exception IOException if anything goes wrong (which should be
 57:      *            a result of the input being NFG)
 58:      */
 59:     public PropertyTable(final HeaderBlock headerBlock,
 60:                          final RawDataBlockList blockList)
 61:         throws IOException
 62:     {
 63:         super(
 64:               headerBlock,
 65:               PropertyFactory.convertToProperties(
 66:                     blockList.fetchBlocks(headerBlock.getPropertyStart(), -1)
 67:               )
 68:         );
 69:         _bigBigBlockSize = headerBlock.getBigBlockSize();
 70:         _blocks      = null;
 71:     }
 72: 
 73:     /**
 74:      * Prepare to be written
 75:      */
 76:     public void preWrite()
 77:     {
 78:         Property[] properties = _properties.toArray(new Property[_properties.size()]);
 79: 
 80:         // give each property its index
 81:         for (int k = 0; k < properties.length; k++)
 82:         {
 83:             properties[ k ].setIndex(k);
 84:         }
 85: 
 86:         // allocate the blocks for the property table
 87:         _blocks = PropertyBlock.createPropertyBlockArray(_bigBigBlockSize, _properties);
 88: 
 89:         // prepare each property for writing
 90:         for (int k = 0; k < properties.length; k++)
 91:         {
 92:             properties[ k ].preWrite();
 93:         }
 94:     }
 95:     
 96:     /**
 97:      * Return the number of BigBlock's this instance uses
 98:      *
 99:      * @return count of BigBlock instances
100:      */
101:     public int countBlocks()
102:     {
103:         return (_blocks == null) ? 0
104:                                  : _blocks.length;
105:     }
106: 
107:     /**
108:      * Write the storage to an OutputStream
109:      *
110:      * @param stream the OutputStream to which the stored data should
111:      *               be written
112:      *
113:      * @exception IOException on problems writing to the specified
114:      *            stream
115:      */
116:     public void writeBlocks(final OutputStream stream)
117:         throws IOException
118:     {
119:         if (_blocks != null)
120:         {
121:             for (int j = 0; j < _blocks.length; j++)
122:             {
123:                 _blocks[ j ].writeBlocks(stream);
124:             }
125:         }
126:     }
