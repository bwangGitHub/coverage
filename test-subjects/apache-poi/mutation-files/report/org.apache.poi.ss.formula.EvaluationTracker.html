 29: /**
 30:  * Instances of this class keep track of multiple dependent cell evaluations due
 31:  * to recursive calls to {@link WorkbookEvaluator#evaluate(EvaluationCell)}}
 32:  * The main purpose of this class is to detect an attempt to evaluate a cell
 33:  * that is already being evaluated. In other words, it detects circular
 34:  * references in spreadsheet formulas.
 35:  *
 36:  * @author Josh Micich
 37:  */
 38: final class EvaluationTracker {
 39: 	// TODO - consider deleting this class and letting CellEvaluationFrame take care of itself
 40: 	private final List<CellEvaluationFrame> _evaluationFrames;
 41: 	private final Set<FormulaCellCacheEntry> _currentlyEvaluatingCells;
 42: 	private final EvaluationCache _cache;
 43: 
 44: 	public EvaluationTracker(EvaluationCache cache) {
 45: 		_cache = cache;
 46: 		_evaluationFrames = new ArrayList<CellEvaluationFrame>();
 47: 		_currentlyEvaluatingCells = new HashSet<FormulaCellCacheEntry>();
 48: 	}
 49: 
 50: 	/**
 51: 	 * Notifies this evaluation tracker that evaluation of the specified cell is
 52: 	 * about to start.<br/>
 53: 	 *
 54: 	 * In the case of a <code>true</code> return code, the caller should
 55: 	 * continue evaluation of the specified cell, and also be sure to call
 56: 	 * <tt>endEvaluate()</tt> when complete.<br/>
 57: 	 *
 58: 	 * In the case of a <code>null</code> return code, the caller should
 59: 	 * return an evaluation result of
 60: 	 * <tt>ErrorEval.CIRCULAR_REF_ERROR<tt>, and not call <tt>endEvaluate()</tt>.
 61: 	 * <br/>
 62: 	 * @return <code>false</code> if the specified cell is already being evaluated
 63: 	 */
 64: 	public boolean startEvaluate(FormulaCellCacheEntry cce) {
 65: 		if (cce == null) {
 66: 			throw new IllegalArgumentException("cellLoc must not be null");
 67: 		}
 68: 		if (_currentlyEvaluatingCells.contains(cce)) {
 69: 			return false;
 70: 		}
 71: 		_currentlyEvaluatingCells.add(cce);
 72: 		_evaluationFrames.add(new CellEvaluationFrame(cce));
 73: 		return true;
 74: 	}
 75: 
 76: 	public void updateCacheResult(ValueEval result) {
 77: 
 78: 		int nFrames = _evaluationFrames.size();
 79: 		if (nFrames < 1) {
 80: 			throw new IllegalStateException("Call to endEvaluate without matching call to startEvaluate");
 81: 		}
 82: 		CellEvaluationFrame frame = _evaluationFrames.get(nFrames-1);
 83: 		if (result == ErrorEval.CIRCULAR_REF_ERROR && nFrames > 1) {
 84: 			// Don't cache a circular ref error result if this cell is not the top evaluated cell.
 85: 			// A true circular ref error will propagate all the way around the loop.  However, it's
 86: 			// possible to have parts of the formula tree (/ parts of the loop) to evaluate to
 87: 			// CIRCULAR_REF_ERROR, and that value not get used in the final cell result (see the
 88: 			// unit tests for a simple example). Thus, the only CIRCULAR_REF_ERROR result that can
 89: 			// safely be cached is that of the top evaluated cell.
 90: 			return;
 91: 		}
 92: 
 93: 		frame.updateFormulaResult(result);
 94: 	}
 95: 
 96: 	/**
 97: 	 * Notifies this evaluation tracker that the evaluation of the specified cell is complete. <p/>
 98: 	 *
 99: 	 * Every successful call to <tt>startEvaluate</tt> must be followed by a call to <tt>endEvaluate</tt> (recommended in a finally block) to enable
100: 	 * proper tracking of which cells are being evaluated at any point in time.<p/>
101: 	 *
102: 	 * Assuming a well behaved client, parameters to this method would not be
103: 	 * required. However, they have been included to assert correct behaviour,
104: 	 * and form more meaningful error messages.
105: 	 */
106: 	public void endEvaluate(CellCacheEntry cce) {
107: 
108: 		int nFrames = _evaluationFrames.size();
109: 		if (nFrames < 1) {
110: 			throw new IllegalStateException("Call to endEvaluate without matching call to startEvaluate");
111: 		}
112: 
113: 		nFrames--;
114: 		CellEvaluationFrame frame = _evaluationFrames.get(nFrames);
115: 		if (cce != frame.getCCE()) {
116: 			throw new IllegalStateException("Wrong cell specified. ");
117: 		}
118: 		// else - no problems so pop current frame
119: 		_evaluationFrames.remove(nFrames);
120: 		_currentlyEvaluatingCells.remove(cce);
121: 	}
122: 
123: 	public void acceptFormulaDependency(CellCacheEntry cce) {
124: 		// Tell the currently evaluating cell frame that it has a dependency on the specified
125: 		int prevFrameIndex = _evaluationFrames.size()-1;
126: 		if (prevFrameIndex < 0) {
127: 			// Top level frame, there is no 'cell' above this frame that is using the current cell
128: 		} else {
129: 			CellEvaluationFrame consumingFrame = _evaluationFrames.get(prevFrameIndex);
130: 			consumingFrame.addSensitiveInputCell(cce);
131: 		}
132: 	}
133: 
134: 	public void acceptPlainValueDependency(int bookIndex, int sheetIndex,
135: 			int rowIndex, int columnIndex, ValueEval value) {
136: 		// Tell the currently evaluating cell frame that it has a dependency on the specified
137: 		int prevFrameIndex = _evaluationFrames.size() - 1;
138: 		if (prevFrameIndex < 0) {
139: 			// Top level frame, there is no 'cell' above this frame that is using the current cell
140: 		} else {
141: 			CellEvaluationFrame consumingFrame = _evaluationFrames.get(prevFrameIndex);
142: 			if (value == BlankEval.instance) {
143: 				consumingFrame.addUsedBlankCell(bookIndex, sheetIndex, rowIndex, columnIndex);
144: 			} else {
145: 				PlainValueCellCacheEntry cce = _cache.getPlainValueEntry(bookIndex, sheetIndex,
146: 						rowIndex, columnIndex, value);
147: 				consumingFrame.addSensitiveInputCell(cce);
148: 			}
149: 		}
150: 	}
