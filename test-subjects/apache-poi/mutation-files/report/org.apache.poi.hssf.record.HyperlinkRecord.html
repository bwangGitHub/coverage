144: 		/**
145: 		 * Read a GUID in standard text form e.g.<br/>
146: 		 * 13579BDF-0246-8ACE-0123-456789ABCDEF 
147: 		 * <br/> -&gt; <br/>
148: 		 *  0x13579BDF, 0x0246, 0x8ACE 0x0123456789ABCDEF
149: 		 */
150: 		public static GUID parse(String rep) {
151: 			char[] cc = rep.toCharArray();
152: 			if (cc.length != TEXT_FORMAT_LENGTH) {
153: 				throw new RecordFormatException("supplied text is the wrong length for a GUID");
154: 			}
155: 			int d0 = (parseShort(cc, 0) << 16) + (parseShort(cc, 4) << 0);
156: 			int d1 = parseShort(cc, 9);
157: 			int d2 = parseShort(cc, 14);
158: 			for (int i = 23; i > 19; i--) {
159: 				cc[i] = cc[i - 1];
160: 			}
161: 			long d3 = parseLELong(cc, 20);
162: 
163: 			return new GUID(d0, d1, d2, d3);
164: 		}
165: 
166: 		private static long parseLELong(char[] cc, int startIndex) {
167: 			long acc = 0;
168: 			for (int i = startIndex + 14; i >= startIndex; i -= 2) {
169: 				acc <<= 4;
170: 				acc += parseHexChar(cc[i + 0]);
171: 				acc <<= 4;
172: 				acc += parseHexChar(cc[i + 1]);
173: 			}
174: 			return acc;
175: 		}
176: 
177: 		private static int parseShort(char[] cc, int startIndex) {
178: 			int acc = 0;
179: 			for (int i = 0; i < 4; i++) {
180: 				acc <<= 4;
181: 				acc += parseHexChar(cc[startIndex + i]);
182: 			}
183: 			return acc;
184: 		}
185: 
186: 		private static int parseHexChar(char c) {
187: 			if (c >= '0' && c <= '9') {
188: 				return c - '0';
189: 			}
190: 			if (c >= 'A' && c <= 'F') {
191: 				return c - 'A' + 10;
192: 			}
193: 			if (c >= 'a' && c <= 'f') {
194: 				return c - 'a' + 10;
195: 			}
196: 			throw new RecordFormatException("Bad hex char '" + c + "'");
197: 		}
198: 	}

434:     public HyperlinkRecord(RecordInputStream in) {
435:         _range = new CellRangeAddress(in);
436: 
437:         _guid = new GUID(in);
438: 
439:         /**
440:          * streamVersion (4 bytes): An unsigned integer that specifies the version number
441:          * of the serialization implementation used to save this structure. This value MUST equal 2.
442:          */
443:         int streamVersion = in.readInt();
444:         if (streamVersion != 0x00000002) {
445:             throw new RecordFormatException("Stream Version must be 0x2 but found " + streamVersion);
446:         }
447:         _linkOpts = in.readInt();
448: 
449:         if ((_linkOpts & HLINK_LABEL) != 0){
450:             int label_len = in.readInt();
451:             _label = in.readUnicodeLEString(label_len);
452:         }
453: 
454:         if ((_linkOpts & HLINK_TARGET_FRAME) != 0){
455:             int len = in.readInt();
456:             _targetFrame = in.readUnicodeLEString(len);
457:         }
458: 
459:         if ((_linkOpts & HLINK_URL) != 0 && (_linkOpts & HLINK_UNC_PATH) != 0) {
460:             _moniker = null;
461:             int nChars = in.readInt();
462:             _address = in.readUnicodeLEString(nChars);
463:         }
464: 
465:         if ((_linkOpts & HLINK_URL) != 0 && (_linkOpts & HLINK_UNC_PATH) == 0) {
466:             _moniker = new GUID(in);
467: 
468:             if(URL_MONIKER.equals(_moniker)){
469:                 int length = in.readInt();
470:                 /**
471:                  * The value of <code>length<code> be either the byte size of the url field
472:                  * (including the terminating NULL character) or the byte size of the url field plus 24.
473:                  * If the value of this field is set to the byte size of the url field,
474:                  * then the tail bytes fields are not present.
475:                  */
476:                 int remaining = in.remaining();
477:                 if (length == remaining) {
478:                     int nChars = length/2;
479:                     _address = in.readUnicodeLEString(nChars);
480:                 } else {
481:                     int nChars = (length - TAIL_SIZE)/2;
482:                     _address = in.readUnicodeLEString(nChars);
483:                     /**
484:                      * TODO: make sense of the remaining bytes
485:                      * According to the spec they consist of:
486:                      * 1. 16-byte  GUID: This field MUST equal
487:                      *    {0xF4815879, 0x1D3B, 0x487F, 0xAF, 0x2C, 0x82, 0x5D, 0xC4, 0x85, 0x27, 0x63}
488:                      * 2. Serial version, this field MUST equal 0 if present.
489:                      * 3. URI Flags
490:                      */
491:                     _uninterpretedTail = readTail(URL_TAIL, in);
492:                 }
493:             } else if (FILE_MONIKER.equals(_moniker)) {
494:                 _fileOpts = in.readShort();
495: 
496:                 int len = in.readInt();
497:                 _shortFilename = StringUtil.readCompressedUnicode(in, len);
498:                 _uninterpretedTail = readTail(FILE_TAIL, in);
499:                 int size = in.readInt();
500:                 if (size > 0) {
501:                     int charDataSize = in.readInt();
502: 
503:                     //From the spec: An optional unsigned integer that MUST be 3 if present
504:                     // but some files has 4
505:                     int usKeyValue = in.readUShort();
506: 
507:                     _address = StringUtil.readUnicodeLE(in, charDataSize/2);
508:                 } else {
509:                     _address = null;
510:                 }
511:             } else if (STD_MONIKER.equals(_moniker)) {
512:                 _fileOpts = in.readShort();
513: 
514:                 int len = in.readInt();
515: 
516:                 byte[] path_bytes = new byte[len];
517:                 in.readFully(path_bytes);
518: 
519:                 _address = new String(path_bytes);
520:             }
521:         }
522: 
523:         if((_linkOpts & HLINK_PLACE) != 0) {
524: 
525:             int len = in.readInt();
526:             _textMark = in.readUnicodeLEString(len);
527:         }
528: 
529:         if (in.remaining() > 0) {
530:            logger.log(POILogger.WARN, 
531:                  "Hyperlink data remains: " + in.remaining() +
532:                  " : " +HexDump.toHex(in.readRemainder())
533:            );
534:         }
535:     }

537:     public void serialize(LittleEndianOutput out) {
538:         _range.serialize(out);
539: 
540:         _guid.serialize(out);
541:         out.writeInt(0x00000002); // TODO const
542:         out.writeInt(_linkOpts);
543: 
544:         if ((_linkOpts & HLINK_LABEL) != 0){
545:             out.writeInt(_label.length());
546:             StringUtil.putUnicodeLE(_label, out);
547:         }
548:         if ((_linkOpts & HLINK_TARGET_FRAME) != 0){
549:             out.writeInt(_targetFrame.length());
550:             StringUtil.putUnicodeLE(_targetFrame, out);
551:         }
552: 
553:         if ((_linkOpts & HLINK_URL) != 0 && (_linkOpts & HLINK_UNC_PATH) != 0) {
554:             out.writeInt(_address.length());
555:             StringUtil.putUnicodeLE(_address, out);
556:         }
557: 
558:         if ((_linkOpts & HLINK_URL) != 0 && (_linkOpts & HLINK_UNC_PATH) == 0) {
559:             _moniker.serialize(out);
560:             if(URL_MONIKER.equals(_moniker)){
561:                 if (_uninterpretedTail == null) {
562:                     out.writeInt(_address.length()*2);
563:                     StringUtil.putUnicodeLE(_address, out);
564:                 } else {
565:                     out.writeInt(_address.length()*2 + TAIL_SIZE);
566:                     StringUtil.putUnicodeLE(_address, out);
567:                     writeTail(_uninterpretedTail, out);
568:                 }
569:             } else if (FILE_MONIKER.equals(_moniker)){
570:                 out.writeShort(_fileOpts);
571:                 out.writeInt(_shortFilename.length());
572:                 StringUtil.putCompressedUnicode(_shortFilename, out);
573:                 writeTail(_uninterpretedTail, out);
574:                 if (_address == null) {
575:                     out.writeInt(0);
576:                 } else {
577:                     int addrLen = _address.length() * 2;
578:                     out.writeInt(addrLen + 6);
579:                     out.writeInt(addrLen);
580:                     out.writeShort(0x0003); // TODO const
581:                     StringUtil.putUnicodeLE(_address, out);
582:                 }
583:             }
584:         }
585:         if((_linkOpts & HLINK_PLACE) != 0){
586:                out.writeInt(_textMark.length());
587:             StringUtil.putUnicodeLE(_textMark, out);
588:         }
589:     }

591:     protected int getDataSize() {
592:         int size = 0;
593:         size += 2 + 2 + 2 + 2;  //rwFirst, rwLast, colFirst, colLast
594:         size += GUID.ENCODED_SIZE;
595:         size += 4;  //label_opts
596:         size += 4;  //link_opts
597:         if ((_linkOpts & HLINK_LABEL) != 0){
598:             size += 4;  //link length
599:             size += _label.length()*2;
600:         }
601:         if ((_linkOpts & HLINK_TARGET_FRAME) != 0){
602:             size += 4;  // int nChars
603:             size += _targetFrame.length()*2;
604:         }
605:         if ((_linkOpts & HLINK_URL) != 0 && (_linkOpts & HLINK_UNC_PATH) != 0) {
606:             size += 4;  // int nChars
607:             size += _address.length()*2;
608:         }
609:         if ((_linkOpts & HLINK_URL) != 0 && (_linkOpts & HLINK_UNC_PATH) == 0) {
610:             size += GUID.ENCODED_SIZE;
611:             if(URL_MONIKER.equals(_moniker)){
612:                 size += 4;  //address length
613:                 size += _address.length()*2;
614:                 if (_uninterpretedTail != null) {
615:                 	size += TAIL_SIZE;
616:                 }
617:             } else if (FILE_MONIKER.equals(_moniker)){
618:                 size += 2;  //file_opts
619:                 size += 4;  //address length
620:                 size += _shortFilename.length();
621:                 size += TAIL_SIZE;
622:                 size += 4;
623:                 if (_address != null) {
624:                     size += 6;
