 27: /**
 28:  * Represents a syntactic element from a formula by encapsulating the corresponding <tt>Ptg</tt>
 29:  * token.  Each <tt>ParseNode</tt> may have child <tt>ParseNode</tt>s in the case when the wrapped
 30:  * <tt>Ptg</tt> is non-atomic.
 31:  *
 32:  * @author Josh Micich
 33:  */
 34: final class ParseNode {
 35: 
 36: 	public static final ParseNode[] EMPTY_ARRAY = { };
 37: 	private final Ptg _token;
 38: 	private final ParseNode[] _children;
 39: 	private boolean _isIf;
 40: 	private final int _tokenCount;
 41: 
 42: 	public ParseNode(Ptg token, ParseNode[] children) {
 43: 		if (token == null) {
 44: 			throw new IllegalArgumentException("token must not be null");
 45: 		}
 46: 		_token = token;
 47: 		_children = children;
 48: 		_isIf = isIf(token);
 49: 		int tokenCount = 1;
 50: 		for (int i = 0; i < children.length; i++) {
 51: 			tokenCount += children[i].getTokenCount();
 52: 		}
 53: 		if (_isIf) {
 54: 			// there will be 2 or 3 extra tAttr tokens according to whether the false param is present
 55: 			tokenCount += children.length;
 56: 		}
 57: 		_tokenCount = tokenCount;
 58: 	}
 59: 	public ParseNode(Ptg token) {
 60: 		this(token, EMPTY_ARRAY);
 61: 	}

 69: 		return _tokenCount;

 71: 	public int getEncodedSize() {
 72: 		int result = _token instanceof ArrayPtg ? ArrayPtg.PLAIN_TOKEN_SIZE : _token.getSize();
 73: 		for (int i = 0; i < _children.length; i++) {
 74: 			result += _children[i].getEncodedSize();
 75: 		}
 76: 		return result;

 79: 	/**
 80: 	 * Collects the array of <tt>Ptg</tt> tokens for the specified tree.
 81: 	 */
 82: 	public static Ptg[] toTokenArray(ParseNode rootNode) {
 83: 		TokenCollector temp = new TokenCollector(rootNode.getTokenCount());
 84: 		rootNode.collectPtgs(temp);
 85: 		return temp.getResult();
 86: 	}
 87: 	private void collectPtgs(TokenCollector temp) {
 88: 		if (isIf(_token)) {
 89: 			collectIfPtgs(temp);
 90: 			return;
 91: 		}
 92: 		boolean isPreFixOperator = _token instanceof MemFuncPtg || _token instanceof MemAreaPtg;
 93: 		if (isPreFixOperator) {
 94: 			temp.add(_token);
 95: 		}
 96: 		for (int i=0; i< getChildren().length; i++) {
 97: 			getChildren()[i].collectPtgs(temp);
 98: 		}
 99: 		if (!isPreFixOperator) {
100: 			temp.add(_token);
101: 		}
102: 	}
103: 	/**
104: 	 * The IF() function gets marked up with two or three tAttr tokens.
105: 	 * Similar logic will be required for CHOOSE() when it is supported
106: 	 *
107: 	 * See excelfileformat.pdf sec 3.10.5 "tAttr (19H)
108: 	 */
109: 	private void collectIfPtgs(TokenCollector temp) {
110: 
111: 		// condition goes first
112: 		getChildren()[0].collectPtgs(temp);
113: 
114: 		// placeholder for tAttrIf
115: 		int ifAttrIndex = temp.createPlaceholder();
116: 
117: 		// true parameter
118: 		getChildren()[1].collectPtgs(temp);
119: 
120: 		// placeholder for first skip attr
121: 		int skipAfterTrueParamIndex = temp.createPlaceholder();
122: 		int trueParamSize = temp.sumTokenSizes(ifAttrIndex+1, skipAfterTrueParamIndex);
123: 
124: 		AttrPtg attrIf = AttrPtg.createIf(trueParamSize + 4); // distance to start of false parameter/tFuncVar. +4 for tAttrSkip after true
125: 
126: 		if (getChildren().length > 2) {
127: 			// false param present
128: 
129: 			// false parameter
130: 			getChildren()[2].collectPtgs(temp);
131: 
132: 			int skipAfterFalseParamIndex = temp.createPlaceholder();
133: 
134: 			int falseParamSize =  temp.sumTokenSizes(skipAfterTrueParamIndex+1, skipAfterFalseParamIndex);
135: 
136: 			AttrPtg attrSkipAfterTrue = AttrPtg.createSkip(falseParamSize + 4 + 4 - 1); // 1 less than distance to end of if FuncVar(size=4). +4 for attr skip before
137: 			AttrPtg attrSkipAfterFalse = AttrPtg.createSkip(4 - 1); // 1 less than distance to end of if FuncVar(size=4).
138: 
139: 			temp.setPlaceholder(ifAttrIndex, attrIf);
140: 			temp.setPlaceholder(skipAfterTrueParamIndex, attrSkipAfterTrue);
141: 			temp.setPlaceholder(skipAfterFalseParamIndex, attrSkipAfterFalse);
142: 		} else {
143: 			// false parameter not present
144: 			AttrPtg attrSkipAfterTrue = AttrPtg.createSkip(4 - 1); // 1 less than distance to end of if FuncVar(size=4).
145: 
146: 			temp.setPlaceholder(ifAttrIndex, attrIf);
147: 			temp.setPlaceholder(skipAfterTrueParamIndex, attrSkipAfterTrue);
148: 		}
149: 		temp.add(_token);
150: 	}

152: 	private static boolean isIf(Ptg token) {
153: 		if (token instanceof FuncVarPtg) {
154: 			FuncVarPtg func = (FuncVarPtg) token;
155: 			if (FunctionMetadataRegistry.FUNCTION_NAME_IF.equals(func.getName())) {
156: 				return true;
157: 			}
158: 		}
159: 		return false;
160: 	}

180: 		public int sumTokenSizes(int fromIx, int toIx) {
181: 			int result = 0;

183: 				result += _ptgs[i].getSize();
184: 			}
185: 			return result;
186: 		}

193: 			if (token == null) {
194: 				throw new IllegalArgumentException("token must not be null");
195: 			}
196: 			_ptgs[_offset] = token;
197: 			_offset++;
198: 		}

201: 			if (_ptgs[index] != null) {
202: 				throw new IllegalStateException("Invalid placeholder index (" + index + ")");
203: 			}
204: 			_ptgs[index] = token;
205: 		}
