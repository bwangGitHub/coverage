 26: /**
 27:  * Implementation of the VLOOKUP() function.<p/>
 28:  *
 29:  * VLOOKUP finds a row in a lookup table by the first column value and returns the value from another column.<br/>
 30:  *
 31:  * <b>Syntax</b>:<br/>
 32:  * <b>VLOOKUP</b>(<b>lookup_value</b>, <b>table_array</b>, <b>col_index_num</b>, range_lookup)<p/>
 33:  *
 34:  * <b>lookup_value</b>  The value to be found in the first column of the table array.<br/>
 35:  * <b>table_array</b> An area reference for the lookup data. <br/>
 36:  * <b>col_index_num</b> a 1 based index specifying which column value of the lookup data will be returned.<br/>
 37:  * <b>range_lookup</b> If TRUE (default), VLOOKUP finds the largest value less than or equal to
 38:  * the lookup_value.  If FALSE, only exact matches will be considered<br/>
 39:  *
 40:  * @author Josh Micich
 41:  */
 42: public final class Vlookup extends Var3or4ArgFunction {
 43: 	private static final ValueEval DEFAULT_ARG3 = BoolEval.TRUE;
 44: 
 45: 	public ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0, ValueEval arg1,
 46: 			ValueEval arg2) {
 47: 		return evaluate(srcRowIndex, srcColumnIndex, arg0, arg1, arg2, DEFAULT_ARG3);
 48: 	}
 49: 
 50: 	public ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0, ValueEval arg1,
 51: 			ValueEval arg2, ValueEval arg3) {
 52: 		try {
 53: 			// Evaluation order:
 54: 			// arg0 lookup_value, arg1 table_array, arg3 range_lookup, find lookup value, arg2 col_index, fetch result
 55: 			ValueEval lookupValue = OperandResolver.getSingleValue(arg0, srcRowIndex, srcColumnIndex);
 56: 			TwoDEval tableArray = LookupUtils.resolveTableArrayArg(arg1);
 57: 			boolean isRangeLookup = LookupUtils.resolveRangeLookupArg(arg3, srcRowIndex, srcColumnIndex);
 58: 			int rowIndex = LookupUtils.lookupIndexOfValue(lookupValue, LookupUtils.createColumnVector(tableArray, 0), isRangeLookup);
 59: 			int colIndex = LookupUtils.resolveRowOrColIndexArg(arg2, srcRowIndex, srcColumnIndex);
 60: 			ValueVector resultCol = createResultColumnVector(tableArray, colIndex);
 61: 			return resultCol.getItem(rowIndex);
 62: 		} catch (EvaluationException e) {
 63: 			return e.getErrorEval();
 64: 		}
 65: 	}
 66: 
 67: 
 68: 	/**
 69: 	 * Returns one column from an <tt>AreaEval</tt>
 70: 	 *
 71: 	 * @param colIndex assumed to be non-negative
 72: 	 *
 73: 	 * @throws EvaluationException (#REF!) if colIndex is too high
 74: 	 */
 75: 	private ValueVector createResultColumnVector(TwoDEval tableArray, int colIndex) throws EvaluationException {
 76: 		if(colIndex >= tableArray.getWidth()) {
 77: 			throw EvaluationException.invalidRef();
 78: 		}
 79: 		return LookupUtils.createColumnVector(tableArray, colIndex);
 80: 	}
