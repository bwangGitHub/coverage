 27: /**
 28:  * Implementation of the cell range address lists,like is described
 29:  * in OpenOffice.org's Excel Documentation: excelfileformat.pdf sec 2.5.14 -
 30:  * 'Cell Range Address List'
 31:  * 
 32:  * In BIFF8 there is a common way to store absolute cell range address lists in
 33:  * several records (not formulas). A cell range address list consists of a field
 34:  * with the number of ranges and the list of the range addresses. Each cell
 35:  * range address (called an ADDR structure) contains 4 16-bit-values.
 36:  * </p>
 37:  * 
 38:  * @author Dragos Buleandra (dragos.buleandra@trade2b.ro)
 39:  */
 40: public class CellRangeAddressList {
 41: 
 42: 	/**
 43: 	 * List of <tt>CellRangeAddress</tt>es. Each structure represents a cell range
 44: 	 */
 45: 	protected final List _list;
 46: 
 47: 	public CellRangeAddressList() {
 48: 		_list = new ArrayList();
 49: 	}
 50: 	/**
 51: 	 * Convenience constructor for creating a <tt>CellRangeAddressList</tt> with a single 
 52: 	 * <tt>CellRangeAddress</tt>.  Other <tt>CellRangeAddress</tt>es may be added later.
 53: 	 */
 54: 	public CellRangeAddressList(int firstRow, int lastRow, int firstCol, int lastCol) {
 55: 		this();
 56: 		addCellRangeAddress(firstRow, firstCol, lastRow, lastCol);
 57: 	}
 58: 	/**
 59: 	 * @param in the RecordInputstream to read the record from
 60: 	 */
 61: 	public CellRangeAddressList(RecordInputStream in) {
 62: 		this();
 63: 		int nItems = in.readUShort();
 64: 
 65: 		for (int k = 0; k < nItems; k++) {
 66: 			_list.add(new CellRangeAddress(in));
 67: 		}
 68: 	}
 69: 	/**
 70: 	 * Get the number of following ADDR structures. The number of this
 71: 	 * structures is automatically set when reading an Excel file and/or
 72: 	 * increased when you manually add a new ADDR structure . This is the reason
 73: 	 * there isn't a set method for this field .
 74: 	 * 
 75: 	 * @return number of ADDR structures
 76: 	 */
 77: 	public int countRanges() {
 78: 		return _list.size();
 79: 	}
 80: 
 81: 	/**
 82: 	 * Add a cell range structure.
 83: 	 * 
 84: 	 * @param firstRow - the upper left hand corner's row
 85: 	 * @param firstCol - the upper left hand corner's col
 86: 	 * @param lastRow - the lower right hand corner's row
 87: 	 * @param lastCol - the lower right hand corner's col
 88: 	 */
 89: 	public void addCellRangeAddress(int firstRow, int firstCol, int lastRow, int lastCol) {
 90: 		CellRangeAddress region = new CellRangeAddress(firstRow, lastRow, firstCol, lastCol);
 91: 		addCellRangeAddress(region);
 92: 	}
 93: 	public void addCellRangeAddress(CellRangeAddress cra) {
 94: 		_list.add(cra);
 95: 	}
 96: 	public CellRangeAddress remove(int rangeIndex) {
 97: 		if (_list.isEmpty()) {
 98: 			throw new RuntimeException("List is empty");
 99: 		}
100: 		if (rangeIndex < 0 || rangeIndex >= _list.size()) {
101: 			throw new RuntimeException("Range index (" + rangeIndex 
102: 					+ ") is outside allowable range (0.." + (_list.size()-1) + ")");
103: 		}
104: 		return (CellRangeAddress) _list.remove(rangeIndex);
105: 	}
106: 
107: 	/**
108: 	 * @return <tt>CellRangeAddress</tt> at the given index
109: 	 */
110: 	public CellRangeAddress getCellRangeAddress(int index) {
111: 		return (CellRangeAddress) _list.get(index);
112: 	}
113: 
114: 	public int getSize() {
115: 		return getEncodedSize(_list.size());
116: 	}
117: 	/**
118: 	 * @return the total size of for the specified number of ranges,
119: 	 *  including the initial 2 byte range count
120: 	 */
121: 	public static int getEncodedSize(int numberOfRanges) {
122: 		return 2 + CellRangeAddress.getEncodedSize(numberOfRanges);
123: 	}
124: 
125: 	public int serialize(int offset, byte[] data) {
126: 		int totalSize = getSize();
127: 		serialize(new LittleEndianByteArrayOutputStream(data, offset, totalSize));
128: 		return totalSize;
129: 	}
130: 	public void serialize(LittleEndianOutput out) {
131: 		int nItems = _list.size();
132: 		out.writeShort(nItems);
133: 		for (int k = 0; k < nItems; k++) {
134: 			CellRangeAddress region = (CellRangeAddress) _list.get(k);
135: 			region.serialize(out);
136: 		}
137: 	}
138: 	
139: 
140: 	public CellRangeAddressList copy() {
141: 		CellRangeAddressList result = new CellRangeAddressList();
142: 		
143: 		int nItems = _list.size();
144: 		for (int k = 0; k < nItems; k++) {
145: 			CellRangeAddress region = (CellRangeAddress) _list.get(k);
146: 			result.addCellRangeAddress(region.copy());
147: 		}
148: 		return result;
149: 	}
150: 	public CellRangeAddress[] getCellRangeAddresses() {
151: 		CellRangeAddress[] result = new CellRangeAddress[_list.size()];
152: 		_list.toArray(result);
153: 		return result;
154: 	}
