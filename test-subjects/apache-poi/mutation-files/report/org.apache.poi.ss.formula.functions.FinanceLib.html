 24:  * This class is a functon library for common fiscal functions.
 25:  * <b>Glossary of terms/abbreviations:</b>
 26:  * <br/>
 27:  * <ul>
 28:  * <li><em>FV:</em> Future Value</li>
 29:  * <li><em>PV:</em> Present Value</li>
 30:  * <li><em>NPV:</em> Net Present Value</li>
 31:  * <li><em>PMT:</em> (Periodic) Payment</li>
 32:  *
 33:  * </ul>
 34:  * For more info on the terms/abbreviations please use the references below
 35:  * (hyperlinks are subject to change):
 36:  * </br>Online References:
 37:  * <ol>
 38:  * <li>GNU Emacs Calc 2.02 Manual: http://theory.uwinnipeg.ca/gnu/calc/calc_203.html</li>
 39:  * <li>Yahoo Financial Glossary: http://biz.yahoo.com/f/g/nn.html#y</li>
 40:  * <li>MS Excel function reference: http://office.microsoft.com/en-us/assistance/CH062528251033.aspx</li>
 41:  * </ol>
 42:  * <h3>Implementation Notes:</h3>
 43:  * Symbols used in the formulae that follow:<br/>
 44:  * <ul>
 45:  * <li>p: present value</li>
 46:  * <li>f: future value</li>
 47:  * <li>n: number of periods</li>
 48:  * <li>y: payment (in each period)</li>
 49:  * <li>r: rate</li>
 50:  * <li>^: the power operator (NOT the java bitwise XOR operator!)</li>
 51:  * </ul>
 52:  * [From MS Excel function reference] Following are some of the key formulas
 53:  * that are used in this implementation:
 54:  * <pre>
 55:  * p(1+r)^n + y(1+rt)((1+r)^n-1)/r + f=0   ...{when r!=0}
 56:  * ny + p + f=0                            ...{when r=0}
 57:  * </pre>
 58:  */
 59: public final class FinanceLib {

 61:     private FinanceLib() {
 62:         // no instances of this class
 63:     }

 65:     /**
 66:      * Future value of an amount given the number of payments, rate, amount
 67:      * of individual payment, present value and boolean value indicating whether
 68:      * payments are due at the beginning of period
 69:      * (false => payments are due at end of period)
 70:      * @param r rate
 71:      * @param n num of periods
 72:      * @param y pmt per period
 73:      * @param p future value
 74:      * @param t type (true=pmt at end of period, false=pmt at begining of period)
 75:      */
 76:     public static double fv(double r, double n, double y, double p, boolean t) {
 77:         double retval = 0;
 78:         if (r == 0) {
 79:             retval = -1*(p+(n*y));
 80:         }
 81:         else {
 82:             double r1 = r + 1;
 83:             retval =((1-Math.pow(r1, n)) * (t ? r1 : 1) * y ) / r
 84:                       -
 85:                    p*Math.pow(r1, n);
 86:         }
 87:         return retval;
 88:     }
 89: 
 90:     /**
 91:      * Present value of an amount given the number of future payments, rate, amount
 92:      * of individual payment, future value and boolean value indicating whether
 93:      * payments are due at the beginning of period
 94:      * (false => payments are due at end of period)
 95:      * @param r
 96:      * @param n
 97:      * @param y
 98:      * @param f
 99:      * @param t
100:      */
101:     public static double pv(double r, double n, double y, double f, boolean t) {
102:         double retval = 0;
103:         if (r == 0) {
104:             retval = -1*((n*y)+f);
105:         }
106:         else {
107:             double r1 = r + 1;
108:             retval =(( ( 1 - Math.pow(r1, n) ) / r ) * (t ? r1 : 1)  * y - f)
109:                      /
110:                     Math.pow(r1, n);
111:         }
112:         return retval;
113:     }
114: 
115:     /**
116:      * calculates the Net Present Value of a principal amount
117:      * given the discount rate and a sequence of cash flows
118:      * (supplied as an array). If the amounts are income the value should
119:      * be positive, else if they are payments and not income, the
120:      * value should be negative.
121:      * @param r
122:      * @param cfs cashflow amounts
123:      */
124:     public static double npv(double r, double[] cfs) {
125:         double npv = 0;
126:         double r1 = r + 1;
127:         double trate = r1;
128:         for (int i=0, iSize=cfs.length; i<iSize; i++) {
129:             npv += cfs[i] / trate;
130:             trate *= r1;
131:         }
132:         return npv;
133:     }
134: 
135:     /**
136:      *
137:      * @param r
138:      * @param n
139:      * @param p
140:      * @param f
141:      * @param t
142:      */
143:     public static double pmt(double r, double n, double p, double f, boolean t) {
144:         double retval = 0;
145:         if (r == 0) {
146:             retval = -1*(f+p)/n;
147:         }
148:         else {
149:         double r1 = r + 1;
150:         retval = ( f + p * Math.pow(r1, n) ) * r
151:                   /
152:                ((t ? r1 : 1) * (1 - Math.pow(r1, n)));
153:         }
154:         return retval;
155:     }
156: 
157:     /**
158:      *
159:      * @param r
160:      * @param y
161:      * @param p
162:      * @param f
163:      * @param t
164:      */
165:     public static double nper(double r, double y, double p, double f, boolean t) {
166:         double retval = 0;
167:         if (r == 0) {
168:             retval = -1 * (f + p) / y;
169:         } else {
170:             double r1 = r + 1;
171:             double ryr = (t ? r1 : 1) * y / r;
172:             double a1 = ((ryr - f) < 0)
173:                     ? Math.log(f - ryr)
174:                     : Math.log(ryr - f);
175:             double a2 = ((ryr - f) < 0)
176:                     ? Math.log(-p - ryr)
177:                     : Math.log(p + ryr);
178:             double a3 = Math.log(r1);
179:             retval = (a1 - a2) / a3;
180:         }
181:         return retval;
182:     }
