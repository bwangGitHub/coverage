 35: /**
 36:  * A proxy HSSFListener that keeps track of the document formatting records, and
 37:  * provides an easy way to look up the format strings used by cells from their
 38:  * ids.
 39:  */
 40: public class FormatTrackingHSSFListener implements HSSFListener {
 41: 	private final HSSFListener _childListener;
 42: 	private final HSSFDataFormatter _formatter;
 43: 	private final NumberFormat _defaultFormat;
 44: 	private final Map<Integer, FormatRecord> _customFormatRecords = new Hashtable<Integer, FormatRecord>();
 45: 	private final List<ExtendedFormatRecord> _xfRecords = new ArrayList<ExtendedFormatRecord>();
 46: 
 47: 	/**
 48: 	 * Creates a format tracking wrapper around the given listener, using
 49: 	 * the {@link Locale#getDefault() default locale} for the formats.
 50: 	 */
 51: 	public FormatTrackingHSSFListener(HSSFListener childListener) {
 52: 		this(childListener, Locale.getDefault());
 53: 	}
 54: 
 55: 	/**
 56: 	 * Creates a format tracking wrapper around the given listener, using
 57: 	 * the given locale for the formats.
 58: 	 */
 59: 	public FormatTrackingHSSFListener(
 60: 			HSSFListener childListener, Locale locale) {
 61: 		_childListener = childListener;
 62: 		_formatter = new HSSFDataFormatter(locale);
 63: 		_defaultFormat = NumberFormat.getInstance(locale);
 64: 	}
 65: 
 66: 	protected int getNumberOfCustomFormats() {
 67: 		return _customFormatRecords.size();
 68: 	}
 69: 
 70: 	protected int getNumberOfExtendedFormats() {
 71: 		return _xfRecords.size();
 72: 	}

 74: 	/**
 75: 	 * Process this record ourselves, and then pass it on to our child listener
 76: 	 */
 77: 	public void processRecord(Record record) {
 78: 		// Handle it ourselves
 79: 		processRecordInternally(record);
 80: 
 81: 		// Now pass on to our child
 82: 		_childListener.processRecord(record);
 83: 	}
 84: 
 85: 	/**
 86: 	 * Process the record ourselves, but do not pass it on to the child
 87: 	 * Listener.
 88: 	 *
 89: 	 * @param record
 90: 	 */
 91: 	public void processRecordInternally(Record record) {
 92: 		if (record instanceof FormatRecord) {
 93: 			FormatRecord fr = (FormatRecord) record;
 94: 			_customFormatRecords.put(Integer.valueOf(fr.getIndexCode()), fr);
 95: 		}
 96: 		if (record instanceof ExtendedFormatRecord) {
 97: 			ExtendedFormatRecord xr = (ExtendedFormatRecord) record;
 98: 			_xfRecords.add(xr);
 99: 		}
100: 	}
101: 
102: 	/**
103: 	 * Formats the given numeric of date Cell's contents as a String, in as
104: 	 * close as we can to the way that Excel would do so. Uses the various
105: 	 * format records to manage this.
106: 	 *
107: 	 * TODO - move this to a central class in such a way that hssf.usermodel can
108: 	 * make use of it too
109: 	 */
110: 	public String formatNumberDateCell(CellValueRecordInterface cell) {
111: 		double value;
112: 		if (cell instanceof NumberRecord) {
113: 			value = ((NumberRecord) cell).getValue();
114: 		} else if (cell instanceof FormulaRecord) {
115: 			value = ((FormulaRecord) cell).getValue();
116: 		} else {
117: 			throw new IllegalArgumentException("Unsupported CellValue Record passed in " + cell);
118: 		}
119: 
120: 		// Get the built in format, if there is one
121: 		int formatIndex = getFormatIndex(cell);
122: 		String formatString = getFormatString(cell);
123: 
124: 		if (formatString == null) {
125: 			return _defaultFormat.format(value);
126: 		}
127: 		// Format, using the nice new
128: 		// HSSFDataFormatter to do the work for us
129: 		return _formatter.formatRawCellContents(value, formatIndex, formatString);
130: 	}
131: 
132: 	/**
133: 	 * Returns the format string, eg $##.##, for the given number format index.
134: 	 */
135: 	public String getFormatString(int formatIndex) {
136: 		String format = null;
137: 		if (formatIndex >= HSSFDataFormat.getNumberOfBuiltinBuiltinFormats()) {
138: 			FormatRecord tfr = _customFormatRecords.get(Integer.valueOf(formatIndex));
139: 			if (tfr == null) {
140: 				System.err.println("Requested format at index " + formatIndex
141: 						+ ", but it wasn't found");
142: 			} else {
143: 				format = tfr.getFormatString();
144: 			}
145: 		} else {
146: 			format = HSSFDataFormat.getBuiltinFormat((short) formatIndex);
147: 		}
148: 		return format;
149: 	}
150: 
151: 	/**
152: 	 * Returns the format string, eg $##.##, used by your cell
153: 	 */
154: 	public String getFormatString(CellValueRecordInterface cell) {
155: 		int formatIndex = getFormatIndex(cell);
156: 		if (formatIndex == -1) {
157: 			// Not found
158: 			return null;
159: 		}
160: 		return getFormatString(formatIndex);
161: 	}
162: 
163: 	/**
164: 	 * Returns the index of the format string, used by your cell, or -1 if none
165: 	 * found
166: 	 */
167: 	public int getFormatIndex(CellValueRecordInterface cell) {
168: 		ExtendedFormatRecord xfr = _xfRecords.get(cell.getXFIndex());
169: 		if (xfr == null) {
170: 			System.err.println("Cell " + cell.getRow() + "," + cell.getColumn()
171: 					+ " uses XF with index " + cell.getXFIndex() + ", but we don't have that");
172: 			return -1;
173: 		}
174: 		return xfr.getFormatIndex();
175: 	}
