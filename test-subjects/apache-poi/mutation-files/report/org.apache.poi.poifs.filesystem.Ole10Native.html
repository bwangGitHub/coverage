 28: /**
 29:  * Represents an Ole10Native record which is wrapped around certain binary
 30:  * files being embedded in OLE2 documents.
 31:  *
 32:  * @author Rainer Schwarze
 33:  */
 34: public class Ole10Native {
 35:   // (the fields as they appear in the raw record:)
 36:   private final int totalSize;                // 4 bytes, total size of record not including this field
 37:   private short flags1;                // 2 bytes, unknown, mostly [02 00]
 38:   private final String label;                // ASCIIZ, stored in this field without the terminating zero
 39:   private final String fileName;        // ASCIIZ, stored in this field without the terminating zero
 40:   private short flags2;                // 2 bytes, unknown, mostly [00 00]
 41:   // private byte unknown1Length;	// 1 byte, specifying the length of the following byte array (unknown1)
 42:   private byte[] unknown1;        // see below
 43:   private byte[] unknown2;        // 3 bytes, unknown, mostly [00 00 00]
 44:   private final String command;                // ASCIIZ, stored in this field without the terminating zero
 45:   private final int dataSize;                // 4 bytes (if space), size of following buffer
 46:   private final byte[] dataBuffer;        // varying size, the actual native data
 47:   private short flags3;                // some final flags? or zero terminators?, sometimes not there
 48:   public static final String OLE10_NATIVE = "\u0001Ole10Native";

 50:   /**
 51:    * Creates an instance of this class from an embedded OLE Object. The OLE Object is expected
 52:    * to include a stream &quot;{01}Ole10Native&quot; which contains the actual
 53:    * data relevant for this class.
 54:    *
 55:    * @param poifs POI Filesystem object
 56:    * @return Returns an instance of this class
 57:    * @throws IOException on IO error
 58:    * @throws Ole10NativeException on invalid or unexcepted data format
 59:    */
 60:   public static Ole10Native createFromEmbeddedOleObject(POIFSFileSystem poifs) throws IOException, Ole10NativeException {
 61:      return createFromEmbeddedOleObject(poifs.getRoot());
 62:   }

 64:   /**
 65:    * Creates an instance of this class from an embedded OLE Object. The OLE Object is expected
 66:    * to include a stream &quot;{01}Ole10Native&quot; which contains the actual
 67:    * data relevant for this class.
 68:    *
 69:    * @param directory POI Filesystem object
 70:    * @return Returns an instance of this class
 71:    * @throws IOException on IO error
 72:    * @throws Ole10NativeException on invalid or unexcepted data format
 73:    */
 74:   public static Ole10Native createFromEmbeddedOleObject(DirectoryNode directory) throws IOException, Ole10NativeException {
 75:      boolean plain = false;
 76: 
 77:      try {
 78:         directory.getEntry("\u0001Ole10ItemName");
 79:         plain = true;
 80:      } catch (FileNotFoundException ex) {
 81:         plain = false;
 82:      }
 83:      
 84:      DocumentEntry nativeEntry = 
 85:         (DocumentEntry)directory.getEntry(OLE10_NATIVE);
 86:      byte[] data = new byte[nativeEntry.getSize()];
 87:      directory.createDocumentInputStream(nativeEntry).read(data);
 88: 
 89:      return new Ole10Native(data, 0, plain);
 90:   }

 92:   /**
 93:    * Creates an instance and fills the fields based on the data in the given buffer.
 94:    *
 95:    * @param data   The buffer containing the Ole10Native record
 96:    * @param offset The start offset of the record in the buffer
 97:    * @throws Ole10NativeException on invalid or unexcepted data format
 98:    */
 99:   public Ole10Native(byte[] data, int offset) throws Ole10NativeException {
100:     this(data, offset, false);
101:   }
102:   /**
103:    * Creates an instance and fills the fields based on the data in the given buffer.
104:    *
105:    * @param data   The buffer containing the Ole10Native record
106:    * @param offset The start offset of the record in the buffer
107:    * @param plain Specified 'plain' format without filename
108:    * @throws Ole10NativeException on invalid or unexcepted data format
109:    */
110:   public Ole10Native(byte[] data, int offset, boolean plain) throws Ole10NativeException {
111:     int ofs = offset;        // current offset, initialized to start
112: 
113:     if (data.length<offset+2) {
114:       throw new Ole10NativeException("data is too small");
115:     }
116: 
117:     totalSize = LittleEndian.getInt(data, ofs);
118:     ofs += LittleEndianConsts.INT_SIZE;
119: 
120:     if (plain) {
121:       dataBuffer = new byte[totalSize-4];
122:       System.arraycopy(data, 4, dataBuffer, 0, dataBuffer.length);
123:       dataSize = totalSize - 4;
124:       
125:       byte[] oleLabel = new byte[8];
126:       System.arraycopy(dataBuffer, 0, oleLabel, 0, Math.min(dataBuffer.length, 8));
127:       label = "ole-"+ HexDump.toHex(oleLabel);
128:       fileName = label;
129:       command = label;
130:     } else {
131:       flags1 = LittleEndian.getShort(data, ofs);
132:       ofs += LittleEndianConsts.SHORT_SIZE;
133:       int len = getStringLength(data, ofs);
134:       label = StringUtil.getFromCompressedUnicode(data, ofs, len - 1);
135:       ofs += len;
136:       len = getStringLength(data, ofs);
137:       fileName = StringUtil.getFromCompressedUnicode(data, ofs, len - 1);
138:       ofs += len;
139:       flags2 = LittleEndian.getShort(data, ofs);
140:       ofs += LittleEndianConsts.SHORT_SIZE;
141:       len = LittleEndian.getUByte(data, ofs);
142:       unknown1 = new byte[len];
143:       ofs += len;
144:       len = 3;
145:       unknown2 = new byte[len];
146:       ofs += len;
147:       len = getStringLength(data, ofs);
148:       command = StringUtil.getFromCompressedUnicode(data, ofs, len - 1);
149:       ofs += len;
150: 
151:       if (totalSize + LittleEndianConsts.INT_SIZE - ofs > LittleEndianConsts.INT_SIZE) {
152:         dataSize = LittleEndian.getInt(data, ofs);
153:         ofs += LittleEndianConsts.INT_SIZE;
154: 
155:         if (dataSize > totalSize || dataSize<0) {
156:           throw new Ole10NativeException("Invalid Ole10Native");
157:         }
158: 
159:         dataBuffer = new byte[dataSize];
160:         System.arraycopy(data, ofs, dataBuffer, 0, dataSize);
161:         ofs += dataSize;
162: 
163:         if (unknown1.length > 0) {
164:           flags3 = LittleEndian.getShort(data, ofs);
165:           ofs += LittleEndianConsts.SHORT_SIZE;
166:         } else {
167:           flags3 = 0;
168:         }
169:       } else {
170:         throw new Ole10NativeException("Invalid Ole10Native");
171:       }
172:     }
173:   }
174: 
175:   /*
176:    * Helper - determine length of zero terminated string (ASCIIZ).
177:    */
178:   private static int getStringLength(byte[] data, int ofs) {
179:     int len = 0;
180:     while (len+ofs<data.length && data[ofs + len] != 0) {
181:       len++;
182:     }
183:     len++;
184:     return len;
185:   }
