 26: /**
 27:  * Formats sheet names for use in formula expressions.
 28:  * 
 29:  * @author Josh Micich
 30:  */
 31: public final class SheetNameFormatter {
 32: 	
 33: 	private static final char DELIMITER = '\'';
 34: 	
 35: 	/**
 36: 	 * Matches a single cell ref with no absolute ('$') markers
 37: 	 */
 38: 	private static final Pattern CELL_REF_PATTERN = Pattern.compile("([A-Za-z]+)([0-9]+)");
 39: 
 40: 	private SheetNameFormatter() {
 41: 		// no instances of this class
 42: 	}
 43: 	/**
 44: 	 * Used to format sheet names as they would appear in cell formula expressions.
 45: 	 * @return the sheet name unchanged if there is no need for delimiting.  Otherwise the sheet
 46: 	 * name is enclosed in single quotes (').  Any single quotes which were already present in the 
 47: 	 * sheet name will be converted to double single quotes ('').  
 48: 	 */
 49: 	public static String format(String rawSheetName) {
 50: 		StringBuffer sb = new StringBuffer(rawSheetName.length() + 2);
 51: 		appendFormat(sb, rawSheetName);
 52: 		return sb.toString();
 53: 	}
 54: 	
 55: 	/**
 56: 	 * Convenience method for ({@link #format(String)}) when a StringBuffer is already available.
 57: 	 * 
 58: 	 * @param out - sheet name will be appended here possibly with delimiting quotes 
 59: 	 */
 60: 	public static void appendFormat(StringBuffer out, String rawSheetName) {
 61: 		boolean needsQuotes = needsDelimiting(rawSheetName);
 62: 		if(needsQuotes) {
 63: 			out.append(DELIMITER);
 64: 			appendAndEscape(out, rawSheetName);
 65: 			out.append(DELIMITER);
 66: 		} else {
 67: 			out.append(rawSheetName);
 68: 		}
 69: 	}
 70: 	public static void appendFormat(StringBuffer out, String workbookName, String rawSheetName) {
 71: 		boolean needsQuotes = needsDelimiting(workbookName) || needsDelimiting(rawSheetName);
 72: 		if(needsQuotes) {
 73: 			out.append(DELIMITER);
 74: 			out.append('[');
 75: 			appendAndEscape(out, workbookName.replace('[', '(').replace(']', ')'));
 76: 			out.append(']');
 77: 			appendAndEscape(out, rawSheetName);
 78: 			out.append(DELIMITER);
 79: 		} else {
 80: 			out.append('[');
 81: 			out.append(workbookName);
 82: 			out.append(']');
 83: 			out.append(rawSheetName);
 84: 		}
 85: 	}

 87: 	private static void appendAndEscape(StringBuffer sb, String rawSheetName) {
 88: 		int len = rawSheetName.length();
 89: 		for(int i=0; i<len; i++) {
 90: 			char ch = rawSheetName.charAt(i);
 91: 			if(ch == DELIMITER) {
 92: 				// single quotes (') are encoded as ('')
 93: 				sb.append(DELIMITER);
 94: 			}
 95: 			sb.append(ch);
 96: 		}
 97: 	}
 98: 
 99: 	private static boolean needsDelimiting(String rawSheetName) {
100: 		int len = rawSheetName.length();
101: 		if(len < 1) {
102: 			throw new RuntimeException("Zero length string is an invalid sheet name");
103: 		}
104: 		if(Character.isDigit(rawSheetName.charAt(0))) {
105: 			// sheet name with digit in the first position always requires delimiting
106: 			return true;
107: 		}
108: 		for(int i=0; i<len; i++) {
109: 			char ch = rawSheetName.charAt(i);
110: 			if(isSpecialChar(ch)) {
111: 				return true;
112: 			}
113: 		}
114: 		if(Character.isLetter(rawSheetName.charAt(0))
115: 				&& Character.isDigit(rawSheetName.charAt(len-1))) {
116: 			// note - values like "A$1:$C$20" don't get this far 
117: 			if(nameLooksLikePlainCellReference(rawSheetName)) {
118: 				return true;
119: 			}
120: 		}
121: 		if (nameLooksLikeBooleanLiteral(rawSheetName)) {
122: 			return true;
123: 		}
124: 		// Error constant literals all contain '#' and other special characters
125: 		// so they don't get this far
126: 		return false;
127: 	}
128: 	
129: 	private static boolean nameLooksLikeBooleanLiteral(String rawSheetName) {
130: 		switch(rawSheetName.charAt(0)) {
131: 			case 'T': case 't':
132: 				return "TRUE".equalsIgnoreCase(rawSheetName);
133: 			case 'F': case 'f':
134: 				return "FALSE".equalsIgnoreCase(rawSheetName);
135: 		}
136: 		return false;
137: 	}
138: 	/**
139: 	 * @return <code>true</code> if the presence of the specified character in a sheet name would 
140: 	 * require the sheet name to be delimited in formulas.  This includes every non-alphanumeric 
141: 	 * character besides underscore '_' and dot '.'.
142: 	 */
143: 	/* package */ static boolean isSpecialChar(char ch) {
144: 		// note - Character.isJavaIdentifierPart() would allow dollars '$'
145: 		if(Character.isLetterOrDigit(ch)) {
146: 			return false;
147: 		}
148: 		switch(ch) {
149: 			case '.': // dot is OK
150: 			case '_': // underscore is OK
151: 				return false;
152: 			case '\n':
153: 			case '\r':
154: 			case '\t':
155: 				throw new RuntimeException("Illegal character (0x" 
156: 						+ Integer.toHexString(ch) + ") found in sheet name");
157: 		}
158: 		return true;
159: 	}

162: 	/**
163: 	 * Used to decide whether sheet names like 'AB123' need delimiting due to the fact that they 
164: 	 * look like cell references.
165: 	 * <p/>
166: 	 * This code is currently being used for translating formulas represented with <code>Ptg</code>
167: 	 * tokens into human readable text form.  In formula expressions, a sheet name always has a 
168: 	 * trailing '!' so there is little chance for ambiguity.  It doesn't matter too much what this 
169: 	 * method returns but it is worth noting the likely consumers of these formula text strings:
170: 	 * <ol>
171: 	 * <li>POI's own formula parser</li>
172: 	 * <li>Visual reading by human</li>
173: 	 * <li>VBA automation entry into Excel cell contents e.g.  ActiveCell.Formula = "=c64!A1"</li>
174: 	 * <li>Manual entry into Excel cell contents</li>
175: 	 * <li>Some third party formula parser</li>
176: 	 * </ol>
177: 	 * 
178: 	 * At the time of writing, POI's formula parser tolerates cell-like sheet names in formulas
179: 	 * with or without delimiters.  The same goes for Excel(2007), both manual and automated entry.  
180: 	 * <p/>
181: 	 * For better or worse this implementation attempts to replicate Excel's formula renderer.
182: 	 * Excel uses range checking on the apparent 'row' and 'column' components.  Note however that
183: 	 * the maximum sheet size varies across versions.
184: 	 * @see org.apache.poi.ss.util.CellReference
185: 	 */
186: 	/* package */ static boolean cellReferenceIsWithinRange(String lettersPrefix, String numbersSuffix) {
187: 		return CellReference.cellReferenceIsWithinRange(lettersPrefix, numbersSuffix, SpreadsheetVersion.EXCEL97);
188: 	}
189: 
190: 	/**
191: 	 * Note - this method assumes the specified rawSheetName has only letters and digits.  It 
192: 	 * cannot be used to match absolute or range references (using the dollar or colon char).
193: 	 * <p/>
194: 	 * Some notable cases:
195: 	 *    <blockquote><table border="0" cellpadding="1" cellspacing="0" 
196: 	 *                 summary="Notable cases.">
197: 	 *      <tr><th>Input&nbsp;</th><th>Result&nbsp;</th><th>Comments</th></tr>
198: 	 *      <tr><td>"A1"&nbsp;&nbsp;</td><td>true</td><td>&nbsp;</td></tr>
199: 	 *      <tr><td>"a111"&nbsp;&nbsp;</td><td>true</td><td>&nbsp;</td></tr>
200: 	 *      <tr><td>"AA"&nbsp;&nbsp;</td><td>false</td><td>&nbsp;</td></tr>
201: 	 *      <tr><td>"aa1"&nbsp;&nbsp;</td><td>true</td><td>&nbsp;</td></tr>
202: 	 *      <tr><td>"A1A"&nbsp;&nbsp;</td><td>false</td><td>&nbsp;</td></tr>
203: 	 *      <tr><td>"A1A1"&nbsp;&nbsp;</td><td>false</td><td>&nbsp;</td></tr>
204: 	 *      <tr><td>"A$1:$C$20"&nbsp;&nbsp;</td><td>false</td><td>Not a plain cell reference</td></tr>
205: 	 *      <tr><td>"SALES20080101"&nbsp;&nbsp;</td><td>true</td>
206: 	 *      		<td>Still needs delimiting even though well out of range</td></tr>
207: 	 *    </table></blockquote>
208: 	 *  
209: 	 * @return <code>true</code> if there is any possible ambiguity that the specified rawSheetName
210: 	 * could be interpreted as a valid cell name.
211: 	 */
212: 	/* package */ static boolean nameLooksLikePlainCellReference(String rawSheetName) {
213: 		Matcher matcher = CELL_REF_PATTERN.matcher(rawSheetName);
214: 		if(!matcher.matches()) {
215: 			return false;
216: 		}
217: 		
218: 		// rawSheetName == "Sheet1" gets this far.
219: 		String lettersPrefix = matcher.group(1);
220: 		String numbersSuffix = matcher.group(2);
221: 		return cellReferenceIsWithinRange(lettersPrefix, numbersSuffix);
222: 	}
