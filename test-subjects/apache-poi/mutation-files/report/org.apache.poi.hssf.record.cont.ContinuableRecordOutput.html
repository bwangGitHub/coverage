 25: /**
 26:  * An augmented {@link LittleEndianOutput} used for serialization of {@link ContinuableRecord}s.
 27:  * This class keeps track of how much remaining space is available in the current BIFF record and
 28:  * can start new {@link ContinueRecord}s as required.
 29:  *
 30:  * @author Josh Micich
 31:  */
 32: public final class ContinuableRecordOutput implements LittleEndianOutput {
 33: 
 34: 	private final LittleEndianOutput _out;
 35: 	private UnknownLengthRecordOutput _ulrOutput;
 36: 	private int _totalPreviousRecordsSize;
 37: 
 38: 	public ContinuableRecordOutput(LittleEndianOutput out, int sid) {
 39: 		_ulrOutput = new UnknownLengthRecordOutput(out, sid);
 40: 		_out = out;
 41: 		_totalPreviousRecordsSize = 0;
 42: 	}

 44: 	public static ContinuableRecordOutput createForCountingOnly() {
 45: 		return new ContinuableRecordOutput(NOPOutput, -777); // fake sid
 46: 	}

 49: 	 * @return total number of bytes written so far (including all BIFF headers)
 50: 	 */
 51: 	public int getTotalSize() {
 52: 		return _totalPreviousRecordsSize + _ulrOutput.getTotalSize();
 53: 	}

 55: 	 * Terminates the last record (also updates its 'ushort size' field)

 58: 		_ulrOutput.terminate();
 59: 	}

 67: 	/**
 68: 	 * Terminates the current record and starts a new {@link ContinueRecord} (regardless
 69: 	 * of how much space is still available in the current record).
 70: 	 */
 71: 	public void writeContinue() {
 72: 		_ulrOutput.terminate();
 73: 		_totalPreviousRecordsSize += _ulrOutput.getTotalSize();
 74: 		_ulrOutput = new UnknownLengthRecordOutput(_out, ContinueRecord.sid);
 75: 	}
 76: 	/**
 77: 	 * Will terminate the current record and start a new {@link ContinueRecord}
 78: 	 *  if there isn't space for the requested number of bytes
 79: 	 * @param requiredContinuousSize The number of bytes that need to be written
 80: 	 */
 81: 	public void writeContinueIfRequired(int requiredContinuousSize) {
 82: 		if (_ulrOutput.getAvailableSpace() < requiredContinuousSize) {
 83: 			writeContinue();
 84: 		}
 85: 	}

 87: 	/**
 88: 	 * Writes the 'optionFlags' byte and encoded character data of a unicode string.  This includes:
 89: 	 * <ul>
 90: 	 * <li>byte optionFlags</li>
 91: 	 * <li>encoded character data (in "ISO-8859-1" or "UTF-16LE" encoding)</li>
 92: 	 * </ul>
 93: 	 *
 94: 	 * Notes:
 95: 	 * <ul>
 96: 	 * <li>The value of the 'is16bitEncoded' flag is determined by the actual character data
 97: 	 * of <tt>text</tt></li>
 98: 	 * <li>The string options flag is never separated (by a {@link ContinueRecord}) from the
 99: 	 * first chunk of character data it refers to.</li>
100: 	 * <li>The 'ushort length' field is assumed to have been explicitly written earlier.  Hence,
101: 	 * there may be an intervening {@link ContinueRecord}</li>
102: 	 * </ul>
103: 	 */
104: 	public void writeStringData(String text) {
105: 		boolean is16bitEncoded = StringUtil.hasMultibyte(text);
106: 		// calculate total size of the header and first encoded char
107: 		int keepTogetherSize = 1 + 1; // ushort len, at least one character byte
108: 		int optionFlags = 0x00;
109: 		if (is16bitEncoded) {
110: 			optionFlags |= 0x01;
111: 			keepTogetherSize += 1; // one extra byte for first char
112: 		}
113: 		writeContinueIfRequired(keepTogetherSize);
114: 		writeByte(optionFlags);
115: 		writeCharacterData(text, is16bitEncoded);
116: 	}
117: 	/**
118: 	 * Writes a unicode string complete with header and character data.  This includes:
119: 	 * <ul>
120: 	 * <li>ushort length</li>
121: 	 * <li>byte optionFlags</li>
122: 	 * <li>ushort numberOfRichTextRuns (optional)</li>
123: 	 * <li>ushort extendedDataSize (optional)</li>
124: 	 * <li>encoded character data (in "ISO-8859-1" or "UTF-16LE" encoding)</li>
125: 	 * </ul>
126: 	 *
127: 	 * The following bits of the 'optionFlags' byte will be set as appropriate:
128: 	 * <table border='1'>
129: 	 * <tr><th>Mask</th><th>Description</th></tr>
130: 	 * <tr><td>0x01</td><td>is16bitEncoded</td></tr>
131: 	 * <tr><td>0x04</td><td>hasExtendedData</td></tr>
132: 	 * <tr><td>0x08</td><td>isRichText</td></tr>
133: 	 * </table>
134: 	 * Notes:
135: 	 * <ul>
136: 	 * <li>The value of the 'is16bitEncoded' flag is determined by the actual character data
137: 	 * of <tt>text</tt></li>
138: 	 * <li>The string header fields are never separated (by a {@link ContinueRecord}) from the
139: 	 * first chunk of character data (i.e. the first character is always encoded in the same
140: 	 * record as the string header).</li>
141: 	 * </ul>
142: 	 */
143: 	public void writeString(String text, int numberOfRichTextRuns, int extendedDataSize) {
144: 		boolean is16bitEncoded = StringUtil.hasMultibyte(text);
145: 		// calculate total size of the header and first encoded char
146: 		int keepTogetherSize = 2 + 1 + 1; // ushort len, byte optionFlags, at least one character byte
147: 		int optionFlags = 0x00;
148: 		if (is16bitEncoded) {
149: 			optionFlags |= 0x01;
150: 			keepTogetherSize += 1; // one extra byte for first char
151: 		}
152: 		if (numberOfRichTextRuns > 0) {
153: 			optionFlags |= 0x08;
154: 			keepTogetherSize += 2;
155: 		}
156: 		if (extendedDataSize > 0) {
157: 			optionFlags |= 0x04;
158: 			keepTogetherSize += 4;
159: 		}
160: 		writeContinueIfRequired(keepTogetherSize);
161: 		writeShort(text.length());
162: 		writeByte(optionFlags);
163: 		if (numberOfRichTextRuns > 0) {
164: 			writeShort(numberOfRichTextRuns);
165: 		}
166: 		if (extendedDataSize > 0) {
167: 			writeInt(extendedDataSize);
168: 		}
169: 		writeCharacterData(text, is16bitEncoded);
170: 	}

173: 	private void writeCharacterData(String text, boolean is16bitEncoded) {
174: 		int nChars = text.length();
175: 		int i=0;
176: 		if (is16bitEncoded) {
177: 			while(true) {
178: 				int nWritableChars = Math.min(nChars-i, _ulrOutput.getAvailableSpace() / 2);
179: 				for ( ; nWritableChars > 0; nWritableChars--) {
180: 					_ulrOutput.writeShort(text.charAt(i++));
181: 				}
182: 				if (i >= nChars) {
183: 					break;
184: 				}
185: 				writeContinue();
186: 				writeByte(0x01);
187: 			}
188: 		} else {
189: 			while(true) {
190: 				int nWritableChars = Math.min(nChars-i, _ulrOutput.getAvailableSpace() / 1);
191: 				for ( ; nWritableChars > 0; nWritableChars--) {
192: 					_ulrOutput.writeByte(text.charAt(i++));
193: 				}
194: 				if (i >= nChars) {
195: 					break;
196: 				}
197: 				writeContinue();
198: 				writeByte(0x00);
199: 			}
200: 		}
201: 	}

204: 		writeContinueIfRequired(b.length);

208: 	public void write(byte[] b, int offset, int len) {
209: 
210:         int i=0;
211:         while(true) {
212:             int nWritableChars = Math.min(len - i, _ulrOutput.getAvailableSpace() / 1);
213:             for ( ; nWritableChars > 0; nWritableChars--) {
214:                 _ulrOutput.writeByte(b[offset + i++]);
215:             }
216:             if (i >= len) {
217:                 break;
218:             }
