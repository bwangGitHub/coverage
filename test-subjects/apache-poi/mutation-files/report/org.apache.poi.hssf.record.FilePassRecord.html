 65: 			case ENCRYPTION_OTHER_CAPI_3:
 66: 				throw new RecordFormatException(
 67: 						"HSSF does not currently support CryptoAPI encryption");
 68: 			default:
 69: 				throw new RecordFormatException("Unknown encryption info " + _encryptionInfo);
 70: 		}

 74: 		}
