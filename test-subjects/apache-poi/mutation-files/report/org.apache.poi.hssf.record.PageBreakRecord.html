 45:     /**
 46:      * Since both records store 2byte integers (short), no point in

 52:     public static final class Break {
 53: 
 54:         public static final int ENCODED_SIZE = 6;

 61:             this.main = main;

 63:             this.subTo = subTo;
 64:         }

 66:         public Break(RecordInputStream in) {
 67:             main = in.readUShort() - 1;
 68:             subFrom = in.readUShort();
 69:             subTo = in.readUShort();
 70:         }

 82:     }

 84:     public PageBreakRecord(RecordInputStream in)
 85:     {
 86:         int nBreaks = in.readShort();
 87:         _breaks = new ArrayList<Break>(nBreaks + 2);
 88:         _breakMap = new HashMap<Integer, Break>();

 91:             Break br = new Break(in);
 92:             _breaks.add(br);
 93:             _breakMap.put(Integer.valueOf(br.main), br);

 96:     }

161:     public void addBreak(int main, int subFrom, int subTo) {
162: 
163:         Integer key = Integer.valueOf(main);
164:         Break region = _breakMap.get(key);
165:         if(region == null) {
166:             region = new Break(main, subFrom, subTo);
167:             _breakMap.put(key, region);
168:             _breaks.add(region);
169:         } else {
170:             region.main = main;
171:             region.subFrom = subFrom;
172:             region.subTo = subTo;
173:         }
174:     }

176:     /**

178:      * @param main (zero-based)
179:      */
180:     public final void removeBreak(int main) {
181:         Integer rowKey = Integer.valueOf(main);
182:         Break region = _breakMap.get(rowKey);
183:         _breaks.remove(region);
184:         _breakMap.remove(rowKey);
185:     }
