 28: /**
 29:  * Translates Graphics calls into escher calls.  The translation is lossy so
 30:  * many features are not supported and some just aren't implemented yet.  If
 31:  * in doubt test the specific calls you wish to make. Graphics calls are
 32:  * always performed into an EscherGroup so one will need to be created.
 33:  * <p>
 34:  * <b>Important:</b>
 35:  * <blockquote>
 36:  * One important concept worth considering is that of font size.  One of the
 37:  * difficulties in converting Graphics calls into escher drawing calls is that
 38:  * Excel does not have the concept of absolute pixel positions.  It measures
 39:  * it's cell widths in 'characters' and the cell heights in points.
 40:  * Unfortunately it's not defined exactly what a type of character it's
 41:  * measuring.  Presumably this is due to the fact that the Excel will be
 42:  * using different fonts on different platforms or even within the same
 43:  * platform.
 44:  * <p>
 45:  * Because of this constraint we've had to calculate the
 46:  * verticalPointsPerPixel.  This the amount the font should be scaled by when
 47:  * you issue commands such as drawString().  A good way to calculate this
 48:  * is to use the follow formula:
 49:  * <p>
 50:  * <pre>
 51:  *      multipler = groupHeightInPoints / heightOfGroup
 52:  * </pre>
 53:  * <p>
 54:  * The height of the group is calculated fairly simply by calculating the
 55:  * difference between the y coordinates of the bounding box of the shape.  The
 56:  * height of the group can be calculated by using a convenience called
 57:  * <code>HSSFClientAnchor.getAnchorHeightInPoints()</code>.
 58:  * </blockquote>
 59:  *
 60:  * @author Glen Stampoultzis (glens at apache.org)
 61:  */

271:     public void drawString(String str, int x, int y)
272:     {
273:         if (str == null || str.equals(""))
274:             return;
275: 
276:         Font excelFont = font;
277:         if ( font.getName().equals( "SansSerif" ) )
278:         {
279:             excelFont = new Font( "Arial", font.getStyle(), (int) ( font.getSize() / verticalPixelsPerPoint ) );
280:         }
281:         else
282:         {
283:             excelFont = new Font( font.getName(), font.getStyle(), (int) ( font.getSize() / verticalPixelsPerPoint ));
284:         }
285:         FontDetails d = StaticFontMetrics.getFontDetails( excelFont );
286:         int width = d.getStringWidth( str ) * 8  + 12;
287:         int height = (int) ( ( font.getSize() / verticalPixelsPerPoint ) + 6 ) * 2;
288:         y -= ( font.getSize() / verticalPixelsPerPoint ) + 2 * verticalPixelsPerPoint;    // we want to draw the shape from the top-left
289:         HSSFTextbox textbox = escherGroup.createTextbox( new HSSFChildAnchor( x, y, x + width, y + height ) );
290:         textbox.setNoFill( true );
291:         textbox.setLineStyle( HSSFShape.LINESTYLE_NONE );
292:         HSSFRichTextString s = new HSSFRichTextString( str );
293:         HSSFFont hssfFont = matchFont( excelFont );
294:         s.applyFont( hssfFont );
295:         textbox.setString( s );
296:     }

298:     private HSSFFont matchFont( Font font )
299:     {
300:         HSSFColor hssfColor = workbook.getCustomPalette()
301:                 .findColor((byte)foreground.getRed(), (byte)foreground.getGreen(), (byte)foreground.getBlue());
302:         if (hssfColor == null)
303:             hssfColor = workbook.getCustomPalette().findSimilarColor((byte)foreground.getRed(), (byte)foreground.getGreen(), (byte)foreground.getBlue());
304:         boolean bold = (font.getStyle() & Font.BOLD) != 0;
305:         boolean italic = (font.getStyle() & Font.ITALIC) != 0;
306:         HSSFFont hssfFont = workbook.findFont(bold ? HSSFFont.BOLDWEIGHT_BOLD : 0,
307:                     hssfColor.getIndex(),
308:                     (short)(font.getSize() * 20),
309:                     font.getName(),
310:                     italic,
311:                     false,
312:                     (short)0,
313:                     (byte)0);
314:         if (hssfFont == null)
315:         {
316:             hssfFont = workbook.createFont();
317:             hssfFont.setBoldweight(bold ? HSSFFont.BOLDWEIGHT_BOLD : 0);
318:             hssfFont.setColor(hssfColor.getIndex());
319:             hssfFont.setFontHeight((short)(font.getSize() * 20));
320:             hssfFont.setFontName(font.getName());
321:             hssfFont.setItalic(italic);
322:             hssfFont.setStrikeout(false);
323:             hssfFont.setTypeOffset((short) 0);
324:             hssfFont.setUnderline((byte) 0);
325:         }
326: 
327:         return hssfFont;
328:     }

354:     /**
355:      * Fills a (closed) polygon, as defined by a pair of arrays, which
356:      *  hold the <i>x</i> and <i>y</i> coordinates.
357:      * <p>
358:      * This draws the polygon, with <code>nPoint</code> line segments.
359:      * The first <code>nPoint&nbsp;-&nbsp;1</code> line segments are
360:      *  drawn between sequential points
361:      *  (<code>xPoints[i],yPoints[i],xPoints[i+1],yPoints[i+1]</code>).
362:      * The final line segment is a closing one, from the last point to
363:      *  the first (assuming they are different).
364:      * <p>
365:      * The area inside of the polygon is defined by using an
366:      *  even-odd fill rule (also known as the alternating rule), and
367:      *  the area inside of it is filled.
368:      * @param xPoints array of the <code>x</code> coordinates.
369:      * @param yPoints array of the <code>y</code> coordinates.
370:      * @param nPoints the total number of points in the polygon.
371:      * @see   java.awt.Graphics#drawPolygon(int[], int[], int)
372:      */
373:     public void fillPolygon(int xPoints[], int yPoints[],
374: 				     int nPoints)
375:     {
376:         int right  = findBiggest(xPoints);
377:         int bottom = findBiggest(yPoints);
378:         int left   = findSmallest(xPoints);
379:         int top    = findSmallest(yPoints);
380:         HSSFPolygon shape = escherGroup.createPolygon(new HSSFChildAnchor(left,top,right,bottom) );
381:         shape.setPolygonDrawArea(right - left, bottom - top);
382:         shape.setPoints(addToAll(xPoints, -left), addToAll(yPoints, -top));
383:         shape.setLineStyleColor(foreground.getRed(), foreground.getGreen(), foreground.getBlue());
384:         shape.setFillColor(foreground.getRed(), foreground.getGreen(), foreground.getBlue());
385:     }
386: 
387:     private int findBiggest( int[] values )
388:     {
389:         int result = Integer.MIN_VALUE;
390:         for ( int i = 0; i < values.length; i++ )
391:         {
392:             if (values[i] > result)
393:                 result = values[i];
394:         }
395:         return result;
396:     }

398:     private int findSmallest( int[] values )
399:     {
400:         int result = Integer.MAX_VALUE;
401:         for ( int i = 0; i < values.length; i++ )
402:         {
403:             if (values[i] < result)
404:                 result = values[i];
405:         }
406:         return result;
407:     }
