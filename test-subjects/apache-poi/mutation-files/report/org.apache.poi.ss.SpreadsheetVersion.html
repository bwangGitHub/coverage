 22: /**
 23:  * This enum allows spreadsheets from multiple Excel versions to be handled by the common code.
 24:  * Properties of this enum correspond to attributes of the <i>spreadsheet</i> that are easily
 25:  * discernable to the user.  It is not intended to deal with low-level issues like file formats.
 26:  * <p/>
 27:  *
 28:  * @author Josh Micich
 29:  * @author Yegor Kozlov
 30:  */
 31: public enum SpreadsheetVersion {
 32: 	/**
 33: 	 * Excel97 format aka BIFF8
 34: 	 * <ul>
 35: 	 * <li>The total number of available columns is 256 (2^8)</li>
 36: 	 * <li>The total number of available rows is 64k (2^16)</li>
 37: 	 * <li>The maximum number of arguments to a function is 30</li>
 38: 	 * <li>Number of conditional format conditions on a cell is 3</li>
 39:      * <li>Length of text cell contents is 32767</li>
 40: 	 * </ul>
 41: 	 */
 42: 	EXCEL97(0x10000, 0x0100, 30, 3, 32767),
 43: 
 44: 	/**
 45: 	 * Excel2007
 46: 	 *
 47: 	 * <ul>
 48: 	 * <li>The total number of available columns is 16K (2^14)</li>
 49: 	 * <li>The total number of available rows is 1M (2^20)</li>
 50: 	 * <li>The maximum number of arguments to a function is 255</li>
 51: 	 * <li>Number of conditional format conditions on a cell is unlimited
 52: 	 * (actually limited by available memory in Excel)</li>
 53:      * <li>Length of text cell contents is 32767</li>
 54: 	 * <ul>
 55: 	 */
 56: 	EXCEL2007(0x100000, 0x4000, 255, Integer.MAX_VALUE, 32767);
 57: 
 58: 	private final int _maxRows;
 59: 	private final int _maxColumns;
 60: 	private final int _maxFunctionArgs;
 61: 	private final int _maxCondFormats;
 62:     private final int _maxTextLength;
 63:

 65: 		_maxRows = maxRows;
 66: 		_maxColumns = maxColumns;
 67: 		_maxFunctionArgs = maxFunctionArgs;
 68: 		_maxCondFormats = maxCondFormats;
 69:         _maxTextLength = maxText;
 70:     }
 71:

 74: 	 */

 76: 		return _maxRows;
 77: 	}

 84: 	}
 85: 
 86: 	/**
 87: 	 * @return the maximum number of usable columns in each spreadsheet

 89: 	public int getMaxColumns() {

 91: 	}
 92: 
 93: 	/**
 94: 	 * @return the last (maximum) valid column index, equals to <code> getMaxColumns() - 1 </code>
 95: 	 */

 97: 		return _maxColumns - 1;
 98: 	}
 99: 
100: 	/**
101: 	 * @return the maximum number arguments that can be passed to a multi-arg function (e.g. COUNTIF)
102: 	 */
103: 	public int getMaxFunctionArgs() {
104: 		return _maxFunctionArgs;
105: 	}
106: 
107: 	/**
108: 	 *
109: 	 * @return the maximum number of conditional format conditions on a cell

113: 	}

122: 	}
123:
