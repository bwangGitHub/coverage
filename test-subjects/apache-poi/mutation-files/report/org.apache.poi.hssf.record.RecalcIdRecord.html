 23: /**
 24:  * Title: Recalc Id Record (0x01C1)<p/>
 25:  * Description:  This record contains an ID that marks when a worksheet was last
 26:  *               recalculated. It's an optimization Excel uses to determine if it
 27:  *               needs to  recalculate the spreadsheet when it's opened. So far, only
 28:  *               the two engine ids <code>0x80 0x38 0x01 0x00</code>
 29:  *               and <code>0x60 0x69 0x01 0x00</code> have been seen.
 30:  *               A value of <code>0x00</code> will cause Excel to recalculate
 31:  *               all formulas on the next load.<p/>
 32:  * REFERENCE:  http://chicago.sourceforge.net/devel/docs/excel/biff8.html<p/>
 33:  * @author Luc Girardin (luc dot girardin at macrofocus dot com)
 34:  */
 35: public final class RecalcIdRecord extends StandardRecord {
 36:     public final static short sid = 0x01C1;
 37:     private final int _reserved0;
 38: 
 39:     /**
 40:      * An unsigned integer that specifies the recalculation engine identifier
 41:      * of the recalculation engine that performed the last recalculation.
 42:      * If the value is less than the recalculation engine identifier associated with the application,
 43:      * the application will recalculate the results of all formulas on
 44:      * this workbook immediately after loading the file
 45:      */
 46:     private int _engineId;
 47: 
 48:     public RecalcIdRecord() {
 49:         _reserved0 = 0;
 50:         _engineId = 0;
 51:     }
 52: 
 53:     public RecalcIdRecord(RecordInputStream in) {
 54:     	in.readUShort(); // field 'rt' should have value 0x01C1, but Excel doesn't care during reading
 55:     	_reserved0 = in.readUShort();
 56:     	_engineId = in.readInt();
 57:     }
 58: 
 59:     public boolean isNeeded() {
 60:         return true;
 61:     }
 62: 
 63:     public void setEngineId(int val) {
 64:         _engineId = val;
 65:     }
 66: 
 67:     public int getEngineId() {
 68:         return _engineId;
 69:     }
 70: 
 71:     public String toString() {
 72:         StringBuffer buffer = new StringBuffer();
 73: 
 74:         buffer.append("[RECALCID]\n");
 75:         buffer.append("    .reserved = ").append(HexDump.shortToHex(_reserved0)).append("\n");
 76:         buffer.append("    .engineId = ").append(HexDump.intToHex(_engineId)).append("\n");
 77:         buffer.append("[/RECALCID]\n");
 78:         return buffer.toString();
 79:     }
 80: 
 81:     public void serialize(LittleEndianOutput out) {
 82:         out.writeShort(sid); // always write 'rt' field as 0x01C1
 83:         out.writeShort(_reserved0);
 84:         out.writeInt(_engineId);
 85:     }
 86: 
 87:     protected int getDataSize() {
 88:         return 8;
 89:     }
 90: 
 91:     public short getSid() {
 92:         return sid;
 93:     }
