 23: /**
 24:  * Title: FeatHdr (Feature Header) Record
 25:  * <P>
 26:  * This record specifies common information for Shared Features, and 
 27:  *  specifies the beginning of a collection of records to define them. 
 28:  * The collection of data (Globals Substream ABNF, macro sheet substream 
 29:  *  ABNF or worksheet substream ABNF) specifies Shared Feature data.
 30:  */
 31: public final class FeatHdrRecord extends StandardRecord  {
 32: 	/**
 33: 	 * Specifies the enhanced protection type. Used to protect a 
 34: 	 * shared workbook by restricting access to some areas of it 
 35: 	 */
 36: 	public static final int SHAREDFEATURES_ISFPROTECTION = 0x02;
 37: 	/**
 38: 	 * Specifies that formula errors should be ignored 
 39: 	 */
 40: 	public static final int SHAREDFEATURES_ISFFEC2       = 0x03;
 41: 	/**
 42: 	 * Specifies the smart tag type. Recognises certain
 43: 	 * types of entries (proper names, dates/times etc) and
 44: 	 * flags them for action 
 45: 	 */
 46: 	public static final int SHAREDFEATURES_ISFFACTOID    = 0x04;
 47: 	/**
 48: 	 * Specifies the shared list type. Used for a table
 49: 	 * within a sheet
 50: 	 */
 51: 	public static final int SHAREDFEATURES_ISFLIST       = 0x05;
 52: 
 53: 	
 54: 	public final static short sid = 0x0867;
 55: 
 56: 	private FtrHeader futureHeader;
 57: 	private int isf_sharedFeatureType; // See SHAREDFEATURES_
 58: 	private byte reserved; // Should always be one
 59: 	/** 
 60: 	 * 0x00000000 = rgbHdrData not present
 61: 	 * 0xffffffff = rgbHdrData present
 62: 	 */
 63: 	private long cbHdrData;
 64: 	/** We need a BOFRecord to make sense of this... */
 65: 	private byte[] rgbHdrData;
 66: 
 67: 	public FeatHdrRecord() {
 68: 		futureHeader = new FtrHeader();
 69: 		futureHeader.setRecordType(sid);
 70: 	}
 71: 
 72: 	public short getSid() {
 73: 		return sid;
 74: 	}
 75: 
 76: 	public FeatHdrRecord(RecordInputStream in) {
 77: 		futureHeader = new FtrHeader(in);
 78: 		
 79: 		isf_sharedFeatureType = in.readShort();
 80: 		reserved = in.readByte();
 81: 		cbHdrData = in.readInt();
 82: 		// Don't process this just yet, need the BOFRecord
 83: 		rgbHdrData = in.readRemainder();
 84: 	}
 85: 
 86: 	public String toString() {
 87: 		StringBuffer buffer = new StringBuffer();
 88: 		buffer.append("[FEATURE HEADER]\n");
 89: 		
 90: 		// TODO ...
 91: 		
 92: 		buffer.append("[/FEATURE HEADER]\n");
 93: 		return buffer.toString();
 94: 	}
 95: 
 96: 	public void serialize(LittleEndianOutput out) {
 97: 		futureHeader.serialize(out);
 98: 		
 99: 		out.writeShort(isf_sharedFeatureType);
100: 		out.writeByte(reserved);
101: 		out.writeInt((int)cbHdrData);
102: 		out.write(rgbHdrData);
103: 	}
104: 
105: 	protected int getDataSize() {
106: 		return 12 + 2+1+4+rgbHdrData.length;
107: 	}
