 26: /**
 27:  * Generates escher records when provided the byte array containing those records.
 28:  *
 29:  * @author Glen Stampoultzis
 30:  * @author Nick Burch   (nick at torchbox . com)
 31:  *
 32:  * @see EscherRecordFactory
 33:  */
 34: public class DefaultEscherRecordFactory implements EscherRecordFactory {
 35:     private static Class<?>[] escherRecordClasses = { EscherBSERecord.class,
 36:             EscherOptRecord.class, EscherTertiaryOptRecord.class,
 37:             EscherClientAnchorRecord.class, EscherDgRecord.class,
 38:             EscherSpgrRecord.class, EscherSpRecord.class,
 39:             EscherClientDataRecord.class, EscherDggRecord.class,
 40:             EscherSplitMenuColorsRecord.class, EscherChildAnchorRecord.class,
 41:             EscherTextboxRecord.class };
 42:     private static Map<Short, Constructor<? extends EscherRecord>> recordsMap = recordsToMap( escherRecordClasses );
 43: 
 44:     /**
 45:      * Creates an instance of the escher record factory
 46:      */
 47:     public DefaultEscherRecordFactory() {
 48:         // no instance initialisation
 49:     }
 50: 
 51:     /**
 52:      * Generates an escher record including the any children contained under that record.
 53:      * An exception is thrown if the record could not be generated.
 54:      *
 55:      * @param data   The byte array containing the records
 56:      * @param offset The starting offset into the byte array
 57:      * @return The generated escher record
 58:      */
 59:     public EscherRecord createRecord(byte[] data, int offset) {
 60:         short options = LittleEndian.getShort( data, offset );
 61:         short recordId = LittleEndian.getShort( data, offset + 2 );
 62:         // int remainingBytes = LittleEndian.getInt( data, offset + 4 );
 63: 
 64:         // Options of 0x000F means container record
 65:         // However, EscherTextboxRecord are containers of records for the
 66:         //  host application, not of other Escher records, so treat them
 67:         //  differently
 68:         if ( ( options & (short) 0x000F ) == (short) 0x000F
 69:              && recordId != EscherTextboxRecord.RECORD_ID ) {
 70:             EscherContainerRecord r = new EscherContainerRecord();
 71:             r.setRecordId( recordId );
 72:             r.setOptions( options );
 73:             return r;
 74:         }
 75: 
 76:         if (recordId >= EscherBlipRecord.RECORD_ID_START
 77:                 && recordId <= EscherBlipRecord.RECORD_ID_END) {
 78:             EscherBlipRecord r;
 79:             if (recordId == EscherBitmapBlip.RECORD_ID_DIB ||
 80:                     recordId == EscherBitmapBlip.RECORD_ID_JPEG ||
 81:                     recordId == EscherBitmapBlip.RECORD_ID_PNG)
 82:             {
 83:                 r = new EscherBitmapBlip();
 84:             }
 85:             else if (recordId == EscherMetafileBlip.RECORD_ID_EMF ||
 86:                     recordId == EscherMetafileBlip.RECORD_ID_WMF ||
 87:                     recordId == EscherMetafileBlip.RECORD_ID_PICT)
 88:             {
 89:                 r = new EscherMetafileBlip();
 90:             } else {
 91:                 r = new EscherBlipRecord();
 92:             }
 93:             r.setRecordId( recordId );
 94:             r.setOptions( options );
 95:             return r;
 96:         }
 97: 
 98:         Constructor<? extends EscherRecord> recordConstructor = recordsMap.get(Short.valueOf(recordId));
 99:         EscherRecord escherRecord = null;
100:         if (recordConstructor == null) {
101:             return new UnknownEscherRecord();
102:         }
103:         try {
104:             escherRecord = recordConstructor.newInstance(new Object[] {});
105:         } catch (Exception e) {
106:             return new UnknownEscherRecord();
107:         }
108:         escherRecord.setRecordId(recordId);
109:         escherRecord.setOptions(options);
110:         return escherRecord;
111:     }
112: 
113:     /**
114:      * Converts from a list of classes into a map that contains the record id as the key and
115:      * the Constructor in the value part of the map.  It does this by using reflection to look up
116:      * the RECORD_ID field then using reflection again to find a reference to the constructor.
117:      *
118:      * @param recClasses The records to convert
119:      * @return The map containing the id/constructor pairs.
120:      */
121:     private static Map<Short, Constructor<? extends EscherRecord>> recordsToMap(Class<?>[] recClasses) {
122:         Map<Short, Constructor<? extends EscherRecord>> result = new HashMap<Short, Constructor<? extends EscherRecord>>();
123:         final Class<?>[] EMPTY_CLASS_ARRAY = new Class[0];
124: 
125:         for (int i = 0; i < recClasses.length; i++) {
126:             @SuppressWarnings("unchecked")
127:             Class<? extends EscherRecord> recCls = (Class<? extends EscherRecord>) recClasses[i];
128:             short sid;
129:             try {
130:                 sid = recCls.getField("RECORD_ID").getShort(null);
131:             } catch (IllegalArgumentException e) {
132:                 throw new RuntimeException(e);
133:             } catch (IllegalAccessException e) {
134:                 throw new RuntimeException(e);
135:             } catch (NoSuchFieldException e) {
136:                 throw new RuntimeException(e);
137:             }
138:             Constructor<? extends EscherRecord> constructor;
139:             try {
140:                 constructor = recCls.getConstructor( EMPTY_CLASS_ARRAY );
141:             } catch (NoSuchMethodException e) {
142:                 throw new RuntimeException(e);
143:             }
144:             result.put(Short.valueOf(sid), constructor);
145:         }
146:         return result;
147:     }
