 91:     private boolean isPlainColumn(String refPart) {
 92:         for(int i=refPart.length()-1; i>=0; i--) {
 93:             int ch = refPart.charAt(i);
 94:             if (ch == '$' && i==0) {
 95:                 continue;
 96:             }

151:      * Is the reference for a contiguous (i.e.
152:      *  unbroken) area, or is it made up of
153:      *  several different parts?
154:      * (If it is, you will need to call
155:      *  {@link #generateContiguous(String)})
156:      */
157:     public static boolean isContiguous(String reference) {
158:        // If there's a sheet name, strip it off
159:        int sheetRefEnd = reference.indexOf('!'); 
160:        if(sheetRefEnd != -1) {
161:           reference = reference.substring(sheetRefEnd);
162:        }
163: 
164:        // Check for the , as a sign of non-coniguous
165:        if(reference.indexOf(',') == -1) {
166:           return true;
167:        }
168:        return false;

305:     /**
306:      * Separates Area refs in two parts and returns them as separate elements in a String array,
307:      * each qualified with the sheet name (if present)
308:      * 
309:      * @return array with one or two elements. never <code>null</code>
310:      */
311:     private static String[] separateAreaRefs(String reference) {
312:         // TODO - refactor cell reference parsing logic to one place.
313:         // Current known incarnations: 
314:         //   FormulaParser.GetName()
315:         //   CellReference.separateRefParts() 
316:         //   AreaReference.separateAreaRefs() (here)
317:         //   SheetNameFormatter.format() (inverse)
318:         
319:         
320:         int len = reference.length();
321:         int delimiterPos = -1;
322:         boolean insideDelimitedName = false;
323:         for(int i=0; i<len; i++) {
324:             switch(reference.charAt(i)) {
325:                 case CELL_DELIMITER:
326:                     if(!insideDelimitedName) {
327:                         if(delimiterPos >=0) {
328:                             throw new IllegalArgumentException("More than one cell delimiter '" 
329:                                     + CELL_DELIMITER + "' appears in area reference '" + reference + "'");
330:                         }
331:                         delimiterPos = i;
332:                     }
333:                 default:
334:                     continue;
335:                 case SPECIAL_NAME_DELIMITER:
336:                     // fall through
337:             }
338:             if(!insideDelimitedName) {
339:                 insideDelimitedName = true;
340:                 continue;
341:             }
342:             
343:             if(i >= len-1) {
344:                 // reference ends with the delimited name. 
345:                 // Assume names like: "Sheet1!'A1'" are never legal.
346:                 throw new IllegalArgumentException("Area reference '" + reference 
347:                         + "' ends with special name delimiter '"  + SPECIAL_NAME_DELIMITER + "'");
348:             }
349:             if(reference.charAt(i+1) == SPECIAL_NAME_DELIMITER) {
350:                 // two consecutive quotes is the escape sequence for a single one
351:                 i++; // skip this and keep parsing the special name
352:             } else {
353:                 // this is the end of the delimited name
354:                 insideDelimitedName = false;
355:             }
356:         }
357:         if(delimiterPos < 0) {
358:             return new String[] { reference, };
359:         }
360: 
361:         String partA = reference.substring(0, delimiterPos);
362:         String partB = reference.substring(delimiterPos+1);
363:         if(partB.indexOf(SHEET_NAME_DELIMITER) >=0) {
364:             // TODO - are references like "Sheet1!A1:Sheet1:B2" ever valid?  
365:             // FormulaParser has code to handle that.
366:             
367:             throw new RuntimeException("Unexpected " + SHEET_NAME_DELIMITER 
368:                     + " in second cell reference of '" + reference + "'");
369:         }
370:         
371:         int plingPos = partA.lastIndexOf(SHEET_NAME_DELIMITER);
372:         if(plingPos < 0) {
373:             return new String [] { partA, partB, };
374:         }
375:         
376:         String sheetName = partA.substring(0, plingPos + 1); // +1 to include delimiter
377:         
378:         return new String [] { partA, sheetName + partB, };
379:     }
