 29:  * Manages a collection of {@link WorkbookEvaluator}s, in order to support evaluation of formulas

 40: 			super(msg);

 50: 	private CollaboratingWorkbooksEnvironment() {
 51: 		_evaluatorsByName = Collections.emptyMap();

 54: 	public static void setup(String[] workbookNames, WorkbookEvaluator[] evaluators) {
 55: 		int nItems = workbookNames.length;
 56: 		if (evaluators.length != nItems) {
 57: 			throw new IllegalArgumentException("Number of workbook names is " + nItems
 58: 					+ " but number of evaluators is " + evaluators.length);
 59: 		}
 60: 		if (nItems < 1) {
 61: 			throw new IllegalArgumentException("Must provide at least one collaborating worbook");
 62: 		}
 63: 		new CollaboratingWorkbooksEnvironment(workbookNames, evaluators, nItems);
 64: 	}

 66: 	private CollaboratingWorkbooksEnvironment(String[] workbookNames, WorkbookEvaluator[] evaluators, int nItems) {
 67: 		Map<String, WorkbookEvaluator> m = new HashMap<String, WorkbookEvaluator>(nItems * 3 / 2);
 68: 		IdentityHashMap<WorkbookEvaluator, String> uniqueEvals = new IdentityHashMap<WorkbookEvaluator, String>(nItems * 3 / 2);
 69: 		for(int i=0; i<nItems; i++) {
 70: 			String wbName = workbookNames[i];
 71: 			WorkbookEvaluator wbEval = evaluators[i];
 72: 			if (m.containsKey(wbName)) {
 73: 				throw new IllegalArgumentException("Duplicate workbook name '" + wbName + "'");
 74: 			}
 75: 			if (uniqueEvals.containsKey(wbEval)) {
 76: 				String msg = "Attempted to register same workbook under names '"
 77: 					+ uniqueEvals.get(wbEval) + "' and '" + wbName + "'";
 78: 				throw new IllegalArgumentException(msg);
 79: 			}
 80: 			uniqueEvals.put(wbEval, wbName);
 81: 			m.put(wbName, wbEval);
 82: 		}
 83: 		unhookOldEnvironments(evaluators);
 84: 		hookNewEnvironment(evaluators, this);
 85: 		_unhooked = false;
 86: 		_evaluators = evaluators;
 87: 		_evaluatorsByName = m;
 88: 	}

 90: 	private static void hookNewEnvironment(WorkbookEvaluator[] evaluators, CollaboratingWorkbooksEnvironment env) {
 91: 
 92: 		// All evaluators will need to share the same cache.
 93: 		// but the cache takes an optional evaluation listener.
 94: 		int nItems = evaluators.length;
 95: 		IEvaluationListener evalListener = evaluators[0].getEvaluationListener();
 96: 		// make sure that all evaluators have the same listener
 97: 		for(int i=0; i<nItems; i++) {
 98: 			if(evalListener != evaluators[i].getEvaluationListener()) {
 99: 				// This would be very complex to support
100: 				throw new RuntimeException("Workbook evaluators must all have the same evaluation listener");
101: 			}
102: 		}
103: 		EvaluationCache cache = new EvaluationCache(evalListener);
104: 
105: 		for(int i=0; i<nItems; i++) {
106: 			evaluators[i].attachToEnvironment(env, cache, i);
107: 		}
108: 	}

110: 	/**
111: 	 * Completely dismantles all workbook environments that the supplied evaluators are part of
112: 	 */
113: 	private void unhookOldEnvironments(WorkbookEvaluator[] evaluators) {
114: 		Set<CollaboratingWorkbooksEnvironment> oldEnvs = new HashSet<CollaboratingWorkbooksEnvironment>();
115: 		for(int i=0; i<evaluators.length; i++) {
116: 			oldEnvs.add(evaluators[i].getEnvironment());
117: 		}
118: 		CollaboratingWorkbooksEnvironment[] oldCWEs = new CollaboratingWorkbooksEnvironment[oldEnvs.size()];
119: 		oldEnvs.toArray(oldCWEs);
120: 		for (int i = 0; i < oldCWEs.length; i++) {
121: 			oldCWEs[i].unhook();
122: 		}
123: 	}
124: 
125: 	/**
126: 	 * Tell all contained evaluators that this environment should be discarded
127: 	 */
128: 	private void unhook() {
129: 		if (_evaluators.length < 1) {
130: 			// Never dismantle the EMPTY environment
131: 			return;
132: 		}
133: 		for (int i = 0; i < _evaluators.length; i++) {
134: 			_evaluators[i].detachFromEnvironment();
135: 		}
136: 		_unhooked = true;
137: 	}

139: 	public WorkbookEvaluator getWorkbookEvaluator(String workbookName) throws WorkbookNotFoundException {
140: 		if (_unhooked) {
141: 			throw new IllegalStateException("This environment has been unhooked");
142: 		}
143: 		WorkbookEvaluator result = _evaluatorsByName.get(workbookName);
144: 		if (result == null) {
145: 			StringBuffer sb = new StringBuffer(256);
146: 			sb.append("Could not resolve external workbook name '").append(workbookName).append("'.");
147: 			if (_evaluators.length < 1) {
148: 				sb.append(" Workbook environment has not been set up.");
149: 			} else {
150: 				sb.append(" The following workbook names are valid: (");
151: 				Iterator<String> i = _evaluatorsByName.keySet().iterator();
152: 				int count=0;
153: 				while(i.hasNext()) {
154: 					if (count++>0) {
155: 						sb.append(", ");
156: 					}
157: 					sb.append("'").append(i.next()).append("'");
158: 				}
159: 				sb.append(")");
160: 			}
161: 			throw new WorkbookNotFoundException(sb.toString());
162: 		}
163: 		return result;
164: 	}
