 66: 	private static String replaceAllOccurrences(String oldStr, String searchStr, String newStr) {
 67: 		StringBuffer sb = new StringBuffer();
 68: 		int startIndex = 0;
 69: 		int nextMatch = -1;
 70: 		while (true) {
 71: 			nextMatch = oldStr.indexOf(searchStr, startIndex);
 72: 			if (nextMatch < 0) {
 73: 				// store everything from end of last match to end of string
 74: 				sb.append(oldStr.substring(startIndex));
 75: 				return sb.toString();
 76: 			}
 77: 			// store everything from end of last match to start of this match
 78: 			sb.append(oldStr.substring(startIndex, nextMatch));
 79: 			sb.append(newStr);

 81: 		}

 85: 		if (searchStr.length() < 1) {
