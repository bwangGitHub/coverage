 28:      * This is really a debugging version of <code>System.arraycopy()</code>.
 29:      * Use it to provide better exception messages when copying arrays around.
 30:      * For production use it's better to use the original for speed.
 31:      */
 32:     public static void arraycopy(byte[] src, int src_position, byte[] dst, int dst_position, int length)
 33:     {
 34:         if (src_position < 0)
 35:             throw new IllegalArgumentException("src_position was less than 0.  Actual value " + src_position);
 36:         if (src_position >= src.length)
 37:             throw new IllegalArgumentException( "src_position was greater than src array size.  Tried to write starting at position " + src_position + " but the array length is " + src.length );
 38:         if (src_position + length > src.length)
 39:             throw new IllegalArgumentException("src_position + length would overrun the src array.  Expected end at " + (src_position + length) + " actual end at " + src.length);
 40:         if (dst_position < 0)
 41:             throw new IllegalArgumentException("dst_position was less than 0.  Actual value " + dst_position);
 42:         if (dst_position >= dst.length)
 43:             throw new IllegalArgumentException( "dst_position was greater than dst array size.  Tried to write starting at position " + dst_position + " but the array length is " + dst.length );
 44:         if (dst_position + length > dst.length)
 45:             throw new IllegalArgumentException("dst_position + length would overrun the dst array.  Expected end at " + (dst_position + length) + " actual end at " + dst.length);
 46: 
 47:         System.arraycopy( src, src_position, dst, dst_position, length);
 48:     }

 50:     /**
 51:      * Moves a number of entries in an array to another point in the array,
 52:      *  shifting those inbetween as required.
 53:      * @param array The array to alter
 54:      * @param moveFrom The (0 based) index of the first entry to move
 55:      * @param moveTo The (0 based) index of the positition to move to
 56:      * @param numToMove The number of entries to move
 57:      */
 58:     public static void arrayMoveWithin(Object[] array, int moveFrom, int moveTo, int numToMove) {
 59:     	// If we're not asked to do anything, return now
 60:     	if(numToMove <= 0) { return; }
 61:     	if(moveFrom == moveTo) { return; }
 62:     	
 63:     	// Check that the values supplied are valid
 64:     	if(moveFrom < 0 || moveFrom >= array.length) {
 65:     		throw new IllegalArgumentException("The moveFrom must be a valid array index");
 66:     	}
 67:     	if(moveTo < 0 || moveTo >= array.length) {
 68:     		throw new IllegalArgumentException("The moveTo must be a valid array index");
 69:     	}
 70:     	if(moveFrom+numToMove > array.length) {
 71:     		throw new IllegalArgumentException("Asked to move more entries than the array has");
 72:     	}
 73:     	if(moveTo+numToMove > array.length) {
 74:     		throw new IllegalArgumentException("Asked to move to a position that doesn't have enough space");
 75:     	}
 76:     	
 77:     	// Grab the bit to move 
 78:     	Object[] toMove = new Object[numToMove];
 79:     	System.arraycopy(array, moveFrom, toMove, 0, numToMove);
 80:     	
 81:     	// Grab the bit to be shifted
 82:     	Object[] toShift;
 83:     	int shiftTo;
 84:     	if(moveFrom > moveTo) {
 85:     		// Moving to an earlier point in the array
 86:     		// Grab everything between the two points
 87:     		toShift = new Object[(moveFrom-moveTo)];
 88:     		System.arraycopy(array, moveTo, toShift, 0, toShift.length);
 89:     		shiftTo = moveTo + numToMove;
 90:     	} else {
 91:     		// Moving to a later point in the array
 92:     		// Grab everything from after the toMove block, to the new point
 93:     		toShift = new Object[(moveTo-moveFrom)];
 94:     		System.arraycopy(array, moveFrom+numToMove, toShift, 0, toShift.length);
 95:     		shiftTo = moveFrom;
 96:     	}
 97:     	
 98:     	// Copy the moved block to its new location
 99:     	System.arraycopy(toMove, 0, array, moveTo, toMove.length);
100:     	
101:     	// And copy the shifted block to the shifted location
102:     	System.arraycopy(toShift, 0, array, shiftTo, toShift.length);
103:     	
104:     	
105:     	// We're done - array will now have everything moved as required
106:     }

108:     /**
109:      * Copies the specified array, truncating or padding with zeros (if
110:      * necessary) so the copy has the specified length. This method is temporary
111:      * replace for Arrays.copyOf() until we start to require JDK 1.6.
112:      * 
113:      * @param source
114:      *            the array to be copied
115:      * @param newLength
116:      *            the length of the copy to be returned
117:      * @return a copy of the original array, truncated or padded with zeros to
118:      *         obtain the specified length
119:      * @throws NegativeArraySizeException
120:      *             if <tt>newLength</tt> is negative
121:      * @throws NullPointerException
122:      *             if <tt>original</tt> is null
123:      */
124:     public static byte[] copyOf( byte[] source, int newLength )
125:     {
126:         byte[] result = new byte[newLength];
127:         System.arraycopy( source, 0, result, 0,
128:                 Math.min( source.length, newLength ) );
129:         return result;
130:     }

132:     /**
133:      * Copies the specified range of the specified array into a new array.
134:      * The initial index of the range (<tt>from</tt>) must lie between zero
135:      * and <tt>original.length</tt>, inclusive.  The value at
136:      * <tt>original[from]</tt> is placed into the initial element of the copy
137:      * (unless <tt>from == original.length</tt> or <tt>from == to</tt>).
138:      * Values from subsequent elements in the original array are placed into
139:      * subsequent elements in the copy.  The final index of the range
140:      * (<tt>to</tt>), which must be greater than or equal to <tt>from</tt>,
141:      * may be greater than <tt>original.length</tt>, in which case
142:      * <tt>(byte)0</tt> is placed in all elements of the copy whose index is
143:      * greater than or equal to <tt>original.length - from</tt>.  The length
144:      * of the returned array will be <tt>to - from</tt>.
145:      *
146:      * This method is temporary
147:      * replace for Arrays.copyOfRange() until we start to require JDK 1.6.
148:      *
149:      * @param original the array from which a range is to be copied
150:      * @param from the initial index of the range to be copied, inclusive
151:      * @param to the final index of the range to be copied, exclusive.
152:      *     (This index may lie outside the array.)
153:      * @return a new array containing the specified range from the original array,
154:      *     truncated or padded with zeros to obtain the required length
155:      * @throws ArrayIndexOutOfBoundsException if <tt>from &lt; 0</tt>
156:      *     or <tt>from &gt; original.length()</tt>
157:      * @throws IllegalArgumentException if <tt>from &gt; to</tt>
158:      * @throws NullPointerException if <tt>original</tt> is null
159:      * @since 1.6
160:      */
161:     public static byte[] copyOfRange(byte[] original, int from, int to) {
162:         int newLength = to - from;
163:         if (newLength < 0)
164:             throw new IllegalArgumentException(from + " > " + to);
165:         byte[] copy = new byte[newLength];
166:         System.arraycopy(original, from, copy, 0,
167:                 Math.min(original.length - from, newLength));
168:         return copy;
169:     }
