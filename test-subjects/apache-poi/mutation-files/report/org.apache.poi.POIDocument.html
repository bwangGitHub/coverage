 97: 	/**
 98: 	 * Will create whichever of SummaryInformation
 99: 	 *  and DocumentSummaryInformation (HPSF) properties
100: 	 *  are not already part of your document.
101: 	 * This is normally useful when creating a new
102: 	 *  document from scratch.
103: 	 * If the information properties are already there,
104: 	 *  then nothing will happen.
105: 	 */
106: 	public void createInformationProperties() {
107:         if(!initialized) readProperties();
108: 		if(sInf == null) {
109: 			sInf = PropertySetFactory.newSummaryInformation();
110: 		}
111: 		if(dsInf == null) {
112: 			dsInf = PropertySetFactory.newDocumentSummaryInformation();
113: 		}
114: 	}

116: 	/**
117: 	 * Find, and create objects for, the standard
118: 	 *  Documment Information Properties (HPSF).
119: 	 * If a given property set is missing or corrupt,
120: 	 *  it will remain null;
121: 	 */
122: 	protected void readProperties() {
123: 		PropertySet ps;
124: 		
125: 		// DocumentSummaryInformation
126: 		ps = getPropertySet(DocumentSummaryInformation.DEFAULT_STREAM_NAME);
127: 		if(ps != null && ps instanceof DocumentSummaryInformation) {
128: 			dsInf = (DocumentSummaryInformation)ps;
129: 		} else if(ps != null) {
130: 			logger.log(POILogger.WARN, "DocumentSummaryInformation property set came back with wrong class - ", ps.getClass());
131: 		}
132: 
133: 		// SummaryInformation
134: 		ps = getPropertySet(SummaryInformation.DEFAULT_STREAM_NAME);
135: 		if(ps instanceof SummaryInformation) {
136: 			sInf = (SummaryInformation)ps;
137: 		} else if(ps != null) {
138: 			logger.log(POILogger.WARN, "SummaryInformation property set came back with wrong class - ", ps.getClass());
139: 		}
140: 
141: 		// Mark the fact that we've now loaded up the properties
142:         initialized = true;
143: 	}
144: 
145: 	/** 
146: 	 * For a given named property entry, either return it or null if
147: 	 *  if it wasn't found
148: 	 */
149: 	protected PropertySet getPropertySet(String setName) {
150: 	   //directory can be null when creating new documents
151: 	   if(directory == null) return null;
152: 
153: 	   DocumentInputStream dis;
154: 	   try {
155: 	      // Find the entry, and get an input stream for it
156: 	      dis = directory.createDocumentInputStream( directory.getEntry(setName) );
157: 	   } catch(IOException ie) {
158: 	      // Oh well, doesn't exist
159: 	      logger.log(POILogger.WARN, "Error getting property set with name " + setName + "\n" + ie);
160: 	      return null;
161: 	   }
162: 
163: 	   try {
164: 	      // Create the Property Set
165: 	      PropertySet set = PropertySetFactory.create(dis);
166: 	      return set;
167: 	   } catch(IOException ie) {
168: 	      // Must be corrupt or something like that
169: 	      logger.log(POILogger.WARN, "Error creating property set with name " + setName + "\n" + ie);
170: 	   } catch(org.apache.poi.hpsf.HPSFException he) {
171: 	      // Oh well, doesn't exist
172: 	      logger.log(POILogger.WARN, "Error creating property set with name " + setName + "\n" + he);
173: 	   }
174: 	   return null;
175: 	}

184: 	/**
185: 	 * Writes out the standard Documment Information Properties (HPSF)
186: 	 * @param outFS the POIFSFileSystem to write the properties into
187: 	 * @param writtenEntries a list of POIFS entries to add the property names too
188: 	 */
189: 	protected void writeProperties(POIFSFileSystem outFS, List<String> writtenEntries) throws IOException {
190:         SummaryInformation si = getSummaryInformation();
191:         if(si != null) {
192: 			writePropertySet(SummaryInformation.DEFAULT_STREAM_NAME, si, outFS);
193: 			if(writtenEntries != null) {
194: 				writtenEntries.add(SummaryInformation.DEFAULT_STREAM_NAME);
195: 			}
196: 		}
197:         DocumentSummaryInformation dsi = getDocumentSummaryInformation();
198:         if(dsi != null) {
199: 			writePropertySet(DocumentSummaryInformation.DEFAULT_STREAM_NAME, dsi, outFS);
200: 			if(writtenEntries != null) {
201: 				writtenEntries.add(DocumentSummaryInformation.DEFAULT_STREAM_NAME);
202: 			}
203: 		}
204: 	}
205: 	
206: 	/**
207: 	 * Writes out a given ProperySet
208: 	 * @param name the (POIFS Level) name of the property to write
209: 	 * @param set the PropertySet to write out 
210: 	 * @param outFS the POIFSFileSystem to write the property into
211: 	 */
212: 	protected void writePropertySet(String name, PropertySet set, POIFSFileSystem outFS) throws IOException {
213: 		try {
214: 			MutablePropertySet mSet = new MutablePropertySet(set);
215: 			ByteArrayOutputStream bOut = new ByteArrayOutputStream();
216: 
217: 			mSet.write(bOut);
218: 			byte[] data = bOut.toByteArray();
219: 			ByteArrayInputStream bIn = new ByteArrayInputStream(data);
220: 			outFS.createDocument(bIn,name);
221: 
222: 			logger.log(POILogger.INFO, "Wrote property set " + name + " of size " + data.length);
223: 		} catch(org.apache.poi.hpsf.WritingNotSupportedException wnse) {
224: 			System.err.println("Couldn't write property set with name " + name + " as not supported by HPSF yet");
225: 		}
226: 	}
