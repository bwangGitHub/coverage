 46: 	 * @return the number of evaluated cells in the range that match the specified criteria
 47: 	 */
 48: 	public static int countMatchingCellsInArea(TwoDEval areaEval, I_MatchPredicate criteriaPredicate) {
 49: 		int result = 0;
 50: 
 51: 		int height = areaEval.getHeight();
 52: 		int width = areaEval.getWidth();
 53: 		for (int rrIx=0; rrIx<height; rrIx++) {
 54: 			for (int rcIx=0; rcIx<width; rcIx++) {
 55: 				ValueEval ve = areaEval.getValue(rrIx, rcIx);
 56: 
 57:                 if(criteriaPredicate instanceof I_MatchAreaPredicate){
 58:                     I_MatchAreaPredicate areaPredicate = (I_MatchAreaPredicate)criteriaPredicate;
 59:                     if(!areaPredicate.matches(areaEval, rrIx, rcIx)) continue;
 60:                 }
 61: 
 62: 				if(criteriaPredicate.matches(ve)) {
 63: 					result++;
 64: 				}
 65: 			}
 66: 		}
 67: 		return result;
 68: 	}

 73: 		if(criteriaPredicate.matches(refEval.getInnerValueEval())) {
 74: 			return 1;

 76: 		return 0;
 77: 	}

 79: 		if (eval == null) {
