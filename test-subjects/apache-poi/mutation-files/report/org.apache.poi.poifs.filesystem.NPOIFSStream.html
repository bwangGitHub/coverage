 31: /**
 32:  * This handles reading and writing a stream within a
 33:  *  {@link NPOIFSFileSystem}. It can supply an iterator
 34:  *  to read blocks, and way to write out to existing and
 35:  *  new blocks.
 36:  * Most users will want a higher level version of this, 
 37:  *  which deals with properties to track which stream
 38:  *  this is.
 39:  * This only works on big block streams, it doesn't
 40:  *  handle small block ones.
 41:  * This uses the new NIO code
 42:  * 
 43:  * TODO Implement a streaming write method, and append
 44:  */
 45: 
 46: public class NPOIFSStream implements Iterable<ByteBuffer>
 47: {
 48: 	private BlockStore blockStore;
 49: 	private int startBlock;
 50: 	
 51: 	/**
 52: 	 * Constructor for an existing stream. It's up to you
 53: 	 *  to know how to get the start block (eg from a 
 54: 	 *  {@link HeaderBlock} or a {@link Property}) 
 55: 	 */
 56: 	public NPOIFSStream(BlockStore blockStore, int startBlock) {
 57: 	   this.blockStore = blockStore;
 58: 	   this.startBlock = startBlock;
 59: 	}
 60: 	
 61: 	/**
 62: 	 * Constructor for a new stream. A start block won't
 63: 	 *  be allocated until you begin writing to it.
 64: 	 */
 65: 	public NPOIFSStream(BlockStore blockStore) {
 66:       this.blockStore = blockStore;
 67: 	   this.startBlock = POIFSConstants.END_OF_CHAIN;
 68: 	}
 69: 	
 70: 	/**
 71: 	 * What block does this stream start at?
 72: 	 * Will be {@link POIFSConstants#END_OF_CHAIN} for a
 73: 	 *  new stream that hasn't been written to yet.
 74: 	 */
 75: 	public int getStartBlock() {
 76: 	   return startBlock;
 77: 	}
 78: 
 79: 	/**
 80: 	 * Returns an iterator that'll supply one {@link ByteBuffer}
 81: 	 *  per block in the stream.
 82: 	 */
 83:    public Iterator<ByteBuffer> iterator() {
 84:       return getBlockIterator();
 85:    }
 86: 	
 87:    public Iterator<ByteBuffer> getBlockIterator() {
 88:       if(startBlock == POIFSConstants.END_OF_CHAIN) {
 89:          throw new IllegalStateException(
 90:                "Can't read from a new stream before it has been written to"
 91:          );
 92:       }
 93:       return new StreamBlockByteBufferIterator(startBlock);
 94:    }
 95:    
 96:    /**
 97:     * Updates the contents of the stream to the new
 98:     *  set of bytes.
 99:     * Note - if this is property based, you'll still
100:     *  need to update the size in the property yourself
101:     */
102:    public void updateContents(byte[] contents) throws IOException {
103:       // How many blocks are we going to need?
104:       int blockSize = blockStore.getBlockStoreBlockSize();
105:       int blocks = (int)Math.ceil( ((double)contents.length) / blockSize );
106:       
107:       // Make sure we don't encounter a loop whilst overwriting
108:       //  the existing blocks
109:       ChainLoopDetector loopDetector = blockStore.getChainLoopDetector();
110:       
111:       // Start writing
112:       int prevBlock = POIFSConstants.END_OF_CHAIN;
113:       int nextBlock = startBlock;
114:       for(int i=0; i<blocks; i++) {
115:          int thisBlock = nextBlock;
116:          
117:          // Allocate a block if needed, otherwise figure
118:          //  out what the next block will be
119:          if(thisBlock == POIFSConstants.END_OF_CHAIN) {
120:             thisBlock = blockStore.getFreeBlock();
121:             loopDetector.claim(thisBlock);
122:             
123:             // We're on the end of the chain
124:             nextBlock = POIFSConstants.END_OF_CHAIN;
125:             
126:             // Mark the previous block as carrying on to us if needed
127:             if(prevBlock != POIFSConstants.END_OF_CHAIN) {
128:                blockStore.setNextBlock(prevBlock, thisBlock);
129:             }
130:             blockStore.setNextBlock(thisBlock, POIFSConstants.END_OF_CHAIN);
131:             
132:             // If we've just written the first block on a 
133:             //  new stream, save the start block offset
134:             if(this.startBlock == POIFSConstants.END_OF_CHAIN) {
135:                this.startBlock = thisBlock;
136:             }
137:          } else {
138:             loopDetector.claim(thisBlock);
139:             nextBlock = blockStore.getNextBlock(thisBlock);
140:          }
141:          
142:          // Write it
143:          ByteBuffer buffer = blockStore.createBlockIfNeeded(thisBlock);
144:          int startAt = i*blockSize;
145:          int endAt = Math.min(contents.length - startAt, blockSize);
146:          buffer.put(contents, startAt, endAt); 
147:          
148:          // Update pointers
149:          prevBlock = thisBlock;
150:       }
151:       int lastBlock = prevBlock;
152:       
153:       // If we're overwriting, free any remaining blocks
154:       NPOIFSStream toFree = new NPOIFSStream(blockStore, nextBlock);
155:       toFree.free(loopDetector);
156:       
157:       // Mark the end of the stream
158:       blockStore.setNextBlock(lastBlock, POIFSConstants.END_OF_CHAIN);
159:    }

165:    /**
166:     * Frees all blocks in the stream
167:     */
168:    public void free() throws IOException {
169:       ChainLoopDetector loopDetector = blockStore.getChainLoopDetector();
170:       free(loopDetector);
171:    }
172:    private void free(ChainLoopDetector loopDetector) {
173:       int nextBlock = startBlock;
174:       while(nextBlock != POIFSConstants.END_OF_CHAIN) {
175:          int thisBlock = nextBlock;
176:          loopDetector.claim(thisBlock);
177:          nextBlock = blockStore.getNextBlock(thisBlock);
178:          blockStore.setNextBlock(thisBlock, POIFSConstants.UNUSED_BLOCK);
179:       }
180:       this.startBlock = POIFSConstants.END_OF_CHAIN;
181:    }
182:    
183:    /**
184:     * Class that handles a streaming read of one stream
185:     */
186:    protected class StreamBlockByteBufferIterator implements Iterator<ByteBuffer> {
187:       private ChainLoopDetector loopDetector;
188:       private int nextBlock;
189:       
190:       protected StreamBlockByteBufferIterator(int firstBlock) {
191:          this.nextBlock = firstBlock;
192:          try {
193:             this.loopDetector = blockStore.getChainLoopDetector();
194:          } catch(IOException e) {
195:             throw new RuntimeException(e);
196:          }
197:       }
198: 
199:       public boolean hasNext() {
200:          if(nextBlock == POIFSConstants.END_OF_CHAIN) {
201:             return false;
202:          }
203:          return true;
204:       }
205: 
206:       public ByteBuffer next() {
207:          if(nextBlock == POIFSConstants.END_OF_CHAIN) {
208:             throw new IndexOutOfBoundsException("Can't read past the end of the stream");
209:          }
210:          
211:          try {
212:             loopDetector.claim(nextBlock);
213:             ByteBuffer data = blockStore.getBlockAt(nextBlock);
214:             nextBlock = blockStore.getNextBlock(nextBlock);
215:             return data;
216:          } catch(IOException e) {
217:             throw new RuntimeException(e);
218:          }
219:       }
220: 
221:       public void remove() {
222:          throw new UnsupportedOperationException();
223:       }
224:    }
