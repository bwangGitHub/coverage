127: 	protected int getDataSize(){
128: 		int result = 2 + 4;  // short and int
129:         result += StringUtil.getEncodedSize(field_4_name) - 1; //size is byte, not short 
130: 
131:         if(!isOLELink() && !isStdDocumentNameIdentifier()){
132:             if(isAutomaticLink()){
133:                 result += 3; // byte, short
134:                 result += ConstantValueParser.getEncodedSize(_ddeValues);
135:             } else {
136:                 result += field_5_name_definition.getEncodedSize();
137:             }
138:         }
139: 		return result;
140: 	}

142: 	public void serialize(LittleEndianOutput out) {
143: 		out.writeShort(field_1_option_flag);
144: 		out.writeShort(field_2_ixals);
145: 		out.writeShort(field_3_not_used);
146: 
147: 		out.writeByte(field_4_name.length());
148: 		StringUtil.writeUnicodeStringFlagAndData(out, field_4_name);
149: 
150:         if(!isOLELink() && !isStdDocumentNameIdentifier()){
151:             if(isAutomaticLink()){
152:                 out.writeByte(_nColumns-1);
153:                 out.writeShort(_nRows-1);
154:                 ConstantValueParser.encode(out, _ddeValues);
155:             } else {
156:                 field_5_name_definition.serialize(out);
157:             }
158:         }
159: 	}

161:     public ExternalNameRecord() {
162:         field_2_ixals = 0;
163:     }
164:     
165:     public ExternalNameRecord(RecordInputStream in) {
166: 		field_1_option_flag = in.readShort();
167: 		field_2_ixals       = in.readShort();
168:       field_3_not_used    = in.readShort();
169: 
170:         int numChars = in.readUByte();
171:         field_4_name = StringUtil.readUnicodeString(in, numChars);
172: 
173:         // the record body can take different forms.
174:         // The form is dictated by the values of 3-th and 4-th bits in field_1_option_flag
175:         if(!isOLELink() && !isStdDocumentNameIdentifier()){
176:             // another switch: the fWantAdvise bit specifies whether the body describes
177:             // an external defined name or a DDE data item
178:             if(isAutomaticLink()){
179:                 if(in.available() > 0) {
180:                    //body specifies DDE data item
181:                    int nColumns = in.readUByte() + 1;
182:                    int nRows = in.readShort() + 1;
183:    
184:                    int totalCount = nRows * nColumns;
185:                    _ddeValues = ConstantValueParser.parse(in, totalCount);
186:                    _nColumns = nColumns;
187:                    _nRows = nRows;
188:                 }
189:             } else {
190:                 //body specifies an external defined name
191:                 int formulaLen = in.readUShort();
192:                 field_5_name_definition = Formula.read(formulaLen, in);
193:             }
194:         }
195:     }

202: 		StringBuffer sb = new StringBuffer();
203: 		sb.append("[EXTERNALNAME]\n");
204:         sb.append("    .options      = ").append(field_1_option_flag).append("\n");
205: 		sb.append("    .ix      = ").append(field_2_ixals).append("\n");
206: 		sb.append("    .name    = ").append(field_4_name).append("\n");
207: 		if(field_5_name_definition != null) {
208:             Ptg[] ptgs = field_5_name_definition.getTokens();
209:             for (int i = 0; i < ptgs.length; i++) {

212:             }
213: 		}
214: 		sb.append("[/EXTERNALNAME]\n");
215: 		return sb.toString();
