 48: 			ValueEval arg2) {

 51: 			TwoDEval aeLookupVector = LookupUtils.resolveTableArrayArg(arg1);

 55: 			ValueVector resultVector = createVector(aeResultVector);
 56: 			if(lookupVector.getSize() > resultVector.getSize()) {

 59: 			}
