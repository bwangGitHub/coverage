 28: /**
 29:  * Implementation for Excel function INDIRECT<p/>
 30:  *
 31:  * INDIRECT() returns the cell or area reference denoted by the text argument.<p/>
 32:  *
 33:  * <b>Syntax</b>:</br>
 34:  * <b>INDIRECT</b>(<b>ref_text</b>,isA1Style)<p/>
 35:  *
 36:  * <b>ref_text</b> a string representation of the desired reference as it would
 37:  * normally be written in a cell formula.<br/>
 38:  * <b>isA1Style</b> (default TRUE) specifies whether the ref_text should be
 39:  * interpreted as A1-style or R1C1-style.
 40:  *
 41:  * @author Josh Micich
 42:  */

 51: 	public ValueEval evaluate(ValueEval[] args, OperationEvaluationContext ec) {
 52: 		if (args.length < 1) {
 53: 			return ErrorEval.VALUE_INVALID;
 54: 		}
 55: 
 56: 		boolean isA1style;
 57: 		String text;
 58: 		try {
 59: 			ValueEval ve = OperandResolver.getSingleValue(args[0], ec.getRowIndex(), ec
 60: 					.getColumnIndex());
 61: 			text = OperandResolver.coerceValueToString(ve);
 62: 			switch (args.length) {
 63: 				case 1:
 64: 					isA1style = true;
 65: 					break;
 66: 				case 2:
 67: 					isA1style = evaluateBooleanArg(args[1], ec);
 68: 					break;
 69: 				default:
 70: 					return ErrorEval.VALUE_INVALID;
 71: 			}
 72: 		} catch (EvaluationException e) {
 73: 			return e.getErrorEval();
 74: 		}
 75: 
 76: 		return evaluateIndirect(ec, text, isA1style);
 77: 	}

 79: 	private static boolean evaluateBooleanArg(ValueEval arg, OperationEvaluationContext ec)
 80: 			throws EvaluationException {
 81: 		ValueEval ve = OperandResolver.getSingleValue(arg, ec.getRowIndex(), ec.getColumnIndex());
 82: 
 83: 		if (ve == BlankEval.instance || ve == MissingArgEval.instance) {
 84: 			return false;
 85: 		}
 86: 		// numeric quantities follow standard boolean conversion rules
 87: 		// for strings, only "TRUE" and "FALSE" (case insensitive) are valid
 88: 		return OperandResolver.coerceValueToBoolean(ve, false).booleanValue();
 89: 	}

 91: 	private static ValueEval evaluateIndirect(OperationEvaluationContext ec, String text,
 92: 			boolean isA1style) {
 93: 		// Search backwards for '!' because sheet names can contain '!'
 94: 		int plingPos = text.lastIndexOf('!');
 95: 
 96: 		String workbookName;
 97: 		String sheetName;
 98: 		String refText; // whitespace around this gets trimmed OK
 99: 		if (plingPos < 0) {
100: 			workbookName = null;
101: 			sheetName = null;
102: 			refText = text;
103: 		} else {
104: 			String[] parts = parseWorkbookAndSheetName(text.subSequence(0, plingPos));
105: 			if (parts == null) {
106: 				return ErrorEval.REF_INVALID;
107: 			}
108: 			workbookName = parts[0];
109: 			sheetName = parts[1];
110: 			refText = text.substring(plingPos + 1);
111: 		}
112: 
113: 		String refStrPart1;
114: 		String refStrPart2;
115: 
116: 		int colonPos = refText.indexOf(':');
117: 		if (colonPos < 0) {
118: 			refStrPart1 = refText.trim();
119: 			refStrPart2 = null;
120: 		} else {
121: 			refStrPart1 = refText.substring(0, colonPos).trim();
122: 			refStrPart2 = refText.substring(colonPos + 1).trim();
123: 		}
124: 		return ec.getDynamicReference(workbookName, sheetName, refStrPart1, refStrPart2, isA1style);
125: 	}

127: 	/**
128: 	 * @return array of length 2: {workbookName, sheetName,}.  Second element will always be
129: 	 * present.  First element may be null if sheetName is unqualified.
130: 	 * Returns <code>null</code> if text cannot be parsed.
131: 	 */
132: 	private static String[] parseWorkbookAndSheetName(CharSequence text) {
133: 		int lastIx = text.length() - 1;
134: 		if (lastIx < 0) {
135: 			return null;
136: 		}
137: 		if (canTrim(text)) {
138: 			return null;
139: 		}
140: 		char firstChar = text.charAt(0);
141: 		if (Character.isWhitespace(firstChar)) {
142: 			return null;
143: 		}
144: 		if (firstChar == '\'') {
145: 			// workbookName or sheetName needs quoting
146: 			// quotes go around both
147: 			if (text.charAt(lastIx) != '\'') {
148: 				return null;
149: 			}
150: 			firstChar = text.charAt(1);
151: 			if (Character.isWhitespace(firstChar)) {
152: 				return null;
153: 			}
154: 			String wbName;
155: 			int sheetStartPos;
156: 			if (firstChar == '[') {
157: 				int rbPos = text.toString().lastIndexOf(']');
158: 				if (rbPos < 0) {
159: 					return null;
160: 				}
161: 				wbName = unescapeString(text.subSequence(2, rbPos));
162: 				if (wbName == null || canTrim(wbName)) {
163: 					return null;
164: 				}
165: 				sheetStartPos = rbPos + 1;
166: 			} else {
167: 				wbName = null;
168: 				sheetStartPos = 1;
169: 			}
170: 
171: 			// else - just sheet name
172: 			String sheetName = unescapeString(text.subSequence(sheetStartPos, lastIx));
173: 			if (sheetName == null) { // note - when quoted, sheetName can
174: 									 // start/end with whitespace
175: 				return null;
176: 			}
177: 			return new String[] { wbName, sheetName, };
178: 		}
179: 
180: 		if (firstChar == '[') {
181: 			int rbPos = text.toString().lastIndexOf(']');
182: 			if (rbPos < 0) {
183: 				return null;
184: 			}
185: 			CharSequence wbName = text.subSequence(1, rbPos);
186: 			if (canTrim(wbName)) {
187: 				return null;
188: 			}
189: 			CharSequence sheetName = text.subSequence(rbPos + 1, text.length());
190: 			if (canTrim(sheetName)) {
191: 				return null;
192: 			}
193: 			return new String[] { wbName.toString(), sheetName.toString(), };
194: 		}
195: 		// else - just sheet name
196: 		return new String[] { null, text.toString(), };
197: 	}

199: 	/**
200: 	 * @return <code>null</code> if there is a syntax error in any escape sequence
201: 	 * (the typical syntax error is a single quote character not followed by another).
202: 	 */
203: 	private static String unescapeString(CharSequence text) {
204: 		int len = text.length();
205: 		StringBuilder sb = new StringBuilder(len);
206: 		int i = 0;
207: 		while (i < len) {
208: 			char ch = text.charAt(i);
209: 			if (ch == '\'') {
210: 				// every quote must be followed by another
211: 				i++;
212: 				if (i >= len) {
213: 					return null;
214: 				}
215: 				ch = text.charAt(i);
216: 				if (ch != '\'') {
217: 					return null;
218: 				}
219: 			}
220: 			sb.append(ch);
221: 			i++;
222: 		}
223: 		return sb.toString();
224: 	}
225: 
226: 	private static boolean canTrim(CharSequence text) {
227: 		int lastIx = text.length() - 1;
228: 		if (lastIx < 0) {
229: 			return false;
230: 		}
231: 		if (Character.isWhitespace(text.charAt(0))) {
232: 			return true;
233: 		}
234: 		if (Character.isWhitespace(text.charAt(lastIx))) {
235: 			return true;
236: 		}
237: 		return false;
238: 	}
