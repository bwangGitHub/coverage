 33: public class POILogFactory

 35: 
 36:     /**
 37:      * Map of POILogger instances, with classes as keys
 38:      */
 39:     private static Map<String,POILogger> _loggers = new HashMap<String,POILogger>();;
 40: 
 41:     /**
 42:      * A common instance of NullLogger, as it does nothing
 43:      *  we only need the one
 44:      */
 45:     private static POILogger _nullLogger = new NullLogger();
 46:     /**
 47:      * The name of the class to use. Initialised the
 48:      *  first time we need it
 49:      */
 50:     private static String _loggerClassName = null;
 51: 
 52:     /**
 53:      * Construct a POILogFactory.
 54:      */
 55:     private POILogFactory()
 56:     {
 57:     }
 58: 
 59:     /**
 60:      * Get a logger, based on a class name
 61:      *
 62:      * @param theclass the class whose name defines the log
 63:      *
 64:      * @return a POILogger for the specified class
 65:      */
 66: 
 67:     public static POILogger getLogger(final Class theclass)
 68:     {
 69:         return getLogger(theclass.getName());
 70:     }
 71:     
 72:     /**
 73:      * Get a logger, based on a String
 74:      *
 75:      * @param cat the String that defines the log
 76:      *
 77:      * @return a POILogger for the specified class
 78:      */
 79: 
 80:     public static POILogger getLogger(final String cat)
 81:     {
 82:         POILogger logger = null;
 83:         
 84:         // If we haven't found out what logger to use yet,
 85:         //  then do so now
 86:         // Don't look it up until we're first asked, so
 87:         //  that our users can set the system property
 88:         //  between class loading and first use
 89:         if(_loggerClassName == null) {
 90:         	try {
 91:         		_loggerClassName = System.getProperty("org.apache.poi.util.POILogger");
 92:         	} catch(Exception e) {}
 93:         	
 94:         	// Use the default logger if none specified,
 95:         	//  or none could be fetched
 96:         	if(_loggerClassName == null) {
 97:         		_loggerClassName = _nullLogger.getClass().getName();
 98:         	}
 99:         }
100:         
101:         // Short circuit for the null logger, which
102:         //  ignores all categories
103:         if(_loggerClassName.equals(_nullLogger.getClass().getName())) {
104:         	return _nullLogger;
105:         }
106: 
107:         
108:         // Fetch the right logger for them, creating
109:         //  it if that's required 
110:         if (_loggers.containsKey(cat)) {
111:             logger = _loggers.get(cat);
112:         } else {
113:             try {
114:               Class<? extends POILogger> loggerClass = 
115:                  (Class<? extends POILogger>)Class.forName(_loggerClassName);
116:               logger = loggerClass.newInstance();
117:               logger.initialize(cat);
118:             } catch(Exception e) {
119:               // Give up and use the null logger
120:               logger = _nullLogger;
121:             }
122:             
123:             // Save for next time
124:             _loggers.put(cat, logger);
125:         }
126:         return logger;
127:     }
128: }   // end public class POILogFactory
