 38: 				ValueEval arg1) {
 39: 			double dn;
 40: 			try {
 41: 				ValueEval ve1 = OperandResolver.getSingleValue(arg1, srcRowIndex, srcColumnIndex);
 42: 				dn = OperandResolver.coerceValueToDouble(ve1);
 43: 			} catch (EvaluationException e1) {
 44: 				// all errors in the second arg translate to #VALUE!
 45: 				return ErrorEval.VALUE_INVALID;
 46: 			}
 47: 			// weird Excel behaviour on second arg
 48: 			if (dn < 1.0) {
 49: 				// values between 0.0 and 1.0 result in #NUM!
 50: 				return ErrorEval.NUM_ERROR;
 51: 			}
 52: 			// all other values are rounded up to the next integer
 53: 			int k = (int) Math.ceil(dn);
 54: 
 55: 			double result;
 56: 			try {
 57: 				double[] ds = ValueCollector.collectValues(arg0);
 58: 				if (k > ds.length) {
 59: 					return ErrorEval.NUM_ERROR;
 60: 				}
 61: 				result = _isLarge ? StatsLib.kthLargest(ds, k) : StatsLib.kthSmallest(ds, k);
 62: 				NumericFunction.checkValue(result);
 63: 			} catch (EvaluationException e) {
 64: 				return e.getErrorEval();
 65: 			}
 66: 
 67: 			return new NumberEval(result);
 68: 		}
