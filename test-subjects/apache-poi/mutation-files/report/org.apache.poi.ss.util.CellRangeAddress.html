 90:     public String formatAsString(String sheetName, boolean useAbsoluteAddress) {
 91:         StringBuffer sb = new StringBuffer();
 92:         if (sheetName != null) {
 93:             sb.append(SheetNameFormatter.format(sheetName));
 94:             sb.append("!");
 95:         }
 96:         CellReference cellRefFrom = new CellReference(getFirstRow(), getFirstColumn(),
 97:                 useAbsoluteAddress, useAbsoluteAddress);
 98:         CellReference cellRefTo = new CellReference(getLastRow(), getLastColumn(),
 99:                 useAbsoluteAddress, useAbsoluteAddress);
100:         sb.append(cellRefFrom.formatAsString());
101: 
102:         //for a single-cell reference return A1 instead of A1:A1
103:         if(!cellRefFrom.equals(cellRefTo)){
104:             sb.append(':');
105:             sb.append(cellRefTo.formatAsString());
106:         }
107:         return sb.toString();
108:     }

113:      */
114:     public static CellRangeAddress valueOf(String ref) {
115:         int sep = ref.indexOf(":");
116:         CellReference a;
117:         CellReference b;
118:         if (sep == -1) {
119:             a = new CellReference(ref);
120:             b = a;
121:         } else {
122:             a = new CellReference(ref.substring(0, sep));
123:             b = new CellReference(ref.substring(sep + 1));
124:         }

126:     }
