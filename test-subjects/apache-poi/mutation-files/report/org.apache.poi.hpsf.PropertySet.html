 29: /**
 30:  * <p>Represents a property set in the Horrible Property Set Format
 31:  * (HPSF). These are usually metadata of a Microsoft Office
 32:  * document.</p>
 33:  *
 34:  * <p>An application that wants to access these metadata should create
 35:  * an instance of this class or one of its subclasses by calling the
 36:  * factory method {@link PropertySetFactory#create} and then retrieve
 37:  * the information its needs by calling appropriate methods.</p>
 38:  *
 39:  * <p>{@link PropertySetFactory#create} does its work by calling one
 40:  * of the constructors {@link PropertySet#PropertySet(InputStream)} or
 41:  * {@link PropertySet#PropertySet(byte[])}. If the constructor's
 42:  * argument is not in the Horrible Property Set Format, i.e. not a
 43:  * property set stream, or if any other error occurs, an appropriate
 44:  * exception is thrown.</p>
 45:  *
 46:  * <p>A {@link PropertySet} has a list of {@link Section}s, and each
 47:  * {@link Section} has a {@link Property} array. Use {@link
 48:  * #getSections} to retrieve the {@link Section}s, then call {@link
 49:  * Section#getProperties} for each {@link Section} to get hold of the
 50:  * {@link Property} arrays.</p> Since the vast majority of {@link
 51:  * PropertySet}s contains only a single {@link Section}, the
 52:  * convenience method {@link #getProperties} returns the properties of
 53:  * a {@link PropertySet}'s {@link Section} (throwing a {@link
 54:  * NoSingleSectionException} if the {@link PropertySet} contains more
 55:  * (or less) than exactly one {@link Section}).</p>
 56:  *
 57:  * @author Rainer Klute <a
 58:  * href="mailto:klute@rainer-klute.de">&lt;klute@rainer-klute.de&gt;</a>

215:     /**
216:      * <p>Creates a {@link PropertySet} instance from an {@link
217:      * InputStream} in the Horrible Property Set Format.</p>
218:      *
219:      * <p>The constructor reads the first few bytes from the stream
220:      * and determines whether it is really a property set stream. If
221:      * it is, it parses the rest of the stream. If it is not, it
222:      * resets the stream to its beginning in order to let other
223:      * components mess around with the data and throws an
224:      * exception.</p>
225:      *
226:      * @param stream Holds the data making out the property set
227:      * stream.
228:      * @throws MarkUnsupportedException if the stream does not support
229:      * the {@link InputStream#markSupported} method.
230:      * @throws IOException if the {@link InputStream} cannot not be
231:      * accessed as needed.
232:      * @exception NoPropertySetStreamException if the input stream does not
233:      * contain a property set.
234:      * @exception UnsupportedEncodingException if a character encoding is not
235:      * supported.
236:      */
237:     public PropertySet(final InputStream stream)
238:         throws NoPropertySetStreamException, MarkUnsupportedException,
239:                IOException, UnsupportedEncodingException
240:     {
241:         if (isPropertySetStream(stream))
242:         {
243:             final int avail = stream.available();
244:             final byte[] buffer = new byte[avail];
245:             stream.read(buffer, 0, buffer.length);
246:             init(buffer, 0, buffer.length);
247:         }
248:         else
249:             throw new NoPropertySetStreamException();
250:     }

300:     /**
301:      * <p>Checks whether an {@link InputStream} is in the Horrible
302:      * Property Set Format.</p>
303:      *
304:      * @param stream The {@link InputStream} to check. In order to
305:      * perform the check, the method reads the first bytes from the
306:      * stream. After reading, the stream is reset to the position it
307:      * had before reading. The {@link InputStream} must support the
308:      * {@link InputStream#mark} method.
309:      * @return <code>true</code> if the stream is a property set
310:      * stream, else <code>false</code>.
311:      * @throws MarkUnsupportedException if the {@link InputStream}
312:      * does not support the {@link InputStream#mark} method.
313:      * @exception IOException if an I/O error occurs
314:      */
315:     public static boolean isPropertySetStream(final InputStream stream)
316:         throws MarkUnsupportedException, IOException
317:     {
318:         /*
319:          * Read at most this many bytes.
320:          */
321:         final int BUFFER_SIZE = 50;
322: 
323:         /*
324:          * Mark the current position in the stream so that we can
325:          * reset to this position if the stream does not contain a
326:          * property set.
327:          */
328:         if (!stream.markSupported())
329:             throw new MarkUnsupportedException(stream.getClass().getName());
330:         stream.mark(BUFFER_SIZE);
331: 
332:         /*
333:          * Read a couple of bytes from the stream.
334:          */
335:         final byte[] buffer = new byte[BUFFER_SIZE];
336:         final int bytes =
337:             stream.read(buffer, 0,
338:                         Math.min(buffer.length, stream.available()));
339:         final boolean isPropertySetStream =
340:             isPropertySetStream(buffer, 0, bytes);
341:         stream.reset();
342:         return isPropertySetStream;
343:     }

347:     /**
348:      * <p>Checks whether a byte array is in the Horrible Property Set
349:      * Format.</p>
350:      *
351:      * @param src The byte array to check.
352:      * @param offset The offset in the byte array.
353:      * @param length The significant number of bytes in the byte
354:      * array. Only this number of bytes will be checked.
355:      * @return <code>true</code> if the byte array is a property set
356:      * stream, <code>false</code> if not.
357:      */
358:     public static boolean isPropertySetStream(final byte[] src,
359:                                               final int offset,
360:                                               final int length)
361:     {
362:         /* FIXME (3): Ensure that at most "length" bytes are read. */
363: 
364:         /*
365:          * Read the header fields of the stream. They must always be
366:          * there.
367:          */
368:         int o = offset;
369:         final int byteOrder = LittleEndian.getUShort(src, o);
370:         o += LittleEndian.SHORT_SIZE;
371:         byte[] temp = new byte[LittleEndian.SHORT_SIZE];
372:         LittleEndian.putShort(temp, 0, (short) byteOrder);
373:         if (!Util.equal(temp, BYTE_ORDER_ASSERTION))
374:             return false;
375:         final int format = LittleEndian.getUShort(src, o);
376:         o += LittleEndian.SHORT_SIZE;
377:         temp = new byte[LittleEndian.SHORT_SIZE];
378:         LittleEndian.putShort(temp, 0, (short) format);
379:         if (!Util.equal(temp, FORMAT_ASSERTION))
380:             return false;
381:         // final long osVersion = LittleEndian.getUInt(src, offset);
382:         o += LittleEndian.INT_SIZE;
383:         // final ClassID classID = new ClassID(src, offset);
384:         o += ClassID.LENGTH;
385:         final long sectionCount = LittleEndian.getUInt(src, o);
386:         o += LittleEndian.INT_SIZE;
387:         if (sectionCount < 0)
388:             return false;
389:         return true;
390:     }

394:     /**
395:      * <p>Initializes this {@link PropertySet} instance from a byte
396:      * array. The method assumes that it has been checked already that
397:      * the byte array indeed represents a property set stream. It does
398:      * no more checks on its own.</p>
399:      *
400:      * @param src Byte array containing the property set stream
401:      * @param offset The property set stream starts at this offset
402:      * from the beginning of <var>src</var>
403:      * @param length Length of the property set stream.
404:      * @throws UnsupportedEncodingException if HPSF does not (yet) support the
405:      * property set's character encoding.
406:      */
407:     private void init(final byte[] src, final int offset, final int length)
408:     throws UnsupportedEncodingException
409:     {
410:         /* FIXME (3): Ensure that at most "length" bytes are read. */
411:         
412:         /*
413:          * Read the stream's header fields.
414:          */
415:         int o = offset;
416:         byteOrder = LittleEndian.getUShort(src, o);
417:         o += LittleEndian.SHORT_SIZE;
418:         format = LittleEndian.getUShort(src, o);
419:         o += LittleEndian.SHORT_SIZE;
420:         osVersion = (int) LittleEndian.getUInt(src, o);
421:         o += LittleEndian.INT_SIZE;
422:         classID = new ClassID(src, o);
423:         o += ClassID.LENGTH;
424:         final int sectionCount = LittleEndian.getInt(src, o);
425:         o += LittleEndian.INT_SIZE;
426:         if (sectionCount < 0)
427:             throw new HPSFRuntimeException("Section count " + sectionCount +
428:                                            " is negative.");
429: 
430:         /*
431:          * Read the sections, which are following the header. They
432:          * start with an array of section descriptions. Each one
433:          * consists of a format ID telling what the section contains
434:          * and an offset telling how many bytes from the start of the
435:          * stream the section begins.
436:          */
437:         /*
438:          * Most property sets have only one section. The Document
439:          * Summary Information stream has 2. Everything else is a rare
440:          * exception and is no longer fostered by Microsoft.
441:          */
442:         sections = new ArrayList<Section>( sectionCount );
443: 
444:         /*
445:          * Loop over the section descriptor array. Each descriptor
446:          * consists of a ClassID and a DWord, and we have to increment
447:          * "offset" accordingly.
448:          */
449:         for (int i = 0; i < sectionCount; i++)
450:         {
451:             final Section s = new Section(src, o);
452:             o += ClassID.LENGTH + LittleEndian.INT_SIZE;
453:             sections.add(s);
454:         }
455:     }

625:     /**
626:      * <p>Returns <code>true</code> if the <code>PropertySet</code> is equal
627:      * to the specified parameter, else <code>false</code>.</p>
628:      *
629:      * @param o the object to compare this <code>PropertySet</code> with
630:      * 
631:      * @return <code>true</code> if the objects are equal, <code>false</code>
632:      * if not
633:      */
634:     public boolean equals(final Object o)
635:     {
636:         if (o == null || !(o instanceof PropertySet))
637:             return false;
638:         final PropertySet ps = (PropertySet) o;
639:         int byteOrder1 = ps.getByteOrder();
640:         int byteOrder2 = getByteOrder();
641:         ClassID classID1 = ps.getClassID();
642:         ClassID classID2 = getClassID();
643:         int format1 = ps.getFormat();
644:         int format2 = getFormat();
645:         int osVersion1 = ps.getOSVersion();
646:         int osVersion2 = getOSVersion();
647:         int sectionCount1 = ps.getSectionCount();
648:         int sectionCount2 = getSectionCount();
649:         if (byteOrder1 != byteOrder2      ||
650:             !classID1.equals(classID2)    ||
651:             format1 != format2            ||
652:             osVersion1 != osVersion2      ||
653:             sectionCount1 != sectionCount2)
654:             return false;
655: 
656:         /* Compare the sections: */
657:         return Util.equals(getSections(), ps.getSections());
