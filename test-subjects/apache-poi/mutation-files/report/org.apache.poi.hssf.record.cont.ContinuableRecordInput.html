 26: /**
 27:  * A decorated {@link RecordInputStream} that can read primitive data types
 28:  * (short, int, long, etc.) spanned across a {@link ContinueRecord } boundary.
 29:  *
 30:  * <p>
 31:  *  Most records construct themselves from {@link RecordInputStream}.
 32:  *  This class assumes that a {@link ContinueRecord} record break always occurs at the type boundary,
 33:  *  however, it is not always so.
 34:  * </p>
 35:  *  Two  attachments to <a href="https://issues.apache.org/bugzilla/show_bug.cgi?id=50779">Bugzilla 50779</a>
 36:  *  demonstrate that a CONTINUE break can appear right in between two bytes of a unicode character
 37:  *  or between two bytes of a <code>short</code>. The problematic portion of the data is
 38:  *  in a Asian Phonetic Settings Block (ExtRst) of a UnicodeString.
 39:  * <p>
 40:  *  {@link RecordInputStream} greedily requests the bytes to be read and stumbles on such files with a
 41:  *  "Not enough data (1) to read requested (2) bytes" exception.  The <code>ContinuableRecordInput</code>
 42:  *   class circumvents this "type boundary" rule and reads data byte-by-byte rolling over CONTINUE if necessary.
 43:  * </p>
 44:  *
 45:  * <p>
 46:  * YK: For now (March 2011) this class is only used to read
 47:  *   @link org.apache.poi.hssf.record.common.UnicodeString.ExtRst} blocks of a UnicodeString.
 48:  *
 49:  * </p>
 50:  *
 51:  * @author Yegor Kozlov
 52:  */
 53: public class ContinuableRecordInput implements LittleEndianInput {
 54:     private final RecordInputStream _in;
 55: 
 56:     public ContinuableRecordInput(RecordInputStream in){
 57:         _in = in;
 58:     }
 59:     public int available(){
 60:         return _in.available();
 61:     }
 62: 
 63:     public byte readByte(){
 64:         return _in.readByte();
 65:     }
 66: 
 67:     public int readUByte(){
 68:         return _in.readUByte();
 69:     }
 70: 
 71:     public short readShort(){
 72:         return _in.readShort();
 73:     }
 74: 
 75:     public int readUShort(){
 76:         int ch1 = readUByte();
 77:         int ch2 = readUByte();
 78:         return (ch2 << 8) + (ch1 << 0);
 79:     }
 80: 
 81:     public int readInt(){
 82:         int ch1 = _in.readUByte();
 83:         int ch2 = _in.readUByte();
 84:         int ch3 = _in.readUByte();
 85:         int ch4 = _in.readUByte();
 86:         return (ch4 << 24) + (ch3 << 16) + (ch2 << 8) + (ch1 << 0);
 87:     }
 88: 
 89:     public long readLong(){
 90:         int b0 = _in.readUByte();
 91:         int b1 = _in.readUByte();
 92:         int b2 = _in.readUByte();
 93:         int b3 = _in.readUByte();
 94:         int b4 = _in.readUByte();
 95:         int b5 = _in.readUByte();
 96:         int b6 = _in.readUByte();
 97:         int b7 = _in.readUByte();
 98:         return (((long)b7 << 56) +
 99:                 ((long)b6 << 48) +
100:                 ((long)b5 << 40) +
101:                 ((long)b4 << 32) +
102:                 ((long)b3 << 24) +
103:                 (b2 << 16) +
104:                 (b1 <<  8) +
105:                 (b0 <<  0));
106:     }
107: 
108:     public double readDouble(){
109:         return _in.readDouble();
110:     }
111:     public void readFully(byte[] buf){
112:         _in.readFully(buf);
113:     }
114:     public void readFully(byte[] buf, int off, int len){
115:         _in.readFully(buf, off, len);
116:     }
