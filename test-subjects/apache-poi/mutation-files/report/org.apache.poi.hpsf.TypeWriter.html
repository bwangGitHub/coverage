 20: import java.io.IOException;

 23: import org.apache.poi.util.LittleEndian;

 25: /**
 26:  * <p>Class for writing little-endian data and more.</p>
 27:  *
 28:  * @author Rainer Klute <a
 29:  * href="mailto:klute@rainer-klute.de">&lt;klute@rainer-klute.de&gt;</a>
 30:  */
 31: public class TypeWriter
 32: {
 33: 
 34:     /**
 35:      * <p>Writes a two-byte value (short) to an output stream.</p>
 36:      *
 37:      * @param out The stream to write to.
 38:      * @param n The value to write.
 39:      * @return The number of bytes that have been written.
 40:      * @exception IOException if an I/O error occurs
 41:      */
 42:     public static int writeToStream( final OutputStream out, final short n )
 43:             throws IOException
 44:     {
 45:         LittleEndian.putShort( out, n ); // FIXME: unsigned
 46:         return LittleEndian.SHORT_SIZE;
 47:     }
 48: 
 49:     /**
 50:      * <p>Writes a four-byte value to an output stream.</p>
 51:      *
 52:      * @param out The stream to write to.
 53:      * @param n The value to write.
 54:      * @exception IOException if an I/O error occurs
 55:      * @return The number of bytes written to the output stream. 
 56:      */
 57:     public static int writeToStream( final OutputStream out, final int n )
 58:             throws IOException
 59:     {
 60:         LittleEndian.putInt( n, out );
 61:         return LittleEndian.INT_SIZE;
 62:     }
 63: 
 64:     /**
 65:      * <p>Writes a eight-byte value to an output stream.</p>
 66:      *
 67:      * @param out The stream to write to.
 68:      * @param n The value to write.
 69:      * @exception IOException if an I/O error occurs
 70:      * @return The number of bytes written to the output stream.
 71:      */
 72:     public static int writeToStream( final OutputStream out, final long n )
 73:             throws IOException
 74:     {
 75:         LittleEndian.putLong( n, out );
 76:         return LittleEndian.LONG_SIZE;
 77:     }
 78: 
 79:     /**
 80:      * <p>Writes an unsigned two-byte value to an output stream.</p>
 81:      *
 82:      * @param out The stream to write to
 83:      * @param n The value to write
 84:      * @exception IOException if an I/O error occurs
 85:      */
 86:     public static void writeUShortToStream( final OutputStream out, final int n )
 87:             throws IOException
 88:     {
 89:         int high = n & 0xFFFF0000;
 90:         if ( high != 0 )
 91:             throw new IllegalPropertySetDataException( "Value " + n
 92:                     + " cannot be represented by 2 bytes." );
 93:         LittleEndian.putUShort( n, out );
 94:     }
 95: 
 96:     /**
 97:      * <p>Writes an unsigned four-byte value to an output stream.</p>
 98:      *
 99:      * @param out The stream to write to.
100:      * @param n The value to write.
101:      * @return The number of bytes that have been written to the output stream.
102:      * @exception IOException if an I/O error occurs
103:      */
104:     public static int writeUIntToStream( final OutputStream out, final long n )
105:             throws IOException
106:     {
107:         long high = n & 0xFFFFFFFF00000000L;
108:         if ( high != 0 && high != 0xFFFFFFFF00000000L )
109:             throw new IllegalPropertySetDataException( "Value " + n
110:                     + " cannot be represented by 4 bytes." );
111:         LittleEndian.putUInt( n, out );
112:         return LittleEndian.INT_SIZE;
113:     }
114: 
115:     /**
116:      * <p>Writes a 16-byte {@link ClassID} to an output stream.</p>
117:      *
118:      * @param out The stream to write to
119:      * @param n The value to write
120:      * @return The number of bytes written
121:      * @exception IOException if an I/O error occurs
122:      */
123:     public static int writeToStream(final OutputStream out, final ClassID n)
124:         throws IOException
125:     {
126:         byte[] b = new byte[16];
127:         n.write(b, 0);
128:         out.write(b, 0, b.length);
129:         return b.length;
130:     }
131: 
132: 
133: 
134:     /**
135:      * <p>Writes an array of {@link Property} instances to an output stream
136:      * according to the Horrible Property Stream Format.</p>
137:      * 
138:      * @param out The stream to write to
139:      * @param properties The array to write to the stream
140:      * @param codepage The codepage number to use for writing strings
141:      * @exception IOException if an I/O error occurs
142:      * @throws UnsupportedVariantTypeException if HPSF does not support some
143:      *         variant type.
144:      */
145:     public static void writeToStream(final OutputStream out,
146:                                      final Property[] properties,
147:                                      final int codepage)
148:         throws IOException, UnsupportedVariantTypeException
149:     {
150:         /* If there are no properties don't write anything. */
151:         if (properties == null)
152:             return;
153: 
154:         /* Write the property list. This is a list containing pairs of property
155:          * ID and offset into the stream. */
156:         for (int i = 0; i < properties.length; i++)
157:         {
158:             final Property p = properties[i];
159:             writeUIntToStream(out, p.getID());
160:             writeUIntToStream(out, p.getSize());
161:         }
162: 
163:         /* Write the properties themselves. */
164:         for (int i = 0; i < properties.length; i++)
165:         {
166:             final Property p = properties[i];
167:             long type = p.getType();
168:             writeUIntToStream(out, type);
169:             VariantSupport.write(out, (int) type, p.getValue(), codepage);
170:         }
171:     }
172: 
173: 
174: 
175:     /**
176:      * <p>Writes a double value value to an output stream.</p>
177:      *
178:      * @param out The stream to write to.
179:      * @param n The value to write.
180:      * @exception IOException if an I/O error occurs
181:      * @return The number of bytes written to the output stream. 
182:      */
183:     public static int writeToStream( final OutputStream out, final double n )
184:             throws IOException
185:     {
186:         LittleEndian.putDouble( n, out );
187:         return LittleEndian.DOUBLE_SIZE;
188:     }
