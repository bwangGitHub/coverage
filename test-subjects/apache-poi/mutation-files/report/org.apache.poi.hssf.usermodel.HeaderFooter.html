138: 	/**
139: 	 * Creates the complete footer string based on the left, center, and middle
140: 	 * strings.
141: 	 */
142: 	private void updateHeaderFooterText(String[] parts) {
143: 		String _left = parts[0];
144: 		String _center = parts[1];
145: 		String _right = parts[2];
146: 		
147: 		if (_center.length() < 1 && _left.length() < 1 && _right.length() < 1) {
148: 			setHeaderFooterText("");
149: 			return;
150: 		}
151: 		StringBuilder sb = new StringBuilder(64);
152: 		sb.append("&C");
153: 		sb.append(_center);
154: 		sb.append("&L");
155: 		sb.append(_left);
156: 		sb.append("&R");
157: 		sb.append(_right);
158: 		String text = sb.toString();
159: 		setHeaderFooterText(text);
160: 	}

275: 	 * Normally used to make some text suitable for showing to humans, and the
276: 	 * resultant text should not normally be saved back into the document!
277: 	 */
278: 	public static String stripFields(String pText) {
279: 		int pos;
280: 
281: 		// Check we really got something to work on
282: 		if (pText == null || pText.length() == 0) {
283: 			return pText;
284: 		}
285: 
286: 		String text = pText;
287: 
288: 		// Firstly, do the easy ones which are static
289: 		for (MarkupTag mt : MarkupTag.values()) {
290: 			String seq = mt.getRepresentation();
291: 			while ((pos = text.indexOf(seq)) > -1) {
292: 				text = text.substring(0, pos) + text.substring(pos + seq.length());
293: 			}
294: 		}
295: 
296: 		// Now do the tricky, dynamic ones
297: 		// These are things like font sizes and font names
298: 		text = text.replaceAll("\\&\\d+", "");
299: 		text = text.replaceAll("\\&\".*?,.*?\"", "");
300: 
301: 		// All done
302: 		return text;
303: 	}
