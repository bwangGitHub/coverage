 44:      *  <var>i</var>&nbsp;&lt;&nbsp;<var>a.length</var> holds
 45:      *  <var>a</var>[<var>i</var>]&nbsp;== <var>b</var>[<var>i</var>].</p></li>

 47:      * </ul>
 48:      *

 54:     public static boolean equal(final byte[] a, final byte[] b)
 55:     {
 56:         if (a.length != b.length)

 61:         return true;

 94:     {
 95:         int capacity = 0;
 96:         for (int i = 0; i < byteArrays.length; i++)
 97:             capacity += byteArrays[i].length;
 98:         final byte[] result = new byte[capacity];
 99:         int r = 0;

138:     /**
139:      * <p>Converts a Windows FILETIME into a {@link Date}. The Windows
140:      * FILETIME structure holds a date and time associated with a
141:      * file. The structure identifies a 64-bit integer specifying the
142:      * number of 100-nanosecond intervals which have passed since
143:      * January 1, 1601. This 64-bit value is split into the two double
144:      * words stored in the structure.</p>
145:      *
146:      * @param high The higher double word of the FILETIME structure.
147:      * @param low The lower double word of the FILETIME structure.
148:      * @return The Windows FILETIME as a {@link Date}.
149:      */
150:     public static Date filetimeToDate(final int high, final int low)
151:     {
152:         final long filetime = ((long) high) << 32 | (low & 0xffffffffL);
153:         return filetimeToDate(filetime);

157:      * <p>Converts a Windows FILETIME into a {@link Date}. The Windows

160:      * number of 100-nanosecond intervals which have passed since
161:      * January 1, 1601.</p>
162:      *
163:      * @param filetime The filetime to convert.
164:      * @return The Windows FILETIME as a {@link Date}.
165:      */
166:     public static Date filetimeToDate(final long filetime)
167:     {

226:      *
227:      * @param c1 The first object array.
228:      * @param c2 The second object array.

230:      * <code>false</code> if they are not.
231:      */
232:     public static boolean equals(Object[] c1, Object[] c2)
233:     {
234:         final Object[] o1 = c1.clone();
235:         final Object[] o2 = c2.clone();
236:         return internalEquals(o1, o2);
237:     }

239:     private static boolean internalEquals(Object[] o1, Object[] o2)
240:     {
241:         for (int i1 = 0; i1 < o1.length; i1++)
242:         {
243:             final Object obj1 = o1[i1];
244:             boolean matchFound = false;
245:             for (int i2 = 0; !matchFound && i2 < o1.length; i2++)
246:             {
247:                 final Object obj2 = o2[i2];
248:                 if (obj1.equals(obj2))
249:                 {
250:                     matchFound = true;
251:                     o2[i2] = null;
252:                 }
253:             }
254:             if (!matchFound)
255:                 return false;
256:         }
257:         return true;
258:     }

264:      * 4.</p>
265:      *
266:      * @param ba The byte array to pad.
267:      * @return The padded byte array.
268:      */
269:     public static byte[] pad4(final byte[] ba)
270:     {
271:         final int PAD = 4;
272:         final byte[] result;
273:         int l = ba.length % PAD;
274:         if (l == 0)
275:             result = ba;
276:         else
277:         {
278:             l = PAD - l;
279:             result = new byte[ba.length + l];
280:             System.arraycopy(ba, 0, result, 0, ba.length);
