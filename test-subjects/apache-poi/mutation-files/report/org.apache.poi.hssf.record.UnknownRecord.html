 76: 	 * construct an unknown record.  No fields are interpreted and the record will
 77: 	 * be serialized in its original form more or less
 78: 	 * @param in the RecordInputstream to read the record from
 79: 	 */
 80: 	public UnknownRecord(RecordInputStream in) {
 81: 		_sid = in.getSid();
 82: 		_rawData = in.readRemainder();
 83: 		if (false && getBiffName(_sid) == null) {
 84: 			// unknown sids in the range 0x0004-0x0013 are probably 'sub-records' of ObjectRecord
 85: 			// those sids are in a different number space.
 86: 			// TODO - put unknown OBJ sub-records in a different class
 87: 			System.out.println("Unknown record 0x" + Integer.toHexString(_sid).toUpperCase());
 88: 		}
 89: 	}

107: 		if (biffName == null) {

109: 		}
110: 		StringBuffer sb = new StringBuffer();

113: 		sb.append(Integer.toHexString(_sid).toUpperCase() + ")\n");
114: 		if (_rawData.length > 0) {
115: 			sb.append("  rawData=").append(HexDump.toHex(_rawData)).append("\n");
116: 		}
117: 		sb.append("[/").append(biffName).append("]\n");
118: 		return sb.toString();
