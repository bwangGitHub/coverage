 25: /**
 26:  * Event-based record factory.  As opposed to RecordFactory
 27:  * this version sends {@link ERFListener#processRecord(Record) } messages to
 28:  * the supplied listener.  Record notifications are sent one record behind
 29:  * to ensure that {@link ContinueRecord}s are processed first.
 30:  *
 31:  * @author Andrew C. Oliver (acoliver@apache.org) - probably to blame for the bugs (so yank his chain on the list)
 32:  * @author Marc Johnson (mjohnson at apache dot org) - methods taken from RecordFactory
 33:  * @author Glen Stampoultzis (glens at apache.org) - methods taken from RecordFactory
 34:  * @author Csaba Nagy (ncsaba at yahoo dot com)
 35:  */
 36: public final class EventRecordFactory {
 37: 
 38: 	private final ERFListener _listener;
 39: 	private final short[] _sids;
 40: 
 41: 	/**
 42: 	 *
 43: 	 * @param sids an array of Record.sid values identifying the records
 44: 	 * the listener will work with.  Alternatively if this is "null" then
 45: 	 * all records are passed. For all 'known' record types use {@link RecordFactory#getAllKnownRecordSIDs()}
 46: 	 */
 47: 	public EventRecordFactory(ERFListener listener, short[] sids) {
 48: 		_listener = listener;
 49: 		if (sids == null) {
 50: 			_sids = null;
 51: 		} else {
 52: 			_sids = sids.clone();
 53: 			Arrays.sort(_sids); // for faster binary search
 54: 		}
 55: 	}
 56: 	private boolean isSidIncluded(short sid) {
 57: 		if (_sids == null) {
 58: 			return true;
 59: 		}
 60: 		return Arrays.binarySearch(_sids, sid) >= 0;
 61: 	}
 62: 
 63: 
 64: 	/**
 65: 	 * sends the record event to all registered listeners.
 66: 	 * @param record the record to be thrown.
 67: 	 * @return <code>false</code> to abort.  This aborts
 68: 	 * out of the event loop should the listener return false
 69: 	 */
 70: 	private boolean processRecord(Record record) {
 71: 		if (!isSidIncluded(record.getSid())) {
 72: 			return true;
 73: 		}
 74: 		return _listener.processRecord(record);
 75: 	}
 76: 
 77: 	/**
 78: 	 * Create an array of records from an input stream
 79: 	 *
 80: 	 * @param in the InputStream from which the records will be
 81: 	 *		   obtained
 82: 	 *
 83: 	 * @exception RecordFormatException on error processing the
 84: 	 *			InputStream
 85: 	 */
 86: 	public void processRecords(InputStream in) throws RecordFormatException {
 87: 		Record last_record = null;
 88: 
 89: 		RecordInputStream recStream = new RecordInputStream(in);
 90: 
 91: 		while (recStream.hasNextRecord()) {
 92: 			recStream.nextRecord();
 93: 			Record[] recs = RecordFactory.createRecord(recStream);   // handle MulRK records
 94: 			if (recs.length > 1) {
 95: 				for (int k = 0; k < recs.length; k++) {
 96: 					if ( last_record != null ) {
 97: 						if (!processRecord(last_record)) {
 98: 							return;
 99: 						}
100: 					}
101: 					last_record = recs[ k ]; // do to keep the algorithm homogeneous...you can't
102: 				}							// actually continue a number record anyhow.
103: 			} else {
104: 				Record record = recs[ 0 ];
105: 
106: 				if (record != null) {
107: 					if (last_record != null) {
108: 						if (!processRecord(last_record)) {
109: 							return;
110: 						}
111: 					}
112: 					 last_record = record;
113: 				}
114: 			}
115: 		}
116: 
117: 		if (last_record != null) {
118: 			processRecord(last_record);
119: 		}
120: 	}
121: }
