 84: 	 * Create an cell ref from a string representation.  Sheet names containing special characters should be
 85: 	 * delimited and escaped as per normal syntax rules for formulas.
 86: 	 */
 87: 	public CellReference(String cellRef) {
 88:       if(cellRef.endsWith("#REF!")) {
 89:          throw new IllegalArgumentException("Cell reference invalid: " + cellRef);
 90:       }
 91: 
 92: 		String[] parts = separateRefParts(cellRef);
 93: 		_sheetName = parts[0];
 94: 		String colRef = parts[1];
 95: 		if (colRef.length() < 1) {
 96: 			throw new IllegalArgumentException("Invalid Formula cell reference: '"+cellRef+"'");
 97: 		}
 98: 		_isColAbs = colRef.charAt(0) == '$';
 99: 		if (_isColAbs) {
100: 			colRef=colRef.substring(1);
101: 		}
102: 		_colIndex = convertColStringToIndex(colRef);
103: 
104: 		String rowRef=parts[2];
105: 		if (rowRef.length() < 1) {
106: 			throw new IllegalArgumentException("Invalid Formula cell reference: '"+cellRef+"'");
107: 		}

109: 		if (_isRowAbs) {
110: 			rowRef=rowRef.substring(1);
111: 		}
112: 		_rowIndex = Integer.parseInt(rowRef)-1; // -1 to convert 1-based to zero-based
113: 	}

160: 	/**
161: 	 * takes in a column reference portion of a CellRef and converts it from
162: 	 * ALPHA-26 number format to 0-based base 10.
163: 	 * 'A' -> 0
164: 	 * 'Z' -> 25
165: 	 * 'AA' -> 26
166: 	 * 'IV' -> 255
167: 	 * @return zero based column index
168: 	 */
169: 	public static int convertColStringToIndex(String ref) {
170: 
171: 		int pos = 0;
172: 		int retval=0;
173: 		for (int k = ref.length()-1; k >= 0; k--) {
174: 			char thechar = ref.charAt(k);
175: 			if (thechar == ABSOLUTE_REFERENCE_MARKER) {
176: 				if (k != 0) {
177: 					throw new IllegalArgumentException("Bad col ref format '" + ref + "'");
178: 				}
179: 				break;
180: 			}
181: 			// Character.getNumericValue() returns the values
182: 			//  10-35 for the letter A-Z
183: 			int shift = (int)Math.pow(26, pos);
184: 			retval += (Character.getNumericValue(thechar)-9) * shift;
185: 			pos++;
186: 		}
187: 		return retval-1;
188: 	}

191: 	 * Classifies an identifier as either a simple (2D) cell reference or a named range name
192: 	 * @return one of the values from <tt>NameType</tt>
193: 	 */
194: 	public static NameType classifyCellReference(String str, SpreadsheetVersion ssVersion) {
195: 		int len = str.length();
196: 		if (len < 1) {
197: 			throw new IllegalArgumentException("Empty string not allowed");
198: 		}
199: 		char firstChar = str.charAt(0);
200: 		switch (firstChar) {
201: 			case ABSOLUTE_REFERENCE_MARKER:
202: 			case '.':
203: 			case '_':
204: 				break;
205: 			default:
206: 				if (!Character.isLetter(firstChar) && !Character.isDigit(firstChar)) {
207: 					throw new IllegalArgumentException("Invalid first char (" + firstChar
208: 							+ ") of cell reference or named range.  Letter expected");
209: 				}
210: 		}
211: 		if (!Character.isDigit(str.charAt(len-1))) {
212: 			// no digits at end of str
213: 			return validateNamedRangeName(str, ssVersion);
214: 		}
215: 		Matcher cellRefPatternMatcher = CELL_REF_PATTERN.matcher(str);
216: 		if (!cellRefPatternMatcher.matches()) {
217: 			return validateNamedRangeName(str, ssVersion);
218: 		}
219: 		String lettersGroup = cellRefPatternMatcher.group(1);
220: 		String digitsGroup = cellRefPatternMatcher.group(2);
221: 		if (cellReferenceIsWithinRange(lettersGroup, digitsGroup, ssVersion)) {
222: 			// valid cell reference
223: 			return NameType.CELL;
224: 		}
225: 		// If str looks like a cell reference, but is out of (row/col) range, it is a valid
226: 		// named range name
227: 		// This behaviour is a little weird.  For example, "IW123" is a valid named range name
228: 		// because the column "IW" is beyond the maximum "IV".  Note - this behaviour is version
229: 		// dependent.  In BIFF12, "IW123" is not a valid named range name, but in BIFF8 it is.
230: 		if (str.indexOf(ABSOLUTE_REFERENCE_MARKER) >= 0) {
231: 			// Of course, named range names cannot have '$'
232: 			return NameType.BAD_CELL_OR_NAMED_RANGE;
233: 		}
234: 		return NameType.NAMED_RANGE;
235: 	}

237: 	private static NameType validateNamedRangeName(String str, SpreadsheetVersion ssVersion) {
238: 		Matcher colMatcher = COLUMN_REF_PATTERN.matcher(str);
239: 		if (colMatcher.matches()) {
240: 			String colStr = colMatcher.group(1);
241: 			if (isColumnWithnRange(colStr, ssVersion)) {
242: 				return NameType.COLUMN;
243: 			}
244: 		}
245: 		Matcher rowMatcher = ROW_REF_PATTERN.matcher(str);
246: 		if (rowMatcher.matches()) {
247: 			String rowStr = rowMatcher.group(1);
248: 			if (isRowWithnRange(rowStr, ssVersion)) {
249: 				return NameType.ROW;
250: 			}
251: 		}
252: 		if (!NAMED_RANGE_NAME_PATTERN.matcher(str).matches()) {
253: 			return NameType.BAD_CELL_OR_NAMED_RANGE;
254: 		}
255: 		return NameType.NAMED_RANGE;

260: 	 * Used to decide whether a name of the form "[A-Z]*[0-9]*" that appears in a formula can be
261: 	 * interpreted as a cell reference.  Names of that form can be also used for sheets and/or
262: 	 * named ranges, and in those circumstances, the question of whether the potential cell
263: 	 * reference is valid (in range) becomes important.
264: 	 * <p/>
265: 	 * Note - that the maximum sheet size varies across Excel versions:
266: 	 * <p/>
267: 	 * <blockquote><table border="0" cellpadding="1" cellspacing="0"
268: 	 *                 summary="Notable cases.">
269: 	 *   <tr><th>Version&nbsp;&nbsp;</th><th>File Format&nbsp;&nbsp;</th>
270: 	 *   	<th>Last Column&nbsp;&nbsp;</th><th>Last Row</th></tr>
271: 	 *   <tr><td>97-2003</td><td>BIFF8</td><td>"IV" (2^8)</td><td>65536 (2^14)</td></tr>
272: 	 *   <tr><td>2007</td><td>BIFF12</td><td>"XFD" (2^14)</td><td>1048576 (2^20)</td></tr>
273: 	 * </table></blockquote>
274: 	 * POI currently targets BIFF8 (Excel 97-2003), so the following behaviour can be observed for
275: 	 * this method:
276: 	 * <blockquote><table border="0" cellpadding="1" cellspacing="0"
277: 	 *                 summary="Notable cases.">
278: 	 *   <tr><th>Input&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;</th>
279: 	 *       <th>Result&nbsp;</th></tr>
280: 	 *   <tr><td>"A", "1"</td><td>true</td></tr>
281: 	 *   <tr><td>"a", "111"</td><td>true</td></tr>
282: 	 *   <tr><td>"A", "65536"</td><td>true</td></tr>
283: 	 *   <tr><td>"A", "65537"</td><td>false</td></tr>
284: 	 *   <tr><td>"iv", "1"</td><td>true</td></tr>
285: 	 *   <tr><td>"IW", "1"</td><td>false</td></tr>
286: 	 *   <tr><td>"AAA", "1"</td><td>false</td></tr>
287: 	 *   <tr><td>"a", "111"</td><td>true</td></tr>
288: 	 *   <tr><td>"Sheet", "1"</td><td>false</td></tr>
289: 	 * </table></blockquote>
290: 	 *
291: 	 * @param colStr a string of only letter characters
292: 	 * @param rowStr a string of only digit characters
293: 	 * @return <code>true</code> if the row and col parameters are within range of a BIFF8 spreadsheet.
294: 	 */

302: 	public static boolean isColumnWithnRange(String colStr, SpreadsheetVersion ssVersion) {
303: 		String lastCol = ssVersion.getLastColumnName();
304: 		int lastColLength = lastCol.length();
305: 
306: 		int numberOfLetters = colStr.length();
307: 		if(numberOfLetters > lastColLength) {
308: 			// "Sheet1" case etc
309: 			return false; // that was easy
310: 		}
311: 		if(numberOfLetters == lastColLength) {
312: 			if(colStr.toUpperCase().compareTo(lastCol) > 0) {
313: 				return false;
314: 			}
315: 		} else {
316: 			// apparent column name has less chars than max
317: 			// no need to check range
318: 		}
319: 		return true;
320: 	}

323: 		int rowNum = Integer.parseInt(rowStr);

337: 	 * Separates the row from the columns and returns an array of three Strings.  The first element
338: 	 * is the sheet name. Only the first element may be null.  The second element in is the column
339: 	 * name still in ALPHA-26 number format.  The third element is the row.
340: 	 */
341: 	private static String[] separateRefParts(String reference) {
342: 		int plingPos = reference.lastIndexOf(SHEET_NAME_DELIMITER);
343: 		String sheetName = parseSheetName(reference, plingPos);
344: 		int start = plingPos+1;
345: 
346: 		int length = reference.length();
347: 
348: 
349: 		int loc = start;
350: 		// skip initial dollars
351: 		if (reference.charAt(loc)==ABSOLUTE_REFERENCE_MARKER) {
352: 			loc++;
353: 		}
354: 		// step over column name chars until first digit (or dollars) for row number.
355: 		for (; loc < length; loc++) {
356: 			char ch = reference.charAt(loc);
357: 			if (Character.isDigit(ch) || ch == ABSOLUTE_REFERENCE_MARKER) {
358: 				break;
359: 			}
360: 		}
361: 		return new String[] {
362: 			sheetName,
363: 			reference.substring(start,loc),
364: 			reference.substring(loc),
365: 		};
366: 	}

368: 	private static String parseSheetName(String reference, int indexOfSheetNameDelimiter) {
369: 		if(indexOfSheetNameDelimiter < 0) {
370: 			return null;
371: 		}
372: 
373: 		boolean isQuoted = reference.charAt(0) == SPECIAL_NAME_DELIMITER;
374: 		if(!isQuoted) {
375: 			return reference.substring(0, indexOfSheetNameDelimiter);
376: 		}
377: 		int lastQuotePos = indexOfSheetNameDelimiter-1;
378: 		if(reference.charAt(lastQuotePos) != SPECIAL_NAME_DELIMITER) {
379: 			throw new RuntimeException("Mismatched quotes: (" + reference + ")");
380: 		}
381: 
382: 		// TODO - refactor cell reference parsing logic to one place.
383: 		// Current known incarnations:
384: 		//   FormulaParser.GetName()
385: 		//   CellReference.parseSheetName() (here)
386: 		//   AreaReference.separateAreaRefs()
387: 		//   SheetNameFormatter.format() (inverse)
388: 
389: 		StringBuffer sb = new StringBuffer(indexOfSheetNameDelimiter);
390: 
391: 		for(int i=1; i<lastQuotePos; i++) { // Note boundaries - skip outer quotes
392: 			char ch = reference.charAt(i);
393: 			if(ch != SPECIAL_NAME_DELIMITER) {
394: 				sb.append(ch);
395: 				continue;
396: 			}
397: 			if(i < lastQuotePos) {
398: 				if(reference.charAt(i+1) == SPECIAL_NAME_DELIMITER) {
399: 					// two consecutive quotes is the escape sequence for a single one
400: 					i++; // skip this and keep parsing the special name
401: 					sb.append(ch);
402: 					continue;
403: 				}
404: 			}
405: 			throw new RuntimeException("Bad sheet name quote escaping: (" + reference + ")");
406: 		}
407: 		return sb.toString();
408: 	}

410: 	/**
411: 	 * Takes in a 0-based base-10 column and returns a ALPHA-26
412: 	 *  representation.
413: 	 * eg column #3 -> D
414: 	 */
415: 	public static String convertNumToColString(int col) {
416: 		// Excel counts column A as the 1st column, we
417: 		//  treat it as the 0th one
418: 		int excelColNum = col + 1;
419: 
420: 		String colRef = "";
421: 		int colRemain = excelColNum;
422: 
423: 		while(colRemain > 0) {
424: 			int thisPart = colRemain % 26;
425: 			if(thisPart == 0) { thisPart = 26; }
426: 			colRemain = (colRemain - thisPart) / 26;
427: 
428: 			// The letter A is at 65
429: 			char colChar = (char)(thisPart+64);
430: 			colRef = colChar + colRef;
431: 		}
432: 
433: 		return colRef;
434: 	}
