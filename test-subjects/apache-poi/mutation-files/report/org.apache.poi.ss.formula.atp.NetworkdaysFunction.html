 28: /**
 29:  * Implementation of Excel 'Analysis ToolPak' function NETWORKDAYS()<br/>
 30:  * Returns the number of workdays given a starting and an ending date, considering an interval of holidays. A workday is any non
 31:  * saturday/sunday date.
 32:  * <p/>
 33:  * <b>Syntax</b><br/>
 34:  * <b>NETWORKDAYS</b>(<b>startDate</b>, <b>endDate</b>, holidays)
 35:  * <p/>
 36:  * 
 37:  * @author jfaenomoto@gmail.com
 38:  */
 39: final class NetworkdaysFunction implements FreeRefFunction {
 40: 
 41:     public static final FreeRefFunction instance = new NetworkdaysFunction(ArgumentsEvaluator.instance);
 42: 
 43:     private ArgumentsEvaluator evaluator;
 44: 
 45:     /**
 46:      * Constructor.
 47:      * 
 48:      * @param anEvaluator an injected {@link ArgumentsEvaluator}.
 49:      */
 50:     private NetworkdaysFunction(ArgumentsEvaluator anEvaluator) {
 51:         // enforces singleton
 52:         this.evaluator = anEvaluator;
 53:     }
 54: 
 55:     /**
 56:      * Evaluate for NETWORKDAYS. Given two dates and a optional date or interval of holidays, determines how many working days are there
 57:      * between those dates.
 58:      * 
 59:      * @return {@link ValueEval} for the number of days between two dates.
 60:      */
 61:     public ValueEval evaluate(ValueEval[] args, OperationEvaluationContext ec) {
 62:         if (args.length < 2 || args.length > 3) {
 63:             return ErrorEval.VALUE_INVALID;
 64:         }
 65: 
 66:         int srcCellRow = ec.getRowIndex();
 67:         int srcCellCol = ec.getColumnIndex();
 68: 
 69:         double start, end;
 70:         double[] holidays;
 71:         try {
 72:             start = this.evaluator.evaluateDateArg(args[0], srcCellRow, srcCellCol);
 73:             end = this.evaluator.evaluateDateArg(args[1], srcCellRow, srcCellCol);
 74:             if (start > end) {
 75:                 return ErrorEval.NAME_INVALID;
 76:             }
 77:             ValueEval holidaysCell = args.length == 3 ? args[2] : null;
 78:             holidays = this.evaluator.evaluateDatesArg(holidaysCell, srcCellRow, srcCellCol);
 79:             return new NumberEval(WorkdayCalculator.instance.calculateWorkdays(start, end, holidays));
 80:         } catch (EvaluationException e) {
 81:             return ErrorEval.VALUE_INVALID;
 82:         }
 83:     }
