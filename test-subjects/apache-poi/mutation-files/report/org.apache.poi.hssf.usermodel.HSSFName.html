 81:     /**
 82:      * Sets the name of the named range
 83:      *
 84:      * <p>The following is a list of syntax rules that you need to be aware of when you create and edit names.</p>
 85:      * <ul>
 86:      *   <li><strong>Valid characters</strong>
 87:      *   The first character of a name must be a letter, an underscore character (_), or a backslash (\).
 88:      *   Remaining characters in the name can be letters, numbers, periods, and underscore characters.
 89:      *   </li>
 90:      *   <li><strong>Cell references disallowed</strong>
 91:      *   Names cannot be the same as a cell reference, such as Z$100 or R1C1.</li>
 92:      *   <li><strong>Spaces are not valid</strong>
 93:      *   Spaces are not allowed as part of a name. Use the underscore character (_) and period (.) as word separators, such as, Sales_Tax or First.Quarter.
 94:      *   </li>
 95:      *   <li><strong>Name length</strong>
 96:      *    A name can contain up to 255 characters.
 97:      *   </li>
 98:      *   <li><strong>Case sensitivity</strong>
 99:      *   Names can contain uppercase and lowercase letters.
100:      *   </li>
101:      * </ul>
102:      *
103:      * <p>
104:      * A name must always be unique within its scope. POI prevents you from defining a name that is not unique
105:      * within its scope. However you can use the same name in different scopes. Example:
106:      * <pre><blockquote>
107:      * //by default names are workbook-global
108:      * HSSFName name;
109:      * name = workbook.createName();
110:      * name.setNameName("sales_08");
111:      *
112:      * name = workbook.createName();
113:      * name.setNameName("sales_08"); //will throw an exception: "The workbook already contains this name (case-insensitive)"
114:      *
115:      * //create sheet-level name
116:      * name = workbook.createName();
117:      * name.setSheetIndex(0); //the scope of the name is the first sheet
118:      * name.setNameName("sales_08");  //ok
119:      *
120:      * name = workbook.createName();
121:      * name.setSheetIndex(0);
122:      * name.setNameName("sales_08");  //will throw an exception: "The sheet already contains this name (case-insensitive)"
123:      *
124:      * </blockquote></pre>
125:     * </p>
126:      *
127:      * @param nameName named range name to set
128:      * @throws IllegalArgumentException if the name is invalid or the name already exists (case-insensitive)
129:      */
130:     public void setNameName(String nameName){
131:         validateName(nameName);
132: 
133:         InternalWorkbook wb = _book.getWorkbook();
134:         _definedNameRec.setNameText(nameName);
135: 
136:         int sheetNumber = _definedNameRec.getSheetNumber();
137: 
138:         //Check to ensure no other names have the same case-insensitive name
139:         for ( int i = wb.getNumNames()-1; i >=0; i-- )
140:         {
141:             NameRecord rec = wb.getNameRecord(i);
142:             if (rec != _definedNameRec) {
143:                 if (rec.getNameText().equalsIgnoreCase(nameName) && sheetNumber == rec.getSheetNumber()){
144:                     String msg = "The "+(sheetNumber == 0 ? "workbook" : "sheet")+" already contains this name: " + nameName;
145:                     _definedNameRec.setNameText(nameName + "(2)");
146:                     throw new IllegalArgumentException(msg);
147:                 }
148:             }
149:         }
150:         
151:         // Update our comment, if there is one
152:         if(_commentRec != null) {
153:            String oldName = _commentRec.getNameText();
154:            _commentRec.setNameText(nameName);
155:            _book.getWorkbook().updateNameCommentRecordCache(_commentRec);
156:         }
157:     }

159:     private static void validateName(String name){
160:         if(name.length() == 0)  throw new IllegalArgumentException("Name cannot be blank");
161: 
162:         char c = name.charAt(0);
163:         if(!(c == '_' || Character.isLetter(c)) || name.indexOf(' ') != -1) {
164:             throw new IllegalArgumentException("Invalid name: '"+name+"'; Names must begin with a letter or underscore and not contain spaces");
165:         }
166:     }
