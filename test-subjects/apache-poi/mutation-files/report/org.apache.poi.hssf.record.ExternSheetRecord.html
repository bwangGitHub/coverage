187: 	/**
188:      * Add a zero-based reference to a {@link org.apache.poi.hssf.record.SupBookRecord}.
189:      * <p>
190:      *  If the type of the SupBook record is same-sheet referencing, Add-In referencing,
191:      *  DDE data source referencing, or OLE data source referencing,
192:      *  then no scope is specified and this value <em>MUST</em> be -2. Otherwise,
193:      *  the scope must be set as follows:
194:      *   <ol>
195:      *    <li><code>-2</code> Workbook-level reference that applies to the entire workbook.</li>
196:      *    <li><code>-1</code> Sheet-level reference. </li>
197:      *    <li><code>&gt;=0</code> Sheet-level reference. This specifies the first sheet in the reference.
198:      *    <p>
199:      *    If the SupBook type is unused or external workbook referencing,
200:      *    then this value specifies the zero-based index of an external sheet name,
201:      *    see {@link org.apache.poi.hssf.record.SupBookRecord#getSheetNames()}.
202:      *    This referenced string specifies the name of the first sheet within the external workbook that is in scope.
203:      *    This sheet MUST be a worksheet or macro sheet.
204:      *    <p>
205:      *    <p>
206:      *    If the supporting link type is self-referencing, then this value specifies the zero-based index of a
207:      *    {@link org.apache.poi.hssf.record.BoundSheetRecord} record in the workbook stream that specifies
208:      *    the first sheet within the scope of this reference. This sheet MUST be a worksheet or a macro sheet.
209:      *    </p>
210:      *    </li>
211:      *  </ol>
212:      *
213:      * @param firstSheetIndex  the scope, must be -2 for add-in references
214:      * @param lastSheetIndex   the scope, must be -2 for add-in references
215: 	 * @return index of newly added ref
216: 	 */
217: 	public int addRef(int extBookIndex, int firstSheetIndex, int lastSheetIndex) {
218: 		_list.add(new RefSubRecord(extBookIndex, firstSheetIndex, lastSheetIndex));
219: 		return _list.size() - 1;
220: 	}

222: 	public int getRefIxForSheet(int externalBookIndex, int sheetIndex) {
223: 		int nItems = _list.size();
224: 		for (int i = 0; i < nItems; i++) {
225: 			RefSubRecord ref = getRef(i);
226: 			if (ref.getExtBookIndex() != externalBookIndex) {
227: 				continue;
228: 			}
229: 			if (ref.getFirstSheetIndex() == sheetIndex && ref.getLastSheetIndex() == sheetIndex) {
230: 				return i;
231: 			}
232: 		}
233: 		return -1;
234: 	}

236: 	public static ExternSheetRecord combine(ExternSheetRecord[] esrs) {
237: 		ExternSheetRecord result = new ExternSheetRecord();
238: 		for (int i = 0; i < esrs.length; i++) {
239: 			ExternSheetRecord esr = esrs[i];
240: 			int nRefs = esr.getNumOfREFRecords();
241: 			for (int j=0; j<nRefs; j++) {
242: 				result.addREFRecord(esr.getRef(j));
243: 			}
244: 		}
245: 		return result;
246: 	}
