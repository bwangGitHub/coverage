 28: /**
 29:  * Low level event based HSSF reader.  Pass either a DocumentInputStream to
 30:  * process events along with a request object or pass a POIFS POIFSFileSystem to
 31:  * processWorkbookEvents along with a request.
 32:  *
 33:  * This will cause your file to be processed a record at a time.  Each record with
 34:  * a static id matching one that you have registered in your HSSFRequest will be passed
 35:  * to your associated HSSFListener.
 36:  *
 37:  * @author Andrew C. Oliver (acoliver at apache dot org)
 38:  * @author Carey Sublette  (careysub@earthling.net)
 39:  */
 40: public class HSSFEventFactory {

 42: 	public HSSFEventFactory() {
 43: 		// no instance fields
 44: 	}

 47: 	 * Processes a file into essentially record events.
 48: 	 *
 49: 	 * @param req an Instance of HSSFRequest which has your registered listeners
 50: 	 * @param fs  a POIFS filesystem containing your workbook
 51: 	 */
 52: 	public void processWorkbookEvents(HSSFRequest req, POIFSFileSystem fs) throws IOException {
 53: 	   processWorkbookEvents(req, fs.getRoot());
 54: 	}

 93: 	/**
 94: 	 * Processes a DocumentInputStream into essentially Record events.
 95: 	 *
 96: 	 * If an <code>AbortableHSSFListener</code> causes a halt to processing during this call
 97: 	 * the method will return just as with <code>abortableProcessEvents</code>, but no
 98: 	 * user code or <code>HSSFUserException</code> will be passed back.
 99: 	 *
100: 	 * @see org.apache.poi.poifs.filesystem.POIFSFileSystem#createDocumentInputStream(String)
101: 	 * @param req an Instance of HSSFRequest which has your registered listeners
102: 	 * @param in  a DocumentInputStream obtained from POIFS's POIFSFileSystem object
103: 	 */
104: 	public void processEvents(HSSFRequest req, InputStream in) {
105: 		try {
106: 			genericProcessEvents(req, in);
107: 		} catch (HSSFUserException hue) {
108: 			/*If an HSSFUserException user exception is thrown, ignore it.*/
109: 		}
110: 	}
111:

113: 	/**
114: 	 * Processes a DocumentInputStream into essentially Record events.
115: 	 *
116: 	 * @see org.apache.poi.poifs.filesystem.POIFSFileSystem#createDocumentInputStream(String)
117: 	 * @param req an Instance of HSSFRequest which has your registered listeners
118: 	 * @param in  a DocumentInputStream obtained from POIFS's POIFSFileSystem object
119: 	 * @return    numeric user-specified result code.
120: 	 */
121: 	public short abortableProcessEvents(HSSFRequest req, InputStream in)
122: 		throws HSSFUserException {
123: 		return genericProcessEvents(req, in);
124: 	}
125: 
126: 	/**
127: 	 * Processes a DocumentInputStream into essentially Record events.
128: 	 *
129: 	 * @see org.apache.poi.poifs.filesystem.POIFSFileSystem#createDocumentInputStream(String)
130: 	 * @param req an Instance of HSSFRequest which has your registered listeners
131: 	 * @param in  a DocumentInputStream obtained from POIFS's POIFSFileSystem object
132: 	 * @return    numeric user-specified result code.
133: 	 */
134: 	private short genericProcessEvents(HSSFRequest req, InputStream in)
135: 		throws HSSFUserException {
136: 		short userCode = 0;
137: 
138: 		// Create a new RecordStream and use that
139: 		RecordFactoryInputStream recordStream = new RecordFactoryInputStream(in, false);
140: 
141: 		// Process each record as they come in
142: 		while(true) {
143: 			Record r = recordStream.nextRecord();
144: 			if(r == null) {
145: 				break;
146: 			}
147: 			userCode = req.processRecord(r);
148: 			if (userCode != 0) {
149: 				break;
150: 			}
151: 		}
152: 
153: 		// All done, return our last code
154: 		return userCode;
155: 	}
