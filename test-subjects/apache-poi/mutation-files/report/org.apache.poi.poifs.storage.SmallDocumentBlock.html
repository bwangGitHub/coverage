 29: /**
 30:  * Storage for documents that are too small to use regular
 31:  * DocumentBlocks for their data
 32:  *
 33:  * @author  Marc Johnson (mjohnson at apache dot org)
 34:  */
 35: public final class SmallDocumentBlock implements BlockWritable, ListManagedBlock {
 36:     private static final int BLOCK_SHIFT = 6;
 37: 
 38:     private byte[]            _data;
 39:     private static final byte _default_fill         = ( byte ) 0xff;
 40:     private static final int  _block_size           = 1 << BLOCK_SHIFT;
 41:     private static final int BLOCK_MASK = _block_size-1;
 42: 
 43:     private final int  _blocks_per_big_block;
 44:     private final POIFSBigBlockSize _bigBlockSize;
 45: 
 46:     private SmallDocumentBlock(final POIFSBigBlockSize bigBlockSize, final byte [] data, final int index)
 47:     {
 48:         this(bigBlockSize);
 49:         System.arraycopy(data, index * _block_size, _data, 0, _block_size);
 50:     }
 51: 
 52:     private SmallDocumentBlock(final POIFSBigBlockSize bigBlockSize)
 53:     {
 54:         _bigBlockSize = bigBlockSize;
 55:         _blocks_per_big_block = getBlocksPerBigBlock(bigBlockSize);
 56:         _data = new byte[ _block_size ];
 57:     }
 58:     
 59:     private static int getBlocksPerBigBlock(final POIFSBigBlockSize bigBlockSize)
 60:     {
 61:        return bigBlockSize.getBigBlockSize() / _block_size;
 62:     }
 63: 
 64:     /**
 65:      * convert a single long array into an array of SmallDocumentBlock
 66:      * instances
 67:      *
 68:      * @param array the byte array to be converted
 69:      * @param size the intended size of the array (which may be smaller)
 70:      *
 71:      * @return an array of SmallDocumentBlock instances, filled from
 72:      *         the array
 73:      */
 74:     public static SmallDocumentBlock [] convert(POIFSBigBlockSize bigBlockSize,
 75:                                                 byte [] array,
 76:                                                 int size)
 77:     {
 78:         SmallDocumentBlock[] rval   =
 79:             new SmallDocumentBlock[ (size + _block_size - 1) / _block_size ];
 80:         int                  offset = 0;
 81: 
 82:         for (int k = 0; k < rval.length; k++)
 83:         {
 84:             rval[ k ] = new SmallDocumentBlock(bigBlockSize);
 85:             if (offset < array.length)
 86:             {
 87:                 int length = Math.min(_block_size, array.length - offset);
 88: 
 89:                 System.arraycopy(array, offset, rval[ k ]._data, 0, length);
 90:                 if (length != _block_size)
 91:                 {
 92:                     Arrays.fill(rval[ k ]._data, length, _block_size,
 93:                                 _default_fill);
 94:                 }
 95:             }
 96:             else
 97:             {
 98:                 Arrays.fill(rval[ k ]._data, _default_fill);
 99:             }
100:             offset += _block_size;
101:         }
102:         return rval;
103:     }
104: 
105:     /**
106:      * fill out a List of SmallDocumentBlocks so that it fully occupies
107:      * a set of big blocks
108:      *
109:      * @param blocks the List to be filled out
110:      *
111:      * @return number of big blocks the list encompasses
112:      */
113:     public static int fill(POIFSBigBlockSize bigBlockSize, List blocks)
114:     {
115:         int _blocks_per_big_block = getBlocksPerBigBlock(bigBlockSize);
116:         
117:         int count           = blocks.size();
118:         int big_block_count = (count + _blocks_per_big_block - 1)
119:                               / _blocks_per_big_block;
120:         int full_count      = big_block_count * _blocks_per_big_block;
121: 
122:         for (; count < full_count; count++)
123:         {
124:             blocks.add(makeEmptySmallDocumentBlock(bigBlockSize));
125:         }
126:         return big_block_count;
127:     }
128: 
129:     /**
130:      * Factory for creating SmallDocumentBlocks from DocumentBlocks
131:      *
132:      * @param store the original DocumentBlocks
133:      * @param size the total document size
134:      *
135:      * @return an array of new SmallDocumentBlocks instances
136:      *
137:      * @exception IOException on errors reading from the DocumentBlocks
138:      * @exception ArrayIndexOutOfBoundsException if, somehow, the store
139:      *            contains less data than size indicates
140:      */
141:     public static SmallDocumentBlock [] convert(POIFSBigBlockSize bigBlockSize,
142:                                                 BlockWritable [] store,
143:                                                 int size)
144:         throws IOException, ArrayIndexOutOfBoundsException
145:     {
146:         ByteArrayOutputStream stream = new ByteArrayOutputStream();
147: 
148:         for (int j = 0; j < store.length; j++)
149:         {
150:             store[ j ].writeBlocks(stream);
151:         }
152:         byte[]               data = stream.toByteArray();
153:         SmallDocumentBlock[] rval =
154:             new SmallDocumentBlock[ convertToBlockCount(size) ];
155: 
156:         for (int index = 0; index < rval.length; index++)
157:         {
158:             rval[ index ] = new SmallDocumentBlock(bigBlockSize, data, index);
159:         }
160:         return rval;
161:     }
162: 
163:     /**
164:      * create a list of SmallDocumentBlock's from raw data
165:      *
166:      * @param blocks the raw data containing the SmallDocumentBlock
167:      *               data
168:      *
169:      * @return a List of SmallDocumentBlock's extracted from the input
170:      */
171:     public static List extract(POIFSBigBlockSize bigBlockSize, ListManagedBlock [] blocks)
172:         throws IOException
173:     {
174:         int _blocks_per_big_block = getBlocksPerBigBlock(bigBlockSize);
175:         
176:         List sdbs = new ArrayList();
177: 
178:         for (int j = 0; j < blocks.length; j++)
179:         {
180:             byte[] data = blocks[ j ].getData();
181: 
182:             for (int k = 0; k < _blocks_per_big_block; k++)
183:             {
184:                 sdbs.add(new SmallDocumentBlock(bigBlockSize, data, k));
185:             }
186:         }
187:         return sdbs;
188:     }

190:     public static DataInputBlock getDataInputBlock(SmallDocumentBlock[] blocks, int offset) {
191:         int firstBlockIndex = offset >> BLOCK_SHIFT;
192:         int firstBlockOffset= offset & BLOCK_MASK;
193:         return new DataInputBlock(blocks[firstBlockIndex]._data, firstBlockOffset);
194:     }

196:     /**
197:      * Calculate the storage size of a set of SmallDocumentBlocks
198:      *
199:      * @param size number of SmallDocumentBlocks
200:      *
201:      * @return total size
202:      */
203:     public static int calcSize(int size)
204:     {
205:         return size * _block_size;
206:     }
207: 
208:     private static SmallDocumentBlock makeEmptySmallDocumentBlock(POIFSBigBlockSize bigBlockSize)
209:     {
210:         SmallDocumentBlock block = new SmallDocumentBlock(bigBlockSize);
211: 
212:         Arrays.fill(block._data, _default_fill);
213:         return block;
214:     }
215: 
216:     private static int convertToBlockCount(int size)
217:     {
218:         return (size + _block_size - 1) / _block_size;
219:     }
220: 
221:     /**
222:      * Write the storage to an OutputStream
223:      *
224:      * @param stream the OutputStream to which the stored data should
225:      *               be written
226:      *
227:      * @exception IOException on problems writing to the specified
228:      *            stream
229:      */
230:     public void writeBlocks(OutputStream stream)
231:         throws IOException
232:     {
233:         stream.write(_data);
234:     }

238:      *

241:      * @exception IOException if there is no data
242:      */
243:     public byte [] getData() {
244:         return _data;
245:     }

248:        return _bigBlockSize;
