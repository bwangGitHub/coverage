 51:     /**
 52:      * Creates a valid sheet name, which is conform to the rules.
 53:      * In any case, the result safely can be used for
 54:      * {@link org.apache.poi.ss.usermodel.Workbook#setSheetName(int, String)}.
 55:      * <br>
 56:      * Rules:
 57:      * <ul>
 58:      * <li>never null</li>
 59:      * <li>minimum length is 1</li>
 60:      * <li>maximum length is 31</li>
 61:      * <li>doesn't contain special chars: : 0x0000, 0x0003, / \ ? * ] [ </li>
 62:      * <li>Sheet names must not begin or end with ' (apostrophe)</li>
 63:      * </ul>
 64:      *
 65:      * @param nameProposal can be any string, will be truncated if necessary,
 66:      *        allowed to be null
 67:      * @param replaceChar the char to replace invalid characters.
 68:      * @return a valid string, "empty" if to short, "null" if null
 69:      */
 70:     public final static String createSafeSheetName(final String nameProposal, char replaceChar) {
 71:         if (nameProposal == null) {
 72:             return "null";
 73:         }
 74:         if (nameProposal.length() < 1) {
 75:             return "empty";
 76:         }
 77:         final int length = Math.min(31, nameProposal.length());
 78:         final String shortenname = nameProposal.substring(0, length);
 79:         final StringBuilder result = new StringBuilder(shortenname);
 80:         for (int i=0; i<length; i++) {
 81:             char ch = result.charAt(i);
 82:             switch (ch) {
 83:                 case '\u0000':
 84:                 case '\u0003':
 85:                 case ':':
 86:                 case '/':
 87:                 case '\\':
 88:                 case '?':
 89:                 case '*':
 90:                 case ']':
 91:                 case '[':
 92:                     result.setCharAt(i, replaceChar);
 93:                     break;
 94:                 case '\'':
 95:                     if (i==0 || i==length-1) {
 96:                         result.setCharAt(i, replaceChar);
 97:                     }
 98:                     break;
 99:                 default:
100:                     // all other chars OK
101:             }
102:         }
103:         return result.toString();
104:     }

111:      * The string MUST NOT contain the any of the following characters:

124:      * </p>
125:      *
126:      * @param sheetName the name to validate
127:      */
128:     public static void validateSheetName(String sheetName) {
129:         if (sheetName == null) {
130:             throw new IllegalArgumentException("sheetName must not be null");
131:         }
132:         int len = sheetName.length();
133:         if (len < 1 || len > 31) {
134:             throw new IllegalArgumentException("sheetName '" + sheetName
135:                     + "' is invalid - character count MUST be greater than or equal to 1 and less than or equal to 31");
136:         }
137: 
138:         for (int i=0; i<len; i++) {
139:             char ch = sheetName.charAt(i);
140:             switch (ch) {
141:                 case '/':
142:                 case '\\':
143:                 case '?':
144:                 case '*':
145:                 case ']':
146:                 case '[':
147:                 case ':':
148:                     break;
149:                 default:
150:                     // all other chars OK
151:                     continue;
152:             }
153:             throw new IllegalArgumentException("Invalid char (" + ch
154:                     + ") found at index (" + i + ") in sheet name '" + sheetName + "'");

157:             throw new IllegalArgumentException("Invalid sheet name '" + sheetName
158:                     + "'. Sheet names must not begin or end with (').");
159:         }
160:     }
