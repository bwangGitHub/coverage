152: 		if (m.size() != index) {

163: 		Map<Integer, String> result = new LinkedHashMap<Integer, String>();

177: 	/**
178: 	 * Get the format string that matches the given format index
179: 	 *
180: 	 * @param index of a built in format
181: 	 * @return string represented at index of format or <code>null</code> if there is not a built-in format at that index
182: 	 */
183: 	public static String getBuiltinFormat(int index) {
184: 		if (index < 0 || index >=_formats.length) {
185: 			return null;
186: 		}
187: 		return _formats[index];
188: 	}
189: 
190: 	/**
191: 	 * Get the format index that matches the given format string
192: 	 * <p/>
193: 	 * Automatically converts "text" to excel's format string to represent text.
194: 	 * </p>
195: 	 * @param pFmt string matching a built-in format
196: 	 * @return index of format or -1 if undefined.
197: 	 */
198: 	public static int getBuiltinFormat(String pFmt) {
199: 		String fmt;
200: 		if (pFmt.equalsIgnoreCase("TEXT")) {
201: 			fmt = "@";
202: 		} else {
203: 			fmt = pFmt;
204: 		}
205: 
206: 		for(int i =0; i< _formats.length; i++) {
207: 			if(fmt.equals(_formats[i])) {
208: 				return i;
209: 			}
210: 		}
211: 		return -1;
212: 	}
