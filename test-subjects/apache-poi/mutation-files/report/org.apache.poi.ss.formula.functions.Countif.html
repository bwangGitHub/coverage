339: 		public boolean matches(ValueEval x) {
340: 			if (x instanceof BlankEval) {
341: 				switch(getCode()) {
342: 					case CmpOp.NONE:
343: 					case CmpOp.EQ:
344: 						return _value.length() == 0;

347:                         // pred '<>ABC'  matches blank and 'not ABC'
348:                         return _value.length() != 0;
349: 				}
350: 				// no other criteria matches a blank cell
351: 				return false;
352: 			}
353: 			if(!(x instanceof StringEval)) {
354: 				// must always be string
355: 				// even if match str is wild, but contains only digits
356: 				// e.g. '4*7', NumberEval(4567) does not match
357: 				return false;
358: 			}
359: 			String testedValue = ((StringEval) x).getStringValue();
360: 			if (testedValue.length() < 1 && _value.length() < 1) {
361: 				// odd case: criteria '=' behaves differently to criteria ''

366: 					case CmpOp.NE:   return true;

377: 		/**
378: 		 * Translates Excel countif wildcard strings into java regex strings
379: 		 * @return <code>null</code> if the specified value contains no special wildcard characters.
380: 		 */
381: 		private static Pattern getWildCardPattern(String value) {
382: 			int len = value.length();
383: 			StringBuffer sb = new StringBuffer(len);
384: 			boolean hasWildCard = false;
385: 			for(int i=0; i<len; i++) {
386: 				char ch = value.charAt(i);
387: 				switch(ch) {
388: 					case '?':
389: 						hasWildCard = true;
390: 						// match exactly one character
391: 						sb.append('.');
392: 						continue;
393: 					case '*':
394: 						hasWildCard = true;
395: 						// match one or more occurrences of any character
396: 						sb.append(".*");
397: 						continue;
398: 					case '~':
399: 						if (i+1<len) {
400: 							ch = value.charAt(i+1);
401: 							switch (ch) {
402: 								case '?':
403: 								case '*':
404: 									hasWildCard = true;
405: 									sb.append('[').append(ch).append(']');
406: 									i++; // Note - incrementing loop variable here
407: 									continue;
408: 							}
409: 						}
410: 						// else not '~?' or '~*'
411: 						sb.append('~'); // just plain '~'
412: 						continue;
413: 					case '.':
414: 					case '$':
415: 					case '^':
416: 					case '[':
417: 					case ']':
418: 					case '(':
419: 					case ')':
420: 						// escape literal characters that would have special meaning in regex
421: 						sb.append("\\").append(ch);
422: 						continue;
423: 				}
424: 				sb.append(ch);
425: 			}
426: 			if (hasWildCard) {
427: 				return Pattern.compile(sb.toString(), Pattern.CASE_INSENSITIVE);
428: 			}
429: 			return null;
430: 		}

521: 	private static ErrorEval parseError(String value) {
522: 		if (value.length() < 4 || value.charAt(0) != '#') {
523: 			return null;
524: 		}
525: 		if (value.equals("#NULL!"))  return ErrorEval.NULL_INTERSECTION;
526: 		if (value.equals("#DIV/0!")) return ErrorEval.DIV_ZERO;
527: 		if (value.equals("#VALUE!")) return ErrorEval.VALUE_INVALID;
528: 		if (value.equals("#REF!"))   return ErrorEval.REF_INVALID;
529: 		if (value.equals("#NAME?"))  return ErrorEval.NAME_INVALID;
530: 		if (value.equals("#NUM!"))   return ErrorEval.NUM_ERROR;
531: 		if (value.equals("#N/A"))    return ErrorEval.NA;
532: 
533: 		return null;
534: 	}
