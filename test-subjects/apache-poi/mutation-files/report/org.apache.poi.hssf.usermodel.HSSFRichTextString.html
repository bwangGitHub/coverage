 27: /**
 28:  * Rich text unicode string.  These strings can have fonts applied to
 29:  * arbitary parts of the string.
 30:  *
 31:  * <p>
 32:  * Note, that in certain cases creating too many HSSFRichTextString cells may cause Excel 2003 and lower to crash
 33:  * when changing the color of the cells and then saving the Excel file. Compare two snippets that produce equivalent output:
 34:  *
 35:  * <p><blockquote><pre>
 36:  *  HSSFCell hssfCell = row.createCell(idx);
 37:  *  //rich text consists of two runs
 38:  *  HSSFRichTextString richString = new HSSFRichTextString( "Hello, World!" );
 39:  *  richString.applyFont( 0, 6, font1 );
 40:  *  richString.applyFont( 6, 13, font2 );
 41:  *  hssfCell.setCellValue( richString );
 42:  * </pre></blockquote>
 43:  *
 44:  * and
 45:  *
 46:  * <p><blockquote><pre>
 47:  *  //create a cell style and assign the first font to it
 48:  *  HSSFCellStyle style = workbook.createCellStyle();
 49:  *  style.setFont(font1);
 50:  *
 51:  *  HSSFCell hssfCell = row.createCell(idx);
 52:  *  hssfCell.setCellStyle(style);
 53:  *
 54:  *  //rich text consists of one run overriding the cell style
 55:  *  HSSFRichTextString richString = new HSSFRichTextString( "Hello, World!" );
 56:  *  richString.applyFont( 6, 13, font2 );
 57:  *  hssfCell.setCellValue( richString );
 58:  * </pre></blockquote><p>
 59:  *
 60:  * Excel always uses the latter approach: for a reach text containing N runs Excel saves the font of the first run in the cell's
 61:  * style and subsequent N-1 runs override this font.
 62:  *
 63:  * <p> For more information regarding this behavior please consult Bugzilla 47543:
 64:  *
 65:  * <a href="https://issues.apache.org/bugzilla/show_bug.cgi?id=47543">
 66:  * https://issues.apache.org/bugzilla/show_bug.cgi?id=47543</a>
 67:  * <p>
 68:  *
 69:  * @author Glen Stampoultzis (glens at apache.org)
 70:  * @author Jason Height (jheight at apache.org)
 71:  */
 72: public final class HSSFRichTextString implements Comparable<HSSFRichTextString>, RichTextString {
 73:     /** Place holder for indicating that NO_FONT has been applied here */
 74:     public static final short NO_FONT = 0;
 75: 
 76:     private UnicodeString _string;
 77:     private InternalWorkbook _book;
 78:     private LabelSSTRecord _record;
 79: 
 80:     public HSSFRichTextString()
 81:     {
 82:         this("");
 83:     }
 84: 
 85:     public HSSFRichTextString(String string) {
 86:         if (string == null) {
 87:             _string = new UnicodeString("");
 88:         } else {
 89:             _string = new UnicodeString(string);
 90:         }
 91:     }
 92: 
 93:     HSSFRichTextString(InternalWorkbook book, LabelSSTRecord record) {
 94:       setWorkbookReferences(book, record);
 95: 
 96:       _string = book.getSSTString(record.getSSTIndex());
 97:     }
 98: 
 99:     /** This must be called to setup the internal work book references whenever
100:      * a RichTextString is added to a cell
101:      */
102:     void setWorkbookReferences(InternalWorkbook book, LabelSSTRecord record) {
103:       _book = book;
104:       _record = record;
105:     }
106: 
107:     /** Called whenever the unicode string is modified. When it is modified
108:      *  we need to create a new SST index, so that other LabelSSTRecords will not
109:      *  be affected by changes that we make to this string.
110:      */
111:     private UnicodeString cloneStringIfRequired() {
112:       if (_book == null)
113:         return _string;
114:       UnicodeString s = (UnicodeString)_string.clone();
115:       return s;
116:     }
117: 
118:     private void addToSSTIfRequired() {
119:       if (_book != null) {
120:         int index = _book.addSSTString(_string);
121:         _record.setSSTIndex(index);
122:         //The act of adding the string to the SST record may have meant that
123:         //an existing string was returned for the index, so update our local version
124:         _string = _book.getSSTString(index);
125:       }
126:     }
127: 
128: 
129:     /**
130:      * Applies a font to the specified characters of a string.
131:      *
132:      * @param startIndex    The start index to apply the font to (inclusive)
133:      * @param endIndex      The end index to apply the font to (exclusive)
134:      * @param fontIndex     The font to use.
135:      */
136:     public void applyFont(int startIndex, int endIndex, short fontIndex)
137:     {
138:         if (startIndex > endIndex)
139:             throw new IllegalArgumentException("Start index must be less than end index.");
140:         if (startIndex < 0 || endIndex > length())
141:             throw new IllegalArgumentException("Start and end index not in range.");
142:         if (startIndex == endIndex)
143:             return;
144: 
145:         //Need to check what the font is currently, so we can reapply it after
146:         //the range is completed
147:         short currentFont = NO_FONT;
148:         if (endIndex != length()) {
149:           currentFont = this.getFontAtIndex(endIndex);
150:         }
151: 
152:         //Need to clear the current formatting between the startIndex and endIndex
153:         _string = cloneStringIfRequired();
154:         Iterator formatting = _string.formatIterator();
155:         if (formatting != null) {
156:           while (formatting.hasNext()) {
157:             UnicodeString.FormatRun r = (UnicodeString.FormatRun)formatting.next();
158:             if ((r.getCharacterPos() >= startIndex) && (r.getCharacterPos() < endIndex))
159:               formatting.remove();
160:           }
161:         }
162: 
163: 
164:         _string.addFormatRun(new UnicodeString.FormatRun((short)startIndex, fontIndex));
165:         if (endIndex != length())
166:           _string.addFormatRun(new UnicodeString.FormatRun((short)endIndex, currentFont));
167: 
168:         addToSSTIfRequired();
169:     }
170: 
171:     /**
172:      * Applies a font to the specified characters of a string.
173:      *
174:      * @param startIndex    The start index to apply the font to (inclusive)
175:      * @param endIndex      The end index to apply to font to (exclusive)
176:      * @param font          The index of the font to use.
177:      */
178:     public void applyFont(int startIndex, int endIndex, Font font)
179:     {
180:         applyFont(startIndex, endIndex, ((HSSFFont) font).getIndex());
181:     }
182: 
183:     /**
184:      * Sets the font of the entire string.
185:      * @param font          The font to use.
186:      */
187:     public void applyFont(Font font)
188:     {
189:         applyFont(0, _string.getCharCount(), font);
190:     }
191: 
192:     /**
193:      * Removes any formatting that may have been applied to the string.
194:      */
195:     public void clearFormatting() {
196:       _string = cloneStringIfRequired();
197:       _string.clearFormatting();
198:       addToSSTIfRequired();
199:     }
200: 
201:     /**
202:      * Returns the plain string representation.
203:      */
204:     public String getString()
205:     {
206:         return _string.getString();
207:     }
208: 
209:     /**
210:      * Used internally by the HSSFCell to get the internal
211:      * string value.
212:      * Will ensure the string is not shared
213:      */
214:     UnicodeString getUnicodeString() {
215:       return cloneStringIfRequired();
216:     }
217:

219:      * Returns the raw, probably shared Unicode String.
220:      * Used when tweaking the styles, eg updating font
221:      *  positions.
222:      * Changes to this string may well effect
223:      *  other RichTextStrings too!
224:      */
225:     UnicodeString getRawUnicodeString() {
226:     	return _string;
227:     }
228: 
229:     /** Used internally by the HSSFCell to set the internal string value*/
230:     void setUnicodeString(UnicodeString str) {
231:       this._string = str;
232:     }
233: 
234: 
235:     /**
236:      * @return  the number of characters in the text.
237:      */
238:     public int length() {
239:         return _string.getCharCount();
240:     }
241: 
242:     /**
243:      * Returns the font in use at a particular index.
244:      *
245:      * @param index         The index.
246:      * @return              The font that's currently being applied at that
247:      *                      index or null if no font is being applied or the
248:      *                      index is out of range.
249:      */
250:     public short getFontAtIndex( int index )
251:     {
252:       int size = _string.getFormatRunCount();
253:       UnicodeString.FormatRun currentRun = null;
254:       for (int i=0;i<size;i++) {
255:         UnicodeString.FormatRun r = _string.getFormatRun(i);
256:         if (r.getCharacterPos() > index) {
257:             break;
258:         }
259:         currentRun = r;
260:       }
261:       if (currentRun == null) {
262:           return NO_FONT;
263:       }
264:       return currentRun.getFontIndex();
265:     }
266: 
267:     /**
268:      * @return  The number of formatting runs used. There will always be at
269:      *          least one of font NO_FONT.
270:      *
271:      * @see #NO_FONT
272:      */
273:     public int numFormattingRuns()
274:     {
275:         return _string.getFormatRunCount();
276:     }
277: 
278:     /**
279:      * The index within the string to which the specified formatting run applies.
280:      * @param index     the index of the formatting run
281:      * @return  the index within the string.
282:      */
283:     public int getIndexOfFormattingRun(int index)
284:     {
285:         UnicodeString.FormatRun r = _string.getFormatRun(index);
286:         return r.getCharacterPos();
287:     }
288: 
289:     /**
290:      * Gets the font used in a particular formatting run.
291:      *
292:      * @param index     the index of the formatting run
293:      * @return  the font number used.
294:      */
295:     public short getFontOfFormattingRun(int index)
296:     {
297:       UnicodeString.FormatRun r = _string.getFormatRun(index);
298:       return r.getFontIndex();
299:     }
300: 
301:     /**
302:      * Compares one rich text string to another.
303:      */
304:     public int compareTo(HSSFRichTextString r) {
305:        return _string.compareTo(r._string);
306:     }
307: 
308:     public boolean equals(Object o) {
309:       if (o instanceof HSSFRichTextString) {
310:         return _string.equals(((HSSFRichTextString)o)._string);
311:       }
312:       return false;
313: 
314:     }
315: 
316:     /**
317:      * @return  the plain text representation of this string.
318:      */
319:     public String toString()
320:     {
321:         return _string.toString();
322:     }

328:      */
329:     public void applyFont( short fontIndex )
330:     {
331:         applyFont(0, _string.getCharCount(), fontIndex);
332:     }
