 29:  * Calculates the number of days between two dates based on a 360-day year

 65:     private static double evaluate(double d0, double d1, boolean method) {
 66:         Calendar startingDate = getStartingDate(d0);
 67:         Calendar endingDate = getEndingDateAccordingToStartingDate(d1, startingDate);
 68:         long startingDay = startingDate.get(Calendar.MONTH) * 30 + startingDate.get(Calendar.DAY_OF_MONTH);
 69:         long endingDay = (endingDate.get(Calendar.YEAR) - startingDate.get(Calendar.YEAR)) * 360
 70:                 + endingDate.get(Calendar.MONTH) * 30 + endingDate.get(Calendar.DAY_OF_MONTH);

 72:     }

102:         clone.add(java.util.Calendar.DAY_OF_MONTH, -1);
103:         int lastDayOfMonth = clone.get(Calendar.DAY_OF_MONTH);
