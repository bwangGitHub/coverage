342:       * This is basically a kludge to deal with the now obsolete Label records.  If
343:       * you have to read in a sheet that contains Label records, be aware that the rest
344:       * of the API doesn't deal with them, the low level structure only provides read-only
345:       * semi-immutable structures (the sets are there for interface conformance with NO
346:       * impelmentation).  In short, you need to call this function passing it a reference
347:       * to the Workbook object.  All labels will be converted to LabelSST records and their
348:       * contained strings will be written to the Shared String tabel (SSTRecord) within
349:       * the Workbook.
350:       *
351:       * @param records a collection of sheet's records.
352:       * @param offset the offset to search at 
353:       * @see org.apache.poi.hssf.record.LabelRecord
354:       * @see org.apache.poi.hssf.record.LabelSSTRecord
355:       * @see org.apache.poi.hssf.record.SSTRecord
356:       */
357: 
358:      private void convertLabelRecords(List records, int offset)
359:      {
360:          if (log.check( POILogger.DEBUG ))
361:              log.log(POILogger.DEBUG, "convertLabelRecords called");
362:          for (int k = offset; k < records.size(); k++)
363:          {
364:              Record rec = ( Record ) records.get(k);
365: 
366:              if (rec.getSid() == LabelRecord.sid)
367:              {
368:                  LabelRecord oldrec = ( LabelRecord ) rec;
369: 
370:                  records.remove(k);
371:                  LabelSSTRecord newrec   = new LabelSSTRecord();
372:                  int            stringid =
373:                      workbook.addSSTString(new UnicodeString(oldrec.getValue()));
374: 
375:                  newrec.setRow(oldrec.getRow());
376:                  newrec.setColumn(oldrec.getColumn());
377:                  newrec.setXFIndex(oldrec.getXFIndex());
378:                  newrec.setSSTIndex(stringid);
379:                        records.add(k, newrec);
380:              }
381:          }
382:          if (log.check( POILogger.DEBUG ))
383:              log.log(POILogger.DEBUG, "convertLabelRecords exit");
384:      }

709:     private String getUniqueSheetName(String srcName) {
710:         int uniqueIndex = 2;
711:         String baseName = srcName;
712:         int bracketPos = srcName.lastIndexOf('(');
713:         if (bracketPos > 0 && srcName.endsWith(")")) {
714:             String suffix = srcName.substring(bracketPos + 1, srcName.length() - ")".length());
