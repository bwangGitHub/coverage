 23: /**
 24:  * Represents a 64 bit IEEE double quantity expressed with both decimal and binary exponents
 25:  * Does not handle negative numbers or zero
 26:  * <p/>
 27:  * The value of a {@link ExpandedDouble} is given by<br/>
 28:  * <tt> a &times; 2<sup>b</sup></tt>
 29:  * <br/>
 30:  * where:<br/>
 31:  *
 32:  * <tt>a</tt> = <i>significand</i><br/>
 33:  * <tt>b</tt> = <i>binaryExponent</i> - bitLength(significand) + 1<br/>
 34:  *
 35:  * @author Josh Micich
 36:  */
 37: final class ExpandedDouble {
 38: 	private static final BigInteger BI_FRAC_MASK = BigInteger.valueOf(FRAC_MASK);
 39: 	private static final BigInteger BI_IMPLIED_FRAC_MSB = BigInteger.valueOf(FRAC_ASSUMED_HIGH_BIT);
 40: 
 41: 	private static BigInteger getFrac(long rawBits) {
 42: 		return BigInteger.valueOf(rawBits).and(BI_FRAC_MASK).or(BI_IMPLIED_FRAC_MSB).shiftLeft(11);
 43: 	}
 44: 
 45: 
 46: 	public static ExpandedDouble fromRawBitsAndExponent(long rawBits, int exp) {
 47: 		return new ExpandedDouble(getFrac(rawBits), exp);
 48: 	}
 49: 
 50: 	/**
 51: 	 * Always 64 bits long (MSB, bit-63 is '1')
 52: 	 */
 53: 	private final BigInteger _significand;
 54: 	private final int _binaryExponent;
 55: 
 56: 	public ExpandedDouble(long rawBits) {
 57: 		int biasedExp = (int) (rawBits >> 52);
 58: 		if (biasedExp == 0) {
 59: 			// sub-normal numbers
 60: 			BigInteger frac = BigInteger.valueOf(rawBits).and(BI_FRAC_MASK);
 61: 			int expAdj = 64 - frac.bitLength();
 62: 			_significand = frac.shiftLeft(expAdj);
 63: 			_binaryExponent = (biasedExp & 0x07FF) - 1023 - expAdj;
 64: 		} else {
 65: 			BigInteger frac = getFrac(rawBits);
 66: 			_significand = frac;
 67: 			_binaryExponent = (biasedExp & 0x07FF) - 1023;
 68: 		}
 69: 	}
 70: 
 71: 	ExpandedDouble(BigInteger frac, int binaryExp) {
 72: 		if (frac.bitLength() != 64) {
 73: 			throw new IllegalArgumentException("bad bit length");
 74: 		}
 75: 		_significand = frac;
 76: 		_binaryExponent = binaryExp;
 77: 	}
 78: 
 79: 
 80: 	/**
 81: 	 * Convert to an equivalent {@link NormalisedDecimal} representation having 15 decimal digits of precision in the
 82: 	 * non-fractional bits of the significand.
 83: 	 */
 84: 	public NormalisedDecimal normaliseBaseTen() {
 85: 		return NormalisedDecimal.create(_significand, _binaryExponent);
 86: 	}
 87: 
 88: 	/**
 89: 	 * @return the number of non-fractional bits after the MSB of the significand
 90: 	 */
 91: 	public int getBinaryExponent() {
 92: 		return _binaryExponent;
 93: 	}
 94: 
 95: 	public BigInteger getSignificand() {
 96: 		return _significand;
 97: 	}
