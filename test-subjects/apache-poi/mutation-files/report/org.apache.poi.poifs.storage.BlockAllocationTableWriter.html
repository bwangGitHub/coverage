 29: /**
 30:  * This class manages and creates the Block Allocation Table, which is
 31:  * basically a set of linked lists of block indices.
 32:  * <P>
 33:  * Each block of the filesystem has an index. The first block, the
 34:  * header, is skipped; the first block after the header is index 0,
 35:  * the next is index 1, and so on.
 36:  * <P>
 37:  * A block's index is also its index into the Block Allocation
 38:  * Table. The entry that it finds in the Block Allocation Table is the
 39:  * index of the next block in the linked list of blocks making up a
 40:  * file, or it is set to -2: end of list.
 41:  *
 42:  * @author Marc Johnson (mjohnson at apache dot org)
 43:  */
 44: public final class BlockAllocationTableWriter implements BlockWritable, BATManaged {
 45:     private IntList    _entries;
 46:     private BATBlock[] _blocks;
 47:     private int        _start_block;
 48:     private POIFSBigBlockSize _bigBlockSize;
 49: 
 50:     /**
 51:      * create a BlockAllocationTableWriter
 52:      */
 53:     public BlockAllocationTableWriter(POIFSBigBlockSize bigBlockSize)
 54:     {
 55:        _bigBlockSize = bigBlockSize; 
 56:         _start_block  = POIFSConstants.END_OF_CHAIN;
 57:         _entries      = new IntList();
 58:         _blocks       = new BATBlock[ 0 ];
 59:     }
 60: 
 61:     /**
 62:      * Create the BATBlocks we need
 63:      *
 64:      * @return start block index of BAT blocks
 65:      */
 66:     public int createBlocks()
 67:     {
 68:         int xbat_blocks = 0;
 69:         int bat_blocks  = 0;
 70: 
 71:         while (true)
 72:         {
 73:             int calculated_bat_blocks  =
 74:                 BATBlock.calculateStorageRequirements(_bigBlockSize,
 75:                                                       bat_blocks
 76:                                                       + xbat_blocks
 77:                                                       + _entries.size());
 78:             int calculated_xbat_blocks =
 79:                 HeaderBlockWriter.calculateXBATStorageRequirements(
 80:                       _bigBlockSize, calculated_bat_blocks);
 81: 
 82:             if ((bat_blocks == calculated_bat_blocks)
 83:                     && (xbat_blocks == calculated_xbat_blocks))
 84:             {
 85: 
 86:                 // stable ... we're OK
 87:                 break;
 88:             }
 89:             bat_blocks  = calculated_bat_blocks;
 90:             xbat_blocks = calculated_xbat_blocks;
 91:         }
 92:         int startBlock = allocateSpace(bat_blocks);
 93: 
 94:         allocateSpace(xbat_blocks);
 95:         simpleCreateBlocks();
 96:         return startBlock;
 97:     }
 98: 
 99:     /**
100:      * Allocate space for a block of indices
101:      *
102:      * @param blockCount the number of blocks to allocate space for
103:      *
104:      * @return the starting index of the blocks
105:      */
106:     public int allocateSpace(final int blockCount)
107:     {
108:         int startBlock = _entries.size();
109: 
110:         if (blockCount > 0)
111:         {
112:             int limit = blockCount - 1;
113:             int index = startBlock + 1;
114: 
115:             for (int k = 0; k < limit; k++)
116:             {
117:                 _entries.add(index++);
118:             }
119:             _entries.add(POIFSConstants.END_OF_CHAIN);
120:         }
121:         return startBlock;
122:     }
123: 
124:     /**
125:      * get the starting block
126:      *
127:      * @return the starting block index
128:      */
129:     public int getStartBlock()
130:     {
131:         return _start_block;
132:     }
133: 
134:     /**
135:      * create the BATBlocks
136:      */
137:     void simpleCreateBlocks()
138:     {
139:         _blocks = BATBlock.createBATBlocks(_bigBlockSize, _entries.toArray());
140:     }
141: 
142:     /**
143:      * Write the storage to an OutputStream
144:      *
145:      * @param stream the OutputStream to which the stored data should
146:      *               be written
147:      *
148:      * @exception IOException on problems writing to the specified
149:      *            stream
150:      */
151:     public void writeBlocks(final OutputStream stream)
152:         throws IOException
153:     {
154:         for (int j = 0; j < _blocks.length; j++)
155:         {
156:             _blocks[ j ].writeBlocks(stream);
157:         }
158:     }

161:      * Write the BAT into its associated block
162:      */

164:         throws IOException
165:     {
166:         bat.writeData(block);
167:     }
