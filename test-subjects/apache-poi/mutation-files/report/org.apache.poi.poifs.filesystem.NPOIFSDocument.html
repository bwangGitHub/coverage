 36:  * This class manages a document in the NIO POIFS filesystem.
 37:  * This is the {@link NPOIFSFileSystem} version.

 39: public final class NPOIFSDocument implements POIFSViewable {

 42:    private NPOIFSFileSystem _filesystem;
 43:    private NPOIFSStream _stream;
 44:    private int _block_size;
 45: 	
 46:    /**
 47:     * Constructor for an existing Document 
 48:     */
 49:    public NPOIFSDocument(DocumentProperty property, NPOIFSFileSystem filesystem) 
 50:       throws IOException
 51:    {
 52:       this._property = property;
 53:       this._filesystem = filesystem;
 54: 
 55:       if(property.getSize() < POIFSConstants.BIG_BLOCK_MINIMUM_DOCUMENT_SIZE) {
 56:          _stream = new NPOIFSStream(_filesystem.getMiniStore(), property.getStartBlock());
 57:          _block_size = _filesystem.getMiniStore().getBlockStoreBlockSize();
 58:       } else {
 59:          _stream = new NPOIFSStream(_filesystem, property.getStartBlock());
 60:          _block_size = _filesystem.getBlockStoreBlockSize();
 61:       }
 62:    }
 63: 
 64:    /**
 65:     * Constructor for a new Document
 66:     *
 67:     * @param name the name of the POIFSDocument
 68:     * @param stream the InputStream we read data from
 69:     */
 70:    public NPOIFSDocument(String name, NPOIFSFileSystem filesystem, InputStream stream) 
 71:       throws IOException 
 72:    {
 73:       this._filesystem = filesystem;
 74: 
 75:       // Buffer the contents into memory. This is a bit icky...
 76:       // TODO Replace with a buffer up to the mini stream size, then streaming write
 77:       byte[] contents;
 78:       if(stream instanceof ByteArrayInputStream) {
 79:          ByteArrayInputStream bais = (ByteArrayInputStream)stream;
 80:          contents = new byte[bais.available()];
 81:          bais.read(contents);
 82:       } else {
 83:          ByteArrayOutputStream baos = new ByteArrayOutputStream();
 84:          IOUtils.copy(stream, baos);
 85:          contents = baos.toByteArray();
 86:       }
 87: 
 88:       // Do we need to store as a mini stream or a full one?
 89:       if(contents.length <= POIFSConstants.BIG_BLOCK_MINIMUM_DOCUMENT_SIZE) {
 90:          _stream = new NPOIFSStream(filesystem.getMiniStore());
 91:          _block_size = _filesystem.getMiniStore().getBlockStoreBlockSize();
 92:       } else {
 93:          _stream = new NPOIFSStream(filesystem);
 94:          _block_size = _filesystem.getBlockStoreBlockSize();
 95:       }
 96: 
 97:       // Store it
 98:       _stream.updateContents(contents);
 99: 
100:       // And build the property for it
101:       this._property = new DocumentProperty(name, contents.length);
102:       _property.setStartBlock(_stream.getStartBlock());     
103:    }

132:    /**
133:     * Get an array of objects, some of which may implement POIFSViewable
134:     *
135:     * @return an array of Object; may not be null, but may be empty
136:     */
137:    public Object[] getViewableArray() {
138:       Object[] results = new Object[1];
139:       String result;
140: 
141:       try {
142:          if(getSize() > 0) {
143:             // Get all the data into a single array
144:             byte[] data = new byte[getSize()];
145:             int offset = 0;
146:             for(ByteBuffer buffer : _stream) {
147:                int length = Math.min(_block_size, data.length-offset); 
148:                buffer.get(data, offset, length);
149:                offset += length;
150:             }
151: 
152:             ByteArrayOutputStream output = new ByteArrayOutputStream();
153:             HexDump.dump(data, 0, output, 0);
154:             result = output.toString();
155:          } else {
156:             result = "<NO DATA>";
157:          }
158:       } catch (IOException e) {
159:          result = e.getMessage();
160:       }
161:       results[0] = result;
162:       return results;
163:    }
