 25:  * This class provides methods to read a DocumentEntry managed by a
 26:  * {@link POIFSFileSystem} instance.

 28:  * @author Marc Johnson (mjohnson at apache dot org)
 29:  */
 30: public final class ODocumentInputStream extends DocumentInputStream {
 31: 	/** current offset into the Document */
 32: 	private int _current_offset;
 33: 
 34: 	/** current marked offset into the Document (used by mark and reset) */
 35: 	private int _marked_offset;
 36: 
 37: 	/** the Document's size */
 38: 	private int _document_size;
 39: 
 40: 	/** have we been closed? */
 41: 	private boolean _closed;
 42: 
 43: 	/** the actual Document */

 46: 	/** the data block containing the current stream pointer */
 47: 	private DataInputBlock _currentBlock;

 49: 	/**
 50: 	 * Create an InputStream from the specified DocumentEntry
 51: 	 * 
 52: 	 * @param document the DocumentEntry to be read
 53: 	 * 
 54: 	 * @exception IOException if the DocumentEntry cannot be opened (like, maybe it has
 55: 	 *                been deleted?)
 56: 	 */
 57: 	public ODocumentInputStream(DocumentEntry document) throws IOException {
 58: 		if (!(document instanceof DocumentNode)) {
 59: 			throw new IOException("Cannot open internal document storage");
 60: 		}
 61: 		DocumentNode documentNode = (DocumentNode)document;
 62: 		if(documentNode.getDocument() == null) {
 63:          throw new IOException("Cannot open internal document storage");
 64: 		}
 65: 		      
 66: 		_current_offset = 0;
 67: 		_marked_offset = 0;
 68: 		_document_size = document.getSize();
 69: 		_closed = false;
 70: 		_document = documentNode.getDocument();
 71: 		_currentBlock = getDataInputBlock(0);
 72: 	}

 74: 	/**
 75: 	 * Create an InputStream from the specified Document
 76: 	 * 
 77: 	 * @param document the Document to be read
 78: 	 */
 79: 	public ODocumentInputStream(POIFSDocument document) {
 80: 		_current_offset = 0;
 81: 		_marked_offset = 0;
 82: 		_document_size = document.getSize();
 83: 		_closed = false;
 84: 		_document = document;
 85: 		_currentBlock = getDataInputBlock(0);
 86: 	}

 88: 	@Override
 89: 	public int available() {
 90: 		if (_closed) {
 91: 			throw new IllegalStateException("cannot perform requested operation on a closed stream");
 92: 		}
 93: 		return _document_size - _current_offset;
 94: 	}

 96:    @Override
 97: 	public void close() {
 98: 		_closed = true;
 99: 	}
100: 
101:    @Override
102: 	public void mark(int ignoredReadlimit) {
103: 		_marked_offset = _current_offset;
104: 	}

106: 	private DataInputBlock getDataInputBlock(int offset) {
107: 		return _document.getDataInputBlock(offset);
108: 	}
109: 
110:    @Override
111: 	public int read() throws IOException {
112: 		dieIfClosed();
113: 		if (atEOD()) {
114: 			return EOF;
115: 		}
116: 		int result = _currentBlock.readUByte();
117: 		_current_offset++;
118: 		if (_currentBlock.available() < 1) {
119: 			_currentBlock = getDataInputBlock(_current_offset);
120: 		}
121: 		return result;
122: 	}
123: 
124:    @Override
125: 	public int read(byte[] b, int off, int len) throws IOException {
126: 		dieIfClosed();
127: 		if (b == null) {
128: 			throw new IllegalArgumentException("buffer must not be null");
129: 		}
130: 		if (off < 0 || len < 0 || b.length < off + len) {
131: 			throw new IndexOutOfBoundsException("can't read past buffer boundaries");
132: 		}
133: 		if (len == 0) {
134: 			return 0;
135: 		}
136: 		if (atEOD()) {
137: 			return EOF;
138: 		}
139: 		int limit = Math.min(available(), len);
140: 		readFully(b, off, limit);
141: 		return limit;
142: 	}

144: 	/**
145: 	 * Repositions this stream to the position at the time the mark() method was
146: 	 * last called on this input stream. If mark() has not been called this
147: 	 * method repositions the stream to its beginning.
148: 	 */
149:    @Override
150: 	public void reset() {
151: 		_current_offset = _marked_offset;
152: 		_currentBlock = getDataInputBlock(_current_offset);
153: 	}
154: 
155:    @Override
156: 	public long skip(long n) throws IOException {
157: 		dieIfClosed();
158: 		if (n < 0) {
159: 			return 0;
160: 		}
161: 		int new_offset = _current_offset + (int) n;
162: 
163: 		if (new_offset < _current_offset) {
164: 
165: 			// wrap around in converting a VERY large long to an int
166: 			new_offset = _document_size;
167: 		} else if (new_offset > _document_size) {
168: 			new_offset = _document_size;
169: 		}
170: 		long rval = new_offset - _current_offset;
171: 
172: 		_current_offset = new_offset;
173: 		_currentBlock = getDataInputBlock(_current_offset);
174: 		return rval;
175: 	}
176: 
177: 	private void dieIfClosed() throws IOException {
178: 		if (_closed) {
179: 			throw new IOException("cannot perform requested operation on a closed stream");
180: 		}
181: 	}
182: 
183: 	private boolean atEOD() {
184: 		return _current_offset == _document_size;
185: 	}
186: 
187: 	private void checkAvaliable(int requestedSize) {
188: 		if (_closed) {
189: 			throw new IllegalStateException("cannot perform requested operation on a closed stream");
190: 		}
191: 		if (requestedSize > _document_size - _current_offset) {
192: 			throw new RuntimeException("Buffer underrun - requested " + requestedSize
193: 					+ " bytes but " + (_document_size - _current_offset) + " was available");
194: 		}
195: 	}

198: 	public byte readByte() {
199: 		return (byte) readUByte();

203: 	public double readDouble() {
204: 		return Double.longBitsToDouble(readLong());

212:    @Override
213: 	public void readFully(byte[] buf, int off, int len) {
214: 		checkAvaliable(len);
215: 		int blockAvailable = _currentBlock.available();
216: 		if (blockAvailable > len) {
217: 			_currentBlock.readFully(buf, off, len);
218: 			_current_offset += len;
219: 			return;
220: 		}
221: 		// else read big amount in chunks
222: 		int remaining = len;
223: 		int writePos = off;
224: 		while (remaining > 0) {
225: 			boolean blockIsExpiring = remaining >= blockAvailable;
226: 			int reqSize;
227: 			if (blockIsExpiring) {
228: 				reqSize = blockAvailable;
229: 			} else {
230: 				reqSize = remaining;
231: 			}
232: 			_currentBlock.readFully(buf, writePos, reqSize);
233: 			remaining -= reqSize;
234: 			writePos += reqSize;
235: 			_current_offset += reqSize;
236: 			if (blockIsExpiring) {
237: 				if (_current_offset == _document_size) {
238: 					if (remaining > 0) {
239: 						throw new IllegalStateException(
240: 								"reached end of document stream unexpectedly");
241: 					}
242: 					_currentBlock = null;
243: 					break;
244: 				}
245: 				_currentBlock = getDataInputBlock(_current_offset);
246: 				blockAvailable = _currentBlock.available();
247: 			}
248: 		}
249: 	}

251:    @Override
252: 	public long readLong() {
253: 		checkAvaliable(SIZE_LONG);
254: 		int blockAvailable = _currentBlock.available();
255: 		long result;
256: 		if (blockAvailable > SIZE_LONG) {
257: 			result = _currentBlock.readLongLE();
258: 		} else {
259: 			DataInputBlock nextBlock = getDataInputBlock(_current_offset + blockAvailable);
260: 			if (blockAvailable == SIZE_LONG) {
261: 				result = _currentBlock.readLongLE();
262: 			} else {
263: 				result = nextBlock.readLongLE(_currentBlock, blockAvailable);
264: 			}
265: 			_currentBlock = nextBlock;
266: 		}
267: 		_current_offset += SIZE_LONG;
268: 		return result;
269: 	}
