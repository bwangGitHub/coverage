 71: 		 * @return <code>null</code> if the double value encoded by <tt>valueLongBits</tt> 
 72: 		 * is a normal (non NaN) double value.
 73: 		 */
 74: 		public static SpecialCachedValue create(long valueLongBits) {
 75: 			if ((BIT_MARKER & valueLongBits) != BIT_MARKER) {
 76: 				return null;
 77: 			}
 78: 
 79: 			byte[] result = new byte[VARIABLE_DATA_LENGTH];
 80: 			long x = valueLongBits;
 81: 			for (int i=0; i<VARIABLE_DATA_LENGTH; i++) {
 82: 				result[i] = (byte) x;
 83: 				x >>= 8;
 84: 			}
 85: 			switch (result[0]) {
 86: 				case STRING:
 87: 				case BOOLEAN:
 88: 				case ERROR_CODE:
 89: 				case EMPTY:
 90: 					break;
 91: 				default:
 92: 					throw new RecordFormatException("Bad special value code (" + result[0] + ")");
 93: 			}
 94: 			return new SpecialCachedValue(result);
 95: 		}
