 47: /**
 48:  * A collection of axes that have been assigned to the TOP, BOTTOM, LEFT or 
 49:  * RIGHT of a chart.  This class is used internally by JFreeChart, you won't
 50:  * normally need to use it yourself.
 51:  */
 52: public class AxisCollection {
 53:     
 54:     /** The axes that need to be drawn at the top of the plot area. */
 55:     private List axesAtTop;
 56:     
 57:     /** The axes that need to be drawn at the bottom of the plot area. */
 58:     private List axesAtBottom;
 59:     
 60:     /** The axes that need to be drawn at the left of the plot area. */
 61:     private List axesAtLeft;
 62:     
 63:     /** The axes that need to be drawn at the right of the plot area. */
 64:     private List axesAtRight;
 65:    
 66:     /**
 67:      * Creates a new empty collection.
 68:      */ 
 69:     public AxisCollection() {
 70:         this.axesAtTop = new java.util.ArrayList();
 71:         this.axesAtBottom = new java.util.ArrayList();
 72:         this.axesAtLeft = new java.util.ArrayList();
 73:         this.axesAtRight = new java.util.ArrayList();
 74:     }
 75:     
 76:     /**
 77:      * Returns a list of the axes (if any) that need to be drawn at the top of 
 78:      * the plot area.
 79:      * 
 80:      * @return A list of axes.

 82:     public List getAxesAtTop() {
 83:         return this.axesAtTop;
 84:     }
 85:     
 86:    /**

 88:     * of the plot area.

 91:     */
 92:    public List getAxesAtBottom() {
 93:         return this.axesAtBottom;
 94:     }
 95:

102:     public List getAxesAtLeft() {

104:     }
105:

108:     * of the plot area.

112:     public List getAxesAtRight() {

115:     
116:     /**
117:      * Adds an axis to the collection.
118:      * 
119:      * @param axis  the axis (<code>null</code> not permitted).
120:      * @param edge  the edge of the plot that the axis should be drawn on

122:      */
123:     public void add(Axis axis, RectangleEdge edge) {
124:         if (axis == null) {
125:             throw new IllegalArgumentException("Null 'axis' argument.");   
126:         }
127:         if (edge == null) {
128:             throw new IllegalArgumentException("Null 'edge' argument.");   
129:         }
130:         if (edge == RectangleEdge.TOP) {
131:             this.axesAtTop.add(axis);
132:         }
133:         else if (edge == RectangleEdge.BOTTOM) {
134:             this.axesAtBottom.add(axis);
135:         }
136:         else if (edge == RectangleEdge.LEFT) {
137:             this.axesAtLeft.add(axis);
138:         }
139:         else if (edge == RectangleEdge.RIGHT) {
140:             this.axesAtRight.add(axis);
141:         }
142:     }
