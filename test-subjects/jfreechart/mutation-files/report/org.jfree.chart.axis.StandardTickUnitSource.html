 53: public class StandardTickUnitSource implements TickUnitSource, Serializable {

 55:     /** Constant for log(10.0). */

 67:      *
 68:      * @param unit  the unit (<code>null</code> not permitted).
 69:      *
 70:      * @return A tick unit that is larger than the supplied unit.
 71:      */

 73:         double x = unit.getSize();
