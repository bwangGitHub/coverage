566:             g2.draw(line);

568:             line.setLine(pt3, pt1);
569:             g2.draw(line);
570:         
571:             line.setLine(pt2, pt1);
572:             g2.draw(line);
573:         
574:             line.setLine(pt2, pt4);
575:             g2.draw(line);
576:
