041    package org.jfree.experimental.chart.annotations.junit;
042    
043    import java.io.ByteArrayInputStream;
044    import java.io.ByteArrayOutputStream;
045    import java.io.ObjectInput;
046    import java.io.ObjectInputStream;
047    import java.io.ObjectOutput;
048    import java.io.ObjectOutputStream;
049    
050    import junit.framework.Test;
051    import junit.framework.TestCase;
052    import junit.framework.TestSuite;
053    
054    import org.jfree.chart.title.TextTitle;
055    import org.jfree.experimental.chart.annotations.XYTitleAnnotation;
056    
057    /**
058     * Tests for the {@link XYTitleAnnotation} class.
059     */
060    public class XYTitleAnnotationTests extends TestCase {
061    
062        /**
063         * Returns the tests as a test suite.
064         *
065         * @return The test suite.
066         */
067        public static Test suite() {
068            return new TestSuite(XYTitleAnnotationTests.class);
069        }
070    
071        /**
072         * Constructs a new set of tests.
073         *
074         * @param name  the name of the tests.
075         */
076        public XYTitleAnnotationTests(String name) {
077            super(name);
078        }
079    
080        /**
081         * Confirm that the equals method can distinguish all the required fields.
082         */
083        public void testEquals() {
084            TextTitle t = new TextTitle("Title");
085            XYTitleAnnotation a1 = new XYTitleAnnotation(1.0, 2.0, t);
086            XYTitleAnnotation a2 = new XYTitleAnnotation(1.0, 2.0, t);
087            assertTrue(a1.equals(a2));
088            
089            a1 = new XYTitleAnnotation(1.1, 2.0, t);
090            assertFalse(a1.equals(a2));
091            a2 = new XYTitleAnnotation(1.1, 2.0, t);
092            assertTrue(a1.equals(a2));
093    
094            a1 = new XYTitleAnnotation(1.1, 2.2, t);
095            assertFalse(a1.equals(a2));
096            a2 = new XYTitleAnnotation(1.1, 2.2, t);
097            assertTrue(a1.equals(a2));
098            
099            TextTitle t2 = new TextTitle("Title 2");
100            a1 = new XYTitleAnnotation(1.1, 2.2, t2);
101            assertFalse(a1.equals(a2));
102            a2 = new XYTitleAnnotation(1.1, 2.2, t2);
103            assertTrue(a1.equals(a2));
104        }
105    
106        /**
107         * Two objects that are equal are required to return the same hashCode. 
108         */
109        public void testHashCode() {
110            TextTitle t = new TextTitle("Title");
111            XYTitleAnnotation a1 = new XYTitleAnnotation(1.0, 2.0, t);
112            XYTitleAnnotation a2 = new XYTitleAnnotation(1.0, 2.0, t);
113            assertTrue(a1.equals(a2));
114            int h1 = a1.hashCode();
115            int h2 = a2.hashCode();
116            assertEquals(h1, h2);
117        }
118        
119        /**
120         * Confirm that cloning works.
121         */
122        public void testCloning() {
123            TextTitle t = new TextTitle("Title");
124            XYTitleAnnotation a1 = new XYTitleAnnotation(1.0, 2.0, t);
125            XYTitleAnnotation a2 = null;
126            try {
127                a2 = (XYTitleAnnotation) a1.clone();
128            }
129            catch (CloneNotSupportedException e) {
130                e.printStackTrace();
131            }
132            assertTrue(a1 != a2);
133            assertTrue(a1.getClass() == a2.getClass());
134            assertTrue(a1.equals(a2));
135        }
136    
137        /**
138         * Serialize an instance, restore it, and check for equality.
139         */
140        public void testSerialization() {
141            TextTitle t = new TextTitle("Title");
142            XYTitleAnnotation a1 = new XYTitleAnnotation(1.0, 2.0, t);
143            XYTitleAnnotation a2 = null;
144            try {
145                ByteArrayOutputStream buffer = new ByteArrayOutputStream();
146                ObjectOutput out = new ObjectOutputStream(buffer);
147                out.writeObject(a1);
148                out.close();
149    
150                ObjectInput in = new ObjectInputStream(new ByteArrayInputStream(
151                        buffer.toByteArray()));
152                a2 = (XYTitleAnnotation) in.readObject();
153                in.close();
154            }
155            catch (Exception e) {
156                e.printStackTrace();
157            }
158            assertEquals(a1, a2);
159        }
160    
161    }
