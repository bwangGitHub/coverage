 24: /**
 25:  * An abstract shape is the lowlevel model for a shape.
 26:  *
 27:  * @author Glen Stampoultzis (glens at apache.org)
 28:  */
 29: public abstract class AbstractShape
 30: {
 31:     /**
 32:      * Create a new shape object used to create the escher records.
 33:      *
 34:      * @param hssfShape     The simple shape this is based on.
 35:      */
 36:     public static AbstractShape createShape( HSSFShape hssfShape, int shapeId )
 37:     {
 38:         AbstractShape shape;
 39:         if (hssfShape instanceof HSSFComment)
 40:         {
 41:             shape = new CommentShape( (HSSFComment)hssfShape, shapeId );
 42:         }
 43:         else if (hssfShape instanceof HSSFTextbox)
 44:         {
 45:             shape = new TextboxShape( (HSSFTextbox)hssfShape, shapeId );
 46:         }
 47:         else if (hssfShape instanceof HSSFPolygon)
 48:         {
 49:             shape = new PolygonShape( (HSSFPolygon) hssfShape, shapeId );
 50:         }
 51:         else if (hssfShape instanceof HSSFSimpleShape)
 52:         {
 53:             HSSFSimpleShape simpleShape = (HSSFSimpleShape) hssfShape;
 54:             switch ( simpleShape.getShapeType() )
 55:             {
 56:                 case HSSFSimpleShape.OBJECT_TYPE_PICTURE:
 57:                     shape = new PictureShape( simpleShape, shapeId );
 58:                     break;
 59:                 case HSSFSimpleShape.OBJECT_TYPE_LINE:
 60:                     shape = new LineShape( simpleShape, shapeId );
 61:                     break;
 62:                 case HSSFSimpleShape.OBJECT_TYPE_OVAL:
 63:                 case HSSFSimpleShape.OBJECT_TYPE_RECTANGLE:
 64:                     shape = new SimpleFilledShape( simpleShape, shapeId );
 65:                     break;
 66:                 case HSSFSimpleShape.OBJECT_TYPE_COMBO_BOX:
 67:                     shape = new ComboboxShape( simpleShape, shapeId );
 68:                     break;
 69:                 default:
 70:                     throw new IllegalArgumentException("Do not know how to handle this type of shape");
 71:             }
 72:         }
 73:         else
 74:         {
 75:             throw new IllegalArgumentException("Unknown shape type");
 76:         }
 77:         EscherSpRecord sp = shape.getSpContainer().getChildById(EscherSpRecord.RECORD_ID);
 78:         if (hssfShape.getParent() != null)
 79:             sp.setFlags(sp.getFlags() | EscherSpRecord.FLAG_CHILD);
 80:         return shape;
 81:     }

109:     /**
110:      * Add standard properties to the opt record.  These properties effect
111:      * all records.
112:      *
113:      * @param shape     The user model shape.
114:      * @param opt       The opt record to add the properties to.
115:      * @return          The number of options added.
116:      */
117:     protected int addStandardOptions( HSSFShape shape, EscherOptRecord opt )
118:     {
119:         opt.addEscherProperty( new EscherBoolProperty( EscherProperties.TEXT__SIZE_TEXT_TO_FIT_SHAPE, 0x080000 ) );
120: //        opt.addEscherProperty( new EscherBoolProperty( EscherProperties.TEXT__SIZE_TEXT_TO_FIT_SHAPE, 0x080008 ) );
121:         if ( shape.isNoFill() )
122:         {
123:             // Wonderful... none of the spec's give any clue as to what these constants mean.
124:             opt.addEscherProperty( new EscherBoolProperty( EscherProperties.FILL__NOFILLHITTEST, 0x00110000 ) );
125:         }
126:         else
127:         {
128:             opt.addEscherProperty( new EscherBoolProperty( EscherProperties.FILL__NOFILLHITTEST, 0x00010000 ) );
129:         }
130:         opt.addEscherProperty( new EscherRGBProperty( EscherProperties.FILL__FILLCOLOR, shape.getFillColor() ) );
131:         opt.addEscherProperty( new EscherBoolProperty( EscherProperties.GROUPSHAPE__PRINT, 0x080000 ) );
132:         opt.addEscherProperty( new EscherRGBProperty( EscherProperties.LINESTYLE__COLOR, shape.getLineStyleColor() ) );
133:         int options = 5;
134:         if (shape.getLineWidth() != HSSFShape.LINEWIDTH_DEFAULT)
135:         {
136:             opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.LINESTYLE__LINEWIDTH, shape.getLineWidth()));
137:             options++;
138:         }
139:         if (shape.getLineStyle() != HSSFShape.LINESTYLE_SOLID)
140:         {
141:             opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.LINESTYLE__LINEDASHING, shape.getLineStyle()));
142:             opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.LINESTYLE__LINEENDCAPSTYLE, 0));
143:             if (shape.getLineStyle() == HSSFShape.LINESTYLE_NONE)
144:                 opt.addEscherProperty( new EscherBoolProperty( EscherProperties.LINESTYLE__NOLINEDRAWDASH, 0x00080000));
145:             else
146:                 opt.addEscherProperty( new EscherBoolProperty( EscherProperties.LINESTYLE__NOLINEDRAWDASH, 0x00080008));
147:             options += 3;
148:         }
149:         opt.sortProperties();
150:         return options;   // # options added
151:     }

153:     /**
154:      * Generate id for the CommonObjectDataSubRecord that stands behind this shape
155:      *
156:      * <p>
157:      *     Typically objectId starts with 1, is unique among all Obj record within the worksheet stream
158:      *     and increments by 1 for every new shape.
159:      *     For most shapes there is a straight relationship between shapeId (generated by DDF) and objectId:
160:      * </p>
161:      * <p>
162:      *     shapeId  is unique and starts with 1024, hence objectId can be derived as <code>shapeId-1024</code>.
163:      * </p>
164:      * <p>
165:      *     An exception from this rule is the CellComment shape whose objectId start with 1024.
166:      *      See {@link CommentShape#getCmoObjectId(int)}
167:      * </p>
168:      *
169:      *
170:      *
171:      * @param  shapeId   shape id as generated by drawing manager
172:      * @return objectId  object id that will be assigned to the Obj record
173:      */
174:     int getCmoObjectId(int shapeId){
175:         return shapeId - 1024;
176:     }
