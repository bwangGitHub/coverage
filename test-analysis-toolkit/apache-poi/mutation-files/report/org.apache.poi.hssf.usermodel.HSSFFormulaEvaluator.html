164: 	/**
165: 	 * If cell contains a formula, the formula is evaluated and returned,
166: 	 * else the CellValue simply copies the appropriate cell value from
167: 	 * the cell and also its cell type. This method should be preferred over
168: 	 * evaluateInCell() when the call should not modify the contents of the
169: 	 * original cell.
170: 	 *
171: 	 * @param cell may be <code>null</code> signifying that the cell is not present (or blank)
172: 	 * @return <code>null</code> if the supplied cell is <code>null</code> or blank
173: 	 */
174: 	public CellValue evaluate(Cell cell) {
175: 		if (cell == null) {
176: 			return null;
177: 		}
178: 
179: 		switch (cell.getCellType()) {
180: 			case HSSFCell.CELL_TYPE_BOOLEAN:
181: 				return CellValue.valueOf(cell.getBooleanCellValue());
182: 			case HSSFCell.CELL_TYPE_ERROR:
183: 				return CellValue.getError(cell.getErrorCellValue());
184: 			case HSSFCell.CELL_TYPE_FORMULA:
185: 				return evaluateFormulaCellValue(cell);
186: 			case HSSFCell.CELL_TYPE_NUMERIC:
187: 				return new CellValue(cell.getNumericCellValue());
188: 			case HSSFCell.CELL_TYPE_STRING:
189: 				return new CellValue(cell.getRichStringCellValue().getString());
190: 			case HSSFCell.CELL_TYPE_BLANK:
191: 				return null;
192: 		}
193: 		throw new IllegalStateException("Bad cell type (" + cell.getCellType() + ")");
194: 	}

197: 	/**
198: 	 * If cell contains formula, it evaluates the formula, and saves the result of the formula. The
199: 	 * cell remains as a formula cell. If the cell does not contain formula, this method returns -1
200: 	 * and leaves the cell unchanged.
201: 	 *
202: 	 * Note that the type of the <em>formula result</em> is returned, so you know what kind of
203: 	 * cached formula result is also stored with  the formula.
204: 	 * <pre>
205: 	 * int evaluatedCellType = evaluator.evaluateFormulaCell(cell);
206: 	 * </pre>
207: 	 * Be aware that your cell will hold both the formula, and the result. If you want the cell
208: 	 * replaced with the result of the formula, use {@link #evaluateInCell(org.apache.poi.ss.usermodel.Cell)}
209: 	 * @param cell The cell to evaluate
210: 	 * @return -1 for non-formula cells, or the type of the <em>formula result</em>
211: 	 */
212: 	public int evaluateFormulaCell(Cell cell) {
213: 		if (cell == null || cell.getCellType() != HSSFCell.CELL_TYPE_FORMULA) {
214: 			return -1;
215: 		}
216: 		CellValue cv = evaluateFormulaCellValue(cell);
217: 		// cell remains a formula cell, but the cached value is changed
218: 		setCellValue(cell, cv);
219: 		return cv.getCellType();
220: 	}
221: 
222: 	/**
223: 	 * If cell contains formula, it evaluates the formula, and
224: 	 *  puts the formula result back into the cell, in place
225: 	 *  of the old formula.
226: 	 * Else if cell does not contain formula, this method leaves
227: 	 *  the cell unchanged.
228: 	 * Note that the same instance of HSSFCell is returned to
229: 	 * allow chained calls like:
230: 	 * <pre>
231: 	 * int evaluatedCellType = evaluator.evaluateInCell(cell).getCellType();
232: 	 * </pre>
233: 	 * Be aware that your cell value will be changed to hold the
234: 	 *  result of the formula. If you simply want the formula
235: 	 *  value computed for you, use {@link #evaluateFormulaCell(Cell)}}
236: 	 */
237: 	public HSSFCell evaluateInCell(Cell cell) {
238: 		if (cell == null) {
239: 			return null;
240: 		}
241: 		HSSFCell result = (HSSFCell) cell;
242: 		if (cell.getCellType() == HSSFCell.CELL_TYPE_FORMULA) {
243: 			CellValue cv = evaluateFormulaCellValue(cell);
244: 			setCellValue(cell, cv);
245: 			setCellType(cell, cv); // cell will no longer be a formula cell
246: 		}
247: 		return result;
248: 	}
249: 	private static void setCellType(Cell cell, CellValue cv) {
250: 		int cellType = cv.getCellType();
251: 		switch (cellType) {
252: 			case HSSFCell.CELL_TYPE_BOOLEAN:
253: 			case HSSFCell.CELL_TYPE_ERROR:
254: 			case HSSFCell.CELL_TYPE_NUMERIC:
255: 			case HSSFCell.CELL_TYPE_STRING:
256: 				cell.setCellType(cellType);
257: 				return;
258: 			case HSSFCell.CELL_TYPE_BLANK:
259: 				// never happens - blanks eventually get translated to zero
260: 			case HSSFCell.CELL_TYPE_FORMULA:
261: 				// this will never happen, we have already evaluated the formula
262: 		}
263: 		throw new IllegalStateException("Unexpected cell value type (" + cellType + ")");
264: 	}
265: 
266: 	private static void setCellValue(Cell cell, CellValue cv) {
267: 		int cellType = cv.getCellType();
268: 		switch (cellType) {
269: 			case HSSFCell.CELL_TYPE_BOOLEAN:
270: 				cell.setCellValue(cv.getBooleanValue());
271: 				break;
272: 			case HSSFCell.CELL_TYPE_ERROR:
273: 				cell.setCellErrorValue(cv.getErrorValue());
274: 				break;
275: 			case HSSFCell.CELL_TYPE_NUMERIC:
276: 				cell.setCellValue(cv.getNumberValue());
277: 				break;
278: 			case HSSFCell.CELL_TYPE_STRING:
279: 				cell.setCellValue(new HSSFRichTextString(cv.getStringValue()));
280: 				break;
281: 			case HSSFCell.CELL_TYPE_BLANK:
282: 				// never happens - blanks eventually get translated to zero
283: 			case HSSFCell.CELL_TYPE_FORMULA:
284: 				// this will never happen, we have already evaluated the formula
285: 			default:
286: 				throw new IllegalStateException("Unexpected cell value type (" + cellType + ")");
287: 		}
288: 	}

349: 	/**
350: 	 * Returns a CellValue wrapper around the supplied ValueEval instance.
351: 	 * @param eval
352: 	 */
353: 	private CellValue evaluateFormulaCellValue(Cell cell) {
354: 		ValueEval eval = _bookEvaluator.evaluate(new HSSFEvaluationCell((HSSFCell)cell));
355: 		if (eval instanceof NumberEval) {
356: 			NumberEval ne = (NumberEval) eval;
357: 			return new CellValue(ne.getNumberValue());
358: 		}
359: 		if (eval instanceof BoolEval) {
360: 			BoolEval be = (BoolEval) eval;
361: 			return CellValue.valueOf(be.getBooleanValue());
362: 		}
363: 		if (eval instanceof StringEval) {
364: 			StringEval ne = (StringEval) eval;
365: 			return new CellValue(ne.getStringValue());
366: 		}
367: 		if (eval instanceof ErrorEval) {
368: 			return CellValue.getError(((ErrorEval)eval).getErrorCode());
369: 		}
370: 		throw new RuntimeException("Unexpected eval class (" + eval.getClass().getName() + ")");
371: 	}
