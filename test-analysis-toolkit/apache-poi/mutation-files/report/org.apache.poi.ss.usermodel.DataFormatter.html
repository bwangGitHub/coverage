 42: /**
 43:  * DataFormatter contains methods for formatting the value stored in an
 44:  * Cell. This can be useful for reports and GUI presentations when you
 45:  * need to display data exactly as it appears in Excel. Supported formats
 46:  * include currency, SSN, percentages, decimals, dates, phone numbers, zip
 47:  * codes, etc.
 48:  * <p>
 49:  * Internally, formats will be implemented using subclasses of {@link Format}
 50:  * such as {@link DecimalFormat} and {@link SimpleDateFormat}. Therefore the
 51:  * formats used by this class must obey the same pattern rules as these Format
 52:  * subclasses. This means that only legal number pattern characters ("0", "#",
 53:  * ".", "," etc.) may appear in number formats. Other characters can be
 54:  * inserted <em>before</em> or <em> after</em> the number pattern to form a
 55:  * prefix or suffix.
 56:  * </p>
 57:  * <p>
 58:  * For example the Excel pattern <code>"$#,##0.00 "USD"_);($#,##0.00 "USD")"
 59:  * </code> will be correctly formatted as "$1,000.00 USD" or "($1,000.00 USD)".
 60:  * However the pattern <code>"00-00-00"</code> is incorrectly formatted by
 61:  * DecimalFormat as "000000--". For Excel formats that are not compatible with
 62:  * DecimalFormat, you can provide your own custom {@link Format} implementation
 63:  * via <code>DataFormatter.addFormat(String,Format)</code>. The following
 64:  * custom formats are already provided by this class:
 65:  * </p>
 66:  * <pre>
 67:  * <ul><li>SSN "000-00-0000"</li>
 68:  *     <li>Phone Number "(###) ###-####"</li>
 69:  *     <li>Zip plus 4 "00000-0000"</li>
 70:  * </ul>
 71:  * </pre>
 72:  * <p>
 73:  * If the Excel format pattern cannot be parsed successfully, then a default
 74:  * format will be used. The default number format will mimic the Excel General
 75:  * format: "#" for whole numbers and "#.##########" for decimal numbers. You
 76:  * can override the default format pattern with <code>
 77:  * DataFormatter.setDefaultNumberFormat(Format)</code>. <b>Note:</b> the
 78:  * default format will only be used when a Format cannot be created from the
 79:  * cell's data format string.
 80:  *
 81:  * <p>
 82:  * Note that by default formatted numeric values are trimmed.
 83:  * Excel formats can contain spacers and padding and the default behavior is to strip them off.
 84:  * </p>
 85:  * <p>Example:</p>
 86:  * <p>
 87:  * Consider a numeric cell with a value <code>12.343</code> and format <code>"##.##_ "</code>.
 88:  *  The trailing underscore and space ("_ ") in the format adds a space to the end and Excel formats this cell as <code>"12.34 "</code>,
 89:  *  but <code>DataFormatter</code> trims the formatted value and returns <code>"12.34"</code>.
 90:  * </p>
 91:  * You can enable spaces by passing the <code>emulateCsv=true</code> flag in the <code>DateFormatter</code> cosntructor.
 92:  * If set to true, then the output tries to conform to what you get when you take an xls or xlsx in Excel and Save As CSV file:
 93:  * <ul>
 94:  *  <li>returned values are not trimmed</li>
 95:  *  <li>Invalid dates are formatted as  255 pound signs ("#")</li>
 96:  *  <li>simulate Excel's handling of a format string of all # when the value is 0.
 97:  *   Excel will output "", <code>DataFormatter</code> will output "0".
 98:  * </ul>
 99:  * @author James May (james dot may at fmr dot com)
100:  * @author Robert Kish
101:   *

306:     private Format createFormat(double cellValue, int formatIndex, String sFormat) {
307:         String formatStr = sFormat;
308:         
309:         // Remove colour formatting if present
310:         Matcher colourM = colorPattern.matcher(formatStr);
311:         while(colourM.find()) {
312:            String colour = colourM.group();
313:            
314:            // Paranoid replacement...
315:            int at = formatStr.indexOf(colour);
316:            if(at == -1) break;
317:            String nFormatStr = formatStr.substring(0,at) +
318:               formatStr.substring(at+colour.length());
319:            if(nFormatStr.equals(formatStr)) break;
320: 
321:            // Try again in case there's multiple
322:            formatStr = nFormatStr;
323:            colourM = colorPattern.matcher(formatStr);
324:         }
325: 
326:         // try to extract special characters like currency
327:         Matcher m = specialPatternGroup.matcher(formatStr);
328:         while(m.find()) {
329:             String match = m.group();
330:             String symbol = match.substring(match.indexOf('$') + 1, match.indexOf('-'));
331:             if (symbol.indexOf('$') > -1) {
332:                 StringBuffer sb = new StringBuffer();
333:                 sb.append(symbol.substring(0, symbol.indexOf('$')));
334:                 sb.append('\\');
335:                 sb.append(symbol.substring(symbol.indexOf('$'), symbol.length()));
336:                 symbol = sb.toString();
337:             }
338:             formatStr = m.replaceAll(symbol);
339:             m = specialPatternGroup.matcher(formatStr);
340:         }
341: 
342:         if(formatStr == null || formatStr.trim().length() == 0) {
343:             return getDefaultFormat(cellValue);
344:         }
345: 
346:         if(DateUtil.isADateFormat(formatIndex,formatStr) &&
347:                 DateUtil.isValidExcelDate(cellValue)) {
348:             return createDateFormat(formatStr, cellValue);
349:         }
350:         
351:         // Excel supports fractions in format strings, which Java doesn't
352:         if (!formatStr.contains("-") &&
353:               (formatStr.indexOf("#/#") >= 0 && formatStr.indexOf("#/#") == formatStr.lastIndexOf("#/#")) ||
354:               (formatStr.indexOf("?/?") >= 0 && formatStr.indexOf("?/?") == formatStr.lastIndexOf("?/?"))) {

363:             return new ConstantStringFormat(cleanFormatForNumber(formatStr));

366:         return null;

370:         String formatStr = pFormatStr;
371:         formatStr = formatStr.replaceAll("\\\\-","-");
372:         formatStr = formatStr.replaceAll("\\\\,",",");
373:         formatStr = formatStr.replaceAll("\\\\\\.","."); // . is a special regexp char
374:         formatStr = formatStr.replaceAll("\\\\ "," ");
375:         formatStr = formatStr.replaceAll("\\\\/","/"); // weird: m\\/d\\/yyyy 
376:         formatStr = formatStr.replaceAll(";@", "");
377:         formatStr = formatStr.replaceAll("\"/\"", "/"); // "/" is escaped for no reason in: mm"/"dd"/"yyyy
378: 
379:         boolean hasAmPm = false;
380:         Matcher amPmMatcher = amPmPattern.matcher(formatStr);
381:         while (amPmMatcher.find()) {
382:             formatStr = amPmMatcher.replaceAll("@");
383:             hasAmPm = true;
384:             amPmMatcher = amPmPattern.matcher(formatStr);
385:         }
386:         formatStr = formatStr.replaceAll("@", "a");
387: 
388: 
389:         Matcher dateMatcher = daysAsText.matcher(formatStr);
390:         if (dateMatcher.find()) {
391:             String match = dateMatcher.group(0);
392:             formatStr = dateMatcher.replaceAll(match.toUpperCase().replaceAll("D", "E"));
393:         }
394: 
395:         // Convert excel date format to SimpleDateFormat.
396:         // Excel uses lower and upper case 'm' for both minutes and months.
397:         // From Excel help:
398:         /*
399:             The "m" or "mm" code must appear immediately after the "h" or"hh"
400:             code or immediately before the "ss" code; otherwise, Microsoft
401:             Excel displays the month instead of minutes."
402:           */
403: 
404:         StringBuffer sb = new StringBuffer();
405:         char[] chars = formatStr.toCharArray();
406:         boolean mIsMonth = true;
407:         List<Integer> ms = new ArrayList<Integer>();
408:         boolean isElapsed = false;
409:         for(int j=0; j<chars.length; j++) {
410:             char c = chars[j];
411:             if (c == '[' && !isElapsed) {
412:                 isElapsed = true;
413:                 mIsMonth = false;
414:                 sb.append(c);
415:             }
416:             else if (c == ']' && isElapsed) {
417:                 isElapsed = false;
418:                 sb.append(c);
419:             }
420:             else if (isElapsed) {

422:                     sb.append('H');
423:                 }
424:                 else if (c == 'm' || c == 'M') {
425:                     sb.append('m');
426:                 }
427:                 else if (c == 's' || c == 'S') {
428:                     sb.append('s');
429:                 }
430:                 else {
431:                     sb.append(c);
432:                 }
433:             }
434:             else if (c == 'h' || c == 'H') {

438:                 } else {

440:                 }

494:     private String cleanFormatForNumber(String formatStr) {
495:         StringBuffer sb = new StringBuffer(formatStr);
496: 
497:         if (emulateCsv) {
498:             // Requested spacers with "_" are replaced by a single space.
499:             // Full-column-width padding "*" are removed.
500:             // Not processing fractions at this time. Replace ? with space.
501:             // This matches CSV output.
502:             for (int i = 0; i < sb.length(); i++) {
503:                 char c = sb.charAt(i);
504:                 if (c == '_' || c == '*' || c == '?') {
505:                     if (i > 0 && sb.charAt((i - 1)) == '\\') {
506:                         // It's escaped, don't worry
507:                         continue;
508:                     }
509:                     if (c == '?') {
510:                         sb.setCharAt(i, ' ');
511:                     } else if (i < sb.length() - 1) {
512:                         // Remove the character we're supposed
513:                         //  to match the space of / pad to the
514:                         //  column width with
515:                         if (c == '_') {
516:                             sb.setCharAt(i + 1, ' ');
517:                         } else {
518:                             sb.deleteCharAt(i + 1);
519:                         }
520:                         // Remove the character too
521:                         sb.deleteCharAt(i);
522:                     }
523:                 }
524:             }
525:         } else {
526:             // If they requested spacers, with "_",
527:             //  remove those as we don't do spacing
528:             // If they requested full-column-width
529:             //  padding, with "*", remove those too
530:             for (int i = 0; i < sb.length(); i++) {
531:                 char c = sb.charAt(i);
532:                 if (c == '_' || c == '*') {
533:                     if (i > 0 && sb.charAt((i - 1)) == '\\') {
534:                         // It's escaped, don't worry
535:                         continue;
536:                     }
537:                     if (i < sb.length() - 1) {
538:                         // Remove the character we're supposed
539:                         //  to match the space of / pad to the
540:                         //  column width with
541:                         sb.deleteCharAt(i + 1);
542:                     }
543:                     // Remove the _ too
544:                     sb.deleteCharAt(i);
545:                 }
546:             }
547:         }
548: 
549:         // Now, handle the other aspects like 
550:         //  quoting and scientific notation
551:         for(int i = 0; i < sb.length(); i++) {
552:            char c = sb.charAt(i);
553:             // remove quotes and back slashes
554:             if (c == '\\' || c == '"') {
555:                 sb.deleteCharAt(i);
556:                 i--;
557: 
558:             // for scientific/engineering notation
559:             } else if (c == '+' && i > 0 && sb.charAt(i - 1) == 'E') {
560:                 sb.deleteCharAt(i);
561:                 i--;
562:             }
563:         }
564: 
565:         return sb.toString();
566:     }
