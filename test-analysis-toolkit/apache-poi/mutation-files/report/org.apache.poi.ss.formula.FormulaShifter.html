105: 		boolean refsWereChanged = false;

116: 	private Ptg adjustPtg(Ptg ptg, int currentExternSheetIx) {
117: 		switch(_mode){

119:                 return adjustPtgDueToRowMove(ptg, currentExternSheetIx);
120:             case Sheet:
121:                 return adjustPtgDueToShiftMove(ptg);
122:             default:
123:                 throw new IllegalStateException("Unsupported shift mode: " + _mode);
124:         }

181: 	private Ptg rowMoveRefPtg(RefPtgBase rptg) {
182: 		int refRow = rptg.getRow();
183: 		if (_firstMovedIndex <= refRow && refRow <= _lastMovedIndex) {
184: 			// Rows being moved completely enclose the ref.
185: 			// - move the area ref along with the rows regardless of destination
186: 			rptg.setRow(refRow + _amountToMove);
187: 			return rptg;
188: 		}
189: 		// else rules for adjusting area may also depend on the destination of the moved rows
190: 
191: 		int destFirstRowIndex = _firstMovedIndex + _amountToMove;
192: 		int destLastRowIndex = _lastMovedIndex + _amountToMove;
193: 
194: 		// ref is outside source rows
195: 		// check for clashes with destination
196: 
197: 		if (destLastRowIndex < refRow || refRow < destFirstRowIndex) {
198: 			// destination rows are completely outside ref
199: 			return null;
200: 		}
201: 
202: 		if (destFirstRowIndex <= refRow && refRow <= destLastRowIndex) {
203: 			// destination rows enclose the area (possibly exactly)
204: 			return createDeletedRef(rptg);
205: 		}
206: 		throw new IllegalStateException("Situation not covered: (" + _firstMovedIndex + ", " +
207: 					_lastMovedIndex + ", " + _amountToMove + ", " + refRow + ", " + refRow + ")");
208: 	}

210: 	private Ptg rowMoveAreaPtg(AreaPtgBase aptg) {
211: 		int aFirstRow = aptg.getFirstRow();
212: 		int aLastRow = aptg.getLastRow();
213: 		if (_firstMovedIndex <= aFirstRow && aLastRow <= _lastMovedIndex) {
214: 			// Rows being moved completely enclose the area ref.
215: 			// - move the area ref along with the rows regardless of destination
216: 			aptg.setFirstRow(aFirstRow + _amountToMove);
217: 			aptg.setLastRow(aLastRow + _amountToMove);
218: 			return aptg;
219: 		}
220: 		// else rules for adjusting area may also depend on the destination of the moved rows
221: 
222: 		int destFirstRowIndex = _firstMovedIndex + _amountToMove;
223: 		int destLastRowIndex = _lastMovedIndex + _amountToMove;
224: 
225: 		if (aFirstRow < _firstMovedIndex && _lastMovedIndex < aLastRow) {
226: 			// Rows moved were originally *completely* within the area ref
227: 
228: 			// If the destination of the rows overlaps either the top
229: 			// or bottom of the area ref there will be a change
230: 			if (destFirstRowIndex < aFirstRow && aFirstRow <= destLastRowIndex) {
231: 				// truncate the top of the area by the moved rows
232: 				aptg.setFirstRow(destLastRowIndex+1);
233: 				return aptg;
234: 			} else if (destFirstRowIndex <= aLastRow && aLastRow < destLastRowIndex) {
235: 				// truncate the bottom of the area by the moved rows
236: 				aptg.setLastRow(destFirstRowIndex-1);
237: 				return aptg;
238: 			}
239: 			// else - rows have moved completely outside the area ref,
240: 			// or still remain completely within the area ref
241: 			return null; // - no change to the area
242: 		}
243: 		if (_firstMovedIndex <= aFirstRow && aFirstRow <= _lastMovedIndex) {
244: 			// Rows moved include the first row of the area ref, but not the last row
245: 			// btw: (aLastRow > _lastMovedIndex)
246: 			if (_amountToMove < 0) {
247: 				// simple case - expand area by shifting top upward
248: 				aptg.setFirstRow(aFirstRow + _amountToMove);
249: 				return aptg;
250: 			}
251: 			if (destFirstRowIndex > aLastRow) {
252: 				// in this case, excel ignores the row move
253: 				return null;
254: 			}
255: 			int newFirstRowIx = aFirstRow + _amountToMove;
256: 			if (destLastRowIndex < aLastRow) {
257: 				// end of area is preserved (will remain exact same row)
258: 				// the top area row is moved simply
259: 				aptg.setFirstRow(newFirstRowIx);
260: 				return aptg;
261: 			}
262: 			// else - bottom area row has been replaced - both area top and bottom may move now
263: 			int areaRemainingTopRowIx = _lastMovedIndex + 1;
264: 			if (destFirstRowIndex > areaRemainingTopRowIx) {
265: 				// old top row of area has moved deep within the area, and exposed a new top row
266: 				newFirstRowIx = areaRemainingTopRowIx;
267: 			}
268: 			aptg.setFirstRow(newFirstRowIx);
269: 			aptg.setLastRow(Math.max(aLastRow, destLastRowIndex));
270: 			return aptg;
271: 		}
272: 		if (_firstMovedIndex <= aLastRow && aLastRow <= _lastMovedIndex) {
273: 			// Rows moved include the last row of the area ref, but not the first
274: 			// btw: (aFirstRow < _firstMovedIndex)
275: 			if (_amountToMove > 0) {
276: 				// simple case - expand area by shifting bottom downward
277: 				aptg.setLastRow(aLastRow + _amountToMove);
278: 				return aptg;
279: 			}
280: 			if (destLastRowIndex < aFirstRow) {
281: 				// in this case, excel ignores the row move
282: 				return null;
283: 			}
284: 			int newLastRowIx = aLastRow + _amountToMove;
285: 			if (destFirstRowIndex > aFirstRow) {
286: 				// top of area is preserved (will remain exact same row)
287: 				// the bottom area row is moved simply
288: 				aptg.setLastRow(newLastRowIx);
289: 				return aptg;
290: 			}
291: 			// else - top area row has been replaced - both area top and bottom may move now
292: 			int areaRemainingBottomRowIx = _firstMovedIndex - 1;
293: 			if (destLastRowIndex < areaRemainingBottomRowIx) {
294: 				// old bottom row of area has moved up deep within the area, and exposed a new bottom row
295: 				newLastRowIx = areaRemainingBottomRowIx;
296: 			}
297: 			aptg.setFirstRow(Math.min(aFirstRow, destFirstRowIndex));
298: 			aptg.setLastRow(newLastRowIx);
299: 			return aptg;
300: 		}
301: 		// else source rows include none of the rows of the area ref
302: 		// check for clashes with destination
303: 
304: 		if (destLastRowIndex < aFirstRow || aLastRow < destFirstRowIndex) {
305: 			// destination rows are completely outside area ref
306: 			return null;
307: 		}
308: 
309: 		if (destFirstRowIndex <= aFirstRow && aLastRow <= destLastRowIndex) {
310: 			// destination rows enclose the area (possibly exactly)
311: 			return createDeletedRef(aptg);
312: 		}
313: 
314: 		if (aFirstRow <= destFirstRowIndex && destLastRowIndex <= aLastRow) {
315: 			// destination rows are within area ref (possibly exact on top or bottom, but not both)
316: 			return null; // - no change to area
317: 		}

320: 			// dest rows overlap top of area
321: 			// - truncate the top
322: 			aptg.setFirstRow(destLastRowIndex+1);
