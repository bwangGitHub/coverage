 36: /**
 37:  * Identifies both built-in and user defined formats within a workbook.<p/>
 38:  * See {@link BuiltinFormats} for a list of supported built-in formats.<p/>
 39:  *
 40:  * <b>International Formats</b><br/>
 41:  * Since version 2003 Excel has supported international formats.  These are denoted
 42:  * with a prefix "[$-xxx]" (where xxx is a 1-7 digit hexadecimal number).
 43:  * See the Microsoft article
 44:  * <a href="http://office.microsoft.com/assistance/hfws.aspx?AssetID=HA010346351033&CTT=6&Origin=EC010272491033">
 45:  *   Creating international number formats
 46:  * </a> for more details on these codes.
 47:  *
 48:  * @author  Andrew C. Oliver (acoliver at apache dot org)
 49:  * @author  Shawn M. Laubach (slaubach at apache dot org)
 50:  */
 51: public final class HSSFDataFormat implements DataFormat {
 52: 	private static final String[] _builtinFormats = BuiltinFormats.getAll();
 53: 
 54: 	private final Vector<String> _formats = new Vector<String>();
 55: 	private final InternalWorkbook _workbook;
 56: 	private boolean _movedBuiltins = false;  // Flag to see if need to
 57: 	// check the built in list
 58: 	// or if the regular list
 59: 	// has all entries.
 60: 
 61: 	/**
 62: 	 * Constructs a new data formatter.  It takes a workbook to have
 63: 	 * access to the workbooks format records.
 64: 	 * @param workbook the workbook the formats are tied to.
 65: 	 */
 66: 	HSSFDataFormat(InternalWorkbook workbook) {
 67: 		_workbook = workbook;
 68: 
 69: 		@SuppressWarnings("unchecked")
 70: 		Iterator<FormatRecord> i = workbook.getFormats().iterator();
 71: 		while (i.hasNext()) {
 72: 			FormatRecord r = i.next();
 73: 			ensureFormatsSize(r.getIndexCode());
 74: 			_formats.set(r.getIndexCode(), r.getFormatString());
 75: 		}
 76: 	}
 77: 
 78: 	public static List<String> getBuiltinFormats() {
 79: 		return Arrays.asList(_builtinFormats);
 80: 	}

 82: 	/**
 83: 	 * get the format index that matches the given format string<p>
 84: 	 * Automatically converts "text" to excel's format string to represent text.
 85: 	 * @param format string matching a built in format
 86: 	 * @return index of format or -1 if undefined.
 87: 	 */
 88: 	public static short getBuiltinFormat(String format) {
 89: 		return (short) BuiltinFormats.getBuiltinFormat(format);
 90: 	}

 92: 	/**
 93: 	 * Get the format index that matches the given format
 94: 	 *  string, creating a new format entry if required.
 95: 	 * Aliases text to the proper format as required.
 96: 	 * @param pFormat string matching a built in format
 97: 	 * @return index of format.
 98: 	 */
 99: 	public short getFormat(String pFormat) {
100: 	   // Normalise the format string
101: 		String format;
102: 		if (pFormat.toUpperCase().equals("TEXT")) {
103: 			format = "@";
104: 		} else {
105: 			format = pFormat;
106: 		}
107: 
108: 		// Merge in the built in formats if we haven't already
109: 		if (!_movedBuiltins) {
110: 			for (int i=0; i<_builtinFormats.length; i++) {
111: 			   ensureFormatsSize(i);
112: 				if (_formats.get(i) == null) {
113: 				   _formats.set(i, _builtinFormats[i]);
114: 				} else {
115: 				   // The workbook overrides this default format
116: 				}
117: 			}
118: 			_movedBuiltins = true;
119: 		}
120: 		
121: 		// See if we can find it
122: 		for(int i=0; i<_formats.size(); i++) {
123: 		   if(format.equals(_formats.get(i))) {
124: 		      return (short)i;
125: 		   }
126: 		}
127: 
128: 		// We can't find it, so add it as a new one
129: 		short index = _workbook.getFormat(format, true);
130: 		ensureFormatsSize(index);
131: 		_formats.set(index, format);
132: 		return index;
133: 	}
134: 
135: 	/**
136: 	 * get the format string that matches the given format index
137: 	 * @param index of a format
138: 	 * @return string represented at index of format or null if there is not a  format at that index
139: 	 */
140: 	public String getFormat(short index) {
141: 		if (_movedBuiltins) {
142: 			return _formats.get(index);
143: 		}
144: 
145:         if(index == -1) {
146:             // YK: formatIndex can be -1, for example, for cell in column Y in test-data/spreadsheet/45322.xls
147:             // return null for those
148:             return null;
149:         }
150: 
151: 		String fmt = _formats.size() > index ? _formats.get(index) : null;
152: 		if (_builtinFormats.length > index && _builtinFormats[index] != null) {
153: 		   // It's in the built in range
154: 		   if (fmt != null) {
155: 		      // It's been overriden, use that value
156: 		      return fmt;
157: 		   } else {
158: 		      // Standard built in format
159: 	         return _builtinFormats[index];
160: 		   }
161: 		}
162: 		return fmt;
163: 	}
164:

167: 	 * @param index of a built in format
168: 	 * @return string represented at index of format or null if there is not a builtin format at that index
169: 	 */
170: 	public static String getBuiltinFormat(short index) {
171: 		return BuiltinFormats.getBuiltinFormat(index);
172: 	}

174: 	/**
175: 	 * get the number of built-in and reserved builtinFormats
176: 	 * @return number of built-in and reserved builtinFormats
177: 	 */
178: 	public static int getNumberOfBuiltinBuiltinFormats() {
179: 		return _builtinFormats.length;
180: 	}

183: 	 * Ensures that the formats list can hold entries

185: 	 */
186: 	private void ensureFormatsSize(int index) {
187: 	   if(_formats.size() <= index) {
188: 	      _formats.setSize(index+1);
189: 	   }
190: 	}
