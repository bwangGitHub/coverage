 55: 	 * Create an InputStream from the specified DocumentEntry

 62: 	public NDocumentInputStream(DocumentEntry document) throws IOException {
 63: 		if (!(document instanceof DocumentNode)) {
 64: 			throw new IOException("Cannot open internal document storage, " + document + " not a Document Node");
 65: 		}
 66: 		_current_offset = 0;
 67: 		_current_block_count = 0;
 68: 		_marked_offset = 0;
 69: 		_marked_offset_count = 0;
 70: 		_document_size = document.getSize();
 71: 		_closed = false;

 73:       DocumentNode doc = (DocumentNode)document;
 74: 		DocumentProperty property = (DocumentProperty)doc.getProperty();
 75: 		_document = new NPOIFSDocument(
 76: 		      property, 
 77: 		      ((DirectoryNode)doc.getParent()).getNFileSystem()
 78: 		);
 79: 		_data = _document.getBlockIterator();

111:    @Override
112: 	public void mark(int ignoredReadlimit) {

117:    @Override
118: 	public int read() throws IOException {
119: 		dieIfClosed();
120: 		if (atEOD()) {
121: 			return EOF;
122: 		}
123: 		byte[] b = new byte[1];
124: 		int result = read(b, 0, 1);
125: 		if(result >= 0) {
126: 		   if(b[0] < 0) {
127: 		      return b[0]+256;
128: 		   }
129: 		   return b[0];
130: 		}
131: 		return result;
132: 	}

134:    @Override
135: 	public int read(byte[] b, int off, int len) throws IOException {
136: 		dieIfClosed();
137: 		if (b == null) {
138: 			throw new IllegalArgumentException("buffer must not be null");
139: 		}
140: 		if (off < 0 || len < 0 || b.length < off + len) {
141: 			throw new IndexOutOfBoundsException("can't read past buffer boundaries");
142: 		}
143: 		if (len == 0) {
144: 			return 0;
145: 		}
146: 		if (atEOD()) {
147: 			return EOF;
148: 		}
149: 		int limit = Math.min(available(), len);
150: 		readFully(b, off, limit);
151: 		return limit;
152: 	}

154: 	/**
155: 	 * Repositions this stream to the position at the time the mark() method was
156: 	 * last called on this input stream. If mark() has not been called this
157: 	 * method repositions the stream to its beginning.
158: 	 */
159:    @Override
160: 	public void reset() {
161: 	   // Special case for reset to the start
162: 	   if(_marked_offset == 0 && _marked_offset_count == 0) {
163: 	      _current_block_count = _marked_offset_count;
164: 	      _current_offset = _marked_offset;
165: 	      _data = _document.getBlockIterator();
166: 	      _buffer = null;
167: 	      return;
168: 	   }
169: 	   
170: 		// Start again, then wind on to the required block
171: 		_data = _document.getBlockIterator();
172: 		_current_offset = 0;
173: 		for(int i=0; i<_marked_offset_count; i++) {
174: 		   _buffer = _data.next();
175: 		   _current_offset += _buffer.remaining();
176: 		}
177: 		
178:       _current_block_count = _marked_offset_count;
179:       
180:       // Do we need to position within it?
181:       if(_current_offset != _marked_offset) {
182:    		// Grab the right block
183:          _buffer = _data.next();
184:          _current_block_count++;
185:          
186:    		// Skip to the right place in it
187:          // (It should be positioned already at the start of the block,
188:          //  we need to move further inside the block)
189:          int skipBy = _marked_offset - _current_offset;
190:    		_buffer.position(_buffer.position() + skipBy);
191:       }
192: 
193:       // All done
194:       _current_offset = _marked_offset;
195: 	}

197:    @Override
198: 	public long skip(long n) throws IOException {
199: 		dieIfClosed();
200: 		if (n < 0) {
201: 			return 0;
202: 		}
203: 		int new_offset = _current_offset + (int) n;
204: 
205: 		if (new_offset < _current_offset) {
206: 			// wrap around in converting a VERY large long to an int
207: 			new_offset = _document_size;
208: 		} else if (new_offset > _document_size) {
209: 			new_offset = _document_size;
210: 		}
211: 		
212: 		long rval = new_offset - _current_offset;
213: 		
214: 		// TODO Do this better
215: 		byte[] skip = new byte[(int)rval];
216: 		readFully(skip);
217: 		return rval;
218: 	}

220: 	private void dieIfClosed() throws IOException {
221: 		if (_closed) {

226: 	private boolean atEOD() {
227: 		return _current_offset == _document_size;
228: 	}

230: 	private void checkAvaliable(int requestedSize) {
231: 		if (_closed) {
232: 			throw new IllegalStateException("cannot perform requested operation on a closed stream");
233: 		}
234: 		if (requestedSize > _document_size - _current_offset) {
235: 			throw new RuntimeException("Buffer underrun - requested " + requestedSize
236: 					+ " bytes but " + (_document_size - _current_offset) + " was available");
237: 		}
238: 	}

240:    @Override
241: 	public void readFully(byte[] buf, int off, int len) {
242: 		checkAvaliable(len);
243: 
244: 		int read = 0;
245: 		while(read < len) {
246: 		   if(_buffer == null || _buffer.remaining() == 0) {
247: 		      _current_block_count++;
248: 		      _buffer = _data.next();
249: 		   }
250: 		   
251: 		   int limit = Math.min(len-read, _buffer.remaining());
252: 		   _buffer.get(buf, off+read, limit);
253:          _current_offset += limit;
254: 		   read += limit;
255: 		}
256: 	}
