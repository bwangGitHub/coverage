 28: /**
 29:  * This class embodies the Property Table for the filesystem,
 30:  *  which looks up entries in the filesystem to their
 31:  *  chain of blocks.
 32:  * This is the core support, there are implementations
 33:  *  for the different block schemes as needed.
 34:  */
 35: public abstract class PropertyTableBase implements BATManaged {
 36:     private   final HeaderBlock    _header_block;
 37:     protected final List<Property> _properties;
 38: 
 39:     public PropertyTableBase(final HeaderBlock header_block)
 40:     {
 41:         _header_block = header_block;
 42:         _properties  = new ArrayList<Property>();
 43:         addProperty(new RootProperty());
 44:     }
 45: 
 46:     /**
 47:      * Reading constructor (used when we've read in a file and we want
 48:      * to extract the property table from it). Populates the
 49:      * properties thoroughly
 50:      *
 51:      * @param header_block the first block to read from
 52:      * @param properties the list to populate
 53:      *
 54:      * @exception IOException if anything goes wrong (which should be
 55:      *            a result of the input being NFG)
 56:      */
 57:     public PropertyTableBase(final HeaderBlock header_block,
 58:                          final List<Property> properties)
 59:         throws IOException
 60:     {
 61:         _header_block = header_block;
 62:         _properties   = properties;
 63:         populatePropertyTree( (DirectoryProperty)_properties.get(0));
 64:     }
 65: 
 66:     /**
 67:      * Add a property to the list of properties we manage
 68:      *
 69:      * @param property the new Property to manage
 70:      */
 71:     public void addProperty(Property property)
 72:     {
 73:         _properties.add(property);
 74:     }
 75: 
 76:     /**
 77:      * Remove a property from the list of properties we manage
 78:      *
 79:      * @param property the Property to be removed
 80:      */
 81:     public void removeProperty(final Property property)
 82:     {
 83:         _properties.remove(property);
 84:     }
 85: 
 86:     /**
 87:      * Get the root property
 88:      *
 89:      * @return the root property
 90:      */
 91:     public RootProperty getRoot()
 92:     {
 93:         // it's always the first element in the List
 94:         return ( RootProperty ) _properties.get(0);
 95:     }
 96:     
 97:     private void populatePropertyTree(DirectoryProperty root)
 98:         throws IOException
 99:     {
100:         int index = root.getChildIndex();
101: 
102:         if (!Property.isValidIndex(index))
103:         {
104: 
105:             // property has no children
106:             return;
107:         }
108:         Stack<Property> children = new Stack<Property>();
109: 
110:         children.push(_properties.get(index));
111:         while (!children.empty())
112:         {
113:             Property property = children.pop();
114: 
115:             root.addChild(property);
116:             if (property.isDirectory())
117:             {
118:                 populatePropertyTree(( DirectoryProperty ) property);
119:             }
120:             index = property.getPreviousChildIndex();
121:             if (Property.isValidIndex(index))
122:             {
123:                 children.push(_properties.get(index));
124:             }
125:             index = property.getNextChildIndex();
126:             if (Property.isValidIndex(index))
127:             {
128:                 children.push(_properties.get(index));
129:             }
130:         }
131:     }
132: 
133:     /**
134:      * Get the start block for the property table
135:      *
136:      * @return start block index
137:      */
138:     public int getStartBlock()
139:     {
140:         return _header_block.getPropertyStart();
141:     }
142: 
143:     /**
144:      * Set the start block for this instance
145:      *
146:      * @param index index into the array of BigBlock instances making
147:      *              up the the filesystem
148:      */
149:     public void setStartBlock(final int index)
150:     {
151:         _header_block.setPropertyStart(index);
152:     }
153: }
