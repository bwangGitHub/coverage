 32: /**
 33:  * When working with the EventUserModel, if you want to
 34:  *  process formulas, you need an instance of
 35:  *  {@link InternalWorkbook} to pass to a {@link HSSFWorkbook},
 36:  *  to finally give to {@link HSSFFormulaParser},
 37:  *  and this will build you stub ones.
 38:  * Since you're working with the EventUserModel, you
 39:  *  wouldn't want to get a full {@link InternalWorkbook} and
 40:  *  {@link HSSFWorkbook}, as they would eat too much memory.
 41:  *  Instead, you should collect a few key records as they
 42:  *  go past, then call this once you have them to build a
 43:  *  stub {@link InternalWorkbook}, and from that a stub
 44:  *  {@link HSSFWorkbook}, to use with the {@link HSSFFormulaParser}.
 45:  *
 46:  * The records you should collect are:
 47:  *  * {@link ExternSheetRecord}
 48:  *  * {@link BoundSheetRecord}
 49:  * You should probably also collect {@link SSTRecord},
 50:  *  but it's not required to pass this in.
 51:  *
 52:  * To help, this class includes a HSSFListener wrapper
 53:  *  that will do the collecting for you.
 54:  */
 55: public class EventWorkbookBuilder {
 56: 
 57: 
 58: 	/**
 59: 	 * Creates a stub Workbook from the supplied records,
 60: 	 *  suitable for use with the {@link HSSFFormulaParser}
 61: 	 * @param externs The ExternSheetRecords in your file
 62: 	 * @param bounds The BoundSheetRecords in your file
 63: 	 * @param sst The SSTRecord in your file.
 64: 	 * @return A stub Workbook suitable for use with {@link HSSFFormulaParser}
 65: 	 */
 66: 	public static InternalWorkbook createStubWorkbook(ExternSheetRecord[] externs,
 67: 			BoundSheetRecord[] bounds, SSTRecord sst) {
 68: 		List wbRecords = new ArrayList();
 69: 
 70: 		// Core Workbook records go first
 71: 		if(bounds != null) {
 72: 			for(int i=0; i<bounds.length; i++) {
 73: 				wbRecords.add(bounds[i]);
 74: 			}
 75: 		}
 76: 		if(sst != null) {
 77: 			wbRecords.add(sst);
 78: 		}
 79: 
 80: 		// Now we can have the ExternSheetRecords,
 81: 		//  preceded by a SupBookRecord
 82: 		if(externs != null) {
 83: 			wbRecords.add(SupBookRecord.createInternalReferences(
 84: 					(short)externs.length));
 85: 			for(int i=0; i<externs.length; i++) {
 86: 				wbRecords.add(externs[i]);
 87: 			}
 88: 		}
 89: 
 90: 		// Finally we need an EoF record
 91: 		wbRecords.add(EOFRecord.instance);
 92: 
 93: 		return InternalWorkbook.createWorkbook(wbRecords);
 94: 	}
 95: 
 96: 	/**
 97: 	 * Creates a stub workbook from the supplied records,
 98: 	 *  suitable for use with the {@link HSSFFormulaParser}
 99: 	 * @param externs The ExternSheetRecords in your file
100: 	 * @param bounds The BoundSheetRecords in your file
101: 	 * @return A stub Workbook suitable for use with {@link HSSFFormulaParser}
102: 	 */
103: 	public static InternalWorkbook createStubWorkbook(ExternSheetRecord[] externs,
104: 			BoundSheetRecord[] bounds) {
105: 		return createStubWorkbook(externs, bounds, null);
106: 	}
107: 
108: 
109: 	/**
110: 	 * A wrapping HSSFListener which will collect
111: 	 *  {@link BoundSheetRecord}s and {@link ExternSheetRecord}s as
112: 	 *  they go past, so you can create a Stub {@link InternalWorkbook} from
113: 	 *  them once required.
114: 	 */
115: 	public static class SheetRecordCollectingListener implements HSSFListener {
116: 		private HSSFListener childListener;
117: 		private List boundSheetRecords = new ArrayList();
118: 		private List externSheetRecords = new ArrayList();
119: 		private SSTRecord sstRecord = null;
120: 
121: 		public SheetRecordCollectingListener(HSSFListener childListener) {
122: 			this.childListener = childListener;
123: 		}
124: 
125: 
126: 		public BoundSheetRecord[] getBoundSheetRecords() {
127: 			return (BoundSheetRecord[])boundSheetRecords.toArray(
128: 					new BoundSheetRecord[boundSheetRecords.size()]
129: 			);
130: 		}
131: 		public ExternSheetRecord[] getExternSheetRecords() {
132: 			return (ExternSheetRecord[])externSheetRecords.toArray(
133: 					new ExternSheetRecord[externSheetRecords.size()]
134: 			);
135: 		}
136: 		public SSTRecord getSSTRecord() {
137: 			return sstRecord;
138: 		}
139: 
140: 		public HSSFWorkbook getStubHSSFWorkbook() {
141: 			return HSSFWorkbook.create(getStubWorkbook());
142: 		}
143: 		public InternalWorkbook getStubWorkbook() {
144: 			return createStubWorkbook(
145: 					getExternSheetRecords(), getBoundSheetRecords(),
146: 					getSSTRecord()
147: 			);
148: 		}
149: 
150: 
151: 		/**
152: 		 * Process this record ourselves, and then
153: 		 *  pass it on to our child listener
154: 		 */
155: 		public void processRecord(Record record) {
156: 			// Handle it ourselves
157: 			processRecordInternally(record);
158: 
159: 			// Now pass on to our child
160: 			childListener.processRecord(record);
161: 		}
162: 
163: 		/**
164: 		 * Process the record ourselves, but do not
165: 		 *  pass it on to the child Listener.
166: 		 */
167: 		public void processRecordInternally(Record record) {
168: 			if(record instanceof BoundSheetRecord) {
169: 				boundSheetRecords.add(record);
170: 			}
171: 			else if(record instanceof ExternSheetRecord) {
172: 				externSheetRecords.add(record);
173: 			}
174: 			else if(record instanceof SSTRecord) {
175: 				sstRecord = (SSTRecord)record;
176: 			}
177: 		}
178: 	}
