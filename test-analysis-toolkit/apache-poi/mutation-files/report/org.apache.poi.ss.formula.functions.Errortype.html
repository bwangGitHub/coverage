 27: /**
 28:  * Implementation for the ERROR.TYPE() Excel function.
 29:  * <p>
 30:  * <b>Syntax:</b><br/>
 31:  * <b>ERROR.TYPE</b>(<b>errorValue</b>)</p>
 32:  * <p>
 33:  * Returns a number corresponding to the error type of the supplied argument.<p/>
 34:  * <p>
 35:  *    <table border="1" cellpadding="1" cellspacing="1" summary="Return values for ERROR.TYPE()">
 36:  *      <tr><td>errorValue</td><td>Return Value</td></tr>
 37:  *      <tr><td>#NULL!</td><td>1</td></tr>
 38:  *      <tr><td>#DIV/0!</td><td>2</td></tr>
 39:  *      <tr><td>#VALUE!</td><td>3</td></tr>
 40:  *      <tr><td>#REF!</td><td>4</td></tr>
 41:  *      <tr><td>#NAME?</td><td>5</td></tr>
 42:  *      <tr><td>#NUM!</td><td>6</td></tr>
 43:  *      <tr><td>#N/A!</td><td>7</td></tr>
 44:  *      <tr><td>everything else</td><td>#N/A!</td></tr>
 45:  *    </table>
 46:  *
 47:  * Note - the results of ERROR.TYPE() are different to the constants defined in
 48:  * <tt>ErrorConstants</tt>.
 49:  * </p>
 50:  *
 51:  * @author Josh Micich
 52:  */
 53: public final class Errortype extends Fixed1ArgFunction {
 54: 
 55: 	public ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0) {
 56: 
 57: 		try {
 58: 			OperandResolver.getSingleValue(arg0, srcRowIndex, srcColumnIndex);
 59: 			return ErrorEval.NA;
 60: 		} catch (EvaluationException e) {
 61: 			int result = translateErrorCodeToErrorTypeValue(e.getErrorEval().getErrorCode());
 62: 			return new NumberEval(result);
 63: 		}
 64: 	}
 65: 
 66: 	private int translateErrorCodeToErrorTypeValue(int errorCode) {
 67: 		switch (errorCode) {
 68: 			case ErrorConstants.ERROR_NULL:  return 1;
 69: 			case ErrorConstants.ERROR_DIV_0: return 2;
 70: 			case ErrorConstants.ERROR_VALUE: return 3;
 71: 			case ErrorConstants.ERROR_REF:   return 4;
 72: 			case ErrorConstants.ERROR_NAME:  return 5;
 73: 			case ErrorConstants.ERROR_NUM:   return 6;
 74: 			case ErrorConstants.ERROR_NA :   return 7;
 75: 		}
 76: 		throw new IllegalArgumentException("Invalid error code (" + errorCode + ")");
 77: 	}
