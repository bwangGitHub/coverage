114: 	private static final long EXCEL_NAN_BITS = 0xFFFF0420003C0000L;
115: 	private static final int MAX_TEXT_LEN = 20;

117: 	private NumberToTextConverter() {
118: 		// no instances of this class

121: 	/**
122: 	 * Converts the supplied <tt>value</tt> to the text representation that Excel would give if
123: 	 * the value were to appear in an unformatted cell, or as a literal number in a formula.<br/>
124: 	 * Note - the results from this method differ slightly from those of <tt>Double.toString()</tt>
125: 	 * In some special cases Excel behaves quite differently.  This function attempts to reproduce
126: 	 * those results.
127: 	 */
128: 	public static String toText(double value) {
129: 		return rawDoubleBitsToText(Double.doubleToLongBits(value));
130: 	}
131: 	/* package */ static String rawDoubleBitsToText(long pRawBits) {
132: 
133: 		long rawBits = pRawBits;
134: 		boolean isNegative = rawBits < 0; // sign bit is in the same place for long and double
135: 		if (isNegative) {
136: 			rawBits &= 0x7FFFFFFFFFFFFFFFL;
137: 		}
138: 		if (rawBits == 0) {
139: 			return isNegative ? "-0" : "0";
140: 		}
141: 		ExpandedDouble ed = new ExpandedDouble(rawBits);
142: 		if (ed.getBinaryExponent() < -1022) {
143: 			// value is 'denormalised' which means it is less than 2^-1022
144: 			// excel displays all these numbers as zero, even though calculations work OK
145: 			return isNegative ? "-0" : "0";
146: 		}
147: 		if (ed.getBinaryExponent() == 1024) {
148: 			// Special number NaN /Infinity
149: 			// Normally one would not create HybridDecimal objects from these values
150: 			// except in these cases Excel really tries to render them as if they were normal numbers
151: 			if(rawBits == EXCEL_NAN_BITS) {
152: 				return "3.484840871308E+308";
153: 			}
154: 			// This is where excel really gets it wrong
155: 			// Special numbers like Infinity and NaN are interpreted according to
156: 			// the standard rules below.
157: 			isNegative = false; // except that the sign bit is ignored
158: 		}
159: 		NormalisedDecimal nd = ed.normaliseBaseTen();
160: 		StringBuilder sb = new StringBuilder(MAX_TEXT_LEN+1);
161: 		if (isNegative) {
162: 			sb.append('-');
163: 		}
164: 		convertToText(sb, nd);
165: 		return sb.toString();
166: 	}
167: 	private static void convertToText(StringBuilder sb, NormalisedDecimal pnd) {
168: 		NormalisedDecimal rnd = pnd.roundUnits();
169: 		int decExponent = rnd.getDecimalExponent();
170: 		String decimalDigits;
171: 		if (Math.abs(decExponent)>98) {
172: 			decimalDigits = rnd.getSignificantDecimalDigitsLastDigitRounded();
173: 			if (decimalDigits.length() == 16) {
174: 				// rounding caused carry
175: 				decExponent++;
176: 			}
177: 		} else {
178: 			decimalDigits = rnd.getSignificantDecimalDigits();
179: 		}
180: 		int countSigDigits = countSignifantDigits(decimalDigits);
181: 		if (decExponent < 0) {
182: 			formatLessThanOne(sb, decimalDigits, decExponent, countSigDigits);
183: 		} else {
184: 			formatGreaterThanOne(sb, decimalDigits, decExponent, countSigDigits);
185: 		}
186: 	}
187: 
188: 	private static void formatLessThanOne(StringBuilder sb, String decimalDigits, int decExponent,
189: 			int countSigDigits) {
190: 		int nLeadingZeros = -decExponent - 1;
191: 		int normalLength = 2 + nLeadingZeros + countSigDigits; // 2 == "0.".length()
192: 
193: 		if (needsScientificNotation(normalLength)) {
194: 			sb.append(decimalDigits.charAt(0));
195: 			if (countSigDigits > 1) {
196:     			sb.append('.');
197:     			sb.append(decimalDigits.subSequence(1, countSigDigits));
198: 			}
199: 			sb.append("E-");
200: 			appendExp(sb, -decExponent);
201: 			return;
202: 		}
203: 		sb.append("0.");
204: 		for (int i=nLeadingZeros; i>0; i--) {
205: 			sb.append('0');
206: 		}
207: 		sb.append(decimalDigits.subSequence(0, countSigDigits));
208: 	}
209: 
210: 	private static void formatGreaterThanOne(StringBuilder sb, String decimalDigits, int decExponent, int countSigDigits) {
211: 
212: 		if (decExponent > 19) {
213: 			// scientific notation
214: 			sb.append(decimalDigits.charAt(0));
215: 			if (countSigDigits>1) {
216: 				sb.append('.');
217: 				sb.append(decimalDigits.subSequence(1, countSigDigits));
218: 			}
219: 			sb.append("E+");
220: 			appendExp(sb, decExponent);
221: 			return;
222: 		}
223: 		int nFractionalDigits = countSigDigits - decExponent-1;
224: 		if (nFractionalDigits > 0) {
225: 			sb.append(decimalDigits.subSequence(0, decExponent+1));
226: 			sb.append('.');
227: 			sb.append(decimalDigits.subSequence(decExponent+1, countSigDigits));
228: 			return;
229: 		}
230: 		sb.append(decimalDigits.subSequence(0, countSigDigits));
231: 		for (int i=-nFractionalDigits; i>0; i--) {
232: 			sb.append('0');
233: 		}
234: 	}
235: 
236: 	private static boolean needsScientificNotation(int nDigits) {
237: 		return nDigits > MAX_TEXT_LEN;
238: 	}
239: 
240: 	private static int countSignifantDigits(String sb) {
241: 		int result=sb.length()-1;
242: 		while(sb.charAt(result) == '0') {
243: 			result--;
244: 			if(result < 0) {
245: 				throw new RuntimeException("No non-zero digits found");
246: 			}
247: 		}
248: 		return result + 1;
249: 	}
250: 
251: 	private static void appendExp(StringBuilder sb, int val) {
252: 		if(val < 10) {
253: 			sb.append('0');
254: 			sb.append((char)('0' + val));
255: 			return;
256: 		}
257: 		sb.append(val);
258: 	}
