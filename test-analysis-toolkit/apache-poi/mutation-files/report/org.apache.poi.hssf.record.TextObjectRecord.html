 33:  * contains the formatting runs.<p/>

 35:  * @author Glen Stampoultzis (glens at apache.org)

 88: 	public TextObjectRecord(RecordInputStream in) {
 89: 		field_1_options = in.readUShort();
 90: 		field_2_textOrientation = in.readUShort();
 91: 		field_3_reserved4 = in.readUShort();
 92: 		field_4_reserved5 = in.readUShort();
 93: 		field_5_reserved6 = in.readUShort();
 94: 		int field_6_textLength = in.readUShort();
 95: 		int field_7_formattingDataLength = in.readUShort();
 96: 		field_8_reserved7 = in.readInt();
 97: 
 98: 		if (in.remaining() > 0) {
 99: 			// Text Objects can have simple reference formulas
100: 			// (This bit not mentioned in the MS document)
101: 			if (in.remaining() < 11) {
102: 				throw new RecordFormatException("Not enough remaining data for a link formula");
103: 			}
104: 			int formulaSize = in.readUShort();
105: 			_unknownPreFormulaInt = in.readInt();
106: 			Ptg[] ptgs = Ptg.readTokens(formulaSize, in);
107: 			if (ptgs.length != 1) {
108: 				throw new RecordFormatException("Read " + ptgs.length
109: 						+ " tokens but expected exactly 1");
110: 			}
111: 			_linkRefPtg = (OperandPtg) ptgs[0];
112: 			if (in.remaining() > 0) {
113: 				_unknownPostFormulaByte = Byte.valueOf(in.readByte());
114: 			} else {
115: 				_unknownPostFormulaByte = null;
116: 			}
117: 		} else {
118: 			_linkRefPtg = null;
119: 		}
120: 		if (in.remaining() > 0) {
121: 			throw new RecordFormatException("Unused " + in.remaining() + " bytes at end of record");
122: 		}
123: 
124: 		String text;
125: 		if (field_6_textLength > 0) {
126: 			text = readRawString(in, field_6_textLength);
127: 		} else {
128: 			text = "";
129: 		}
130: 		_text = new HSSFRichTextString(text);
131: 
132: 		if (field_7_formattingDataLength > 0) {
133: 			processFontRuns(in, _text, field_7_formattingDataLength);
134: 		}
135: 	}

137: 	private static String readRawString(RecordInputStream in, int textLength) {
138: 		byte compressByte = in.readByte();
139: 		boolean isCompressed = (compressByte & 0x01) == 0;
140: 		if (isCompressed) {
141: 			return in.readCompressedUnicode(textLength);
142: 		}
143: 		return in.readUnicodeLEString(textLength);
144: 	}

146: 	private static void processFontRuns(RecordInputStream in, HSSFRichTextString str,
147: 			int formattingRunDataLength) {
148: 		if (formattingRunDataLength % FORMAT_RUN_ENCODED_SIZE != 0) {
149: 			throw new RecordFormatException("Bad format run data length " + formattingRunDataLength
150: 					+ ")");
151: 		}
152: 		int nRuns = formattingRunDataLength / FORMAT_RUN_ENCODED_SIZE;
153: 		for (int i = 0; i < nRuns; i++) {
154: 			short index = in.readShort();
155: 			short iFont = in.readShort();
156: 			in.readInt(); // skip reserved.
157: 			str.applyFont(index, str.length(), iFont);
158: 		}
159: 	}

165: 	private void serializeTXORecord(ContinuableRecordOutput out) {
166: 
167: 		out.writeShort(field_1_options);
168: 		out.writeShort(field_2_textOrientation);
169: 		out.writeShort(field_3_reserved4);
170: 		out.writeShort(field_4_reserved5);
171: 		out.writeShort(field_5_reserved6);
172: 		out.writeShort(_text.length());
173: 		out.writeShort(getFormattingDataLength());
174: 		out.writeInt(field_8_reserved7);
175: 
176: 		if (_linkRefPtg != null) {
177: 			int formulaSize = _linkRefPtg.getSize();
178: 			out.writeShort(formulaSize);
179: 			out.writeInt(_unknownPreFormulaInt);
180: 			_linkRefPtg.write(out);
181: 			if (_unknownPostFormulaByte != null) {
182: 				out.writeByte(_unknownPostFormulaByte.byteValue());
183: 			}
184: 		}
185: 	}

202: 	private int getFormattingDataLength() {
203: 		if (_text.length() < 1) {
204: 			// important - no formatting data if text is empty
205: 			return 0;
206: 		}
207: 		return (_text.numFormattingRuns() + 1) * FORMAT_RUN_ENCODED_SIZE;
208: 	}

210: 	private static void writeFormatData(ContinuableRecordOutput out , HSSFRichTextString str) {
211: 		int nRuns = str.numFormattingRuns();
212: 		for (int i = 0; i < nRuns; i++) {
213: 			out.writeShort(str.getIndexOfFormattingRun(i));
214: 			int fontIndex = str.getFontOfFormattingRun(i);
215: 			out.writeShort(fontIndex == str.NO_FONT ? 0 : fontIndex);
216: 			out.writeInt(0); // skip reserved
217: 		}
218: 		out.writeShort(str.length());
219: 		out.writeShort(0);
220: 		out.writeInt(0); // skip reserved
221: 	}
