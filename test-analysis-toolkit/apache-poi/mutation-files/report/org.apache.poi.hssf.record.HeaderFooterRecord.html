 25: /**
 26:  * The HEADERFOOTER record stores information added in Office Excel 2007 for headers/footers.
 27:  *
 28:  * @author Yegor Kozlov
 29:  */
 30: public final class HeaderFooterRecord extends StandardRecord {
 31: 
 32:     private static final byte[] BLANK_GUID = new byte[16];
 33: 
 34:     public final static short sid = 0x089C;
 35: 	private byte[] _rawData;
 36: 
 37:     public HeaderFooterRecord(byte[] data) {
 38:         _rawData = data;
 39:     }
 40: 
 41: 	/**
 42: 	 * construct a HeaderFooterRecord record.  No fields are interpreted and the record will
 43: 	 * be serialized in its original form more or less
 44: 	 * @param in the RecordInputstream to read the record from
 45: 	 */
 46: 	public HeaderFooterRecord(RecordInputStream in) {
 47: 		_rawData = in.readRemainder();
 48: 	}
 49: 
 50: 	/**
 51: 	 * spit the record out AS IS. no interpretation or identification
 52: 	 */
 53: 	public void serialize(LittleEndianOutput out) {
 54: 		out.write(_rawData);
 55: 	}
 56: 
 57: 	protected int getDataSize() {
 58: 		return _rawData.length;
 59: 	}
 60:     
 61:     public short getSid()
 62:     {
 63:         return sid;
 64:     }
 65: 
 66:     /**
 67:      * If this header belongs to a specific sheet view , the sheet view?s GUID will be saved here.
 68:      * <p>
 69:      * If it is zero, it means the current sheet. Otherwise, this field MUST match the guid field
 70:      * of the preceding {@link UserSViewBegin} record.
 71:      *
 72:      * @return the sheet view?s GUID
 73:      */
 74:     public byte[] getGuid(){
 75:         byte[] guid = new byte[16];
 76:         System.arraycopy(_rawData, 12, guid, 0, guid.length);
 77:         return guid;
 78:     }
 79: 
 80:     /**
 81:      * @return whether this record belongs to the current sheet 
 82:      */
 83:     public boolean isCurrentSheet(){
 84:         return Arrays.equals(getGuid(), BLANK_GUID);
 85:     }
 86: 
 87:     public String toString() {
 88:         StringBuffer sb = new StringBuffer();
 89: 
 90:         sb.append("[").append("HEADERFOOTER").append("] (0x");
 91:         sb.append(Integer.toHexString(sid).toUpperCase() + ")\n");
 92:         sb.append("  rawData=").append(HexDump.toHex(_rawData)).append("\n");
 93:         sb.append("[/").append("HEADERFOOTER").append("]\n");
 94:         return sb.toString();
 95:     }
