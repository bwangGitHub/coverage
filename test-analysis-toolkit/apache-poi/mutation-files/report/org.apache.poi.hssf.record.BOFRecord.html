 23: /**
 24:  * Title: Beginning Of File (0x0809)<P>
 25:  * Description: Somewhat of a misnomer, its used for the beginning of a set of
 26:  *              records that have a particular purpose or subject.
 27:  *              Used in sheets and workbooks.<P>
 28:  * REFERENCE:  PG 289 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<P>
 29:  * @author Andrew C. Oliver
 30:  * @author Jason Height (jheight at chariot dot net dot au)
 31:  */
 32: public final class BOFRecord extends StandardRecord {
 33:     /**
 34:      * for BIFF8 files the BOF is 0x809.  For earlier versions it was 0x09 or 0x(biffversion)09
 35:      */
 36:     public final static short sid = 0x809;
 37: 
 38:     /** suggested default (0x0600 - BIFF8) */
 39:     public final static int VERSION             = 0x0600;
 40:     /** suggested default 0x10d3 */
 41:     public final static int BUILD               = 0x10d3;
 42:     /** suggested default  0x07CC (1996) */
 43:     public final static int BUILD_YEAR          = 0x07CC;   // 1996
 44:     /** suggested default for a normal sheet (0x41) */
 45:     public final static int HISTORY_MASK        = 0x41;
 46: 
 47:     public final static int TYPE_WORKBOOK       = 0x05;
 48:     public final static int TYPE_VB_MODULE      = 0x06;
 49:     public final static int TYPE_WORKSHEET      = 0x10;
 50:     public final static int TYPE_CHART          = 0x20;
 51:     public final static int TYPE_EXCEL_4_MACRO  = 0x40;
 52:     public final static int TYPE_WORKSPACE_FILE = 0x100;
 53: 
 54:     private int field_1_version;
 55:     private int field_2_type;
 56:     private int field_3_build;
 57:     private int field_4_year;
 58:     private int field_5_history;
 59:     private int field_6_rversion;
 60: 
 61:     /**
 62:      * Constructs an empty BOFRecord with no fields set.
 63:      */
 64:     public BOFRecord() {
 65:     }
 66:     
 67:     private BOFRecord(int type) {
 68:         field_1_version = VERSION;
 69:         field_2_type = type;
 70:         field_3_build = BUILD;
 71:         field_4_year = BUILD_YEAR;
 72:         field_5_history = 0x01;
 73:         field_6_rversion = VERSION;
 74:     }
 75:     
 76:     public static BOFRecord createSheetBOF() {
 77:         return new BOFRecord(TYPE_WORKSHEET);
 78:     }
 79: 
 80:     public BOFRecord(RecordInputStream in) {
 81:         field_1_version  = in.readShort();
 82:         field_2_type     = in.readShort();
 83: 
 84:         // Some external tools don't generate all of
 85:         //  the remaining fields
 86:         if (in.remaining() >= 2) {
 87:             field_3_build = in.readShort();
 88:         }
 89:         if (in.remaining() >= 2) {
 90:             field_4_year = in.readShort();
 91:         }
 92:         if (in.remaining() >= 4) {
 93:             field_5_history  = in.readInt();
 94:         }
 95:         if (in.remaining() >= 4) {
 96:             field_6_rversion = in.readInt();
 97:         }
 98:     }

220:     public String toString() {
221:         StringBuffer buffer = new StringBuffer();
222: 
223:         buffer.append("[BOF RECORD]\n");
224:         buffer.append("    .version  = ").append(HexDump.shortToHex(getVersion())).append("\n");
225:         buffer.append("    .type     = ").append(HexDump.shortToHex(getType()));
226:         buffer.append(" (").append(getTypeName()).append(")").append("\n");
227:         buffer.append("    .build    = ").append(HexDump.shortToHex(getBuild())).append("\n");
228:         buffer.append("    .buildyear= ").append(getBuildYear()).append("\n");
229:         buffer.append("    .history  = ").append(HexDump.intToHex(getHistoryBitMask())).append("\n");
230:         buffer.append("    .reqver   = ").append(HexDump.intToHex(getRequiredVersion())).append("\n");
231:         buffer.append("[/BOF RECORD]\n");
232:         return buffer.toString();
233:     }

235:     private String getTypeName() {
236:         switch(field_2_type) {
237:             case TYPE_CHART: return "chart";
238:             case TYPE_EXCEL_4_MACRO: return "excel 4 macro";
239:             case TYPE_VB_MODULE: return "vb module";
240:             case TYPE_WORKBOOK: return "workbook";
241:             case TYPE_WORKSHEET: return "worksheet";
242:             case TYPE_WORKSPACE_FILE: return "workspace file";
243:         }
244:         return "#error unknown type#";
245:     }

248:         out.writeShort(getVersion());

252:         out.writeInt(getHistoryBitMask());
253:         out.writeInt(getRequiredVersion());
254:     }
