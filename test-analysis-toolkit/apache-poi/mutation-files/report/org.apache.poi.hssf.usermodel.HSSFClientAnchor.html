 75:     /**
 76:      * Calculates the height of a client anchor in points.
 77:      *
 78:      * @param sheet     the sheet the anchor will be attached to
 79:      * @return          the shape height.
 80:      */
 81:     public float getAnchorHeightInPoints(HSSFSheet sheet )
 82:     {
 83:         int y1 = getDy1();
 84:         int y2 = getDy2();
 85:         int row1 = Math.min( getRow1(), getRow2() );
 86:         int row2 = Math.max( getRow1(), getRow2() );
 87: 
 88:         float points = 0;
 89:         if (row1 == row2)
 90:         {
 91:             points = ((y2 - y1) / 256.0f) * getRowHeightInPoints(sheet, row2);
 92:         }
 93:         else
 94:         {
 95:             points += ((256.0f - y1) / 256.0f) * getRowHeightInPoints(sheet, row1);
 96:             for (int i = row1 + 1; i < row2; i++)
 97:             {
 98:                 points += getRowHeightInPoints(sheet, i);
 99:             }
100:             points += (y2 / 256.0f) * getRowHeightInPoints(sheet, row2);
101:         }
102: 
103:         return points;
104:     }
105: 
106:     private float getRowHeightInPoints(HSSFSheet sheet, int rowNum)
107:     {
108:         HSSFRow row = sheet.getRow(rowNum);
109:         if (row == null) {
110:             return sheet.getDefaultRowHeightInPoints();
111:         }
112:         return row.getHeightInPoints();
113:     }
