 28: /**
 29:  * This class manages and creates the Block Allocation Table, which is
 30:  * basically a set of linked lists of block indices.
 31:  * <P>
 32:  * Each block of the filesystem has an index. The first block, the
 33:  * header, is skipped; the first block after the header is index 0,
 34:  * the next is index 1, and so on.
 35:  * <P>
 36:  * A block's index is also its index into the Block Allocation
 37:  * Table. The entry that it finds in the Block Allocation Table is the
 38:  * index of the next block in the linked list of blocks making up a
 39:  * file, or it is set to -2: end of list.
 40:  *
 41:  * @author Marc Johnson (mjohnson at apache dot org)
 42:  */
 43: public final class BlockAllocationTableReader {
 44:     private static final POILogger _logger = POILogFactory.getLogger(BlockAllocationTableReader.class);
 45: 
 46:     /**
 47:      * Maximum number size (in blocks) of the allocation table as supported by
 48:      * POI.<br/>
 49:      *
 50:      * This constant has been chosen to help POI identify corrupted data in the
 51:      * header block (rather than crash immediately with {@link OutOfMemoryError}
 52:      * ). It's not clear if the compound document format actually specifies any
 53:      * upper limits. For files with 512 byte blocks, having an allocation table
 54:      * of 65,335 blocks would correspond to a total file size of 4GB. Needless
 55:      * to say, POI probably cannot handle files anywhere near that size.
 56:      */
 57:     private static final int MAX_BLOCK_COUNT = 65535;
 58:     private final IntList _entries;
 59:     private POIFSBigBlockSize bigBlockSize;
 60: 
 61:     /**
 62:      * create a BlockAllocationTableReader for an existing filesystem. Side
 63:      * effect: when this method finishes, the BAT blocks will have
 64:      * been removed from the raw block list, and any blocks labeled as
 65:      * 'unused' in the block allocation table will also have been
 66:      * removed from the raw block list.
 67:      *
 68:      * @param block_count the number of BAT blocks making up the block
 69:      *                    allocation table
 70:      * @param block_array the array of BAT block indices from the
 71:      *                    filesystem's header
 72:      * @param xbat_count the number of XBAT blocks
 73:      * @param xbat_index the index of the first XBAT block
 74:      * @param raw_block_list the list of RawDataBlocks
 75:      *
 76:      * @exception IOException if, in trying to create the table, we
 77:      *            encounter logic errors
 78:      */
 79:     public BlockAllocationTableReader(POIFSBigBlockSize bigBlockSize, int block_count, int [] block_array,
 80:             int xbat_count, int xbat_index, BlockList raw_block_list) throws IOException {
 81:         this(bigBlockSize);
 82:         
 83:         sanityCheckBlockCount(block_count);
 84: 
 85:         // We want to get the whole of the FAT table
 86:         // To do this:
 87:         //  * Work through raw_block_list, which points to the 
 88:         //     first (up to) 109 BAT blocks
 89:         //  * Jump to the XBAT offset, and read in XBATs which
 90:         //     point to more BAT blocks
 91:         int          limit    = Math.min(block_count, block_array.length);
 92:         int          block_index;
 93:         
 94:         // This will hold all of the BAT blocks in order
 95:         RawDataBlock blocks[] = new RawDataBlock[ block_count ];
 96: 
 97:         // Process the first (up to) 109 BAT blocks
 98:         for (block_index = 0; block_index < limit; block_index++)
 99:         {
100:             // Check that the sector number of the BAT block is a valid one
101:             int nextOffset = block_array[ block_index ];
102:             if(nextOffset > raw_block_list.blockCount()) {
103:                throw new IOException("Your file contains " + raw_block_list.blockCount() + 
104:                      " sectors, but the initial DIFAT array at index " + block_index +
105:                      " referenced block # " + nextOffset + ". This isn't allowed and " +
106:                      " your file is corrupt");
107:             }
108:             // Record the sector number of this BAT block 
109:             blocks[ block_index ] =
110:                 ( RawDataBlock ) raw_block_list.remove(nextOffset);
111:         }
112:         
113:         // Process additional BAT blocks via the XBATs
114:         if (block_index < block_count)
115:         {
116: 
117:             // must have extended blocks
118:             if (xbat_index < 0)
119:             {
120:                 throw new IOException(
121:                     "BAT count exceeds limit, yet XBAT index indicates no valid entries");
122:             }
123:             int chain_index           = xbat_index;
124:             int max_entries_per_block = bigBlockSize.getXBATEntriesPerBlock(); 
125:             int chain_index_offset    = bigBlockSize.getNextXBATChainOffset(); 
126: 
127:             // Each XBAT block contains either:
128:             //  (maximum number of sector indexes) + index of next XBAT
129:             //  some sector indexes + FREE sectors to max # + EndOfChain
130:             for (int j = 0; j < xbat_count; j++)
131:             {
132:                 limit = Math.min(block_count - block_index,
133:                                  max_entries_per_block);
134:                 byte[] data   = raw_block_list.remove(chain_index).getData();
135:                 int    offset = 0;
136: 
137:                 for (int k = 0; k < limit; k++)
138:                 {
139:                     blocks[ block_index++ ] =
140:                         ( RawDataBlock ) raw_block_list
141:                             .remove(LittleEndian.getInt(data, offset));
142:                     offset                  += LittleEndianConsts.INT_SIZE;
143:                 }
144:                 chain_index = LittleEndian.getInt(data, chain_index_offset);
145:                 if (chain_index == POIFSConstants.END_OF_CHAIN)
146:                 {
147:                     break;
148:                 }
149:             }
150:         }
151:         if (block_index != block_count)
152:         {
153:             throw new IOException("Could not find all blocks");
154:         }
155: 
156:         // Now that we have all of the raw data blocks which make
157:         //  up the FAT, go through and create the indices
158:         setEntries(blocks, raw_block_list);
159:     }
160: 
161:     /**
162:      * create a BlockAllocationTableReader from an array of raw data blocks
163:      *
164:      * @param blocks the raw data
165:      * @param raw_block_list the list holding the managed blocks
166:      *
167:      * @exception IOException
168:      */
169:     BlockAllocationTableReader(POIFSBigBlockSize bigBlockSize, ListManagedBlock[] blocks, BlockList raw_block_list)
170:             throws IOException {
171:         this(bigBlockSize);
172:         setEntries(blocks, raw_block_list);
173:     }
174: 
175:     BlockAllocationTableReader(POIFSBigBlockSize bigBlockSize) {
176:         this.bigBlockSize = bigBlockSize;
177:         _entries = new IntList();
178:     }
179:     
180:     public static void sanityCheckBlockCount(int block_count) throws IOException {
181:        if (block_count <= 0) {
182:           throw new IOException(
183:                 "Illegal block count; minimum count is 1, got " + 
184:                 block_count + " instead"
185:           );
186:        }
187:        if (block_count > MAX_BLOCK_COUNT) {
188:           throw new IOException(
189:                 "Block count " + block_count + 
190:                 " is too high. POI maximum is " + MAX_BLOCK_COUNT + "."
191:           );
192:        }
193:     }
194: 
195:     /**
196:      * walk the entries from a specified point and return the
197:      * associated blocks. The associated blocks are removed from the
198:      * block list
199:      *
200:      * @param startBlock the first block in the chain
201:      * @param blockList the raw data block list
202:      *
203:      * @return array of ListManagedBlocks, in their correct order
204:      *
205:      * @exception IOException if there is a problem acquiring the blocks
206:      */
207:     ListManagedBlock[] fetchBlocks(int startBlock, int headerPropertiesStartBlock,
208:             BlockList blockList) throws IOException {
209:         List<ListManagedBlock> blocks = new ArrayList<ListManagedBlock>();
210:         int  currentBlock = startBlock;
211:         boolean firstPass = true;
212:         ListManagedBlock dataBlock = null;
213: 
214:         // Process the chain from the start to the end
215:         // Normally we have header, data, end
216:         // Sometimes we have data, header, end
217:         // For those cases, stop at the header, not the end
218:         while (currentBlock != POIFSConstants.END_OF_CHAIN) {
219:             try {
220:                 // Grab the data at the current block offset
221:                 dataBlock = blockList.remove(currentBlock);
222:                 blocks.add(dataBlock);
223:                 // Now figure out which block we go to next
224:                 currentBlock = _entries.get(currentBlock);
225:                 firstPass = false;
226:             } catch(IOException e) {
227:                 if(currentBlock == headerPropertiesStartBlock) {
228:                     // Special case where things are in the wrong order
229:                     _logger.log(POILogger.WARN, "Warning, header block comes after data blocks in POIFS block listing");
230:                     currentBlock = POIFSConstants.END_OF_CHAIN;
231:                 } else if(currentBlock == 0 && firstPass) {
232:                     // Special case where the termination isn't done right
233:                     //  on an empty set
234:                     _logger.log(POILogger.WARN, "Warning, incorrectly terminated empty data blocks in POIFS block listing (should end at -2, ended at 0)");
235:                     currentBlock = POIFSConstants.END_OF_CHAIN;
236:                 } else {
237:                     // Ripple up
238:                     throw e;
239:                 }
240:             }
241:         }
242: 
243:         return blocks.toArray(new ListManagedBlock[blocks.size()]);
244:     }

246:     // methods for debugging reader
247: 
248:     /**
249:      * determine whether the block specified by index is used or not
250:      *
251:      * @param index index of block in question
252:      *
253:      * @return true if the specific block is used, else false
254:      */
255:     boolean isUsed(int index) {
256: 
257:         try {
258:             return _entries.get(index) != -1;
259:         } catch (IndexOutOfBoundsException e) {
260:             // ignored
261:             return false;
262:         }
263:     }
264: 
265:     /**
266:      * return the next block index
267:      *
268:      * @param index of the current block
269:      *
270:      * @return index of the next block (may be
271:      *         POIFSConstants.END_OF_CHAIN, indicating end of chain
272:      *         (duh))
273:      *
274:      * @exception IOException if the current block is unused
275:      */
276:     int getNextBlockIndex(int index) throws IOException {
277:         if (isUsed(index)) {
278:             return _entries.get(index);
279:         }
280:         throw new IOException("index " + index + " is unused");
281:     }
282: 
283:     /**
284:      * Convert an array of blocks into a set of integer indices
285:      *
286:      * @param blocks the array of blocks containing the indices
287:      * @param raw_blocks the list of blocks being managed. Unused
288:      *                   blocks will be eliminated from the list
289:      */
290:     private void setEntries(ListManagedBlock[] blocks, BlockList raw_blocks) throws IOException {
291:         int limit = bigBlockSize.getBATEntriesPerBlock(); 
292: 
293:         for (int block_index = 0; block_index < blocks.length; block_index++)
294:         {
295:             byte[] data   = blocks[ block_index ].getData();
296:             int    offset = 0;
297: 
298:             for (int k = 0; k < limit; k++)
299:             {
300:                 int entry = LittleEndian.getInt(data, offset);
301: 
302:                 if (entry == POIFSConstants.UNUSED_BLOCK)
303:                 {
304:                     raw_blocks.zap(_entries.size());
305:                 }
306:                 _entries.add(entry);
307:                 offset += LittleEndianConsts.INT_SIZE;
308:             }
309: 
310:             // discard block
311:             blocks[ block_index ] = null;
312:         }
313:         raw_blocks.setBAT(this);
314:     }
