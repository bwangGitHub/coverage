 24:  * Subclasses of this class (the majority of BIFF records) are non-continuable.  This allows for

 32: 		return 4 + getDataSize();
 33: 	}

 35:     /**
 36:      * Write the data content of this BIFF record including the sid and record length.<br/>
 37:      *
 38:      * The subclass must write the exact number of bytes as reported by
 39:      *  {@link org.apache.poi.hssf.record.Record#getRecordSize()}}
 40:      */
 41: 	@Override
 42: 	public final int serialize(int offset, byte[] data) {
 43: 		int dataSize = getDataSize();
 44: 		int recSize = 4 + dataSize;
 45: 		LittleEndianByteArrayOutputStream out = new LittleEndianByteArrayOutputStream(data, offset, recSize);
 46: 		out.writeShort(getSid());
 47: 		out.writeShort(dataSize);
 48: 		serialize(out);
 49: 		if (out.getWriteIndex() - offset != recSize) {
 50: 			throw new IllegalStateException("Error in serialization of (" + getClass().getName() + "): "
 51: 					+ "Incorrect number of bytes written - expected "
 52: 					+ recSize + " but got " + (out.getWriteIndex() - offset));
 53: 		}
 54: 		return recSize;
 55: 	}
