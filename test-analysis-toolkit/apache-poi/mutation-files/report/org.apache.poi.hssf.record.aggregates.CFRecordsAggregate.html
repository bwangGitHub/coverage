 33: /**
 34:  * CFRecordsAggregate - aggregates Conditional Formatting records CFHeaderRecord 
 35:  * and number of up to three CFRuleRecord records together to simplify

 50: 	private CFRecordsAggregate(CFHeaderRecord pHeader, CFRuleRecord[] pRules) {
 51: 		if(pHeader == null) {
 52: 			throw new IllegalArgumentException("header must not be null");
 53: 		}
 54: 		if(pRules == null) {
 55: 			throw new IllegalArgumentException("rules must not be null");
 56: 		}
 57: 		if(pRules.length > MAX_CONDTIONAL_FORMAT_RULES) {
 58: 			throw new IllegalArgumentException("No more than " 
 59: 					+ MAX_CONDTIONAL_FORMAT_RULES + " rules may be specified");
 60: 		}
 61: 		if (pRules.length != pHeader.getNumberOfConditionalFormats()) {
 62: 			throw new RuntimeException("Mismatch number of rules");
 63: 		}
 64: 		header = pHeader;
 65: 		rules = new ArrayList(3);
 66: 		for (int i = 0; i < pRules.length; i++) {
 67: 			rules.add(pRules[i]);
 68: 		}
 69: 	}

 75: 	/**
 76: 	 * Create CFRecordsAggregate from a list of CF Records
 77: 	 * @param rs - the stream to read from
 78: 	 * @return CFRecordsAggregate object
 79: 	 */
 80: 	public static CFRecordsAggregate createCFAggregate(RecordStream rs) {
 81: 		Record rec = rs.getNext();
 82: 		if (rec.getSid() != CFHeaderRecord.sid) {
 83: 			throw new IllegalStateException("next record sid was " + rec.getSid() 
 84: 					+ " instead of " + CFHeaderRecord.sid + " as expected");
 85: 		}
 86: 
 87: 		CFHeaderRecord header = (CFHeaderRecord)rec;
 88: 		int nRules = header.getNumberOfConditionalFormats();
 89: 
 90: 		CFRuleRecord[] rules = new CFRuleRecord[nRules];
 91: 		for (int i = 0; i < rules.length; i++) {
 92: 			rules[i] = (CFRuleRecord) rs.getNext();
 93: 		}
 94: 		
 95: 		return new CFRecordsAggregate(header, rules);
 96: 	}

180: 	/**
181: 	 * @return <code>false</code> if this whole {@link CFHeaderRecord} / {@link CFRuleRecord}s should be deleted
182: 	 */
183: 	public boolean updateFormulasAfterCellShift(FormulaShifter shifter, int currentExternSheetIx) {
184: 		CellRangeAddress[] cellRanges = header.getCellRanges();
185: 		boolean changed = false;
186: 		List temp = new ArrayList();
187: 		for (int i = 0; i < cellRanges.length; i++) {
188: 			CellRangeAddress craOld = cellRanges[i];
189: 			CellRangeAddress craNew = shiftRange(shifter, craOld, currentExternSheetIx);
190: 			if (craNew == null) {
191: 				changed = true;
192: 				continue;
193: 			}
194: 			temp.add(craNew);
195: 			if (craNew != craOld) {
196: 				changed = true;
197: 			}
198: 		}
199: 
200: 		if (changed) {
201: 			int nRanges = temp.size();
202: 			if (nRanges == 0) {
203: 				return false;
204: 			}
205: 			CellRangeAddress[] newRanges = new CellRangeAddress[nRanges];
206: 			temp.toArray(newRanges);
207: 			header.setCellRanges(newRanges);
208: 		}
209: 		
210: 		for(int i=0; i<rules.size(); i++) {
211: 			CFRuleRecord rule = (CFRuleRecord)rules.get(i);
212: 			Ptg[] ptgs;
213: 			ptgs = rule.getParsedExpression1();
214: 			if (ptgs != null && shifter.adjustFormula(ptgs, currentExternSheetIx)) {
215: 				rule.setParsedExpression1(ptgs);
216: 			}
217: 			ptgs = rule.getParsedExpression2();
218: 			if (ptgs != null && shifter.adjustFormula(ptgs, currentExternSheetIx)) {
219: 				rule.setParsedExpression2(ptgs);
220: 			}
221: 		}
222: 		return true;
223: 	}

225: 	private static CellRangeAddress shiftRange(FormulaShifter shifter, CellRangeAddress cra, int currentExternSheetIx) {
226: 		// FormulaShifter works well in terms of Ptgs - so convert CellRangeAddress to AreaPtg (and back) here
227: 		AreaPtg aptg = new AreaPtg(cra.getFirstRow(), cra.getLastRow(), cra.getFirstColumn(), cra.getLastColumn(), false, false, false, false);
228: 		Ptg[] ptgs = { aptg, };
229: 		
230: 		if (!shifter.adjustFormula(ptgs, currentExternSheetIx)) {
231: 			return cra;
232: 		}
233: 		Ptg ptg0 = ptgs[0];
234: 		if (ptg0 instanceof AreaPtg) {
235: 			AreaPtg bptg = (AreaPtg) ptg0;
236: 			return new CellRangeAddress(bptg.getFirstRow(), bptg.getLastRow(), bptg.getFirstColumn(), bptg.getLastColumn());
237: 		}
238: 		if (ptg0 instanceof AreaErrPtg) {
239: 			return null;
240: 		}
241: 		throw new IllegalStateException("Unexpected shifted ptg class (" + ptg0.getClass().getName() + ")");
242: 	}
