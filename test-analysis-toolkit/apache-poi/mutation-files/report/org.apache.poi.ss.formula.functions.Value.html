 26: /**
 27:  * Implementation for Excel VALUE() function.<p/>
 28:  *
 29:  * <b>Syntax</b>:<br/> <b>VALUE</b>(<b>text</b>)<br/>
 30:  *
 31:  * Converts the text argument to a number. Leading and/or trailing whitespace is
 32:  * ignored. Currency symbols and thousands separators are stripped out.
 33:  * Scientific notation is also supported. If the supplied text does not convert
 34:  * properly the result is <b>#VALUE!</b> error. Blank string converts to zero.
 35:  *
 36:  * @author Josh Micich
 37:  */
 38: public final class Value extends Fixed1ArgFunction {
 39: 
 40: 	/** "1,0000" is valid, "1,00" is not */
 41: 	private static final int MIN_DISTANCE_BETWEEN_THOUSANDS_SEPARATOR = 4;
 42: 	private static final Double ZERO = new Double(0.0);
 43: 
 44: 	public ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0) {
 45: 		ValueEval veText;
 46: 		try {
 47: 			veText = OperandResolver.getSingleValue(arg0, srcRowIndex, srcColumnIndex);
 48: 		} catch (EvaluationException e) {
 49: 			return e.getErrorEval();
 50: 		}
 51: 		String strText = OperandResolver.coerceValueToString(veText);
 52: 		Double result = convertTextToNumber(strText);
 53: 		if (result == null) {
 54: 			return ErrorEval.VALUE_INVALID;
 55: 		}
 56: 		return new NumberEval(result.doubleValue());
 57: 	}
 58: 
 59: 	/**
 60: 	 * TODO see if the same functionality is needed in {@link OperandResolver#parseDouble(String)}
 61: 	 *
 62: 	 * @return <code>null</code> if there is any problem converting the text
 63: 	 */
 64: 	private static Double convertTextToNumber(String strText) {
 65: 		boolean foundCurrency = false;
 66: 		boolean foundUnaryPlus = false;
 67: 		boolean foundUnaryMinus = false;
 68: 
 69: 		int len = strText.length();
 70: 		int i;
 71: 		for (i = 0; i < len; i++) {
 72: 			char ch = strText.charAt(i);
 73: 			if (Character.isDigit(ch) || ch == '.') {
 74: 				break;
 75: 			}
 76: 			switch (ch) {
 77: 				case ' ':
 78: 					// intervening spaces between '$', '-', '+' are OK
 79: 					continue;
 80: 				case '$':
 81: 					if (foundCurrency) {
 82: 						// only one currency symbols is allowed
 83: 						return null;
 84: 					}
 85: 					foundCurrency = true;
 86: 					continue;
 87: 				case '+':
 88: 					if (foundUnaryMinus || foundUnaryPlus) {
 89: 						return null;
 90: 					}
 91: 					foundUnaryPlus = true;
 92: 					continue;
 93: 				case '-':
 94: 					if (foundUnaryMinus || foundUnaryPlus) {
 95: 						return null;
 96: 					}
 97: 					foundUnaryMinus = true;
 98: 					continue;
 99: 				default:
100: 					// all other characters are illegal
101: 					return null;
102: 			}
103: 		}
104: 		if (i >= len) {
105: 			// didn't find digits or '.'
106: 			if (foundCurrency || foundUnaryMinus || foundUnaryPlus) {
107: 				return null;
108: 			}
109: 			return ZERO;
110: 		}
111: 
112: 		// remove thousands separators
113: 
114: 		boolean foundDecimalPoint = false;
115: 		int lastThousandsSeparatorIndex = Short.MIN_VALUE;
116: 
117: 		StringBuffer sb = new StringBuffer(len);
118: 		for (; i < len; i++) {
119: 			char ch = strText.charAt(i);
120: 			if (Character.isDigit(ch)) {
121: 				sb.append(ch);
122: 				continue;
123: 			}
124: 			switch (ch) {
125: 				case ' ':
126: 					String remainingText = strText.substring(i);
127: 					if (remainingText.trim().length() > 0) {
128: 						// intervening spaces not allowed once the digits start
129: 						return null;
130: 					}
131: 					break;
132: 				case '.':
133: 					if (foundDecimalPoint) {
134: 						return null;
135: 					}
136: 					if (i - lastThousandsSeparatorIndex < MIN_DISTANCE_BETWEEN_THOUSANDS_SEPARATOR) {
137: 						return null;
138: 					}
139: 					foundDecimalPoint = true;
140: 					sb.append('.');
141: 					continue;
142: 				case ',':
143: 					if (foundDecimalPoint) {
144: 						// thousands separators not allowed after '.' or 'E'
145: 						return null;
146: 					}
147: 					int distanceBetweenThousandsSeparators = i - lastThousandsSeparatorIndex;
148: 					// as long as there are 3 or more digits between
149: 					if (distanceBetweenThousandsSeparators < MIN_DISTANCE_BETWEEN_THOUSANDS_SEPARATOR) {
150: 						return null;
151: 					}
152: 					lastThousandsSeparatorIndex = i;
153: 					// don't append ','
154: 					continue;
155: 
156: 				case 'E':
157: 				case 'e':
158: 					if (i - lastThousandsSeparatorIndex < MIN_DISTANCE_BETWEEN_THOUSANDS_SEPARATOR) {
159: 						return null;
160: 					}
161: 					// append rest of strText and skip to end of loop
162: 					sb.append(strText.substring(i));
163: 					i = len;
164: 					break;
165: 				default:
166: 					// all other characters are illegal
167: 					return null;
168: 			}
169: 		}
170: 		if (!foundDecimalPoint) {
171: 			if (i - lastThousandsSeparatorIndex < MIN_DISTANCE_BETWEEN_THOUSANDS_SEPARATOR) {
172: 				return null;
173: 			}
174: 		}
175: 		double d;
176: 		try {
177: 			d = Double.parseDouble(sb.toString());
178: 		} catch (NumberFormatException e) {
179: 			// still a problem parsing the number - probably out of range
180: 			return null;
181: 		}
182: 		return new Double(foundUnaryMinus ? -d : d);
183: 	}
