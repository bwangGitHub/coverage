 28: /**
 29:  * Intends to provide support for the very evil index to triplet issue and
 30:  * will likely replace the color constants interface for HSSF 2.0.
 31:  * This class contains static inner class members for representing colors.
 32:  * Each color has an index (for the standard palette in Excel (tm) ),
 33:  * native (RGB) triplet and string triplet.  The string triplet is as the
 34:  * color would be represented by Gnumeric.  Having (string) this here is a bit of a
 35:  * collusion of function between HSSF and the HSSFSerializer but I think its
 36:  * a reasonable one in this case.
 37:  *
 38:  * @author  Andrew C. Oliver (acoliver at apache dot org)
 39:  * @author  Brian Sanders (bsanders at risklabs dot com) - full default color palette
 40:  */
 41: public class HSSFColor implements Color {

 49:     /**
 50:      * This function returns all the colours in an unmodifiable Map.
 51:      * The map is cached on first use.
 52:      *
 53:      * @return a Map containing all colours keyed by <tt>Integer</tt> excel-style palette indexes
 54:      */
 55:     public final static Map<Integer,HSSFColor> getIndexHash() {
 56:         if(indexHash == null) {
 57:            indexHash = Collections.unmodifiableMap( createColorsByIndexMap() );
 58:         }
 59: 
 60:         return indexHash;
 61:     }
 62:     /**
 63:      * This function returns all the Colours, stored in a Hashtable that
 64:      *  can be edited. No caching is performed. If you don't need to edit
 65:      *  the table, then call {@link #getIndexHash()} which returns a
 66:      *  statically cached imuatable map of colours.
 67:      */
 68:     public final static Hashtable<Integer,HSSFColor> getMutableIndexHash() {

 70:     }

 72:     private static Hashtable<Integer,HSSFColor> createColorsByIndexMap() {
 73:         HSSFColor[] colors = getAllColors();
 74:         Hashtable<Integer,HSSFColor> result = new Hashtable<Integer,HSSFColor>(colors.length * 3 / 2);
 75: 
 76:         for (int i = 0; i < colors.length; i++) {
 77:             HSSFColor color = colors[i];
 78: 
 79:             Integer index1 = Integer.valueOf(color.getIndex());
 80:             if (result.containsKey(index1)) {
 81:                 HSSFColor prevColor = (HSSFColor)result.get(index1);
 82:                 throw new RuntimeException("Dup color index (" + index1
 83:                         + ") for colors (" + prevColor.getClass().getName()
 84:                         + "),(" + color.getClass().getName() + ")");
 85:             }
 86:             result.put(index1, color);
 87:         }
 88: 
 89:         for (int i = 0; i < colors.length; i++) {
 90:             HSSFColor color = colors[i];
 91:             Integer index2 = getIndex2(color);
 92:             if (index2 == null) {
 93:                 // most colors don't have a second index
 94:                 continue;
 95:             }
 96:             if (result.containsKey(index2)) {
 97:                 if (false) { // Many of the second indexes clash
 98:                     HSSFColor prevColor = (HSSFColor)result.get(index2);
 99:                     throw new RuntimeException("Dup color index (" + index2
100:                             + ") for colors (" + prevColor.getClass().getName()
101:                             + "),(" + color.getClass().getName() + ")");
102:                 }
103:             }
104:             result.put(index2, color);
105:         }
106:         return result;
107:     }
108: 
109:     private static Integer getIndex2(HSSFColor color) {
110: 
111:         Field f;
112:         try {
113:             f = color.getClass().getDeclaredField("index2");
114:         } catch (NoSuchFieldException e) {
115:             // can happen because not all colors have a second index
116:             return null;
117:         }
118: 
119:         Short s;
120:         try {
121:             s = (Short) f.get(color);
122:         } catch (IllegalArgumentException e) {
123:             throw new RuntimeException(e);
124:         } catch (IllegalAccessException e) {
125:             throw new RuntimeException(e);
126:         }
127:         return Integer.valueOf(s.intValue());
128:     }

147:     /**
148:      * this function returns all colors in a hastable.  Its not implemented as a
149:      * static member/staticly initialized because that would be dirty in a
150:      * server environment as it is intended.  This means you'll eat the time
151:      * it takes to create it once per request but you will not hold onto it
152:      * if you have none of those requests.
153:      *
154:      * @return a hashtable containing all colors keyed by String gnumeric-like triplets
155:      */
156:     public final static Hashtable<String,HSSFColor> getTripletHash()
157:     {
158:         return createColorsByHexStringMap();
159:     }

161:     private static Hashtable<String,HSSFColor> createColorsByHexStringMap() {
162:         HSSFColor[] colors = getAllColors();
163:         Hashtable<String,HSSFColor> result = new Hashtable<String,HSSFColor>(colors.length * 3 / 2);
164: 
165:         for (int i = 0; i < colors.length; i++) {
166:             HSSFColor color = colors[i];
167: 
168:             String hexString = color.getHexString();
169:             if (result.containsKey(hexString)) {
170:             	HSSFColor other = (HSSFColor)result.get(hexString);
171:                 throw new RuntimeException(
172:                 		"Dup color hexString (" + hexString
173:                         + ") for color (" + color.getClass().getName() + ") - "
174:                         + " already taken by (" + other.getClass().getName() + ")"
175:                 );
176:             }
177:             result.put(hexString, color);
178:         }
179:         return result;
180:     }
