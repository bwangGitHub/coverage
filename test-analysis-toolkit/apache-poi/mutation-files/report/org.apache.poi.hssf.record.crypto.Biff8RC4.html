 24: /**
 25:  * Used for both encrypting and decrypting BIFF8 streams. The internal
 26:  * {@link RC4} instance is renewed (re-keyed) every 1024 bytes.
 27:  *
 28:  * @author Josh Micich
 29:  */
 30: final class Biff8RC4 {
 31: 
 32: 	private static final int RC4_REKEYING_INTERVAL = 1024;
 33: 
 34: 	private RC4 _rc4;
 35: 	/**
 36: 	 * This field is used to keep track of when to change the {@link RC4}
 37: 	 * instance. The change occurs every 1024 bytes. Every byte passed over is
 38: 	 * counted.
 39: 	 */
 40: 	private int _streamPos;
 41: 	private int _nextRC4BlockStart;
 42: 	private int _currentKeyIndex;
 43: 	private boolean _shouldSkipEncryptionOnCurrentRecord;
 44: 
 45: 	private final Biff8EncryptionKey _key;
 46: 
 47: 	public Biff8RC4(int initialOffset, Biff8EncryptionKey key) {
 48: 		if (initialOffset >= RC4_REKEYING_INTERVAL) {
 49: 			throw new RuntimeException("initialOffset (" + initialOffset + ")>"
 50: 					+ RC4_REKEYING_INTERVAL + " not supported yet");
 51: 		}
 52: 		_key = key;
 53: 		_streamPos = 0;
 54: 		rekeyForNextBlock();
 55: 		_streamPos = initialOffset;
 56: 		for (int i = initialOffset; i > 0; i--) {
 57: 			_rc4.output();
 58: 		}
 59: 		_shouldSkipEncryptionOnCurrentRecord = false;
 60: 	}
 61: 
 62: 	private void rekeyForNextBlock() {
 63: 		_currentKeyIndex = _streamPos / RC4_REKEYING_INTERVAL;
 64: 		_rc4 = _key.createRC4(_currentKeyIndex);
 65: 		_nextRC4BlockStart = (_currentKeyIndex + 1) * RC4_REKEYING_INTERVAL;
 66: 	}
 67: 
 68: 	private int getNextRC4Byte() {
 69: 		if (_streamPos >= _nextRC4BlockStart) {
 70: 			rekeyForNextBlock();
 71: 		}
 72: 		byte mask = _rc4.output();
 73: 		_streamPos++;
 74: 		if (_shouldSkipEncryptionOnCurrentRecord) {
 75: 			return 0;
 76: 		}
 77: 		return mask & 0xFF;
 78: 	}
 79: 
 80: 	public void startRecord(int currentSid) {
 81: 		_shouldSkipEncryptionOnCurrentRecord = isNeverEncryptedRecord(currentSid);
 82: 	}
 83: 
 84: 	/**
 85: 	 * TODO: Additionally, the lbPlyPos (position_of_BOF) field of the BoundSheet8 record MUST NOT be encrypted.
 86: 	 *
 87: 	 * @return <code>true</code> if record type specified by <tt>sid</tt> is never encrypted
 88: 	 */
 89: 	private static boolean isNeverEncryptedRecord(int sid) {
 90: 		switch (sid) {
 91: 			case BOFRecord.sid:
 92: 				// sheet BOFs for sure
 93: 				// TODO - find out about chart BOFs
 94: 
 95: 			case InterfaceHdrRecord.sid:
 96: 				// don't know why this record doesn't seem to get encrypted
 97: 
 98: 			case FilePassRecord.sid:
 99: 				// this only really counts when writing because FILEPASS is read early
100: 
101: 			// UsrExcl(0x0194)
102: 			// FileLock
103: 			// RRDInfo(0x0196)
104: 			// RRDHead(0x0138)
105: 
106: 				return true;
107: 		}
108: 		return false;
109: 	}
110: 
111: 	/**
112: 	 * Used when BIFF header fields (sid, size) are being read. The internal
113: 	 * {@link RC4} instance must step even when unencrypted bytes are read
114: 	 */
115: 	public void skipTwoBytes() {
116: 		getNextRC4Byte();
117: 		getNextRC4Byte();
118: 	}
119: 
120: 	public void xor(byte[] buf, int pOffset, int pLen) {
121: 		int nLeftInBlock;
122: 		nLeftInBlock = _nextRC4BlockStart - _streamPos;
123: 		if (pLen <= nLeftInBlock) {
124: 			// simple case - this read does not cross key blocks
125: 			_rc4.encrypt(buf, pOffset, pLen);
126: 			_streamPos += pLen;
127: 			return;
128: 		}
129: 
130: 		int offset = pOffset;
131: 		int len = pLen;
132: 
133: 		// start by using the rest of the current block
134: 		if (len > nLeftInBlock) {
135: 			if (nLeftInBlock > 0) {
136: 				_rc4.encrypt(buf, offset, nLeftInBlock);
137: 				_streamPos += nLeftInBlock;
138: 				offset += nLeftInBlock;
139: 				len -= nLeftInBlock;
140: 			}
141: 			rekeyForNextBlock();
142: 		}
143: 		// all full blocks following
144: 		while (len > RC4_REKEYING_INTERVAL) {
145: 			_rc4.encrypt(buf, offset, RC4_REKEYING_INTERVAL);
146: 			_streamPos += RC4_REKEYING_INTERVAL;
147: 			offset += RC4_REKEYING_INTERVAL;
148: 			len -= RC4_REKEYING_INTERVAL;
149: 			rekeyForNextBlock();
150: 		}
151: 		// finish with incomplete block
152: 		_rc4.encrypt(buf, offset, len);
153: 		_streamPos += len;
154: 	}
155: 
156: 	public int xorByte(int rawVal) {
157: 		int mask = getNextRC4Byte();
158: 		return (byte) (rawVal ^ mask);
159: 	}
160: 
161: 	public int xorShort(int rawVal) {
162: 		int b0 = getNextRC4Byte();
163: 		int b1 = getNextRC4Byte();
164: 		int mask = (b1 << 8) + (b0 << 0);
165: 		return rawVal ^ mask;
166: 	}
167: 
168: 	public int xorInt(int rawVal) {
169: 		int b0 = getNextRC4Byte();
170: 		int b1 = getNextRC4Byte();
171: 		int b2 = getNextRC4Byte();
172: 		int b3 = getNextRC4Byte();
173: 		int mask = (b3 << 24) + (b2 << 16) + (b1 << 8) + (b0 << 0);
174: 		return rawVal ^ mask;
175: 	}
176: 
177: 	public long xorLong(long rawVal) {
178: 		int b0 = getNextRC4Byte();
179: 		int b1 = getNextRC4Byte();
180: 		int b2 = getNextRC4Byte();
181: 		int b3 = getNextRC4Byte();
182: 		int b4 = getNextRC4Byte();
183: 		int b5 = getNextRC4Byte();
184: 		int b6 = getNextRC4Byte();
185: 		int b7 = getNextRC4Byte();
186: 		long mask =
187: 			  (((long)b7) << 56)
188: 			+ (((long)b6) << 48)
189: 			+ (((long)b5) << 40)
190: 			+ (((long)b4) << 32)
191: 			+ (((long)b3) << 24)
192: 			+ (b2 << 16)
193: 			+ (b1 << 8)
194: 			+ (b0 << 0);
195: 		return rawVal ^ mask;
196: 	}
