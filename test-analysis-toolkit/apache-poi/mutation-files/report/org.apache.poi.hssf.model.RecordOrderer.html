 93: 		int index = findSheetInsertPos(sheetRecords, newRecord.getClass());

 97: 	private static int findSheetInsertPos(List<RecordBase> records, Class<? extends RecordBase> recClass) {
 98: 		if (recClass == DataValidityTable.class) {
 99: 			return findDataValidationTableInsertPos(records);
100: 		}
101: 		if (recClass == MergedCellsTable.class) {
102: 			return findInsertPosForNewMergedRecordTable(records);
103: 		}
104: 		if (recClass == ConditionalFormattingTable.class) {
105: 			return findInsertPosForNewCondFormatTable(records);
106: 		}
107: 		if (recClass == GutsRecord.class) {
108: 			return getGutsRecordInsertPos(records);
109: 		}
110: 		if (recClass == PageSettingsBlock.class) {
111: 			return getPageBreakRecordInsertPos(records);
112: 		}
113: 		if (recClass == WorksheetProtectionBlock.class) {
114: 			return getWorksheetProtectionBlockInsertPos(records);
115: 		}
116: 		throw new RuntimeException("Unexpected record class (" + recClass.getName() + ")");
117: 	}

120: 	 * Finds the index where the protection block should be inserted
121: 	 * @param records the records for this sheet
122: 	 * <pre>
123: 	 * + BOF
124: 	 * o INDEX
125: 	 * o Calculation Settings Block
126: 	 * o PRINTHEADERS
127: 	 * o PRINTGRIDLINES
128: 	 * o GRIDSET
129: 	 * o GUTS
130: 	 * o DEFAULTROWHEIGHT
131: 	 * o SHEETPR

133: 	 * o Worksheet Protection Block

140: 	private static int getWorksheetProtectionBlockInsertPos(List<RecordBase> records) {
141: 		int i = getDimensionsIndex(records);
142: 		while (i > 0) {
143: 			i--;
144: 			Object rb = records.get(i);
145: 			if (!isProtectionSubsequentRecord(rb)) {
146: 				return i+1;
147: 			}
148: 		}
149: 		throw new IllegalStateException("did not find insert pos for protection block");

220: 
221: 		for (int i = records.size() - 2; i >= 0; i--) { // -2 to skip EOF record
222: 			Object rb = records.get(i);
223: 			if (rb instanceof MergedCellsTable) {
224: 				return i + 1;
225: 			}
226: 			if (rb instanceof DataValidityTable) {
227: 				continue;
228: 			}
229: 
230: 			Record rec = (Record) rb;
231: 			switch (rec.getSid()) {
232: 				case WindowTwoRecord.sid:
233: 				case SCLRecord.sid:
234: 				case PaneRecord.sid:
235: 				case SelectionRecord.sid:
236: 				case UnknownRecord.STANDARDWIDTH_0099:
237: 				// MergedCellsTable usually here
238: 				case UnknownRecord.LABELRANGES_015F:
239: 				case UnknownRecord.PHONETICPR_00EF:
240: 					// ConditionalFormattingTable goes here
241: 					return i + 1;
242: 				// HyperlinkTable (not aggregated by POI yet)
243: 				// DataValidityTable
244: 			}
245: 		}
246: 		throw new RuntimeException("Did not find Window2 record");
247: 	}

249: 	private static int findInsertPosForNewMergedRecordTable(List<RecordBase> records) {
250: 		for (int i = records.size() - 2; i >= 0; i--) { // -2 to skip EOF record
251: 			Object rb = records.get(i);
252: 			if (!(rb instanceof Record)) {
253: 				// DataValidityTable, ConditionalFormattingTable,
254: 				// even PageSettingsBlock (which doesn't normally appear after 'View Settings')
255: 				continue;
256: 			}
257: 			Record rec = (Record) rb;
258: 			switch (rec.getSid()) {
259: 				// 'View Settings' (4 records)
260: 				case WindowTwoRecord.sid:
261: 				case SCLRecord.sid:
262: 				case PaneRecord.sid:
263: 				case SelectionRecord.sid:
264: 
265: 				case UnknownRecord.STANDARDWIDTH_0099:
266: 					return i + 1;
267: 			}
268: 		}
269: 		throw new RuntimeException("Did not find Window2 record");
270: 	}

273: 	/**
274: 	 * Finds the index where the sheet validations header record should be inserted
275: 	 * @param records the records for this sheet
276: 	 *
277: 	 * + WINDOW2
278: 	 * o SCL
279: 	 * o PANE
280: 	 * oo SELECTION
281: 	 * o STANDARDWIDTH
282: 	 * oo MERGEDCELLS
283: 	 * o LABELRANGES
284: 	 * o PHONETICPR
285: 	 * o Conditional Formatting Table
286: 	 * o Hyperlink Table
287: 	 * o Data Validity Table
288: 	 * o SHEETLAYOUT
289: 	 * o SHEETPROTECTION
290: 	 * o RANGEPROTECTION
291: 	 * + EOF
292: 	 */
293: 	private static int findDataValidationTableInsertPos(List<RecordBase> records) {
294: 		int i = records.size() - 1;
295: 		if (!(records.get(i) instanceof EOFRecord)) {
296: 			throw new IllegalStateException("Last sheet record should be EOFRecord");
297: 		}
298: 		while (i > 0) {
299: 			i--;
300: 			RecordBase rb = records.get(i);
301: 			if (isDVTPriorRecord(rb)) {
302: 				Record nextRec = (Record) records.get(i + 1);
303: 				if (!isDVTSubsequentRecord(nextRec.getSid())) {
304: 					throw new IllegalStateException("Unexpected (" + nextRec.getClass().getName()
305: 							+ ") found after (" + rb.getClass().getName() + ")");
306: 				}
307: 				return i+1;
308: 			}
309: 			Record rec = (Record) rb;
310: 			if (!isDVTSubsequentRecord(rec.getSid())) {
311: 				throw new IllegalStateException("Unexpected (" + rec.getClass().getName()
312: 						+ ") while looking for DV Table insert pos");
313: 			}
314: 		}
315: 		return 0;
316: 	}

319: 	private static boolean isDVTPriorRecord(RecordBase rb) {
320: 		if (rb instanceof MergedCellsTable || rb instanceof ConditionalFormattingTable) {
321: 			return true;
322: 		}
323: 		short sid = ((Record)rb).getSid();
324: 		switch(sid) {
325: 			case WindowTwoRecord.sid:
326: 			case UnknownRecord.SCL_00A0:
327: 			case PaneRecord.sid:
328: 			case SelectionRecord.sid:
329: 			case UnknownRecord.STANDARDWIDTH_0099:
330: 			// MergedCellsTable
331: 			case UnknownRecord.LABELRANGES_015F:
332: 			case UnknownRecord.PHONETICPR_00EF:
333: 			// ConditionalFormattingTable
334: 			case HyperlinkRecord.sid:
335: 			case UnknownRecord.QUICKTIP_0800:
336:             // name of a VBA module    
337:             case UnknownRecord.CODENAME_1BA:
338: 				return true;
339: 		}
340: 		return false;
341: 	}
