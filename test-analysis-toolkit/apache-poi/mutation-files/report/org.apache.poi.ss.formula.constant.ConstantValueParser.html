 20: import org.apache.poi.util.LittleEndianInput;
 21: import org.apache.poi.util.LittleEndianOutput;
 22: import org.apache.poi.util.StringUtil;

 24: /**
 25:  * To support Constant Values (2.5.7) as required by the CRN record.
 26:  * This class is also used for two dimensional arrays which are encoded by 
 27:  * EXTERNALNAME (5.39) records and Array tokens.<p/>
 28:  * 
 29:  * @author Josh Micich
 30:  */
 31: public final class ConstantValueParser {
 32: 	// note - these (non-combinable) enum values are sparse.
 33: 	private static final int TYPE_EMPTY = 0;
 34: 	private static final int TYPE_NUMBER = 1;
 35: 	private static final int TYPE_STRING = 2;
 36: 	private static final int TYPE_BOOLEAN = 4; 
 37: 	private static final int TYPE_ERROR_CODE = 16; // TODO - update OOO document to include this value
 38: 	
 39: 	private static final int TRUE_ENCODING = 1; 
 40: 	private static final int FALSE_ENCODING = 0;
 41: 	
 42: 	// TODO - is this the best way to represent 'EMPTY'?
 43: 	private static final Object EMPTY_REPRESENTATION = null;
 44: 
 45: 	private ConstantValueParser() {
 46: 		// no instances of this class
 47: 	}
 48: 
 49: 	public static Object[] parse(LittleEndianInput in, int nValues) {
 50: 		Object[] result = new Object[nValues];
 51: 		for (int i = 0; i < result.length; i++) {
 52: 			result[i] = readAConstantValue(in);
 53: 		}
 54: 		return result;
 55: 	}
 56: 
 57: 	private static Object readAConstantValue(LittleEndianInput in) {
 58: 		byte grbit = in.readByte();
 59: 		switch(grbit) {
 60: 			case TYPE_EMPTY:
 61: 				in.readLong(); // 8 byte 'not used' field
 62: 				return EMPTY_REPRESENTATION; 
 63: 			case TYPE_NUMBER:
 64: 				return new Double(in.readDouble());
 65: 			case TYPE_STRING:
 66: 				return StringUtil.readUnicodeString(in);
 67: 			case TYPE_BOOLEAN:
 68: 				return readBoolean(in);
 69: 			case TYPE_ERROR_CODE:
 70: 				int errCode = in.readUShort();
 71: 				// next 6 bytes are unused
 72: 				in.readUShort();
 73: 				in.readInt();
 74: 				return ErrorConstant.valueOf(errCode);
 75: 		}
 76: 		throw new RuntimeException("Unknown grbit value (" + grbit + ")");
 77: 	}
 78: 
 79: 	private static Object readBoolean(LittleEndianInput in) {
 80: 		byte val = (byte)in.readLong(); // 7 bytes 'not used'
 81: 		switch(val) {
 82: 			case FALSE_ENCODING:
 83: 				return Boolean.FALSE;
 84: 			case TRUE_ENCODING:
 85: 				return Boolean.TRUE;
 86: 		}
 87: 		// Don't tolerate unusual boolean encoded values (unless it becomes evident that they occur)
 88: 		throw new RuntimeException("unexpected boolean encoding (" + val + ")");
 89: 	}
 90: 
 91: 	public static int getEncodedSize(Object[] values) {
 92: 		// start with one byte 'type' code for each value
 93: 		int result = values.length * 1;
 94: 		for (int i = 0; i < values.length; i++) {
 95: 			result += getEncodedSize(values[i]);
 96: 		}
 97: 		return result;
 98: 	}
 99: 
100: 	/**
101: 	 * @return encoded size without the 'type' code byte
102: 	 */
103: 	private static int getEncodedSize(Object object) {
104: 		if(object == EMPTY_REPRESENTATION) {
105: 			return 8;
106: 		}
107: 		Class cls = object.getClass();
108: 		
109: 		if(cls == Boolean.class || cls == Double.class || cls == ErrorConstant.class) {
110: 			return 8;
111: 		}
112: 		String strVal = (String)object;
113: 		return StringUtil.getEncodedSize(strVal);
114: 	}
115: 
116: 	public static void encode(LittleEndianOutput out, Object[] values) {
117: 		for (int i = 0; i < values.length; i++) {
118: 			encodeSingleValue(out, values[i]);
119: 		}
120: 	}
121: 
122: 	private static void encodeSingleValue(LittleEndianOutput out, Object value) {
123: 		if (value == EMPTY_REPRESENTATION) {
124: 			out.writeByte(TYPE_EMPTY);
125: 			out.writeLong(0L);
126: 			return;
127: 		}
128: 		if (value instanceof Boolean) {
129: 			Boolean bVal = ((Boolean)value);
130: 			out.writeByte(TYPE_BOOLEAN);
131: 			long longVal = bVal.booleanValue() ? 1L : 0L;
132: 			out.writeLong(longVal);
133: 			return;
134: 		}
135: 		if (value instanceof Double) {
136: 			Double dVal = (Double) value;
137: 			out.writeByte(TYPE_NUMBER);
138: 			out.writeDouble(dVal.doubleValue());
139: 			return;
140: 		}
141: 		if (value instanceof String) {
142: 			String val = (String) value;
143: 			out.writeByte(TYPE_STRING);
144: 			StringUtil.writeUnicodeString(out, val);
145: 			return;
146: 		}
147: 		if (value instanceof ErrorConstant) {
148: 			ErrorConstant ecVal = (ErrorConstant) value;
149: 			out.writeByte(TYPE_ERROR_CODE);
150: 			long longVal = ecVal.getErrorCode();
151: 			out.writeLong(longVal);
152: 			return;
153: 		}
154: 
155: 		throw new IllegalStateException("Unexpected value type (" + value.getClass().getName() + "'");
156: 	}
157: }
