 46: 	/**
 47: 	 * Retrieves a single value from a variety of different argument types according to standard
 48: 	 * Excel rules.  Does not perform any type conversion.
 49: 	 * @param arg the evaluated argument as passed to the function or operator.
 50: 	 * @param srcCellRow used when arg is a single column AreaRef
 51: 	 * @param srcCellCol used when arg is a single row AreaRef
 52: 	 * @return a <tt>NumberEval</tt>, <tt>StringEval</tt>, <tt>BoolEval</tt> or <tt>BlankEval</tt>.
 53: 	 * Never <code>null</code> or <tt>ErrorEval</tt>.
 54: 	 * @throws EvaluationException(#VALUE!) if srcCellRow or srcCellCol do not properly index into
 55: 	 *  an AreaEval.  If the actual value retrieved is an ErrorEval, a corresponding
 56: 	 *  EvaluationException is thrown.
 57: 	 */
 58: 	public static ValueEval getSingleValue(ValueEval arg, int srcCellRow, int srcCellCol)
 59: 			throws EvaluationException {
 60: 		ValueEval result;
 61: 		if (arg instanceof RefEval) {
 62: 			result = ((RefEval) arg).getInnerValueEval();
 63: 		} else if (arg instanceof AreaEval) {
 64: 			result = chooseSingleElementFromArea((AreaEval) arg, srcCellRow, srcCellCol);
 65: 		} else {
 66: 			result = arg;
 67: 		}
 68: 		if (result instanceof ErrorEval) {
 69: 			throw new EvaluationException((ErrorEval) result);
 70: 		}
 71: 		return result;
 72: 	}

 74: 	/**
 75: 	 * Implements (some perhaps not well known) Excel functionality to select a single cell from an
 76: 	 * area depending on the coordinates of the calling cell.  Here is an example demonstrating
 77: 	 * both selection from a single row area and a single column area in the same formula.
 78: 	 *
 79: 	 *    <table border="1" cellpadding="1" cellspacing="1" summary="sample spreadsheet">
 80: 	 *      <tr><th>&nbsp;</th><th>&nbsp;A&nbsp;</th><th>&nbsp;B&nbsp;</th><th>&nbsp;C&nbsp;</th><th>&nbsp;D&nbsp;</th></tr>
 81: 	 *      <tr><th>1</th><td>15</td><td>20</td><td>25</td><td>&nbsp;</td></tr>
 82: 	 *      <tr><th>2</th><td>&nbsp;</td><td>&nbsp;</td><td>&nbsp;</td><td>200</td></tr>
 83: 	 *      <tr><th>3</th><td>&nbsp;</td><td>&nbsp;</td><td>&nbsp;</td><td>300</td></tr>
 84: 	 *      <tr><th>3</th><td>&nbsp;</td><td>&nbsp;</td><td>&nbsp;</td><td>400</td></tr>
 85: 	 *    </table>
 86: 	 *
 87: 	 * If the formula "=1000+A1:B1+D2:D3" is put into the 9 cells from A2 to C4, the spreadsheet
 88: 	 * will look like this:
 89: 	 *
 90: 	 *    <table border="1" cellpadding="1" cellspacing="1" summary="sample spreadsheet">
 91: 	 *      <tr><th>&nbsp;</th><th>&nbsp;A&nbsp;</th><th>&nbsp;B&nbsp;</th><th>&nbsp;C&nbsp;</th><th>&nbsp;D&nbsp;</th></tr>
 92: 	 *      <tr><th>1</th><td>15</td><td>20</td><td>25</td><td>&nbsp;</td></tr>
 93: 	 *      <tr><th>2</th><td>1215</td><td>1220</td><td>#VALUE!</td><td>200</td></tr>
 94: 	 *      <tr><th>3</th><td>1315</td><td>1320</td><td>#VALUE!</td><td>300</td></tr>
 95: 	 *      <tr><th>4</th><td>#VALUE!</td><td>#VALUE!</td><td>#VALUE!</td><td>400</td></tr>
 96: 	 *    </table>
 97: 	 *
 98: 	 * Note that the row area (A1:B1) does not include column C and the column area (D2:D3) does
 99: 	 * not include row 4, so the values in C1(=25) and D4(=400) are not accessible to the formula
100: 	 * as written, but in the 4 cells A2:B3, the row and column selection works ok.<p/>
101: 	 *
102: 	 * The same concept is extended to references across sheets, such that even multi-row,
103: 	 * multi-column areas can be useful.<p/>
104: 	 *
105: 	 * Of course with carefully (or carelessly) chosen parameters, cyclic references can occur and
106: 	 * hence this method <b>can</b> throw a 'circular reference' EvaluationException.  Note that
107: 	 * this method does not attempt to detect cycles.  Every cell in the specified Area <tt>ae</tt>
108: 	 * has already been evaluated prior to this method call.  Any cell (or cell<b>s</b>) part of
109: 	 * <tt>ae</tt> that would incur a cyclic reference error if selected by this method, will
110: 	 * already have the value <t>ErrorEval.CIRCULAR_REF_ERROR</tt> upon entry to this method.  It
111: 	 * is assumed logic exists elsewhere to produce this behaviour.
112: 	 *
113: 	 * @return whatever the selected cell's evaluated value is.  Never <code>null</code>. Never
114: 	 *  <tt>ErrorEval</tt>.
115: 	 * @throws EvaluationException if there is a problem with indexing into the area, or if the
116: 	 *  evaluated cell has an error.
117: 	 */
118: 	public static ValueEval chooseSingleElementFromArea(AreaEval ae,
119: 			int srcCellRow, int srcCellCol) throws EvaluationException {
120: 		ValueEval result = chooseSingleElementFromAreaInternal(ae, srcCellRow, srcCellCol);
121: 		if (result instanceof ErrorEval) {
122: 			throw new EvaluationException((ErrorEval) result);
123: 		}
124: 		return result;
125: 	}

127: 	/**
128: 	 * @return possibly <tt>ErrorEval</tt>, and <code>null</code>
129: 	 */
130: 	private static ValueEval chooseSingleElementFromAreaInternal(AreaEval ae,
131: 			int srcCellRow, int srcCellCol) throws EvaluationException {
132: 
133: 		if(false) {
134: 			// this is too simplistic
135: 			if(ae.containsRow(srcCellRow) && ae.containsColumn(srcCellCol)) {
136: 				throw new EvaluationException(ErrorEval.CIRCULAR_REF_ERROR);
137: 			}
138: 		/*
139: 		Circular references are not dealt with directly here, but it is worth noting some issues.
140: 
141: 		ANY one of the return statements in this method could return a cell that is identical
142: 		to the one immediately being evaluated.  The evaluating cell is identified by srcCellRow,
143: 		srcCellRow AND sheet.  The sheet is not available in any nearby calling method, so that's
144: 		one reason why circular references are not easy to detect here. (The sheet of the returned
145: 		cell can be obtained from ae if it is an Area3DEval.)
146: 
147: 		Another reason there's little value in attempting to detect circular references here is
148: 		that only direct circular references could be detected.  If the cycle involved two or more
149: 		cells this method could not detect it.
150: 
151: 		Logic to detect evaluation cycles of all kinds has been coded in EvaluationCycleDetector
152: 		(and FormulaEvaluator).
153: 		 */
154: 		}
155: 
156: 		if (ae.isColumn()) {
157: 			if(ae.isRow()) {
158: 				return ae.getRelativeValue(0, 0);
159: 			}
160: 			if(!ae.containsRow(srcCellRow)) {
161: 				throw EvaluationException.invalidValue();
162: 			}
163: 			return ae.getAbsoluteValue(srcCellRow, ae.getFirstColumn());
164: 		}
165: 		if(!ae.isRow()) {
166: 			// multi-column, multi-row area
167: 			if(ae.containsRow(srcCellRow) && ae.containsColumn(srcCellCol)) {
168: 				return ae.getAbsoluteValue(ae.getFirstRow(), ae.getFirstColumn());
169: 			}
170: 			throw EvaluationException.invalidValue();
171: 		}
172: 		if(!ae.containsColumn(srcCellCol)) {
173: 			throw EvaluationException.invalidValue();
174: 		}
175: 		return ae.getAbsoluteValue(ae.getFirstRow(), srcCellCol);
176: 	}

178: 	/**
179: 	 * Applies some conversion rules if the supplied value is not already an integer.<br/>
180: 	 * Value is first coerced to a <tt>double</tt> ( See <tt>coerceValueToDouble()</tt> ).
181: 	 * Note - <tt>BlankEval</tt> is converted to <code>0</code>.<p/>
182: 	 *
183: 	 * Excel typically converts doubles to integers by truncating toward negative infinity.<br/>
184: 	 * The equivalent java code is:<br/>
185: 	 * &nbsp;&nbsp;<code>return (int)Math.floor(d);</code><br/>
186: 	 * <b>not</b>:<br/>
187: 	 * &nbsp;&nbsp;<code>return (int)d; // wrong - rounds toward zero</code>
188: 	 *
189: 	 */
190: 	public static int coerceValueToInt(ValueEval ev) throws EvaluationException {
191: 		if (ev == BlankEval.instance) {
192: 			return 0;
193: 		}
194: 		double d = coerceValueToDouble(ev);
195: 		// Note - the standard java type conversion from double to int truncates toward zero.
196: 		// but Math.floor() truncates toward negative infinity
197: 		return (int)Math.floor(d);
198: 	}

200: 	/**
201: 	 * Applies some conversion rules if the supplied value is not already a number.
202: 	 * Note - <tt>BlankEval</tt> is converted to {@link NumberEval#ZERO}.
203: 	 * @param ev must be a {@link NumberEval}, {@link StringEval}, {@link BoolEval} or
204: 	 * {@link BlankEval}
205: 	 * @return actual, parsed or interpreted double value (respectively).
206: 	 * @throws EvaluationException(#VALUE!) only if a StringEval is supplied and cannot be parsed
207: 	 * as a double (See <tt>parseDouble()</tt> for allowable formats).
208: 	 * @throws RuntimeException if the supplied parameter is not {@link NumberEval},
209: 	 * {@link StringEval}, {@link BoolEval} or {@link BlankEval}
210: 	 */
211: 	public static double coerceValueToDouble(ValueEval ev) throws EvaluationException {
212: 
213: 		if (ev == BlankEval.instance) {
214: 			return 0.0;
215: 		}
216: 		if (ev instanceof NumericValueEval) {
217: 			// this also handles booleans
218: 			return ((NumericValueEval)ev).getNumberValue();
219: 		}
220: 		if (ev instanceof StringEval) {
221: 			Double dd = parseDouble(((StringEval) ev).getStringValue());
222: 			if (dd == null) {
223: 				throw EvaluationException.invalidValue();
224: 			}
225: 			return dd.doubleValue();
226: 		}
227: 		throw new RuntimeException("Unexpected arg eval type (" + ev.getClass().getName() + ")");
228: 	}

230: 	/**
231: 	 * Converts a string to a double using standard rules that Excel would use.<br/>
232: 	 * Tolerates leading and trailing spaces, <p/>
233: 	 * 
234: 	 * Doesn't support currency prefixes, commas, percentage signs or arithmetic operations strings.  
235: 	 *
236: 	 *  Some examples:<br/>
237: 	 *  " 123 " -&gt; 123.0<br/>
238: 	 *  ".123" -&gt; 0.123<br/>
239: 	 *  "1E4" -&gt; 1000<br/>
240: 	 *  "-123" -&gt; -123.0<br/>
241: 	 *  These not supported yet:<br/>
242: 	 *  " $ 1,000.00 " -&gt; 1000.0<br/>
243: 	 *  "$1.25E4" -&gt; 12500.0<br/>
244: 	 *  "5**2" -&gt; 500<br/>
245: 	 *  "250%" -&gt; 2.5<br/>
246: 	 *
247: 	 * @return <code>null</code> if the specified text cannot be parsed as a number
248: 	 */
249: 	public static Double parseDouble(String pText) {
250: 		
251: 	    if (Pattern.matches(fpRegex, pText))
252: 			try {
253: 				return Double.parseDouble(pText);
254: 			} catch (NumberFormatException e) {
255: 				return null;
256: 			}
257: 	    else {
258: 	        return null;
259: 	    }
260: 		
261: 	}

263: 	/**
264: 	 * @param ve must be a <tt>NumberEval</tt>, <tt>StringEval</tt>, <tt>BoolEval</tt>, or <tt>BlankEval</tt>
265: 	 * @return the converted string value. never <code>null</code>
266: 	 */
267: 	public static String coerceValueToString(ValueEval ve) {
268: 		if (ve instanceof StringValueEval) {
269: 			StringValueEval sve = (StringValueEval) ve;
270: 			return sve.getStringValue();
271: 		}
272: 		if (ve == BlankEval.instance) {
273: 			return "";

278: 	/**
279: 	 * @return <code>null</code> to represent blank values
280: 	 * @throws EvaluationException if ve is an ErrorEval, or if a string value cannot be converted
281: 	 */
282: 	public static Boolean coerceValueToBoolean(ValueEval ve, boolean stringsAreBlanks) throws EvaluationException {
283: 
284: 		if (ve == null || ve == BlankEval.instance) {
285: 			// TODO - remove 've == null' condition once AreaEval is fixed
286: 			return null;
287: 		}
288: 		if (ve instanceof BoolEval) {
289: 			return Boolean.valueOf(((BoolEval) ve).getBooleanValue());
290: 		}
291: 
292: 		if (ve == BlankEval.instance) {
293: 			return null;
294: 		}
295: 
296: 		if (ve instanceof StringEval) {
297: 			if (stringsAreBlanks) {
298: 				return null;
299: 			}
300: 			String str = ((StringEval) ve).getStringValue();
301: 			if (str.equalsIgnoreCase("true")) {
302: 				return Boolean.TRUE;
303: 			}
304: 			if (str.equalsIgnoreCase("false")) {
305: 				return Boolean.FALSE;
306: 			}
307: 			// else - string cannot be converted to boolean
308: 			throw new EvaluationException(ErrorEval.VALUE_INVALID);
309: 		}
310: 
311: 		if (ve instanceof NumericValueEval) {
312: 			NumericValueEval ne = (NumericValueEval) ve;
313: 			double d = ne.getNumberValue();
314: 			if (Double.isNaN(d)) {
315: 				throw new EvaluationException(ErrorEval.VALUE_INVALID);
316: 			}
317: 			return Boolean.valueOf(d != 0);
318: 		}
319: 		if (ve instanceof ErrorEval) {
320: 			throw new EvaluationException((ErrorEval) ve);
321: 		}
322: 		throw new RuntimeException("Unexpected eval (" + ve.getClass().getName() + ")");
323: 	}
