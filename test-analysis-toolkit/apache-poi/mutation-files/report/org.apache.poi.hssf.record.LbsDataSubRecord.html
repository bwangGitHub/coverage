 93:     public LbsDataSubRecord(LittleEndianInput in, int cbFContinued, int cmoOt) {
 94:         _cbFContinued = cbFContinued;
 95: 
 96:         int encodedTokenLen = in.readUShort();
 97:         if (encodedTokenLen > 0) {
 98:             int formulaSize = in.readUShort();
 99:             _unknownPreFormulaInt = in.readInt();
100: 
101:             Ptg[] ptgs = Ptg.readTokens(formulaSize, in);
102:             if (ptgs.length != 1) {
103:                 throw new RecordFormatException("Read " + ptgs.length
104:                         + " tokens but expected exactly 1");
105:             }
106:             _linkPtg = ptgs[0];
107:             switch (encodedTokenLen - formulaSize - 6) {
108:                 case 1:
109:                     _unknownPostFormulaByte = in.readByte();
110:                     break;
111:                 case 0:
112:                     _unknownPostFormulaByte = null;
113:                     break;
114:                 default:
115:                     throw new RecordFormatException("Unexpected leftover bytes");
116:             }
117:         }
118: 
119:         _cLines = in.readUShort();
120:         _iSel = in.readUShort();
121:         _flags = in.readUShort();
122:         _idEdit = in.readUShort();
123: 
124:         // From [MS-XLS].pdf 2.5.147 FtLbsData:
125:         // This field MUST exist if and only if the containing Obj?s cmo.ot is equal to 0x14.
126:         if(cmoOt == 0x14) {
127:             _dropData = new LbsDropData(in);
128:         }
129: 
130:         // From [MS-XLS].pdf 2.5.147 FtLbsData:
131:         // This array MUST exist if and only if the fValidPlex flag (0x2) is set
132:         if((_flags & 0x2) != 0) {
133:             _rgLines = new String[_cLines];
134:             for(int i=0; i < _cLines; i++) {
135:                 _rgLines[i] = StringUtil.readUnicodeString(in);
136:             }
137:         }
138: 
139:         // bits 5-6 in the _flags specify the type
140:         // of selection behavior this list control is expected to support
141: 
142:         // From [MS-XLS].pdf 2.5.147 FtLbsData:
143:         // This array MUST exist if and only if the wListType field is not equal to 0.
144:         if(((_flags >> 4) & 0x2) != 0) {
145:             _bsels = new boolean[_cLines];
146:             for(int i=0; i < _cLines; i++) {
147:                 _bsels[i] = in.readByte() == 1;
148:             }
149:         }
150: 
151:     }
