 45: 	public ValueEval evaluate(ValueEval[] args, OperationEvaluationContext ec) {
 46:         double number, multiple, result;
 47: 
 48:         if (args.length != 2) {
 49:             return ErrorEval.VALUE_INVALID;
 50:         }
 51: 
 52:         try {
 53:             number = OperandResolver.coerceValueToDouble(OperandResolver.getSingleValue(args[0], ec.getRowIndex(), ec.getColumnIndex()));
 54:             multiple = OperandResolver.coerceValueToDouble(OperandResolver.getSingleValue(args[1], ec.getRowIndex(), ec.getColumnIndex()));
 55: 
 56:             if( multiple == 0.0 ) {
 57:                 result = 0.0;
 58:             } else {
 59:                 if(number*multiple < 0) {
 60:                     // Returns #NUM! because the number and the multiple have different signs
 61:                     throw new EvaluationException(ErrorEval.NUM_ERROR);
 62:                 }
 63:                 result = multiple * Math.round( number / multiple );
 64:             }
 65:             NumericFunction.checkValue(result);
