 56:     public static final class SpaceType {
 57:         private SpaceType() {
 58:             // no instances of this class

 61:         /** 00H = Spaces before the next token (not allowed before tParen token) */
 62:         public static final int SPACE_BEFORE = 0x00;
 63:         /** 01H = Carriage returns before the next token (not allowed before tParen token) */
 64:         public static final int CR_BEFORE = 0x01;
 65:         /** 02H = Spaces before opening parenthesis (only allowed before tParen token) */
 66:         public static final int SPACE_BEFORE_OPEN_PAREN = 0x02;
 67:         /** 03H = Carriage returns before opening parenthesis (only allowed before tParen token) */
 68:         public static final int CR_BEFORE_OPEN_PAREN = 0x03;
 69:         /** 04H = Spaces before closing parenthesis (only allowed before tParen, tFunc, and tFuncVar tokens) */
 70:         public static final int SPACE_BEFORE_CLOSE_PAREN = 0x04;
 71:         /** 05H = Carriage returns before closing parenthesis (only allowed before tParen, tFunc, and tFuncVar tokens) */
 72:         public static final int CR_BEFORE_CLOSE_PAREN = 0x05;
 73:         /** 06H = Spaces following the equality sign (only in macro sheets) */
 74:         public static final int SPACE_AFTER_EQUALITY = 0x06;

 77:     public AttrPtg(LittleEndianInput in) {
 78:         _options = in.readByte();
 79:         _data    = in.readShort();
 80:         if (isOptimizedChoose()) {
 81:             int nCases = _data;
 82:             int[] jumpTable = new int[nCases];
 83:             for (int i = 0; i < jumpTable.length; i++) {
 84:                 jumpTable[i] = in.readUShort();
 85:             }
 86:             _jumpTable = jumpTable;
 87:             _chooseFuncOffset = in.readUShort();
 88:         } else {
 89:             _jumpTable = null;
 90:             _chooseFuncOffset = -1;
 91:         }
 92: 
 93:     }
 94:     private AttrPtg(int options, int data, int[] jt, int chooseFuncOffset) {
 95:         _options = (byte) options;
 96:         _data = (short) data;
 97:         _jumpTable = jt;
 98:         _chooseFuncOffset = chooseFuncOffset;
 99:     }

171:     public String toString() {
172:         StringBuffer sb = new StringBuffer(64);
173:         sb.append(getClass().getName()).append(" [");
174: 
175:         if(isSemiVolatile()) {
176:             sb.append("volatile ");
177:         }
178:         if(isSpace()) {
179:             sb.append("space count=").append((_data >> 8) & 0x00FF);
180:             sb.append(" type=").append(_data & 0x00FF).append(" ");
181:         }
182:         // the rest seem to be mutually exclusive
183:         if(isOptimizedIf()) {
184:             sb.append("if dist=").append(_data);
185:         } else if(isOptimizedChoose()) {
186:             sb.append("choose nCases=").append(_data);
187:         } else if(isSkip()) {
188:             sb.append("skip dist=").append(_data);
189:         } else if(isSum()) {
190:             sb.append("sum ");
191:         } else if(isBaxcel()) {
192:             sb.append("assign ");
193:         }
194:         sb.append("]");
195:         return sb.toString();

198:     public void write(LittleEndianOutput out) {
199:         out.writeByte(sid + getPtgClass());
200:         out.writeByte(_options);
201:         out.writeShort(_data);
202:         int[] jt = _jumpTable;
203:         if (jt != null) {
204:             for (int i = 0; i < jt.length; i++) {
205:                 out.writeShort(jt[i]);
206:             }
207:             out.writeShort(_chooseFuncOffset);
208:         }
