 75: 	/**
 76: 	 * @param rs record stream with all {@link SharedFormulaRecord}
 77: 	 * {@link ArrayRecord}, {@link TableRecord} {@link MergeCellsRecord} Records removed
 78: 	 * @param svm an initialised {@link SharedValueManager} (from the shared formula, array
 79: 	 * and table records of the current sheet).  Never <code>null</code>.
 80: 	 */
 81: 	public RowRecordsAggregate(RecordStream rs, SharedValueManager svm) {
 82: 		this(svm);
 83: 		while(rs.hasNext()) {
 84: 			Record rec = rs.getNext();
 85: 			switch (rec.getSid()) {
 86: 				case RowRecord.sid:
 87: 					insertRow((RowRecord) rec);
 88: 					continue;
 89: 				case DBCellRecord.sid:
 90: 					// end of 'Row Block'.  Should only occur after cell records
 91: 					// ignore DBCELL records because POI generates them upon re-serialization
 92: 					continue;
 93: 			}
 94: 			if (rec instanceof UnknownRecord) {
 95: 				// might need to keep track of where exactly these belong
 96: 				addUnknownRecord(rec);
 97: 				while (rs.peekNextSid() == ContinueRecord.sid) {
 98: 					addUnknownRecord(rs.getNext());
 99: 				}
100: 				continue;
101: 			}
102: 			if (rec instanceof MulBlankRecord) {
103: 				_valuesAgg.addMultipleBlanks((MulBlankRecord) rec);
104: 				continue;
105: 			}
106: 			if (!(rec instanceof CellValueRecordInterface)) {
107: 				throw new RuntimeException("Unexpected record type (" + rec.getClass().getName() + ")");
108: 			}
109: 			_valuesAgg.construct((CellValueRecordInterface)rec, rs, svm);
110: 		}
111: 	}

235: 	private int visitRowRecordsForBlock(int blockIndex, RecordVisitor rv) {
236: 		final int startIndex = blockIndex*DBCellRecord.BLOCK_SIZE;
237: 		final int endIndex = startIndex + DBCellRecord.BLOCK_SIZE;
238: 
239: 		Iterator<RowRecord> rowIterator = _rowRecords.values().iterator();
240: 
241: 		//Given that we basically iterate through the rows in order,
242: 		//For a performance improvement, it would be better to return an instance of
243: 		//an iterator and use that instance throughout, rather than recreating one and
244: 		//having to move it to the right position.
245: 		int i=0;
246: 		for (;i<startIndex;i++)
247: 		  rowIterator.next();
248: 		int result = 0;
249: 		while(rowIterator.hasNext() && (i++ < endIndex)) {
250: 		  Record rec = (Record)rowIterator.next();
251: 		  result += rec.getRecordSize();
252: 		  rv.visitRecord(rec);
253: 		}
254: 		return result;
255: 	}

257: 	public void visitContainedRecords(RecordVisitor rv) {
258: 
259: 		PositionTrackingVisitor stv = new PositionTrackingVisitor(rv, 0);
260: 		//DBCells are serialized before row records.
261: 		final int blockCount = getRowBlockCount();
262: 		for (int blockIndex = 0; blockIndex < blockCount; blockIndex++) {
263: 			// Serialize a block of rows.
264: 			// Hold onto the position of the first row in the block
265: 			int pos=0;
266: 			// Hold onto the size of this block that was serialized
267: 			final int rowBlockSize = visitRowRecordsForBlock(blockIndex, rv);
268: 			pos += rowBlockSize;
269: 			// Serialize a block of cells for those rows
270: 			final int startRowNumber = getStartRowNumberForBlock(blockIndex);
271: 			final int endRowNumber = getEndRowNumberForBlock(blockIndex);
272: 			DBCellRecord.Builder dbcrBuilder = new DBCellRecord.Builder();
273: 			// Note: Cell references start from the second row...
274: 			int cellRefOffset = (rowBlockSize - RowRecord.ENCODED_SIZE);
275: 			for (int row = startRowNumber; row <= endRowNumber; row++) {
276: 				if (_valuesAgg.rowHasCells(row)) {
277: 					stv.setPosition(0);
278: 					_valuesAgg.visitCellsForRow(row, stv);
279: 					int rowCellSize = stv.getPosition();
280: 					pos += rowCellSize;
281: 					// Add the offset to the first cell for the row into the
282: 					// DBCellRecord.
283: 					dbcrBuilder.addCellOffset(cellRefOffset);
284: 					cellRefOffset = rowCellSize;
285: 				}
286: 			}
287: 			// Calculate Offset from the start of a DBCellRecord to the first Row
288: 			rv.visitRecord(dbcrBuilder.build(pos));
289: 		}
290: 		for (int i=0; i< _unknownRecords.size(); i++) {
291: 			// Potentially breaking the file here since we don't know exactly where to write these records
292: 			rv.visitRecord(_unknownRecords.get(i));
293: 		}
294: 	}

300: 	public int findStartOfRowOutlineGroup(int row) {
301: 		// Find the start of the group.
302: 		RowRecord rowRecord = this.getRow( row );
303: 		int level = rowRecord.getOutlineLevel();
304: 		int currentRow = row;
305: 		while (this.getRow( currentRow ) != null) {
306: 			rowRecord = this.getRow( currentRow );
307: 			if (rowRecord.getOutlineLevel() < level) {
308: 				return currentRow + 1;
309: 			}
310: 			currentRow--;
311: 		}
312: 
313: 		return currentRow + 1;
314: 	}

316: 	public int findEndOfRowOutlineGroup(int row) {
317: 		int level = getRow( row ).getOutlineLevel();
318: 		int currentRow;
319: 		for (currentRow = row; currentRow < getLastRowNum(); currentRow++) {
320: 			if (getRow(currentRow) == null || getRow(currentRow).getOutlineLevel() < level) {
321: 				break;
322: 			}
323: 		}
324: 
325: 		return currentRow-1;
326: 	}

330: 	 * @return index of the <em>next<em> row after the last row that gets hidden

332: 	private int writeHidden(RowRecord pRowRecord, int row) {
333: 		int rowIx = row;

335: 		int level = rowRecord.getOutlineLevel();
336: 		while (rowRecord != null && getRow(rowIx).getOutlineLevel() >= level) {
337: 			rowRecord.setZeroHeight(true);

344: 	public void collapseRow(int rowNumber) {
345: 
346: 		// Find the start of the group.
347: 		int startRow = findStartOfRowOutlineGroup(rowNumber);
348: 		RowRecord rowRecord = getRow(startRow);
349: 
350: 		// Hide all the columns until the end of the group
351: 		int nextRowIx = writeHidden(rowRecord, startRow);
352: 
353: 		RowRecord row = getRow(nextRowIx);
354: 		if (row == null) {
355: 			row = createRow(nextRowIx);
356: 			insertRow(row);
357: 		}
358: 		// Write collapse field
359: 		row.setColapsed(true);
360: 	}

382: 	public void expandRow(int rowNumber) {
383: 		int idx = rowNumber;
384: 		if (idx == -1)
385: 			return;
386: 
387: 		// If it is already expanded do nothing.
388: 		if (!isRowGroupCollapsed(idx)) {
389: 			return;
390: 		}
391: 
392: 		// Find the start of the group.
393: 		int startIdx = findStartOfRowOutlineGroup(idx);
394: 		RowRecord row = getRow(startIdx);
395: 
396: 		// Find the end of the group.
397: 		int endIdx = findEndOfRowOutlineGroup(idx);
398: 
399: 		// expand:
400: 		// collapsed bit must be unset
401: 		// hidden bit gets unset _if_ surrounding groups are expanded you can determine
402: 		//   this by looking at the hidden bit of the enclosing group.  You will have
403: 		//   to look at the start and the end of the current group to determine which
404: 		//   is the enclosing group
405: 		// hidden bit only is altered for this outline level.  ie.  don't un-collapse contained groups
406: 		if (!isRowGroupHiddenByParent(idx)) {
407: 			for (int i = startIdx; i <= endIdx; i++) {
408: 				RowRecord otherRow = getRow(i);
409: 				if (row.getOutlineLevel() == otherRow.getOutlineLevel() || !isRowGroupCollapsed(i)) {
410: 					otherRow.setZeroHeight(false);
411: 				}
412: 			}
413: 		}
414: 
415: 		// Write collapse field
416: 		getRow(endIdx + 1).setColapsed(false);
417: 	}

419: 	public boolean isRowGroupHiddenByParent(int row) {
420: 		// Look out outline details of end
421: 		int endLevel;
422: 		boolean endHidden;
423: 		int endOfOutlineGroupIdx = findEndOfRowOutlineGroup(row);
424: 		if (getRow(endOfOutlineGroupIdx + 1) == null) {
425: 			endLevel = 0;
426: 			endHidden = false;
427: 		} else {
428: 			endLevel = getRow(endOfOutlineGroupIdx + 1).getOutlineLevel();
429: 			endHidden = getRow(endOfOutlineGroupIdx + 1).getZeroHeight();
430: 		}
431: 
432: 		// Look out outline details of start
433: 		int startLevel;
434: 		boolean startHidden;
435: 		int startOfOutlineGroupIdx = findStartOfRowOutlineGroup( row );
436: 		if (startOfOutlineGroupIdx - 1 < 0 || getRow(startOfOutlineGroupIdx - 1) == null) {
437: 			startLevel = 0;
438: 			startHidden = false;
439: 		} else {
440: 			startLevel = getRow(startOfOutlineGroupIdx - 1).getOutlineLevel();
441: 			startHidden = getRow(startOfOutlineGroupIdx - 1).getZeroHeight();
442: 		}
443: 
444: 		if (endLevel > startLevel) {

465: 	public IndexRecord createIndexRecord(int indexRecordOffset, int sizeOfInitialSheetRecords) {
466: 		IndexRecord result = new IndexRecord();
467: 		result.setFirstRow(_firstrow);
468: 		result.setLastRowAdd1(_lastrow + 1);
469: 		// Calculate the size of the records from the end of the BOF
470: 		// and up to the RowRecordsAggregate...
471: 
472: 		// Add the references to the DBCells in the IndexRecord (one for each block)
473: 		// Note: The offsets are relative to the Workbook BOF. Assume that this is
474: 		// 0 for now.....
475: 
476: 		int blockCount = getRowBlockCount();
477: 		// Calculate the size of this IndexRecord
478: 		int indexRecSize = IndexRecord.getRecordSizeForBlockCount(blockCount);
479: 
480: 		int currentOffset = indexRecordOffset + indexRecSize + sizeOfInitialSheetRecords;
481: 
482: 		for (int block = 0; block < blockCount; block++) {
483: 			// each row-block has a DBCELL record.
484: 			// The offset of each DBCELL record needs to be updated in the INDEX record
485: 
486: 			// account for row records in this row-block
487: 			currentOffset += getRowBlockSize(block);
488: 			// account for cell value records after those
489: 			currentOffset += _valuesAgg.getRowCellBlockSize(
490: 					getStartRowNumberForBlock(block), getEndRowNumberForBlock(block));
491: 
492: 			// currentOffset is now the location of the DBCELL record for this row-block
493: 			result.addDbcell(currentOffset);
494: 			// Add space required to write the DBCELL record (whose reference was just added).
495: 			currentOffset += (8 + (getRowCountForBlock(block) * 2));
496: 		}
497: 		return result;
498: 	}
