112:     /**
113:      * Opens a sample file from the test data directory
114:      *
115:      * @param  sampleFileName the file to open
116:      * @return an open <tt>InputStream</tt> for the specified sample file
117:      */
118:     public InputStream openResourceAsStream(String sampleFileName) {
119: 
120:         if (_sampleDataIsAvaliableOnClassPath) {
121:             InputStream result = sampleFileName == null ? null :
122:                     openClasspathResource(sampleFileName);
123:             if(result == null) {
124:                 throw new RuntimeException("specified test sample file '" + sampleFileName
125:                         + "' not found on the classpath");
126:             }
127:             // wrap to avoid temp warning method about auto-closing input stream
128:             return new NonSeekableInputStream(result);
129:         }
130:         if (_resolvedDataDir == null) {
131:             throw new RuntimeException("Must set system property '"
132:                     + TEST_PROPERTY
133:                     + "' properly before running tests");
134:         }
135: 
136:         File f = getFile(sampleFileName);
137:         try {
138:             return new FileInputStream(f);
139:         } catch (FileNotFoundException e) {
140:             throw new RuntimeException(e);
141:         }
142:     }
143: 
144:     /**
145:      *
146:      * @param sampleFileName    the name of the test file
147:      * @return
148:      * @throws RuntimeException if the file was not found
149:      */
150:     public File getFile(String sampleFileName) {
151:         File f = new File(_resolvedDataDir, sampleFileName);
152:         if (!f.exists()) {
153:             throw new RuntimeException("Sample file '" + sampleFileName
154:                     + "' not found in data dir '" + _resolvedDataDir.getAbsolutePath() + "'");
155:         }
156:         try {
157:             if(sampleFileName.length() > 0) {
158:                String fn = sampleFileName;
159:                if(fn.indexOf('/') > 0) {
160:                   fn = fn.substring(fn.indexOf('/')+1);
161:                }
162:                if(!fn.equals(f.getCanonicalFile().getName())){
163:                    throw new RuntimeException("File name is case-sensitive: requested '" + fn
164:                         + "' but actual file is '" + f.getCanonicalFile().getName() + "'");
165:                }
166:             }
167:         } catch (IOException e){
168:             throw new RuntimeException(e);
169:         }
170:         return f;
171:     }

173:     private void initialise() {
174:         String dataDirName = System.getProperty(TEST_PROPERTY);
175:         if (dataDirName == null) {
176:             // check to see if we can just get the resources from the classpath
177:             InputStream is = openClasspathResource("");
178:             if (is != null) {
179:                 try {
180:                     is.close(); // be nice
181:                 } catch (IOException e) {
182:                     throw new RuntimeException(e);
183:                 }
184:                 _sampleDataIsAvaliableOnClassPath = true;
185:                 return;
186:             }
187: 
188:             throw new RuntimeException("Must set system property '" +
189:                     TEST_PROPERTY + "' before running tests");
190:         }
191:         File dataDir = new File(dataDirName, _moduleDir);
192:         if (!dataDir.exists()) {
193:             throw new RuntimeException("Data dir '" + _moduleDir + " does not exist");
194:         }
195:         // convert to canonical file, to make any subsequent error messages
196:         // clearer.
197:         try {
198:             _resolvedDataDir = dataDir.getCanonicalFile();
199:         } catch (IOException e) {
200:             throw new RuntimeException(e);
201:         }
202:     }

238:      * @return byte array of sample file content from file found in standard hssf test data dir

240:     public byte[] readFile(String fileName) {
241:         ByteArrayOutputStream bos = new ByteArrayOutputStream();
242: 
243:         try {
244:             InputStream fis = openResourceAsStream(fileName);
245: 
246:             byte[] buf = new byte[512];
247:             while (true) {
248:                 int bytesRead = fis.read(buf);
249:                 if (bytesRead < 1) {
250:                     break;

253:             }
254:             fis.close();
255:         } catch (IOException e) {
256:             throw new RuntimeException(e);
257:         }
258:         return bos.toByteArray();
