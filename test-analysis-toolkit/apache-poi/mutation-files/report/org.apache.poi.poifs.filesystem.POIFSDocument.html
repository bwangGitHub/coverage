225: 	 * read data from the internal stores
226: 	 *
227: 	 * @param buffer the buffer to write to
228: 	 * @param offset the offset into our storage to read from
229: 	 * This method is currently (Oct 2008) only used by test code. Perhaps it can be deleted
230: 	 */
231: 	void read(byte[] buffer, int offset) {
232: 		int len = buffer.length;
233: 
234: 		DataInputBlock currentBlock = getDataInputBlock(offset);
235: 
236: 		int blockAvailable = currentBlock.available();
237: 		if (blockAvailable > len) {
238: 			currentBlock.readFully(buffer, 0, len);
239: 			return;
240: 		}
241: 		// else read big amount in chunks
242: 		int remaining = len;
243: 		int writePos = 0;
244: 		int currentOffset = offset;
245: 		while (remaining > 0) {
246: 			boolean blockIsExpiring = remaining >= blockAvailable;
247: 			int reqSize;
248: 			if (blockIsExpiring) {
249: 				reqSize = blockAvailable;
250: 			} else {
251: 				reqSize = remaining;
252: 			}
253: 			currentBlock.readFully(buffer, writePos, reqSize);
254: 			remaining-=reqSize;
255: 			writePos+=reqSize;
256: 			currentOffset += reqSize;
257: 			if (blockIsExpiring) {
258: 				if (currentOffset == _size) {
259: 					if (remaining > 0) {
260: 						throw new IllegalStateException("reached end of document stream unexpectedly");
261: 					}
262: 					currentBlock = null;
263: 					break;
264: 				}
265: 				currentBlock = getDataInputBlock(currentOffset);
266: 				blockAvailable = currentBlock.available();
267: 			}
268: 		}
269: 	}
