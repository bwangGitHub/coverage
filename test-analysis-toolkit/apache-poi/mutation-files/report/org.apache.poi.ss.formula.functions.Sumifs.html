 26: /**
 27:  * Implementation for the Excel function SUMIFS<p>
 28:  *
 29:  * Syntax : <br/>
 30:  *  SUMIFS ( <b>sum_range</b>, <b>criteria_range1</b>, <b>criteria1</>,
 31:  *  [<b>criteria_range2</b>,  <b>criteria2</b>], ...) <br/>
 32:  *    <ul>
 33:  *      <li><b>sum_range</b> Required. One or more cells to sum, including numbers or names, ranges,
 34:  *      or cell references that contain numbers. Blank and text values are ignored.</li>
 35:  *      <li><b>criteria1_range</b> Required. The first range in which
 36:  *      to evaluate the associated criteria.</li>
 37:  *      <li><b>criteria1</b> Required. The criteria in the form of a number, expression,
 38:  *        cell reference, or text that define which cells in the criteria_range1
 39:  *        argument will be added</li>
 40:  *      <li><b> criteria_range2, criteria2, ...</b>    Optional. Additional ranges and their associated criteria.
 41:  *      Up to 127 range/criteria pairs are allowed.
 42:  *    </ul>
 43:  * </p>
 44:  *
 45:  * @author Yegor Kozlov
 46:  */
 47: public final class Sumifs implements FreeRefFunction {
 48:     public static final FreeRefFunction instance = new Sumifs();
 49: 
 50: 	public ValueEval evaluate(ValueEval[] args, OperationEvaluationContext ec) {
 51:         if(args.length < 3 || args.length % 2 == 0) {
 52:             return ErrorEval.VALUE_INVALID;
 53:         }
 54: 
 55: 		try {
 56:             AreaEval sumRange = convertRangeArg(args[0]);
 57: 
 58:             // collect pairs of ranges and criteria
 59:             AreaEval[] ae = new AreaEval[(args.length - 1)/2];
 60:             I_MatchPredicate[] mp = new I_MatchPredicate[ae.length];
 61:             for(int i = 1, k=0; i < args.length; i += 2, k++){
 62:                 ae[k] = convertRangeArg(args[i]);
 63:                 mp[k] = Countif.createCriteriaPredicate(args[i+1], ec.getRowIndex(), ec.getColumnIndex());
 64:             }
 65: 
 66:             validateCriteriaRanges(ae, sumRange);
 67: 
 68:             double result = sumMatchingCells(ae, mp, sumRange);
 69:             return new NumberEval(result);
 70: 		} catch (EvaluationException e) {
 71: 			return e.getErrorEval();
 72: 		}
 73: 	}
 74: 
 75:     /**
 76:      * Verify that each <code>criteriaRanges</code> argument contains the same number of rows and columns
 77:      * as the <code>sumRange</code> argument
 78:      *
 79:      * @throws EvaluationException if
 80:      */
 81:     private void validateCriteriaRanges(AreaEval[] criteriaRanges, AreaEval sumRange) throws EvaluationException {
 82:         for(AreaEval r : criteriaRanges){
 83:             if(r.getHeight() != sumRange.getHeight() ||
 84:                r.getWidth() != sumRange.getWidth() ) {
 85:                 throw EvaluationException.invalidValue();
 86:             }
 87:         }
 88:     }
 89: 
 90:     /**
 91:      *
 92:      * @param ranges  criteria ranges, each range must be of the same dimensions as <code>aeSum</code>
 93:      * @param predicates  array of predicates, a predicate for each value in <code>ranges</code>
 94:      * @param aeSum  the range to sum
 95:      *
 96:      * @return the computed value
 97:      */
 98:     private static double sumMatchingCells(AreaEval[] ranges, I_MatchPredicate[] predicates, AreaEval aeSum) {
 99:         int height = aeSum.getHeight();
100:         int width = aeSum.getWidth();
101: 
102:         double result = 0.0;
103:         for (int r = 0; r < height; r++) {
104:             for (int c = 0; c < width; c++) {
105: 
106:                 boolean matches = true;
107:                 for(int i = 0; i < ranges.length; i++){
108:                     AreaEval aeRange = ranges[i];
109:                     I_MatchPredicate mp = predicates[i];
110: 
111:                     if (!mp.matches(aeRange.getRelativeValue(r, c))) {
112:                         matches = false;
113:                         break;
114:                     }
115: 
116:                 }
117: 
118:                 if(matches) { // sum only if all of the corresponding criteria specified are true for that cell.
119:                     result += accumulate(aeSum, r, c);
120:                 }
121:             }
122:         }
123:         return result;
124:     }
125: 
126: 	private static double accumulate(AreaEval aeSum, int relRowIndex,
127: 			int relColIndex) {
128: 
129: 		ValueEval addend = aeSum.getRelativeValue(relRowIndex, relColIndex);
130: 		if (addend instanceof NumberEval) {
131: 			return ((NumberEval)addend).getNumberValue();
132: 		}
133: 		// everything else (including string and boolean values) counts as zero
134: 		return 0.0;
135: 	}
136: 
137: 	private static AreaEval convertRangeArg(ValueEval eval) throws EvaluationException {
138: 		if (eval instanceof AreaEval) {
139: 			return (AreaEval) eval;
140: 		}
141: 		if (eval instanceof RefEval) {
142: 			return ((RefEval)eval).offset(0, 0, 0, 0);
143: 		}
144: 		throw new EvaluationException(ErrorEval.VALUE_INVALID);
145: 	}
