 33: /**
 34:  * Represents a workbook being used for forked evaluation. Most operations are delegated to the
 35:  * shared master workbook, except those that potentially involve cell values that may have been
 36:  * updated after a call to {@link #getOrCreateUpdatableCell(String, int, int)}.
 37:  *
 38:  * @author Josh Micich
 39:  */
 40: final class ForkedEvaluationWorkbook implements EvaluationWorkbook {
 41: 
 42: 	private final EvaluationWorkbook _masterBook;
 43: 	private final Map<String, ForkedEvaluationSheet> _sharedSheetsByName;
 44: 
 45: 	public ForkedEvaluationWorkbook(EvaluationWorkbook master) {
 46: 		_masterBook = master;
 47: 		_sharedSheetsByName = new HashMap<String, ForkedEvaluationSheet>();
 48: 	}

 50: 	public ForkedEvaluationCell getOrCreateUpdatableCell(String sheetName, int rowIndex,
 51: 			int columnIndex) {
 52: 		ForkedEvaluationSheet sheet = getSharedSheet(sheetName);
 53: 		return sheet.getOrCreateUpdatableCell(rowIndex, columnIndex);
 54: 	}
 55: 
 56: 	public EvaluationCell getEvaluationCell(String sheetName, int rowIndex, int columnIndex) {
 57: 		ForkedEvaluationSheet sheet = getSharedSheet(sheetName);
 58: 		return sheet.getCell(rowIndex, columnIndex);
 59: 	}

 61: 	private ForkedEvaluationSheet getSharedSheet(String sheetName) {
 62: 		ForkedEvaluationSheet result = _sharedSheetsByName.get(sheetName);
 63: 		if (result == null) {
 64: 			result = new ForkedEvaluationSheet(_masterBook.getSheet(_masterBook
 65: 					.getSheetIndex(sheetName)));
 66: 			_sharedSheetsByName.put(sheetName, result);
 67: 		}
 68: 		return result;
 69: 	}
 70: 
 71: 	public void copyUpdatedCells(Workbook workbook) {
 72: 		String[] sheetNames = new String[_sharedSheetsByName.size()];
 73: 		_sharedSheetsByName.keySet().toArray(sheetNames);
 74: 		OrderedSheet[] oss = new OrderedSheet[sheetNames.length];
 75: 		for (int i = 0; i < sheetNames.length; i++) {
 76: 			String sheetName = sheetNames[i];
 77: 			oss[i] = new OrderedSheet(sheetName, _masterBook.getSheetIndex(sheetName));
 78: 		}
 79: 		for (int i = 0; i < oss.length; i++) {
 80: 			String sheetName = oss[i].getSheetName();
 81: 			ForkedEvaluationSheet sheet = _sharedSheetsByName.get(sheetName);
 82: 			sheet.copyUpdatedCells(workbook.getSheet(sheetName));
 83: 		}
 84: 	}
