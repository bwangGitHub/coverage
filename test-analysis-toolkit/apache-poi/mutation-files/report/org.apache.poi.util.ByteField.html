 41:     /**

 54:     }

 56:     /**

 62:      *
 63:      * @exception ArrayIndexOutOfBoundsException if offset is negative
 64:      */
 65: 
 66:     public ByteField(final int offset, final byte value)
 67:         throws ArrayIndexOutOfBoundsException

163:      *            of range of the bte array

165: 
166:     public void readFromBytes(final byte [] data)
167:         throws ArrayIndexOutOfBoundsException
168:     {
169:         _value = data[ _offset ];
170:     }

183:

205: 
206:     public void writeToBytes(final byte [] data)
