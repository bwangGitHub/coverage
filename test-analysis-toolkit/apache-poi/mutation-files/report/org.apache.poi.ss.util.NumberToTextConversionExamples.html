 30: 	public static final class ExampleConversion {
 31: 		private final String _javaRendering;
 32: 		private final String _excelRendering;
 33: 		private final double _doubleValue;
 34: 		private final long _rawDoubleBits;
 35: 
 36: 		ExampleConversion(long rawDoubleBits, String javaRendering, String excelRendering) {
 37: 			double d = Double.longBitsToDouble(rawDoubleBits);
 38: 			if ("NaN".equals(javaRendering)) {
 39: 				if (!Double.isNaN(d)) {
 40: 					throw new IllegalArgumentException("value must be NaN");
 41: 				}
 42: 			} else {
 43: 				if (Double.isNaN(d)) {
 44: 					throw new IllegalArgumentException("value must not be NaN");
 45: 				}
 46: 				// just to be dead sure test conversion in java both ways
 47: 				boolean javaToStringOk = javaRendering.equals(Double.toString(d));
 48: 				boolean javaParseOk = Double.parseDouble(javaRendering) == d;
 49: 				if(!javaToStringOk || !javaParseOk) {
 50: 					String msgA = "Specified rawDoubleBits " + doubleToHexString(d) + " encodes to double '" + d + "'.";
 51: 					String msgB = "Specified javaRendering '" + javaRendering+ "' parses as double with rawDoubleBits "
 52: 						+ doubleToHexString(Double.parseDouble(javaRendering));
 53: 					System.err.println(msgA);
 54: 					System.err.println(msgB);
 55: 
 56: 					throw new RuntimeException(msgA + msgB);
 57: 				}
 58: 			}
 59: 			_rawDoubleBits = rawDoubleBits;
 60: 			_javaRendering = javaRendering;
 61: 			_excelRendering = excelRendering;
 62: 			_doubleValue = d;
 63: 		}
 64: 		private static String doubleToHexString(double d) {
 65: 			return "0x" + Long.toHexString(Double.doubleToLongBits(d)).toUpperCase() + "L";
 66: 		}
 67: 		public String getJavaRendering() {
 68: 			return _javaRendering;
 69: 		}
 70: 		public String getExcelRendering() {
 71: 			return _excelRendering;
 72: 		}
 73: 		public double getDoubleValue() {
 74: 			return _doubleValue;
 75: 		}
 76: 		public boolean isNaN() {
 77: 			return Double.isNaN(_doubleValue);
 78: 		}
 79: 		public long getRawDoubleBits() {
 80: 			return _rawDoubleBits;
 81: 		}
 82: 	}
