 45: 	 * Create an InputStream from the specified DocumentEntry
 46: 	 * 
 47: 	 * @param document the DocumentEntry to be read
 48: 	 * 
 49: 	 * @exception IOException if the DocumentEntry cannot be opened (like, maybe it has
 50: 	 *                been deleted?)
 51: 	 */
 52: 	public DocumentInputStream(DocumentEntry document) throws IOException {
 53: 	   if (!(document instanceof DocumentNode)) {
 54: 	      throw new IOException("Cannot open internal document storage");
 55: 	   }
 56: 	   DocumentNode documentNode = (DocumentNode)document;
 57: 	   DirectoryNode parentNode = (DirectoryNode)document.getParent();
 58: 
 59: 	   if(documentNode.getDocument() != null) {
 60: 	      delegate = new ODocumentInputStream(document);
 61: 	   } else if(parentNode.getFileSystem() != null) {
 62: 	      delegate = new ODocumentInputStream(document);
 63: 	   } else if(parentNode.getNFileSystem() != null) {
 64: 	      delegate = new NDocumentInputStream(document);
 65: 	   } else {
 66: 	      throw new IOException("No FileSystem bound on the parent, can't read contents");
 67: 	   }
 68: 	}
