 28: /**
 29:  * A stream based way to get at complete records, with
 30:  * as low a memory footprint as possible.
 31:  * This handles reading from a RecordInputStream, turning
 32:  * the data into full records, processing continue records
 33:  * etc.
 34:  * Most users should use {@link HSSFEventFactory} /
 35:  * {@link HSSFListener} and have new records pushed to
 36:  * them, but this does allow for a "pull" style of coding.
 37:  */

 41: 	 * Keeps track of the sizes of the initial records up to and including {@link FilePassRecord}
 42: 	 * Needed for protected files because each byte is encrypted with respect to its absolute
 43: 	 * position from the start of the stream.
 44: 	 */

 51: 		public StreamEncryptionInfo(RecordInputStream rs, List<Record> outputRecs) {
 52: 			Record rec;
 53: 			rs.nextRecord();
 54: 			int recSize = 4 + rs.remaining();
 55: 			rec = RecordFactory.createSingleRecord(rs);
 56: 			outputRecs.add(rec);
 57: 			FilePassRecord fpr = null;
 58: 			if (rec instanceof BOFRecord) {
 59: 				_hasBOFRecord = true;
 60: 				
 61: 				// Fetch the next record, and see if it indicates whether
 62: 				//  the document is encrypted or not
 63: 				if (rs.hasNextRecord()) {
 64: 					rs.nextRecord();
 65: 					rec = RecordFactory.createSingleRecord(rs);
 66: 					recSize += rec.getRecordSize();
 67: 					outputRecs.add(rec);
 68: 					
 69: 					// Encrypted is normally BOF then FILEPASS
 70: 					// May sometimes be BOF, WRITEPROTECT, FILEPASS
 71: 					if (rec instanceof WriteProtectRecord && rs.hasNextRecord()) {
 72: 	               rs.nextRecord();
 73: 	               rec = RecordFactory.createSingleRecord(rs);
 74: 	               recSize += rec.getRecordSize();
 75: 	               outputRecs.add(rec);
 76: 					}
 77: 					
 78: 					// If it's a FILEPASS, track it specifically but
 79: 					//  don't include it in the main stream
 80: 					if (rec instanceof FilePassRecord) {
 81: 						fpr = (FilePassRecord) rec;
 82: 						outputRecs.remove(outputRecs.size()-1);
 83: 						// TODO - add fpr not added to outputRecs
 84: 						rec = outputRecs.get(0);
 85: 					} else {
 86: 						// workbook not encrypted (typical case)
 87: 						if (rec instanceof EOFRecord) {
 88: 							// A workbook stream is never empty, so crash instead
 89: 							// of trying to keep track of nesting level
 90: 							throw new IllegalStateException("Nothing between BOF and EOF");
 91: 						}
 92: 					}
 93: 				}
 94: 			} else {
 95: 				// Invalid in a normal workbook stream.
 96: 				// However, some test cases work on sub-sections of
 97: 				// the workbook stream that do not begin with BOF
 98: 				_hasBOFRecord = false;
 99: 			}
100: 			_initialRecordsSize = recSize;
101: 			_filePassRec = fpr;
102: 			_lastRecord = rec;
103: 		}

105: 		public RecordInputStream createDecryptingStream(InputStream original) {
106: 			FilePassRecord fpr = _filePassRec;
107: 			String userPassword = Biff8EncryptionKey.getCurrentUserPassword();
108: 
109: 			Biff8EncryptionKey key;
110: 			if (userPassword == null) {
111: 				key = Biff8EncryptionKey.create(fpr.getDocId());
112: 			} else {
113: 				key = Biff8EncryptionKey.create(userPassword, fpr.getDocId());
114: 			}
115: 			if (!key.validate(fpr.getSaltData(), fpr.getSaltHash())) {
116: 				throw new EncryptedDocumentException(
117: 						(userPassword == null ? "Default" : "Supplied")
118: 						+ " password is invalid for docId/saltData/saltHash");
119: 			}
120: 			return new RecordInputStream(original, key, _initialRecordsSize);
121: 		}

148: 	/**
149: 	 * Temporarily stores a group of {@link Record}s, for future return by {@link #nextRecord()}.
150: 	 * This is used at the start of the workbook stream, and also when the most recently read
151: 	 * underlying record is a {@link MulRKRecord}
152: 	 */
153: 	private Record[] _unreadRecordBuffer;
154: 
155: 	/**
156: 	 * used to help iterating over the unread records
157: 	 */
158: 	private int _unreadRecordIndex = -1;

162: 	 */

165: 	 * The most recent DrawingRecord seen

174: 	/**
175: 	 * @param shouldIncludeContinueRecords caller can pass <code>false</code> if loose
176: 	 * {@link ContinueRecord}s should be skipped (this is sometimes useful in event based
177: 	 * processing).
178: 	 */
179: 	public RecordFactoryInputStream(InputStream in, boolean shouldIncludeContinueRecords) {
180: 		RecordInputStream rs = new RecordInputStream(in);
181: 		List<Record> records = new ArrayList<Record>();
182: 		StreamEncryptionInfo sei = new StreamEncryptionInfo(rs, records);
183: 		if (sei.hasEncryption()) {
184: 			rs = sei.createDecryptingStream(in);
185: 		} else {
186: 			// typical case - non-encrypted stream
187: 		}
188: 
189: 		if (!records.isEmpty()) {
190: 			_unreadRecordBuffer = new Record[records.size()];
191: 			records.toArray(_unreadRecordBuffer);
192: 			_unreadRecordIndex =0;
193: 		}
194: 		_recStream = rs;
195: 		_shouldIncludeContinueRecords = shouldIncludeContinueRecords;
196: 		_lastRecord = sei.getLastRecord();
197: 
198: 		/*
199: 		* How to recognise end of stream?
200: 		* In the best case, the underlying input stream (in) ends just after the last EOF record
201: 		* Usually however, the stream is padded with an arbitrary byte count.  Excel and most apps
202: 		* reliably use zeros for padding and if this were always the case, this code could just
203: 		* skip all the (zero sized) records with sid==0.  However, bug 46987 shows a file with
204: 		* non-zero padding that is read OK by Excel (Excel also fixes the padding).
205: 		*
206: 		* So to properly detect the workbook end of stream, this code has to identify the last
207: 		* EOF record.  This is not so easy because the worbook bof+eof pair do not bracket the
208: 		* whole stream.  The worksheets follow the workbook, but it is not easy to tell how many
209: 		* sheet sub-streams should be present.  Hence we are looking for an EOF record that is not
210: 		* immediately followed by a BOF record.  One extra complication is that bof+eof sub-
211: 		* streams can be nested within worksheet streams and it's not clear in these cases what
212: 		* record might follow any EOF record.  So we also need to keep track of the bof/eof
213: 		* nesting level.
214: 		*/
215: 		_bofDepth = sei.hasBOFRecord() ? 1 : 0;
216: 		_lastRecordWasEOFLevelZero = false;
217: 	}

219: 	/**
220: 	 * Returns the next (complete) record from the
221: 	 * stream, or null if there are no more.
222: 	 */
223: 	public Record nextRecord() {
224: 		Record r;
225: 		r = getNextUnreadRecord();
226: 		if (r != null) {
227: 			// found an unread record
228: 			return r;
229: 		}
230: 		while (true) {
231: 			if (!_recStream.hasNextRecord()) {
232: 				// recStream is exhausted;
233: 				return null;
234: 			}
235: 
236: 			if (_lastRecordWasEOFLevelZero) {
237: 				// Potential place for ending the workbook stream
238: 				// Check that the next record is not BOFRecord(0x0809)
239: 				// Normally the input stream contains only zero padding after the last EOFRecord,
240: 				// but bug 46987 and 48068 suggests that the padding may be garbage.
241: 				// This code relies on the padding bytes not starting with BOFRecord.sid
242: 				if (_recStream.getNextSid() != BOFRecord.sid) {
243: 					return null;
244: 				}
245: 				// else - another sheet substream starting here
246: 			}
247: 
248:             // step underlying RecordInputStream to the next record
249:             _recStream.nextRecord();
250: 
251: 			r = readNextRecord();
252: 			if (r == null) {
253: 				// some record types may get skipped (e.g. DBCellRecord and ContinueRecord)
254: 				continue;
255: 			}
256: 			return r;
257: 		}
258: 	}

260: 	/**
261: 	 * @return the next {@link Record} from the multiple record group as expanded from
262: 	 * a recently read {@link MulRKRecord}. <code>null</code> if not present.
263: 	 */
264: 	private Record getNextUnreadRecord() {
265: 		if (_unreadRecordBuffer != null) {
266: 			int ix = _unreadRecordIndex;
267: 			if (ix < _unreadRecordBuffer.length) {
268: 				Record result = _unreadRecordBuffer[ix];
269: 				_unreadRecordIndex = ix + 1;
270: 				return result;
271: 			}
272: 			_unreadRecordIndex = -1;
273: 			_unreadRecordBuffer = null;
274: 		}
275: 		return null;
276: 	}
277: 
278: 	/**
279: 	 * @return the next available record, or <code>null</code> if
280: 	 * this pass didn't return a record that's
281: 	 * suitable for returning (eg was a continue record).
282: 	 */
283: 	private Record readNextRecord() {
284: 
285: 		Record record = RecordFactory.createSingleRecord(_recStream);
286: 		_lastRecordWasEOFLevelZero = false;
287: 
288: 		if (record instanceof BOFRecord) {
289: 			_bofDepth++;
290: 			return record;
291: 		}
292: 
293: 		if (record instanceof EOFRecord) {
294: 			_bofDepth--;
295: 			if (_bofDepth < 1) {
296: 				_lastRecordWasEOFLevelZero = true;
297: 			}
298: 
299: 			return record;
300: 		}
301: 
302: 		if (record instanceof DBCellRecord) {
303: 			// Not needed by POI.  Regenerated from scratch by POI when spreadsheet is written
304: 			return null;
305: 		}
306: 
307: 		if (record instanceof RKRecord) {
308: 			return RecordFactory.convertToNumberRecord((RKRecord) record);
309: 		}
310: 
311: 		if (record instanceof MulRKRecord) {
312: 			Record[] records = RecordFactory.convertRKRecords((MulRKRecord) record);
313: 
314: 			_unreadRecordBuffer = records;
315: 			_unreadRecordIndex = 1;
316: 			return records[0];
317: 		}
318: 
319: 		if (record.getSid() == DrawingGroupRecord.sid
320: 				&& _lastRecord instanceof DrawingGroupRecord) {
321: 			DrawingGroupRecord lastDGRecord = (DrawingGroupRecord) _lastRecord;
322: 			lastDGRecord.join((AbstractEscherHolderRecord) record);
323: 			return null;
324: 		}
325: 		if (record.getSid() == ContinueRecord.sid) {
326: 			ContinueRecord contRec = (ContinueRecord) record;
327: 
328: 			if (_lastRecord instanceof ObjRecord || _lastRecord instanceof TextObjectRecord) {
329: 				// Drawing records have a very strange continue behaviour.
330: 				//There can actually be OBJ records mixed between the continues.
331: 				_lastDrawingRecord.processContinueRecord(contRec.getData());
332: 				//we must remember the position of the continue record.
333: 				//in the serialization procedure the original structure of records must be preserved
334: 				if (_shouldIncludeContinueRecords) {
335: 					return record;
336: 				}
337: 				return null;
338: 			}
339: 			if (_lastRecord instanceof DrawingGroupRecord) {
340: 				((DrawingGroupRecord) _lastRecord).processContinueRecord(contRec.getData());
341: 				return null;
342: 			}
343: 			if (_lastRecord instanceof DrawingRecord) {
344: 				((DrawingRecord) _lastRecord).processContinueRecord(contRec.getData());
345: 				return null;
346: 			}
347: 			if (_lastRecord instanceof UnknownRecord) {
348: 				//Gracefully handle records that we don't know about,
349: 				//that happen to be continued
350: 				return record;
351: 			}
352: 			if (_lastRecord instanceof EOFRecord) {
353: 				// This is really odd, but excel still sometimes
354: 				//  outputs a file like this all the same
355: 				return record;
356: 			}
357: 			throw new RecordFormatException("Unhandled Continue Record followining " + _lastRecord.getClass());
358: 		}
359: 		_lastRecord = record;
360: 		if (record instanceof DrawingRecord) {
361: 			_lastDrawingRecord = (DrawingRecord) record;
362: 		}
363: 		return record;
364: 	}
