 30: /**
 31:  * A block of block allocation table entries. BATBlocks are created
 32:  * only through a static factory method: createBATBlocks.
 33:  *
 34:  * @author Marc Johnson (mjohnson at apache dot org)
 35:  */
 36: public final class BATBlock extends BigBlock {
 37:     /**
 38:      * For a regular fat block, these are 128 / 1024 
 39:      *  next sector values.
 40:      * For a XFat (DIFat) block, these are 127 / 1023
 41:      *  next sector values, then a chaining value.
 42:      */
 43:     private int[] _values;
 44:     
 45:     /**
 46:      * Does this BATBlock have any free sectors in it?
 47:      */
 48:     private boolean _has_free_sectors;

 51:      * Where in the file are we?

 55:     /**
 56:      * Create a single instance initialized with default values
 57:      */
 58:     private BATBlock(POIFSBigBlockSize bigBlockSize)
 59:     {
 60:         super(bigBlockSize);
 61:         
 62:         int _entries_per_block = bigBlockSize.getBATEntriesPerBlock();
 63:         _values = new int[_entries_per_block];
 64:         _has_free_sectors = true;
 65: 
 66:         Arrays.fill(_values, POIFSConstants.UNUSED_BLOCK);
 67:     }
 68: 
 69:     /**
 70:      * Create a single instance initialized (perhaps partially) with entries
 71:      *
 72:      * @param entries the array of block allocation table entries
 73:      * @param start_index the index of the first entry to be written
 74:      *                    to the block
 75:      * @param end_index the index, plus one, of the last entry to be
 76:      *                  written to the block (writing is for all index
 77:      *                  k, start_index <= k < end_index)
 78:      */
 79: 
 80:     private BATBlock(POIFSBigBlockSize bigBlockSize, final int [] entries,
 81:                      final int start_index, final int end_index)
 82:     {
 83:         this(bigBlockSize);
 84:         for (int k = start_index; k < end_index; k++) {
 85:            _values[k - start_index] = entries[k];
 86:         }
 87:         
 88:         // Do we have any free sectors?
 89:         if(end_index - start_index == _values.length) {
 90:            recomputeFree();
 91:         }
 92:     }
 93:     
 94:     private void recomputeFree() {
 95:        boolean hasFree = false;
 96:        for(int k=0; k<_values.length; k++) {
 97:           if(_values[k] == POIFSConstants.UNUSED_BLOCK) {
 98:              hasFree = true;
 99:              break;
100:           }
101:        }
102:        _has_free_sectors = hasFree;
103:     }

105:     /**
106:      * Create a single BATBlock from the byte buffer, which must hold at least
107:      *  one big block of data to be read.
108:      */
109:     public static BATBlock createBATBlock(final POIFSBigBlockSize bigBlockSize, ByteBuffer data)
110:     {
111:        // Create an empty block
112:        BATBlock block = new BATBlock(bigBlockSize);
113:        
114:        // Fill it
115:        byte[] buffer = new byte[LittleEndian.INT_SIZE];
116:        for(int i=0; i<block._values.length; i++) {
117:           data.get(buffer);
118:           block._values[i] = LittleEndian.getInt(buffer);
119:        }
120:        block.recomputeFree();
121:        
122:        // All done
123:        return block;
124:     }

137:     /**
138:      * Create an array of BATBlocks from an array of int block
139:      * allocation table entries
140:      *
141:      * @param entries the array of int entries
142:      *
143:      * @return the newly created array of BATBlocks
144:      */
145:     public static BATBlock [] createBATBlocks(final POIFSBigBlockSize bigBlockSize, final int [] entries)
146:     {
147:         int        block_count = calculateStorageRequirements(bigBlockSize, entries.length);
148:         BATBlock[] blocks      = new BATBlock[ block_count ];
149:         int        index       = 0;
150:         int        remaining   = entries.length;
151: 
152:         int _entries_per_block = bigBlockSize.getBATEntriesPerBlock();
153:         for (int j = 0; j < entries.length; j += _entries_per_block)
154:         {
155:             blocks[ index++ ] = new BATBlock(bigBlockSize, entries, j,
156:                                              (remaining > _entries_per_block)
157:                                              ? j + _entries_per_block
158:                                              : entries.length);
159:             remaining         -= _entries_per_block;
160:         }
161:         return blocks;
162:     }

164:     /**
165:      * Create an array of XBATBlocks from an array of int block
166:      * allocation table entries
167:      *
168:      * @param entries the array of int entries
169:      * @param startBlock the start block of the array of XBAT blocks
170:      *
171:      * @return the newly created array of BATBlocks
172:      */
173: 
174:     public static BATBlock [] createXBATBlocks(final POIFSBigBlockSize bigBlockSize,
175:                                                final int [] entries,
176:                                                final int startBlock)
177:     {
178:         int        block_count =
179:             calculateXBATStorageRequirements(bigBlockSize, entries.length);
180:         BATBlock[] blocks      = new BATBlock[ block_count ];
181:         int        index       = 0;
182:         int        remaining   = entries.length;
183: 
184:         int _entries_per_xbat_block = bigBlockSize.getXBATEntriesPerBlock();
185:         if (block_count != 0)
186:         {
187:             for (int j = 0; j < entries.length; j += _entries_per_xbat_block)
188:             {
189:                 blocks[ index++ ] =
190:                     new BATBlock(bigBlockSize, entries, j,
191:                                  (remaining > _entries_per_xbat_block)
192:                                  ? j + _entries_per_xbat_block
193:                                  : entries.length);
194:                 remaining         -= _entries_per_xbat_block;
195:             }
196:             for (index = 0; index < blocks.length - 1; index++)
197:             {
198:                 blocks[ index ].setXBATChain(bigBlockSize, startBlock + index + 1);
199:             }
200:             blocks[ index ].setXBATChain(bigBlockSize, POIFSConstants.END_OF_CHAIN);
201:         }
202:         return blocks;
203:     }

205:     /**
206:      * Calculate how many BATBlocks are needed to hold a specified
207:      * number of BAT entries.
208:      *
209:      * @param entryCount the number of entries
210:      *
211:      * @return the number of BATBlocks needed
212:      */
213:     public static int calculateStorageRequirements(final POIFSBigBlockSize bigBlockSize, final int entryCount)
214:     {
215:         int _entries_per_block = bigBlockSize.getBATEntriesPerBlock();
216:         return (entryCount + _entries_per_block - 1) / _entries_per_block;
217:     }

219:     /**
220:      * Calculate how many XBATBlocks are needed to hold a specified
221:      * number of BAT entries.
222:      *
223:      * @param entryCount the number of entries
224:      *
225:      * @return the number of XBATBlocks needed
226:      */
227:     public static int calculateXBATStorageRequirements(final POIFSBigBlockSize bigBlockSize, final int entryCount)
228:     {
229:         int _entries_per_xbat_block = bigBlockSize.getXBATEntriesPerBlock();
230:         return (entryCount + _entries_per_xbat_block - 1)
231:                / _entries_per_xbat_block;
232:     }

234:     /**
235:      * Calculates the maximum size of a file which is addressable given the
236:      *  number of FAT (BAT) sectors specified. (We don't care if those BAT
237:      *  blocks come from the 109 in the header, or from header + XBATS, it
238:      *  won't affect the calculation)
239:      *  
240:      * The actual file size will be between [size of fatCount-1 blocks] and
241:      *   [size of fatCount blocks].
242:      *  For 512 byte block sizes, this means we may over-estimate by up to 65kb.
243:      *  For 4096 byte block sizes, this means we may over-estimate by up to 4mb
244:      */
245:     public static int calculateMaximumSize(final POIFSBigBlockSize bigBlockSize,
246:           final int numBATs) {
247:        int size = 1; // Header isn't FAT addressed
248:        
249:        // The header has up to 109 BATs, and extra ones are referenced
250:        //  from XBATs
251:        // However, all BATs can contain 128/1024 blocks
252:        size += (numBATs * bigBlockSize.getBATEntriesPerBlock());
253:        
254:        // So far we've been in sector counts, turn into bytes
255:        return size * bigBlockSize.getBigBlockSize();
256:     }
257:     public static int calculateMaximumSize(final HeaderBlock header)
258:     {
259:        return calculateMaximumSize(header.getBigBlockSize(), header.getBATCount());
260:     }

263:      * Returns the BATBlock that handles the specified offset,
264:      *  and the relative index within it.
265:      * The List of BATBlocks must be in sequential order
266:      */
267:     public static BATBlockAndIndex getBATBlockAndIndex(final int offset, 
268:                 final HeaderBlock header, final List<BATBlock> bats) {
269:        POIFSBigBlockSize bigBlockSize = header.getBigBlockSize();
270:        
271:        int whichBAT = (int)Math.floor(offset / bigBlockSize.getBATEntriesPerBlock());
272:        int index = offset % bigBlockSize.getBATEntriesPerBlock();
273:        return new BATBlockAndIndex( index, bats.get(whichBAT) );

305:     public int getValueAt(int relativeOffset) {
306:        if(relativeOffset >= _values.length) {
307:           throw new ArrayIndexOutOfBoundsException(
308:                 "Unable to fetch offset " + relativeOffset + " as the " + 
309:                 "BAT only contains " + _values.length + " entries"
310:           ); 
311:        }
312:        return _values[relativeOffset];
313:     }
314:     public void setValueAt(int relativeOffset, int value) {
315:        int oldValue = _values[relativeOffset];
316:        _values[relativeOffset] = value;
317:        
318:        // Do we need to re-compute the free?
319:        if(value == POIFSConstants.UNUSED_BLOCK) {
320:           _has_free_sectors = true;
321:           return;
322:        }
323:        if(oldValue == POIFSConstants.UNUSED_BLOCK) {
324:           recomputeFree();
325:        }
326:     }

367:     private byte[] serialize() {
368:        // Create the empty array
369:        byte[] data = new byte[ bigBlockSize.getBigBlockSize() ];
370:        
371:        // Fill in the values
372:        int offset = 0;
373:        for(int i=0; i<_values.length; i++) {
374:           LittleEndian.putInt(data, offset, _values[i]);
375:           offset += LittleEndian.INT_SIZE;
376:        }
377:        
378:        // Done
379:        return data;
380:     }
