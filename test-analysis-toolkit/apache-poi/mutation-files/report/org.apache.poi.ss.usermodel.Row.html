 22: /**
 23:  * High level representation of a row of a spreadsheet.
 24:  */
 25: public interface Row extends Iterable<Cell> {
 26: 
 27:     /**
 28:      * Use this to create new cells within the row and return it.
 29:      * <p>
 30:      * The cell that is returned is a {@link Cell#CELL_TYPE_BLANK}. The type can be changed
 31:      * either through calling <code>setCellValue</code> or <code>setCellType</code>.
 32:      *

 49:      * (255 for *.xls, 1048576 for *.xlsx)
 50:      */
 51:     Cell createCell(int column, int type);

 53:     /**

 55:      *

 66:     void setRowNum(int rowNum);

 96:     /**

 98:      *
 99:      * @return short representing the first logical cell in the row,

102:     short getFirstCellNum();
103: 
104:     /**
105:      * Gets the index of the last cell contained in this row <b>PLUS ONE</b>. The result also
106:      * happens to be the 1-based column number of the last cell.  This value can be used as a
107:      * standard upper bound when iterating over cells:
108:      * <pre>
109:      * short minColIx = row.getFirstCellNum();
110:      * short maxColIx = row.getLastCellNum();

112:      *   Cell cell = row.getCell(colIx);
113:      *   if(cell == null) {
114:      *     continue;
115:      *   }
116:      *   //... do something with cell
117:      * }
118:      * </pre>
119:      *
120:      * @return short representing the last logical cell in the row <b>PLUS ONE</b>,

123:     short getLastCellNum();

196:     void setRowStyle(CellStyle style);

202:     Iterator<Cell> cellIterator();
