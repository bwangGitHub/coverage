 78:     /**
 79:      * Resize the image
 80:      * <p>
 81:      * Please note, that this method works correctly only for workbooks
 82:      * with default font size (Arial 10pt for .xls).
 83:      * If the default font is changed the resized image can be streched vertically or horizontally.
 84:      * </p>
 85:      *
 86:      * @param scale the amount by which image dimensions are multiplied relative to the original size.
 87:      * <code>resize(1.0)</code> sets the original size, <code>resize(0.5)</code> resize to 50% of the original,
 88:      * <code>resize(2.0)</code> resizes to 200% of the original.
 89:      */
 90:     public void resize(double scale){
 91:         HSSFClientAnchor anchor = (HSSFClientAnchor)getAnchor();
 92:         anchor.setAnchorType(2);
 93: 
 94:         HSSFClientAnchor pref = getPreferredSize(scale);
 95: 
 96:         int row2 = anchor.getRow1() + (pref.getRow2() - pref.getRow1());
 97:         int col2 = anchor.getCol1() + (pref.getCol2() - pref.getCol1());
 98: 
 99:         anchor.setCol2((short)col2);
100:         anchor.setDx1(0);
101:         anchor.setDx2(pref.getDx2());
102: 
103:         anchor.setRow2(row2);
104:         anchor.setDy1(0);
105:         anchor.setDy2(pref.getDy2());
106:     }

131:     /**
132:      * Calculate the preferred size for this picture.
133:      *
134:      * @param scale the amount by which image dimensions are multiplied relative to the original size.
135:      * @return HSSFClientAnchor with the preferred size for this image
136:      * @since POI 3.0.2
137:      */
138:     public HSSFClientAnchor getPreferredSize(double scale){
139:         HSSFClientAnchor anchor = (HSSFClientAnchor)getAnchor();
140: 
141:         Dimension size = getImageDimension();
142:         double scaledWidth = size.getWidth() * scale;
143:         double scaledHeight = size.getHeight() * scale;
144: 
145:         float w = 0;
146: 
147:         //space in the leftmost cell
148:         w += getColumnWidthInPixels(anchor.col1)*(1 - (float)anchor.dx1/1024);
149:         short col2 = (short)(anchor.col1 + 1);
150:         int dx2 = 0;
151: 
152:         while(w < scaledWidth){
153:             w += getColumnWidthInPixels(col2++);
154:         }
155: 
156:         if(w > scaledWidth) {
157:             //calculate dx2, offset in the rightmost cell
158:             col2--;
159:             double cw = getColumnWidthInPixels(col2);
160:             double delta = w - scaledWidth;
161:             dx2 = (int)((cw-delta)/cw*1024);
162:         }
163:         anchor.col2 = col2;
164:         anchor.dx2 = dx2;
165: 
166:         float h = 0;
167:         h += (1 - (float)anchor.dy1/256)* getRowHeightInPixels(anchor.row1);
168:         int row2 = anchor.row1 + 1;
169:         int dy2 = 0;
170: 
171:         while(h < scaledHeight){
172:             h += getRowHeightInPixels(row2++);
173:         }
174:         if(h > scaledHeight) {
175:             row2--;
176:             double ch = getRowHeightInPixels(row2);
177:             double delta = h - scaledHeight;
178:             dy2 = (int)((ch-delta)/ch*256);
179:         }
180:         anchor.row2 = row2;
181:         anchor.dy2 = dy2;
182: 
183:         return anchor;
184:     }

186:     private float getColumnWidthInPixels(int column){
187: 
188:         int cw = _patriarch._sheet.getColumnWidth(column);
189:         float px = getPixelWidth(column);
190: 
191:         return cw/px;
192:     }
193: 
194:     private float getRowHeightInPixels(int i){
195: 
196:         HSSFRow row = _patriarch._sheet.getRow(i);
197:         float height;
198:         if(row != null) height = row.getHeight();
199:         else height = _patriarch._sheet.getDefaultRowHeight();
200: 
201:         return height/PX_ROW;
202:     }
203: 
204:     private float getPixelWidth(int column){
205: 
206:         int def = _patriarch._sheet.getDefaultColumnWidth()*256;
207:         int cw = _patriarch._sheet.getColumnWidth(column);
208: 
209:         return cw == def ? PX_DEFAULT : PX_MODIFIED;
210:     }
