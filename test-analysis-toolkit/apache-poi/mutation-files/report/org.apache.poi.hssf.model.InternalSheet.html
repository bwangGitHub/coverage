374:     /**
375:      * Clones the low level records of this sheet and returns the new sheet instance.
376:      * This method is implemented by adding methods for deep cloning to all records that
377:      * can be added to a sheet. The <b>Record</b> object does not implement cloneable.
378:      * When adding a new record, implement a public clone method if and only if the record
379:      * belongs to a sheet.
380:      */
381:     public InternalSheet cloneSheet() {
382:         List<Record> clonedRecords = new ArrayList<Record>(_records.size());
383:         for (int i = 0; i < _records.size(); i++) {
384:             RecordBase rb = _records.get(i);
385:             if (rb instanceof RecordAggregate) {
386:                 ((RecordAggregate) rb).visitContainedRecords(new RecordCloner(clonedRecords));
387:                 continue;
388:             }
389:             if (rb instanceof EscherAggregate){
390:                 // EscherAggregate is used only as a container for SODRAWING and OBJ record combinations
391:                 // So, if the container is empty, there is no reason to clone this record
392:                 // See https://issues.apache.org/bugzilla/show_bug.cgi?id=49529
393:                 if (0 == rb.getRecordSize()){
394:                     continue;
395:                 }
396:             }
397:             Record rec = (Record) ((Record) rb).clone();
398:             clonedRecords.add(rec);
399:         }
400:         return createSheet(new RecordStream(clonedRecords, 0));
401:     }

561:     public void visitContainedRecords(RecordVisitor rv, int offset) {
562: 
563:         PositionTrackingVisitor ptv = new PositionTrackingVisitor(rv, offset);
564: 
565:         boolean haveSerializedIndex = false;
566: 
567:         for (int k = 0; k < _records.size(); k++) {
568:             RecordBase record = _records.get(k);
569: 
570:             if (record instanceof RecordAggregate) {
571:                 RecordAggregate agg = (RecordAggregate) record;
572:                 agg.visitContainedRecords(ptv);
573:             } else {
574:                 ptv.visitRecord((Record) record);
575:             }
576: 
577:             // If the BOF record was just serialized then add the IndexRecord
578:             if (record instanceof BOFRecord) {
579:               if (!haveSerializedIndex) {
580:                 haveSerializedIndex = true;
581:                 // Add an optional UncalcedRecord. However, we should add
582:                 //  it in only the once, after the sheet's own BOFRecord.
583:                 // If there are diagrams, they have their own BOFRecords,
584:                 //  and one shouldn't go in after that!
585:                 if (_isUncalced) {
586:                     ptv.visitRecord(new UncalcedRecord());
587:                 }
588:                 //Can there be more than one BOF for a sheet? If not then we can
589:                 //remove this guard. So be safe it is left here.
590:                 if (_rowsAggregate != null) {
591:                     // find forward distance to first RowRecord
592:                     int initRecsSize = getSizeOfInitialSheetRecords(k);
593:                     int currentPos = ptv.getPosition();
594:                     ptv.visitRecord(_rowsAggregate.createIndexRecord(currentPos, initRecsSize));
595:                 }
596:               }
597:             }
598:         }
