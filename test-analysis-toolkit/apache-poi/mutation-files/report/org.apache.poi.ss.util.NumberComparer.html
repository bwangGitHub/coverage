 20: import static org.apache.poi.ss.util.IEEEDouble.*;

 22: /**
 23:  * Excel compares numbers using different rules to those of java, so
 24:  *  {@link Double#compare(double, double)} won't do.
 25:  *
 26:  *
 27:  * @author Josh Micich
 28:  */
 29: public final class NumberComparer {
 30: 
 31: 	/**
 32: 	 * This class attempts to reproduce Excel's behaviour for comparing numbers.  Results are
 33: 	 * mostly the same as those from {@link Double#compare(double, double)} but with some
 34: 	 * rounding.  For numbers that are very close, this code converts to a format having 15
 35: 	 * decimal digits of precision and a decimal exponent, before completing the comparison.
 36: 	 * <p/>
 37: 	 * In Excel formula evaluation, expressions like "(0.06-0.01)=0.05" evaluate to "TRUE" even
 38: 	 * though the equivalent java expression is <code>false</code>.  In examples like this,
 39: 	 * Excel achieves the effect by having additional logic for comparison operations.
 40: 	 * <p/>
 41: 	 * <p/>
 42: 	 * Note - Excel also gives special treatment to expressions like "0.06-0.01-0.05" which
 43: 	 * evaluates to "0" (in java, rounding anomalies give a result of 6.9E-18).  The special
 44: 	 * behaviour here is for different reasons to the example above:  If the last operator in a
 45: 	 * cell formula is '+' or '-' and the result is less than 2<sup>50</sup> times smaller than
 46: 	 * first operand, the result is rounded to zero.
 47: 	 * Needless to say, the two rules are not consistent and it is relatively easy to find
 48: 	 * examples that satisfy<br/>
 49: 	 * "A=B" is "TRUE" but "A-B" is not "0"<br/>
 50: 	 * and<br/>
 51: 	 * "A=B" is "FALSE" but "A-B" is "0"<br/>
 52: 	 * <br/>
 53: 	 * This rule (for rounding the result of a final addition or subtraction), has not been
 54: 	 * implemented in POI (as of Jul-2009).
 55: 	 *
 56: 	 * @return <code>negative, 0, or positive</code> according to the standard Excel comparison
 57: 	 * of values <tt>a</tt> and <tt>b</tt>.
 58: 	 */
 59: 	public static int compare(double a, double b) {
 60: 		long rawBitsA = Double.doubleToLongBits(a);
 61: 		long rawBitsB = Double.doubleToLongBits(b);
 62: 
 63: 		int biasedExponentA = getBiasedExponent(rawBitsA);
 64: 		int biasedExponentB = getBiasedExponent(rawBitsB);
 65: 
 66: 		if (biasedExponentA == BIASED_EXPONENT_SPECIAL_VALUE) {
 67: 			throw new IllegalArgumentException("Special double values are not allowed: " + toHex(a));
 68: 		}
 69: 		if (biasedExponentB == BIASED_EXPONENT_SPECIAL_VALUE) {
 70: 			throw new IllegalArgumentException("Special double values are not allowed: " + toHex(a));
 71: 		}
 72: 
 73: 		int cmp;
 74: 
 75: 		// sign bit is in the same place for long and double:
 76: 		boolean aIsNegative = rawBitsA < 0;
 77: 		boolean bIsNegative = rawBitsB < 0;
 78: 
 79: 		// compare signs
 80: 		if (aIsNegative != bIsNegative) {
 81: 			// Excel seems to have 'normal' comparison behaviour around zero (no rounding)
 82: 			// even -0.0 < +0.0 (which is not quite the initial conclusion of bug 47198)
 83: 			return aIsNegative ? -1 : +1;
 84: 		}
 85: 
 86: 		// then compare magnitudes (IEEE 754 has exponent bias specifically to allow this)
 87: 		cmp = biasedExponentA - biasedExponentB;
 88: 		int absExpDiff = Math.abs(cmp);
 89: 		if (absExpDiff > 1) {
 90: 			return aIsNegative ? -cmp : cmp;
 91: 		}
 92: 
 93: 		if (absExpDiff == 1) {
 94: 			// special case exponent differs by 1.  There is still a chance that with rounding the two quantities could end up the same
 95: 
 96: 		} else {
 97: 			// else - sign and exponents equal
 98: 			if (rawBitsA == rawBitsB) {
 99: 				// fully equal - exit here
100: 				return 0;
101: 			}
102: 		}
103: 		if (biasedExponentA == 0) {
104: 			if (biasedExponentB == 0) {
105: 				return compareSubnormalNumbers(rawBitsA & FRAC_MASK, rawBitsB & FRAC_MASK, aIsNegative);
106: 			}
107: 			// else biasedExponentB is 1
108: 			return -compareAcrossSubnormalThreshold(rawBitsB, rawBitsA, aIsNegative);
109: 		}
110: 		if (biasedExponentB == 0) {
111: 			// else biasedExponentA is 1
112: 			return +compareAcrossSubnormalThreshold(rawBitsA, rawBitsB, aIsNegative);
113: 		}
114: 
115: 		// sign and exponents same, but fractional bits are different
116: 
117: 		ExpandedDouble edA = ExpandedDouble.fromRawBitsAndExponent(rawBitsA, biasedExponentA - EXPONENT_BIAS);
118: 		ExpandedDouble edB = ExpandedDouble.fromRawBitsAndExponent(rawBitsB, biasedExponentB - EXPONENT_BIAS);
119: 		NormalisedDecimal ndA = edA.normaliseBaseTen().roundUnits();
120: 		NormalisedDecimal ndB = edB.normaliseBaseTen().roundUnits();
121: 		cmp = ndA.compareNormalised(ndB);
122: 		if (aIsNegative) {
123: 			return -cmp;
124: 		}
125: 		return cmp;
126: 	}
127: 
128: 	/**
129: 	 * If both numbers are subnormal, Excel seems to use standard comparison rules
130: 	 */
131: 	private static int compareSubnormalNumbers(long fracA, long fracB, boolean isNegative) {
132: 		int cmp = fracA > fracB ? +1 : fracA < fracB ? -1 : 0;
133: 
134: 		return isNegative ? -cmp : cmp;
135: 	}
136: 
137: 
138: 
139: 	/**
140: 	 * Usually any normal number is greater (in magnitude) than any subnormal number.
141: 	 * However there are some anomalous cases around the threshold where Excel produces screwy results
142: 	 * @param isNegative both values are either negative or positive. This parameter affects the sign of the comparison result
143: 	 * @return usually <code>isNegative ? -1 : +1</code>
144: 	 */
145: 	private static int compareAcrossSubnormalThreshold(long normalRawBitsA, long subnormalRawBitsB, boolean isNegative) {
146: 		long fracB = subnormalRawBitsB & FRAC_MASK;
147: 		if (fracB == 0) {
148: 			// B is zero, so A is definitely greater than B
149: 			return isNegative ? -1 : +1;
150: 		}
151: 		long fracA = normalRawBitsA & FRAC_MASK;
152: 		if (fracA <= 0x0000000000000007L && fracB >= 0x000FFFFFFFFFFFFAL) {
153: 			// Both A and B close to threshold - weird results
154: 			if (fracA == 0x0000000000000007L && fracB == 0x000FFFFFFFFFFFFAL) {
155: 				// special case
156: 				return 0;
157: 			}
158: 			// exactly the opposite
159: 			return isNegative ? +1 : -1;
160: 		}
161: 		// else - typical case A and B is not close to threshold
162: 		return isNegative ? -1 : +1;
163: 	}

167: 	/**
168: 	 * for formatting double values in error messages
169: 	 */
170: 	private static String toHex(double a) {
171: 		return "0x" + Long.toHexString(Double.doubleToLongBits(a)).toUpperCase();
172: 	}
