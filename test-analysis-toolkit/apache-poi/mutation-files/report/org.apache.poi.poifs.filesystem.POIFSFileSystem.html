 99:     /**
100:      * Create a POIFSFileSystem from an <tt>InputStream</tt>.  Normally the stream is read until
101:      * EOF.  The stream is always closed.<p/>
102:      *
103:      * Some streams are usable after reaching EOF (typically those that return <code>true</code>
104:      * for <tt>markSupported()</tt>).  In the unlikely case that the caller has such a stream
105:      * <i>and</i> needs to use it after this constructor completes, a work around is to wrap the
106:      * stream in order to trap the <tt>close()</tt> call.  A convenience method (
107:      * <tt>createNonClosingInputStream()</tt>) has been provided for this purpose:
108:      * <pre>
109:      * InputStream wrappedStream = POIFSFileSystem.createNonClosingInputStream(is);
110:      * HSSFWorkbook wb = new HSSFWorkbook(wrappedStream);
111:      * is.reset();
112:      * doSomethingElse(is);
113:      * </pre>
114:      * Note also the special case of <tt>ByteArrayInputStream</tt> for which the <tt>close()</tt>
115:      * method does nothing.
116:      * <pre>
117:      * ByteArrayInputStream bais = ...
118:      * HSSFWorkbook wb = new HSSFWorkbook(bais); // calls bais.close() !
119:      * bais.reset(); // no problem
120:      * doSomethingElse(bais);
121:      * </pre>
122:      *
123:      * @param stream the InputStream from which to read the data
124:      *
125:      * @exception IOException on errors reading, or on invalid data
126:      */
127: 
128:     public POIFSFileSystem(InputStream stream)
129:         throws IOException
130:     {
131:         this();
132:         boolean success = false;
133: 
134:         HeaderBlock header_block;
135:         RawDataBlockList data_blocks;
136:         try {
137:             // read the header block from the stream
138:             header_block = new HeaderBlock(stream);
139:             bigBlockSize = header_block.getBigBlockSize();
140: 
141:             // read the rest of the stream into blocks
142:             data_blocks = new RawDataBlockList(stream, bigBlockSize);
143:             success = true;
144:         } finally {
145:             closeInputStream(stream, success);
146:         }
147: 
148: 
149:         // set up the block allocation table (necessary for the
150:         // data_blocks to be manageable
151:         new BlockAllocationTableReader(header_block.getBigBlockSize(),
152:                                        header_block.getBATCount(),
153:                                        header_block.getBATArray(),
154:                                        header_block.getXBATCount(),
155:                                        header_block.getXBATIndex(),
156:                                        data_blocks);
157: 
158:         // get property table from the document
159:         PropertyTable properties =
160:             new PropertyTable(header_block, data_blocks);
161: 
162:         // init documents
163:         processProperties(
164:         		SmallBlockTableReader.getSmallDocumentBlocks(
165:         		      bigBlockSize, data_blocks, properties.getRoot(),
166:         				header_block.getSBATStart()
167:         		),
168:         		data_blocks,
169:         		properties.getRoot().getChildren(),
170:         		null,
171:         		header_block.getPropertyStart()
172:         );
173: 
174:         // For whatever reason CLSID of root is always 0.
175:         getRoot().setStorageClsid(properties.getRoot().getStorageClsid());
176:     }

203:     /**
204:      * Checks that the supplied InputStream (which MUST
205:      *  support mark and reset, or be a PushbackInputStream)
206:      *  has a POIFS (OLE2) header at the start of it.
207:      * If your InputStream does not support mark / reset,
208:      *  then wrap it in a PushBackInputStream, then be
209:      *  sure to always use that, and not the original!
210:      * @param inp An InputStream which supports either mark/reset, or is a PushbackInputStream
211:      */
212:     public static boolean hasPOIFSHeader(InputStream inp) throws IOException {
213:         // We want to peek at the first 8 bytes
214:         inp.mark(8);
215: 
216:         byte[] header = new byte[8];
217:         IOUtils.readFully(inp, header);
218:         LongField signature = new LongField(HeaderBlockConstants._signature_offset, header);
219: 
220:         // Wind back those 8 bytes
221:         if(inp instanceof PushbackInputStream) {
222:             PushbackInputStream pin = (PushbackInputStream)inp;
223:             pin.unread(header);
224:         } else {
225:             inp.reset();
226:         }
227: 
228:         // Did it match the signature?
229:         return (signature.get() == HeaderBlockConstants._signature);
230:     }

287:     /**
288:      * Write the filesystem out
289:      *
290:      * @param stream the OutputStream to which the filesystem will be
291:      *               written
292:      *
293:      * @exception IOException thrown on errors writing to the stream
294:      */
295: 
296:     public void writeFilesystem(final OutputStream stream)
297:         throws IOException
298:     {
299: 
300:         // get the property table ready
301:         _property_table.preWrite();
302: 
303:         // create the small block store, and the SBAT
304:         SmallBlockTableWriter      sbtw       =
305:             new SmallBlockTableWriter(bigBlockSize, _documents, _property_table.getRoot());
306: 
307:         // create the block allocation table
308:         BlockAllocationTableWriter bat        =
309:             new BlockAllocationTableWriter(bigBlockSize);
310: 
311:         // create a list of BATManaged objects: the documents plus the
312:         // property table and the small block table
313:         List bm_objects = new ArrayList();
314: 
315:         bm_objects.addAll(_documents);
316:         bm_objects.add(_property_table);
317:         bm_objects.add(sbtw);
318:         bm_objects.add(sbtw.getSBAT());
319: 
320:         // walk the list, allocating space for each and assigning each
321:         // a starting block number
322:         Iterator iter = bm_objects.iterator();
323: 
324:         while (iter.hasNext())
325:         {
326:             BATManaged bmo         = ( BATManaged ) iter.next();
327:             int        block_count = bmo.countBlocks();
328: 
329:             if (block_count != 0)
330:             {
331:                 bmo.setStartBlock(bat.allocateSpace(block_count));
332:             }
333:             else
334:             {
335: 
336:                 // Either the BATManaged object is empty or its data
337:                 // is composed of SmallBlocks; in either case,
338:                 // allocating space in the BAT is inappropriate
339:             }
340:         }
341: 
342:         // allocate space for the block allocation table and take its
343:         // starting block
344:         int               batStartBlock       = bat.createBlocks();
345: 
346:         // get the extended block allocation table blocks
347:         HeaderBlockWriter header_block_writer = new HeaderBlockWriter(bigBlockSize);
348:         BATBlock[]        xbat_blocks         =
349:             header_block_writer.setBATBlocks(bat.countBlocks(),
350:                                              batStartBlock);
351: 
352:         // set the property table start block
353:         header_block_writer.setPropertyStart(_property_table.getStartBlock());
354: 
355:         // set the small block allocation table start block
356:         header_block_writer.setSBATStart(sbtw.getSBAT().getStartBlock());
357: 
358:         // set the small block allocation table block count
359:         header_block_writer.setSBATBlockCount(sbtw.getSBATBlockCount());
360: 
361:         // the header is now properly initialized. Make a list of
362:         // writers (the header block, followed by the documents, the
363:         // property table, the small block store, the small block
364:         // allocation table, the block allocation table, and the
365:         // extended block allocation table blocks)
366:         List writers = new ArrayList();
367: 
368:         writers.add(header_block_writer);
369:         writers.addAll(_documents);
370:         writers.add(_property_table);
371:         writers.add(sbtw);
372:         writers.add(sbtw.getSBAT());
373:         writers.add(bat);
374:         for (int j = 0; j < xbat_blocks.length; j++)
375:         {
376:             writers.add(xbat_blocks[ j ]);
377:         }
378: 
379:         // now, write everything out
380:         iter = writers.iterator();
381:         while (iter.hasNext())
382:         {
383:             BlockWritable writer = ( BlockWritable ) iter.next();
384: 
385:             writer.writeBlocks(stream);
386:         }
387:     }

398:     public static void main(String args[])
399:         throws IOException
400:     {

486:     private void processProperties(final BlockList small_blocks,
487:                                    final BlockList big_blocks,
488:                                    final Iterator properties,
489:                                    final DirectoryNode dir,
490:                                    final int headerPropertiesStartAt)
491:         throws IOException
492:     {
493:         while (properties.hasNext())
494:         {
495:             Property      property = ( Property ) properties.next();
496:             String        name     = property.getName();
497:             DirectoryNode parent   = (dir == null)
498:                                      ? (( DirectoryNode ) getRoot())
499:                                      : dir;
500: 
501:             if (property.isDirectory())
502:             {
503:                 DirectoryNode new_dir =
504:                     ( DirectoryNode ) parent.createDirectory(name);
505: 
506:                 new_dir.setStorageClsid( property.getStorageClsid() );
507: 
508:                 processProperties(
509:                     small_blocks, big_blocks,
510:                     (( DirectoryProperty ) property).getChildren(),
511:                     new_dir, headerPropertiesStartAt);
512:             }
513:             else
514:             {
515:                 int           startBlock = property.getStartBlock();
516:                 int           size       = property.getSize();
517:                 POIFSDocument document   = null;
518: 
519:                 if (property.shouldUseSmallBlocks())
520:                 {
521:                     document =
522:                         new POIFSDocument(name,
523:                                           small_blocks.fetchBlocks(startBlock, headerPropertiesStartAt),
524:                                           size);
525:                 }
526:                 else
527:                 {
528:                     document =
529:                         new POIFSDocument(name,
530:                                           big_blocks.fetchBlocks(startBlock, headerPropertiesStartAt),
531:                                           size);
532:                 }
533:                 parent.createDocument(document);
534:             }
535:         }
536:     }
