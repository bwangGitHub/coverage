 52: 		/**
 53: 		 * Coordinates of the first cell having a formula that uses this shared formula.
 54: 		 * This is often <i>but not always</i> the top left cell in the range covered by
 55: 		 * {@link #_sfr}
 56: 		 */

 59: 		public SharedFormulaGroup(SharedFormulaRecord sfr, CellReference firstCell) {
 60: 			if (!sfr.isInRange(firstCell.getRow(), firstCell.getCol())) {
 61: 				throw new IllegalArgumentException("First formula cell " + firstCell.formatAsString()
 62: 						+ " is not shared formula range " + sfr.getRange().toString() + ".");
 63: 			}
 64: 			_sfr = sfr;
 65: 			_firstCell = firstCell;
 66: 			int width = sfr.getLastColumn() - sfr.getFirstColumn() + 1;
 67: 			int height = sfr.getLastRow() - sfr.getFirstRow() + 1;
 68: 			_frAggs = new FormulaRecordAggregate[width * height];
 69: 			_numberOfFormulas = 0;
 70: 		}
 71: 
 72: 		public void add(FormulaRecordAggregate agg) {
 73: 			if (_numberOfFormulas == 0) {
 74: 				if (_firstCell.getRow() != agg.getRow() || _firstCell.getCol() != agg.getColumn()) {
 75: 					throw new IllegalStateException("shared formula coding error: "+_firstCell.getCol()+'/'+_firstCell.getRow()+" != "+agg.getColumn()+'/'+agg.getRow());
 76: 				}
 77: 			}
 78: 			if (_numberOfFormulas >= _frAggs.length) {
 79: 				throw new RuntimeException("Too many formula records for shared formula group");
 80: 			}
 81: 			_frAggs[_numberOfFormulas++] = agg;
 82: 		}
 83: 
 84: 		public void unlinkSharedFormulas() {
 85: 			for (int i = 0; i < _numberOfFormulas; i++) {
 86: 				_frAggs[i].unlinkSharedFormula();
 87: 			}
 88: 		}

106: 	public static SharedValueManager createEmpty() {
107: 		// Note - must create distinct instances because they are assumed to be mutable.

112: 	private final TableRecord[] _tableRecords;
113: 	private final Map<SharedFormulaRecord, SharedFormulaGroup> _groupsBySharedFormulaRecord;
114: 	/** cached for optimization purposes */

117: 	private SharedValueManager(SharedFormulaRecord[] sharedFormulaRecords,
118: 			CellReference[] firstCells, ArrayRecord[] arrayRecords, TableRecord[] tableRecords) {
119: 		int nShF = sharedFormulaRecords.length;
120: 		if (nShF != firstCells.length) {
121: 			throw new IllegalArgumentException("array sizes don't match: " + nShF + "!=" + firstCells.length + ".");
122: 		}
123: 		_arrayRecords = toList(arrayRecords);
124: 		_tableRecords = tableRecords;
125: 		Map<SharedFormulaRecord, SharedFormulaGroup> m = new HashMap<SharedFormulaRecord, SharedFormulaGroup>(nShF * 3 / 2);
126: 		for (int i = 0; i < nShF; i++) {
127: 			SharedFormulaRecord sfr = sharedFormulaRecords[i];
128: 			m.put(sfr, new SharedFormulaGroup(sfr, firstCells[i]));
129: 		}
130: 		_groupsBySharedFormulaRecord = m;
131: 	}

156: 	 * @param firstCell as extracted from the {@link ExpPtg} from the cell's formula.
157: 	 * @return never <code>null</code>
158: 	 */
159: 	public SharedFormulaRecord linkSharedFormulaRecord(CellReference firstCell, FormulaRecordAggregate agg) {
160: 		SharedFormulaGroup result = findFormulaGroupForCell(firstCell);
161:         if(null == result) {
162:             throw new RuntimeException("Failed to find a matching shared formula record");
163:         }
164: 		result.add(agg);
165: 		return result.getSFR();
166: 	}

168:     private SharedFormulaGroup findFormulaGroupForCell(final CellReference cellRef) {
169:         if(null == _groupsCache) {
170:             _groupsCache = new HashMap<Integer,SharedFormulaGroup>(_groupsBySharedFormulaRecord.size());
171:             for(SharedFormulaGroup group: _groupsBySharedFormulaRecord.values()) {
172:                 _groupsCache.put(getKeyForCache(group._firstCell),group);
173:             }
174:         }
175:         SharedFormulaGroup sfg = _groupsCache.get(getKeyForCache(cellRef));
176:         return sfg;
177:     }
178: 
179:     private Integer getKeyForCache(final CellReference cellRef) {
180:         // The HSSF has a max of 2^16 rows and 2^8 cols
181:         return new Integer((cellRef.getCol()+1)<<16 | cellRef.getRow());
182:     }

184: 	/**
185: 	 * Gets the {@link SharedValueRecordBase} record if it should be encoded immediately after the
186: 	 * formula record contained in the specified {@link FormulaRecordAggregate} agg.  Note - the
187: 	 * shared value record always appears after the first formula record in the group.  For arrays
188: 	 * and tables the first formula is always the in the top left cell.  However, since shared
189: 	 * formula groups can be sparse and/or overlap, the first formula may not actually be in the
190: 	 * top left cell.
191: 	 *
192: 	 * @return the SHRFMLA, TABLE or ARRAY record for the formula cell, if it is the first cell of
193: 	 * a table or array region. <code>null</code> if the formula cell is not shared/array/table,
194: 	 * or if the specified formula is not the the first in the group.
195: 	 */
196: 	public SharedValueRecordBase getRecordForFirstCell(FormulaRecordAggregate agg) {
197: 		CellReference firstCell = agg.getFormulaRecord().getFormula().getExpReference();
198: 		// perhaps this could be optimised by consulting the (somewhat unreliable) isShared flag
199: 		// and/or distinguishing between tExp and tTbl.
200: 		if (firstCell == null) {
201: 			// not a shared/array/table formula
202: 			return null;
203: 		}
204: 
205: 
206: 		int row = firstCell.getRow();
207: 		int column = firstCell.getCol();
208: 		if (agg.getRow() != row || agg.getColumn() != column) {
209: 			// not the first formula cell in the group
210: 			return null;
211: 		}
212: 
213:         if(!_groupsBySharedFormulaRecord.isEmpty()) {
214:             SharedFormulaGroup sfg = findFormulaGroupForCell(firstCell);
215:             if(null != sfg) {
216:                 return sfg.getSFR();
217:             }
218:         }
219: 
220: 		// Since arrays and tables cannot be sparse (all cells in range participate)
221: 		// The first cell will be the top left in the range.  So we can match the
222: 		// ARRAY/TABLE record directly.
223: 
224: 		for (TableRecord tr : _tableRecords) {
225: 			if (tr.isFirstCell(row, column)) {
226: 				return tr;
227: 			}
228: 		}
229: 		for (ArrayRecord ar : _arrayRecords) {
230: 			if (ar.isFirstCell(row, column)) {
231: 				return ar;
232: 			}
233: 		}
234: 		return null;
235: 	}

238: 	 * Converts all {@link FormulaRecord}s handled by <tt>sharedFormulaRecord</tt>
239: 	 * to plain unshared formulas
240: 	 */
241: 	public void unlink(SharedFormulaRecord sharedFormulaRecord) {
242: 		SharedFormulaGroup svg = _groupsBySharedFormulaRecord.remove(sharedFormulaRecord);
243: 		if (svg == null) {
244: 			throw new IllegalStateException("Failed to find formulas for shared formula");
245: 		}
246: 		_groupsCache = null; // be sure to reset cached value
247: 		svg.unlinkSharedFormulas();
248: 	}
