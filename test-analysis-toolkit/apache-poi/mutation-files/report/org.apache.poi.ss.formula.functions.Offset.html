 26: /**
 27:  * Implementation for Excel function OFFSET()<p/>
 28:  *
 29:  * OFFSET returns an area reference that is a specified number of rows and columns from a
 30:  * reference cell or area.<p/>
 31:  *
 32:  * <b>Syntax</b>:<br/>
 33:  * <b>OFFSET</b>(<b>reference</b>, <b>rows</b>, <b>cols</b>, height, width)<p/>
 34:  * <b>reference</b> is the base reference.<br/>
 35:  * <b>rows</b> is the number of rows up or down from the base reference.<br/>
 36:  * <b>cols</b> is the number of columns left or right from the base reference.<br/>
 37:  * <b>height</b> (default same height as base reference) is the row count for the returned area reference.<br/>
 38:  * <b>width</b> (default same width as base reference) is the column count for the returned area reference.<br/>
 39:  *
 40:  * @author Josh Micich
 41:  */
 42: public final class Offset implements Function {
 43: 	// These values are specific to BIFF8
 44: 	private static final int LAST_VALID_ROW_INDEX = 0xFFFF;
 45: 	private static final int LAST_VALID_COLUMN_INDEX = 0xFF;
 46: 
 47: 
 48: 	/**
 49: 	 * A one dimensional base + offset.  Represents either a row range or a column range.
 50: 	 * Two instances of this class together specify an area range.
 51: 	 */
 52: 	/* package */ static final class LinearOffsetRange {
 53: 
 54: 		private final int _offset;
 55: 		private final int _length;
 56: 
 57: 		public LinearOffsetRange(int offset, int length) {
 58: 			if(length == 0) {
 59: 				// handled that condition much earlier
 60: 				throw new RuntimeException("length may not be zero");
 61: 			}
 62: 			_offset = offset;
 63: 			_length = length;
 64: 		}
 65: 
 66: 		public short getFirstIndex() {
 67: 			return (short) _offset;
 68: 		}
 69: 		public short getLastIndex() {
 70: 			return (short) (_offset + _length - 1);
 71: 		}
 72: 		/**
 73: 		 * Moves the range by the specified translation amount.<p/>
 74: 		 *
 75: 		 * This method also 'normalises' the range: Excel specifies that the width and height
 76: 		 * parameters (length field here) cannot be negative.  However, OFFSET() does produce
 77: 		 * sensible results in these cases.  That behavior is replicated here. <p/>
 78: 		 *
 79: 		 * @param translationAmount may be zero negative or positive
 80: 		 *
 81: 		 * @return the equivalent <tt>LinearOffsetRange</tt> with a positive length, moved by the
 82: 		 * specified translationAmount.
 83: 		 */
 84: 		public LinearOffsetRange normaliseAndTranslate(int translationAmount) {
 85: 			if (_length > 0) {
 86: 				if(translationAmount == 0) {
 87: 					return this;
 88: 				}
 89: 				return new LinearOffsetRange(translationAmount + _offset, _length);
 90: 			}
 91: 			return new LinearOffsetRange(translationAmount + _offset + _length + 1, -_length);
 92: 		}
 93: 
 94: 		public boolean isOutOfBounds(int lowValidIx, int highValidIx) {
 95: 			if(_offset < lowValidIx) {
 96: 				return true;
 97: 			}
 98: 			if(getLastIndex() > highValidIx) {
 99: 				return true;
100: 			}
101: 			return false;
102: 		}
103: 		public String toString() {
104: 			StringBuffer sb = new StringBuffer(64);
105: 			sb.append(getClass().getName()).append(" [");
106: 			sb.append(_offset).append("...").append(getLastIndex());
107: 			sb.append("]");
108: 			return sb.toString();
109: 		}
110: 	}

112: 	/**
113: 	 * Encapsulates either an area or cell reference which may be 2d or 3d.
114: 	 */
115: 	private static final class BaseRef {
116: 		private final int _firstRowIndex;
117: 		private final int _firstColumnIndex;
118: 		private final int _width;
119: 		private final int _height;
120: 		private final RefEval _refEval;
121: 		private final AreaEval _areaEval;
122: 
123: 		public BaseRef(RefEval re) {
124: 			_refEval = re;
125: 			_areaEval = null;
126: 			_firstRowIndex = re.getRow();
127: 			_firstColumnIndex = re.getColumn();
128: 			_height = 1;
129: 			_width = 1;
130: 		}
131: 
132: 		public BaseRef(AreaEval ae) {
133: 			_refEval = null;
134: 			_areaEval = ae;
135: 			_firstRowIndex = ae.getFirstRow();
136: 			_firstColumnIndex = ae.getFirstColumn();
137: 			_height = ae.getLastRow() - ae.getFirstRow() + 1;
138: 			_width = ae.getLastColumn() - ae.getFirstColumn() + 1;
139: 		}
140: 
141: 		public int getWidth() {
142: 			return _width;
143: 		}
144: 		public int getHeight() {
145: 			return _height;
146: 		}
147: 		public int getFirstRowIndex() {
148: 			return _firstRowIndex;
149: 		}
150: 		public int getFirstColumnIndex() {
151: 			return _firstColumnIndex;
152: 		}
153: 
154: 		public AreaEval offset(int relFirstRowIx, int relLastRowIx,
155: 				int relFirstColIx, int relLastColIx) {
156: 			if (_refEval == null) {
157: 				return _areaEval.offset(relFirstRowIx, relLastRowIx, relFirstColIx, relLastColIx);
158: 			}
159: 			return _refEval.offset(relFirstRowIx, relLastRowIx, relFirstColIx, relLastColIx);
160: 		}
161: 	}

163: 	public ValueEval evaluate(ValueEval[] args, int srcCellRow, int srcCellCol) {
164: 		if(args.length < 3 || args.length > 5) {
165: 			return ErrorEval.VALUE_INVALID;
166: 		}
167: 
168: 		try {
169: 			BaseRef baseRef = evaluateBaseRef(args[0]);
170: 			int rowOffset = evaluateIntArg(args[1], srcCellRow, srcCellCol);
171: 			int columnOffset = evaluateIntArg(args[2], srcCellRow, srcCellCol);
172: 			int height = baseRef.getHeight();
173: 			int width = baseRef.getWidth();
174: 			switch(args.length) {
175: 				case 5:
176: 					width = evaluateIntArg(args[4], srcCellRow, srcCellCol);
177: 				case 4:
178: 					height = evaluateIntArg(args[3], srcCellRow, srcCellCol);
179: 			}
180: 			// Zero height or width raises #REF! error
181: 			if(height == 0 || width == 0) {
182: 				return ErrorEval.REF_INVALID;
183: 			}
184: 			LinearOffsetRange rowOffsetRange = new LinearOffsetRange(rowOffset, height);
185: 			LinearOffsetRange colOffsetRange = new LinearOffsetRange(columnOffset, width);
186: 			return createOffset(baseRef, rowOffsetRange, colOffsetRange);
187: 		} catch (EvaluationException e) {
188: 			return e.getErrorEval();
189: 		}
190: 	}
191: 
192: 	private static AreaEval createOffset(BaseRef baseRef,
193: 			LinearOffsetRange orRow, LinearOffsetRange orCol) throws EvaluationException {
194: 		LinearOffsetRange absRows = orRow.normaliseAndTranslate(baseRef.getFirstRowIndex());
195: 		LinearOffsetRange absCols = orCol.normaliseAndTranslate(baseRef.getFirstColumnIndex());
196: 
197: 		if(absRows.isOutOfBounds(0, LAST_VALID_ROW_INDEX)) {
198: 			throw new EvaluationException(ErrorEval.REF_INVALID);
199: 		}
200: 		if(absCols.isOutOfBounds(0, LAST_VALID_COLUMN_INDEX)) {
201: 			throw new EvaluationException(ErrorEval.REF_INVALID);
202: 		}
203: 		return baseRef.offset(orRow.getFirstIndex(), orRow.getLastIndex(), orCol.getFirstIndex(), orCol.getLastIndex());
204: 	}
205: 
206: 	private static BaseRef evaluateBaseRef(ValueEval eval) throws EvaluationException {
207: 
208: 		if(eval instanceof RefEval) {
209: 			return new BaseRef((RefEval)eval);
210: 		}
211: 		if(eval instanceof AreaEval) {
212: 			return new BaseRef((AreaEval)eval);
213: 		}
214: 		if (eval instanceof ErrorEval) {
215: 			throw new EvaluationException((ErrorEval) eval);
216: 		}
217: 		throw new EvaluationException(ErrorEval.VALUE_INVALID);
218: 	}
219: 
220: 	/**
221: 	 * OFFSET's numeric arguments (2..5) have similar processing rules
222: 	 */
223: 	static int evaluateIntArg(ValueEval eval, int srcCellRow, int srcCellCol) throws EvaluationException {
224: 		ValueEval ve = OperandResolver.getSingleValue(eval, srcCellRow, srcCellCol);
225: 		return OperandResolver.coerceValueToInt(ve);
226: 	}
