 38: 	/**
 39: 	 * PLS is potentially a <i>continued</i> record, but is currently uninterpreted by POI
 40: 	 */
 41: 	private static final class PLSAggregate extends RecordAggregate {
 42: 		private static final ContinueRecord[] EMPTY_CONTINUE_RECORD_ARRAY = { };
 43: 		private final Record _pls;
 44: 		/**
 45: 		 * holds any continue records found after the PLS record.<br/>
 46: 		 * This would not be required if PLS was properly interpreted.
 47: 		 * Currently, PLS is an {@link UnknownRecord} and does not automatically
 48: 		 * include any trailing {@link ContinueRecord}s.
 49: 		 */
 50: 		private ContinueRecord[] _plsContinues;
 51: 
 52: 		public PLSAggregate(RecordStream rs) {
 53: 			_pls = rs.getNext();
 54: 			if (rs.peekNextSid()==ContinueRecord.sid) {
 55: 				List<ContinueRecord> temp = new ArrayList<ContinueRecord>();
 56: 				while (rs.peekNextSid()==ContinueRecord.sid) {
 57: 					temp.add((ContinueRecord)rs.getNext());
 58: 				}
 59: 				_plsContinues = new ContinueRecord[temp.size()];
 60: 				temp.toArray(_plsContinues);
 61: 			} else {
 62: 				_plsContinues = EMPTY_CONTINUE_RECORD_ARRAY;
 63: 			}
 64: 		}
 65: 
 66: 		@Override
 67: 		public void visitContainedRecords(RecordVisitor rv) {
 68: 			rv.visitRecord(_pls);
 69: 			for (int i = 0; i < _plsContinues.length; i++) {
 70: 				rv.visitRecord(_plsContinues[i]);
 71: 			}
 72: 		}
 73: 	}

148: 	private boolean readARecord(RecordStream rs) {
149: 		switch (rs.peekNextSid()) {
150: 			case HorizontalPageBreakRecord.sid:
151: 				checkNotPresent(_rowBreaksRecord);
152: 				_rowBreaksRecord = (PageBreakRecord) rs.getNext();
153: 				break;
154: 			case VerticalPageBreakRecord.sid:
155: 				checkNotPresent(_columnBreaksRecord);
156: 				_columnBreaksRecord = (PageBreakRecord) rs.getNext();
157: 				break;
158: 			case HeaderRecord.sid:
159: 				checkNotPresent(_header);
160: 				_header = (HeaderRecord) rs.getNext();
161: 				break;
162: 			case FooterRecord.sid:
163: 				checkNotPresent(_footer);
164: 				_footer = (FooterRecord) rs.getNext();
165: 				break;
166: 			case HCenterRecord.sid:
167: 				checkNotPresent(_hCenter);
168: 				_hCenter = (HCenterRecord) rs.getNext();
169: 				break;
170: 			case VCenterRecord.sid:
171: 				checkNotPresent(_vCenter);
172: 				_vCenter = (VCenterRecord) rs.getNext();
173: 				break;
174: 			case LeftMarginRecord.sid:
175: 				checkNotPresent(_leftMargin);
176: 				_leftMargin = (LeftMarginRecord) rs.getNext();
177: 				break;
178: 			case RightMarginRecord.sid:
179: 				checkNotPresent(_rightMargin);
180: 				_rightMargin = (RightMarginRecord) rs.getNext();
181: 				break;
182: 			case TopMarginRecord.sid:
183: 				checkNotPresent(_topMargin);
184: 				_topMargin = (TopMarginRecord) rs.getNext();
185: 				break;
186: 			case BottomMarginRecord.sid:
187: 				checkNotPresent(_bottomMargin);
188: 				_bottomMargin = (BottomMarginRecord) rs.getNext();
189: 				break;
190: 			case UnknownRecord.PLS_004D:
191: 				_plsRecords.add(new PLSAggregate(rs));
192: 				break;
193: 			case PrintSetupRecord.sid:
194: 				checkNotPresent(_printSetup);
195: 				_printSetup = (PrintSetupRecord)rs.getNext();
196: 				break;
197: 			case UnknownRecord.BITMAP_00E9:
198: 				checkNotPresent(_bitmap);
199: 				_bitmap = rs.getNext();
200: 				break;
201: 			case UnknownRecord.PRINTSIZE_0033:
202: 				checkNotPresent(_printSize);
203: 				_printSize = rs.getNext();
204: 				break;
205: 			case HeaderFooterRecord.sid:
206: 				//there can be multiple HeaderFooterRecord records belonging to different sheet views
207: 				HeaderFooterRecord hf = (HeaderFooterRecord)rs.getNext();
208:                 if(hf.isCurrentSheet()) _headerFooter = hf;
209:                 else {
210:                     _sviewHeaderFooters.add(hf);
211:                 }
212:                 break;
213: 			default:
214: 				// all other record types are not part of the PageSettingsBlock
215: 				return false;
216: 		}
217: 		return true;
218: 	}

261: 	public void visitContainedRecords(RecordVisitor rv) {
262: 		// Replicates record order from Excel 2007, though this is not critical
263: 
264: 		visitIfPresent(_rowBreaksRecord, rv);
265: 		visitIfPresent(_columnBreaksRecord, rv);
266: 		// Write out empty header / footer records if these are missing
267: 		if (_header == null) {
268: 			rv.visitRecord(new HeaderRecord(""));
269: 		} else {
270: 			rv.visitRecord(_header);
271: 		}
272: 		if (_footer == null) {
273: 			rv.visitRecord(new FooterRecord(""));
274: 		} else {
275: 			rv.visitRecord(_footer);
276: 		}
277: 		visitIfPresent(_hCenter, rv);
278: 		visitIfPresent(_vCenter, rv);
279: 		visitIfPresent(_leftMargin, rv);
280: 		visitIfPresent(_rightMargin, rv);
281: 		visitIfPresent(_topMargin, rv);
282: 		visitIfPresent(_bottomMargin, rv);
283: 		for (RecordAggregate pls : _plsRecords) {
284: 			pls.visitContainedRecords(rv);
285: 		}
286: 		visitIfPresent(_printSetup, rv);
287: 		visitIfPresent(_bitmap, rv);
288: 		visitIfPresent(_printSize, rv);
289: 		visitIfPresent(_headerFooter, rv);
290: 	}
291: 	private static void visitIfPresent(Record r, RecordVisitor rv) {
292: 		if (r != null) {
293: 			rv.visitRecord(r);
294: 		}
295: 	}

297: 		if (r != null) {

301: 			}

467: 	/**
468: 	 * Shifts all the page breaks in the range "count" number of rows/columns
469: 	 * @param breaks The page record to be shifted
470: 	 * @param start Starting "main" value to shift breaks
471: 	 * @param stop Ending "main" value to shift breaks
472: 	 * @param count number of units (rows/columns) to shift by
473: 	 */
474: 	private static void shiftBreaks(PageBreakRecord breaks, int start, int stop, int count) {
475: 
476: 		Iterator<PageBreakRecord.Break> iterator = breaks.getBreaksIterator();
477: 		List<PageBreakRecord.Break> shiftedBreak = new ArrayList<PageBreakRecord.Break>();
478: 		while(iterator.hasNext())
479: 		{
480: 			PageBreakRecord.Break breakItem = iterator.next();
481: 			int breakLocation = breakItem.main;
482: 			boolean inStart = (breakLocation >= start);
483: 			boolean inEnd = (breakLocation <= stop);
484: 			if(inStart && inEnd)
485: 				shiftedBreak.add(breakItem);
486: 		}
487: 
488: 		iterator = shiftedBreak.iterator();
489: 		while (iterator.hasNext()) {
490: 			PageBreakRecord.Break breakItem = iterator.next();
491: 			breaks.removeBreak(breakItem.main);
492: 			breaks.addBreak((short)(breakItem.main+count), breakItem.subFrom, breakItem.subTo);
493: 		}
494: 	}

604: 	/**
605: 	 * This method reads PageSettingsBlock records from the supplied RecordStream until the first
606: 	 * non-PageSettingsBlock record is encountered.  As each record is read, it is incorporated
607: 	 * into this PageSettingsBlock.
608: 	 * <p/>
609: 	 * The latest Excel version seems to write the PageSettingsBlock uninterrupted. However there
610: 	 * are several examples (that Excel reads OK) where these records are not written together:
611: 	 * <ul>
612: 	 * <li><b>HEADER_FOOTER(0x089C) after WINDOW2</b> - This record is new in 2007.  Some apps
613: 	 * seem to have scattered this record long after the PageSettingsBlock where it belongs
614: 	 * test samples: SharedFormulaTest.xls, ex44921-21902.xls, ex42570-20305.xls</li>
615: 	 * <li><b>PLS, WSBOOL, PageSettingsBlock</b> - WSBOOL is not a PSB record.
616: 	 * This happens in the test sample file "NoGutsRecords.xls" and "WORKBOOK_in_capitals.xls"</li>
617: 	 * <li><b>Margins after DIMENSION</b> - All of PSB should be before DIMENSION. (Bug-47199)</li>
618: 	 * </ul>
619: 	 * These were probably written by other applications (or earlier versions of Excel). It was
620: 	 * decided to not write specific code for detecting each of these cases.  POI now tolerates
621: 	 * PageSettingsBlock records scattered all over the sheet record stream, and in any order, but
