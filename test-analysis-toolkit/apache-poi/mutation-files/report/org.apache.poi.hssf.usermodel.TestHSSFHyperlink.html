195:      * Test that HSSFSheet#shiftRows moves hyperlinks,
196:      * see bugs #46445 and #29957
197:      */
198:     public void testShiftRows(){
199:         HSSFWorkbook wb = HSSFTestDataSamples.openSampleWorkbook("46445.xls");
200: 
201: 
202:         HSSFSheet sheet = wb.getSheetAt(0);
203: 
204:         //verify existing hyperlink in A3
205:         HSSFCell cell1 = sheet.getRow(2).getCell(0);
206:         HSSFHyperlink link1 = cell1.getHyperlink();
207:         assertNotNull(link1);
208:         assertEquals(2, link1.getFirstRow());
209:         assertEquals(2, link1.getLastRow());
210: 
211:         //assign a hyperlink to A4
212:         HSSFHyperlink link2 = new HSSFHyperlink(HSSFHyperlink.LINK_DOCUMENT);
213:         link2.setAddress("Sheet2!A2");
214:         HSSFCell cell2 = sheet.getRow(3).getCell(0);
215:         cell2.setHyperlink(link2);
216:         assertEquals(3, link2.getFirstRow());
217:         assertEquals(3, link2.getLastRow());
218: 
219:         //move the 3rd row two rows down
220:         sheet.shiftRows(sheet.getFirstRowNum(), sheet.getLastRowNum(), 2);
221: 
222:         //cells A3 and A4 don't contain hyperlinks anymore
223:         assertNull(sheet.getRow(2).getCell(0).getHyperlink());
224:         assertNull(sheet.getRow(3).getCell(0).getHyperlink());
225: 
226:         //the first hypelink now belongs to A5
227:         HSSFHyperlink link1_shifted = sheet.getRow(2+2).getCell(0).getHyperlink();
228:         assertNotNull(link1_shifted);
229:         assertEquals(4, link1_shifted.getFirstRow());
230:         assertEquals(4, link1_shifted.getLastRow());
231: 
232:         //the second hypelink now belongs to A6
233:         HSSFHyperlink link2_shifted = sheet.getRow(3+2).getCell(0).getHyperlink();
