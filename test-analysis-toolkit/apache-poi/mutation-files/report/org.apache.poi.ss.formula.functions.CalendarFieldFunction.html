 50: 	public final ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0) {
 51: 		double val;
 52: 		try {
 53: 			ValueEval ve = OperandResolver.getSingleValue(arg0, srcRowIndex, srcColumnIndex);
 54: 			val = OperandResolver.coerceValueToDouble(ve);
 55: 		} catch (EvaluationException e) {
 56: 			return e.getErrorEval();
 57: 		}
 58: 		if (val < 0) {
 59: 			return ErrorEval.NUM_ERROR;
 60: 		}
 61: 		return new NumberEval(getCalField(val));
 62: 	}

 64: 	private int getCalField(double serialDate) {
 65: 	   // For some reason, a date of 0 in Excel gets shown
 66: 	   //  as the non existant 1900-01-00
 67: 		if(((int)serialDate) == 0) {
 68: 			switch (_dateFieldId) {
 69: 				case Calendar.YEAR: return 1900;
 70: 				case Calendar.MONTH: return 1;
 71: 				case Calendar.DAY_OF_MONTH: return 0;
 72: 			}
 73: 			// They want time, that's normal
 74: 		}
 75: 
 76: 		// TODO Figure out if we're in 1900 or 1904
 77: 		Date d = DateUtil.getJavaDate(serialDate, false);
 78: 
 79: 		Calendar c = new GregorianCalendar();
 80: 		c.setTime(d);
 81: 		int result = c.get(_dateFieldId);
 82: 		
 83: 		// Month is a special case due to C semantics
 84: 		if (_dateFieldId == Calendar.MONTH) {
 85: 			result++;
 86: 		}
 87: 		
 88: 		return result;
 89: 	}
