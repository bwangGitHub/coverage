 23: /**
 24:  * Utility for delaying the concatenation of multiple byte arrays.  Doing this up-front
 25:  * causes significantly more copying, which for a large number of byte arrays can cost
 26:  * a large amount of time.
 27:  */
 28: public class LazilyConcatenatedByteArray {
 29:     private final List<byte[]> arrays = new ArrayList<byte[]>(1);
 30: 
 31:     /**
 32:      * Clears the array (sets the concatenated length back to zero.
 33:      */
 34:     public void clear() {
 35:         arrays.clear();
 36:     }
 37: 
 38:     /**
 39:      * Concatenates an array onto the end of our array.
 40:      * This is a relatively fast operation.
 41:      *
 42:      * @param array the array to concatenate.
 43:      * @throws IllegalArgumentException if {@code array} is {@code null}.
 44:      */
 45:     public void concatenate(byte[] array) {
 46:         if (array == null) {
 47:             throw new IllegalArgumentException("array cannot be null");
 48:         }
 49:         arrays.add(array);
 50:     }
 51: 
 52:     /**
 53:      * Gets the concatenated contents as a single byte array.
 54:      *
 55:      * This is a slower operation, but the concatenated array is stored off as a single
 56:      * array again so that subsequent calls will not perform additional copying.
 57:      *
 58:      * @return the byte array.  Returns {@code null} if no data has been placed into it.
 59:      */
 60:     public byte[] toArray() {
 61:         if (arrays.isEmpty()) {
 62:             return null;
 63:         } else if (arrays.size() > 1) {
 64:             int totalLength = 0;
 65:             for (byte[] array : arrays) {
 66:                 totalLength += array.length;
 67:             }
 68: 
 69:             byte[] concatenated = new byte[totalLength];
 70:             int destPos = 0;
 71:             for (byte[] array : arrays) {
 72:                 System.arraycopy(array, 0, concatenated, destPos, array.length);
 73:                 destPos += array.length;
 74:             }
 75: 
 76:             arrays.clear();
 77:             arrays.add(concatenated);
 78:         }
 79: 
 80:         return arrays.get(0);
 81:     }
 82: }
