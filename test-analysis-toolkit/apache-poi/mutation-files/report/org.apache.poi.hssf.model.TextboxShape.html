 25: /**
 26:  * Represents an textbox shape and converts between the highlevel records
 27:  * and lowlevel records for an oval.
 28:  *
 29:  * @author Glen Stampoultzis (glens at apache.org)
 30:  */
 31: public class TextboxShape
 32:         extends AbstractShape
 33: {
 34:     private EscherContainerRecord spContainer;
 35:     private TextObjectRecord textObjectRecord;
 36:     private ObjRecord objRecord;
 37:     private EscherTextboxRecord escherTextbox;
 38: 
 39:     /**
 40:      * Creates the low evel records for an textbox.
 41:      *
 42:      * @param hssfShape  The highlevel shape.
 43:      * @param shapeId    The shape id to use for this shape.
 44:      */
 45:     TextboxShape( HSSFTextbox hssfShape, int shapeId )
 46:     {
 47:         spContainer = createSpContainer( hssfShape, shapeId );
 48:         objRecord = createObjRecord( hssfShape, shapeId );
 49:         textObjectRecord = createTextObjectRecord( hssfShape, shapeId );
 50:     }
 51: 
 52:     /**
 53:      * Creates the low level OBJ record for this shape.
 54:      */
 55:     private ObjRecord createObjRecord( HSSFTextbox hssfShape, int shapeId )
 56:     {
 57:         HSSFShape shape = hssfShape;
 58: 
 59:         ObjRecord obj = new ObjRecord();
 60:         CommonObjectDataSubRecord c = new CommonObjectDataSubRecord();
 61:         c.setObjectType( (short) ( (HSSFSimpleShape) shape ).getShapeType() );
 62:         c.setObjectId( getCmoObjectId(shapeId) );
 63:         c.setLocked( true );
 64:         c.setPrintable( true );
 65:         c.setAutofill( true );
 66:         c.setAutoline( true );
 67:         EndSubRecord e = new EndSubRecord();
 68: 
 69:         obj.addSubRecord( c );
 70:         obj.addSubRecord( e );
 71: 
 72:         return obj;
 73:     }
 74: 
 75:     /**
 76:      * Generates the escher shape records for this shape.
 77:      *
 78:      * @param hssfShape
 79:      * @param shapeId
 80:      */
 81:     private EscherContainerRecord createSpContainer( HSSFTextbox hssfShape, int shapeId )
 82:     {
 83:         HSSFTextbox shape = hssfShape;
 84: 
 85:         EscherContainerRecord spContainer = new EscherContainerRecord();
 86:         EscherSpRecord sp = new EscherSpRecord();
 87:         EscherOptRecord opt = new EscherOptRecord();
 88:         EscherRecord anchor = new EscherClientAnchorRecord();
 89:         EscherClientDataRecord clientData = new EscherClientDataRecord();
 90:         escherTextbox = new EscherTextboxRecord();
 91: 
 92:         spContainer.setRecordId( EscherContainerRecord.SP_CONTAINER );
 93:         spContainer.setOptions( (short) 0x000F );
 94:         sp.setRecordId( EscherSpRecord.RECORD_ID );
 95:         sp.setOptions( (short) ( ( EscherAggregate.ST_TEXTBOX << 4 ) | 0x2 ) );
 96: 
 97:         sp.setShapeId( shapeId );
 98:         sp.setFlags( EscherSpRecord.FLAG_HAVEANCHOR | EscherSpRecord.FLAG_HASSHAPETYPE );
 99:         opt.setRecordId( EscherOptRecord.RECORD_ID );
100:         //        opt.addEscherProperty( new EscherBoolProperty( EscherProperties.PROTECTION__LOCKAGAINSTGROUPING, 262144 ) );
101:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.TEXT__TEXTID, 0 ) );
102:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.TEXT__TEXTLEFT, shape.getMarginLeft() ) );
103:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.TEXT__TEXTRIGHT, shape.getMarginRight() ) );
104:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.TEXT__TEXTBOTTOM, shape.getMarginBottom() ) );
105:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.TEXT__TEXTTOP, shape.getMarginTop() ) );
106: 
107:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.TEXT__WRAPTEXT, 0 ) );
108:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.TEXT__ANCHORTEXT, 0 ) );
109:         opt.addEscherProperty( new EscherSimpleProperty( EscherProperties.GROUPSHAPE__PRINT, 0x00080000 ) );
110: 
111:         addStandardOptions( shape, opt );
112:         HSSFAnchor userAnchor = shape.getAnchor();
113:         //        if (userAnchor.isHorizontallyFlipped())
114:         //            sp.setFlags(sp.getFlags() | EscherSpRecord.FLAG_FLIPHORIZ);
115:         //        if (userAnchor.isVerticallyFlipped())
116:         //            sp.setFlags(sp.getFlags() | EscherSpRecord.FLAG_FLIPVERT);
117:         anchor = createAnchor( userAnchor );
118:         clientData.setRecordId( EscherClientDataRecord.RECORD_ID );
119:         clientData.setOptions( (short) 0x0000 );
120:         escherTextbox.setRecordId( EscherTextboxRecord.RECORD_ID );
121:         escherTextbox.setOptions( (short) 0x0000 );
122: 
123:         spContainer.addChildRecord( sp );
124:         spContainer.addChildRecord( opt );
125:         spContainer.addChildRecord( anchor );
126:         spContainer.addChildRecord( clientData );
127:         spContainer.addChildRecord( escherTextbox );
128: 
129:         return spContainer;
130:     }
131: 
132:     /**
133:      * Textboxes also have an extra TXO record associated with them that most
134:      * other shapes dont have.
135:      */
136:     private TextObjectRecord createTextObjectRecord( HSSFTextbox hssfShape, int shapeId )
137:     {
138:         HSSFTextbox shape = hssfShape;
139: 
140:         TextObjectRecord obj = new TextObjectRecord();
141:         obj.setHorizontalTextAlignment(hssfShape.getHorizontalAlignment());
142:         obj.setVerticalTextAlignment(hssfShape.getVerticalAlignment());
143:         obj.setTextLocked(true);
144:         obj.setTextOrientation(TextObjectRecord.TEXT_ORIENTATION_NONE);
145:         obj.setStr(shape.getString());
146: 
147:         return obj;
148:     }
