 82:     /**
 83:      * Given a Date in the form of a Calendar, converts it into a double
 84:      *  representing its internal Excel representation, which is the
 85:      *  number of days since 1/1/1900. Fractional days represent hours,
 86:      *  minutes, and seconds.
 87:      *
 88:      * @return Excel representation of Date (-1 if error - test for error by checking for less than 0.1)
 89:      * @param date the Calendar holding the date to convert
 90:      * @param use1904windowing Should 1900 or 1904 date windowing be used?
 91:      */
 92:     public static double getExcelDate(Calendar date, boolean use1904windowing) {
 93:         // Don't alter the supplied Calendar as we do our work
 94:         return internalGetExcelDate( (Calendar)date.clone(), use1904windowing );

 96:     private static double internalGetExcelDate(Calendar date, boolean use1904windowing) {
 97:         if ((!use1904windowing && date.get(Calendar.YEAR) < 1900) ||
 98:             (use1904windowing && date.get(Calendar.YEAR) < 1904))
 99:         {
100:             return BAD_DATE;
101:         }
102:         // Because of daylight time saving we cannot use
103:         //     date.getTime() - calStart.getTimeInMillis()
104:         // as the difference in milliseconds between 00:00 and 04:00
105:         // can be 3, 4 or 5 hours but Excel expects it to always
106:         // be 4 hours.
107:         // E.g. 2004-03-28 04:00 CEST - 2004-03-28 00:00 CET is 3 hours
108:         // and 2004-10-31 04:00 CET - 2004-10-31 00:00 CEST is 5 hours
109:         double fraction = (((date.get(Calendar.HOUR_OF_DAY) * 60
110:                              + date.get(Calendar.MINUTE)
111:                             ) * 60 + date.get(Calendar.SECOND)
112:                            ) * 1000 + date.get(Calendar.MILLISECOND)
113:                           ) / ( double ) DAY_MILLISECONDS;
114:         Calendar calStart = dayStart(date);
115: 
116:         double value = fraction + absoluteDay(calStart, use1904windowing);
117: 
118:         if (!use1904windowing && value >= 60) {
119:             value++;
120:         } else if (use1904windowing) {
121:             value--;
122:         }
123: 
124:         return value;
125:     }

147:     /**
148:      *  Given an Excel date with either 1900 or 1904 date windowing,
149:      *  converts it to a java.util.Date.
150:      *
151:      *  NOTE: If the default <code>TimeZone</code> in Java uses Daylight
152:      *  Saving Time then the conversion back to an Excel date may not give
153:      *  the same value, that is the comparison
154:      *  <CODE>excelDate == getExcelDate(getJavaDate(excelDate,false))</CODE>
155:      *  is not always true. For example if default timezone is
156:      *  <code>Europe/Copenhagen</code>, on 2004-03-28 the minute after
157:      *  01:59 CET is 03:00 CEST, if the excel date represents a time between
158:      *  02:00 and 03:00 then it is converted to past 03:00 summer time
159:      *
160:      *  @param date  The Excel date.
161:      *  @param use1904windowing  true if date uses 1904 windowing,
162:      *   or false if using 1900 date windowing.
163:      *  @return Java representation of the date, or null if date is not a valid Excel date
164:      *  @see java.util.TimeZone
165:      */
166:     public static Date getJavaDate(double date, boolean use1904windowing) {
167:         if (!isValidExcelDate(date)) {
168:             return null;
169:         }
170:         int wholeDays = (int)Math.floor(date);
171:         int millisecondsInDay = (int)((date - wholeDays) * DAY_MILLISECONDS + 0.5);
172:         Calendar calendar = new GregorianCalendar(); // using default time-zone
173:         setCalendar(calendar, wholeDays, millisecondsInDay, use1904windowing);
174:         return calendar.getTime();
175:     }
176:     public static void setCalendar(Calendar calendar, int wholeDays,
177:             int millisecondsInDay, boolean use1904windowing) {
178:         int startYear = 1900;
179:         int dayAdjust = -1; // Excel thinks 2/29/1900 is a valid date, which it isn't
180:         if (use1904windowing) {
181:             startYear = 1904;
182:             dayAdjust = 1; // 1904 date windowing uses 1/2/1904 as the first day
183:         }
184:         else if (wholeDays < 61) {
185:             // Date is prior to 3/1/1900, so adjust because Excel thinks 2/29/1900 exists
186:             // If Excel date == 2/29/1900, will become 3/1/1900 in Java representation
187:             dayAdjust = 0;
188:         }
189:         calendar.set(startYear,0, wholeDays + dayAdjust, 0, 0, 0);
190:         calendar.set(GregorianCalendar.MILLISECOND, millisecondsInDay);
191:     }

194:     /**
195:      * Given a format ID and its format String, will check to see if the
196:      *  format represents a date format or not.
197:      * Firstly, it will check to see if the format ID corresponds to an
198:      *  internal excel date format (eg most US date formats)
199:      * If not, it will check to see if the format string only contains
200:      *  date formatting characters (ymd-/), which covers most
201:      *  non US date formats.
202:      *
203:      * @param formatIndex The index of the format, eg from ExtendedFormatRecord.getFormatIndex
204:      * @param formatString The format string, eg from FormatRecord.getFormatString
205:      * @see #isInternalDateFormat(int)
206:      */
207:     public static boolean isADateFormat(int formatIndex, String formatString) {
208:         // First up, is this an internal date format?
209:         if(isInternalDateFormat(formatIndex)) {
210:             return true;
211:         }
212: 
213:         // If we didn't get a real string, it can't be
214:         if(formatString == null || formatString.length() == 0) {
215:             return false;
216:         }
217: 
218:         String fs = formatString;
219:         if (false) {
220:             // Normalize the format string. The code below is equivalent
221:             // to the following consecutive regexp replacements:
222: 
223:              // Translate \- into just -, before matching
224:              fs = fs.replaceAll("\\\\-","-");
225:              // And \, into ,
226:              fs = fs.replaceAll("\\\\,",",");
227:              // And \. into .
228:              fs = fs.replaceAll("\\\\\\.",".");
229:              // And '\ ' into ' '
230:              fs = fs.replaceAll("\\\\ "," ");
231: 
232:              // If it end in ;@, that's some crazy dd/mm vs mm/dd
233:              //  switching stuff, which we can ignore
234:              fs = fs.replaceAll(";@", "");
235: 
236:              // The code above was reworked as suggested in bug 48425:
237:              // simple loop is more efficient than consecutive regexp replacements.
238:         }
239:         StringBuilder sb = new StringBuilder(fs.length());
240:         for (int i = 0; i < fs.length(); i++) {
241:             char c = fs.charAt(i);
242:             if (i < fs.length() - 1) {
243:                 char nc = fs.charAt(i + 1);
244:                 if (c == '\\') {
245:                     switch (nc) {
246:                         case '-':
247:                         case ',':
248:                         case '.':
249:                         case ' ':
250:                         case '\\':
251:                             // skip current '\' and continue to the next char
252:                             continue;
253:                     }
254:                 } else if (c == ';' && nc == '@') {
255:                     i++;
256:                     // skip ";@" duplets
257:                     continue;
258:                 }
259:             }
260:             sb.append(c);
261:         }
262:         fs = sb.toString();
263: 
264:         // short-circuit if it indicates elapsed time: [h], [m] or [s]
265:         if(date_ptrn4.matcher(fs).matches()){
266:             return true;
267:         }
268: 
269:         // If it starts with [$-...], then could be a date, but
270:         //  who knows what that starting bit is all about
271:         fs = date_ptrn1.matcher(fs).replaceAll("");
272:         // If it starts with something like [Black] or [Yellow],
273:         //  then it could be a date
274:         fs = date_ptrn2.matcher(fs).replaceAll("");
275:         // You're allowed something like dd/mm/yy;[red]dd/mm/yy
276:         //  which would place dates before 1900/1904 in red
277:         // For now, only consider the first one
278:         if(fs.indexOf(';') > 0 && fs.indexOf(';') < fs.length()-1) {
279:            fs = fs.substring(0, fs.indexOf(';'));
280:         }
281: 
282:         // Otherwise, check it's only made up, in any case, of:
283:         //  y m d h s - \ / , . :
284:         // optionally followed by AM/PM
285:         return date_ptrn3.matcher(fs).matches();
286:     }

382:     /**
383:      * Return the number of days in prior years since 1900
384:      *
385:      * @return    days  number of days in years prior to yr.
386:      * @param     yr    a year (1900 < yr < 4000)
387:      * @param use1904windowing
388:      * @exception IllegalArgumentException if year is outside of range.
389:      */
390: 
391:     private static int daysInPriorYears(int yr, boolean use1904windowing)
392:     {
393:         if ((!use1904windowing && yr < 1900) || (use1904windowing && yr < 1900)) {
394:             throw new IllegalArgumentException("'year' must be 1900 or greater");
395:         }
396: 
397:         int yr1  = yr - 1;
398:         int leapDays =   yr1 / 4   // plus julian leap days in prior years
399:                        - yr1 / 100 // minus prior century years
400:                        + yr1 / 400 // plus years divisible by 400
401:                        - 460;      // leap days in previous 1900 years
402: 
403:         return 365 * (yr - (use1904windowing ? 1904 : 1900)) + leapDays;
404:     }

428:      * Converts a string of format "HH:MM" or "HH:MM:SS" to its (Excel) numeric equivalent
429:      *
430:      * @return a double between 0 and 1 representing the fraction of the day
431:      */
432:     public static double convertTime(String timeStr) {
433:         try {
434:             return convertTimeInternal(timeStr);
435:         } catch (FormatException e) {
436:             String msg = "Bad time format '" + timeStr
437:                 + "' expected 'HH:MM' or 'HH:MM:SS' - " + e.getMessage();
438:             throw new IllegalArgumentException(msg);
439:         }
440:     }
441:     private static double convertTimeInternal(String timeStr) throws FormatException {
442:         int len = timeStr.length();
443:         if (len < 4 || len > 8) {
444:             throw new FormatException("Bad length");
445:         }
446:         String[] parts = TIME_SEPARATOR_PATTERN.split(timeStr);
447: 
448:         String secStr;
449:         switch (parts.length) {
450:             case 2: secStr = "00"; break;
451:             case 3: secStr = parts[2]; break;
452:             default:
453:                 throw new FormatException("Expected 2 or 3 fields but got (" + parts.length + ")");
454:         }
455:         String hourStr = parts[0];
456:         String minStr = parts[1];
457:         int hours = parseInt(hourStr, "hour", HOURS_PER_DAY);
458:         int minutes = parseInt(minStr, "minute", MINUTES_PER_HOUR);
459:         int seconds = parseInt(secStr, "second", SECONDS_PER_MINUTE);
460: 
461:         double totalSeconds = seconds + (minutes + (hours) * 60) * 60;
462:         return totalSeconds / (SECONDS_PER_DAY);
463:     }

465:      * Converts a string of format "YYYY/MM/DD" to its (Excel) numeric equivalent
466:      *
467:      * @return a double representing the (integer) number of days since the start of the Excel epoch
468:      */
469:     public static Date parseYYYYMMDDDate(String dateStr) {
470:         try {
471:             return parseYYYYMMDDDateInternal(dateStr);
472:         } catch (FormatException e) {
473:             String msg = "Bad time format " + dateStr
474:                 + " expected 'YYYY/MM/DD' - " + e.getMessage();
475:             throw new IllegalArgumentException(msg);
476:         }
477:     }
478:     private static Date parseYYYYMMDDDateInternal(String timeStr) throws FormatException {
479:         if(timeStr.length() != 10) {
480:             throw new FormatException("Bad length");
481:         }
482: 
483:         String yearStr = timeStr.substring(0, 4);
484:         String monthStr = timeStr.substring(5, 7);
485:         String dayStr = timeStr.substring(8, 10);
486:         int year = parseInt(yearStr, "year", Short.MIN_VALUE, Short.MAX_VALUE);
487:         int month = parseInt(monthStr, "month", 1, 12);
488:         int day = parseInt(dayStr, "day", 1, 31);
489: 
490:         Calendar cal = new GregorianCalendar(year, month-1, day, 0, 0, 0);
491:         cal.set(Calendar.MILLISECOND, 0);
492:         return cal.getTime();
493:     }
