 53:     }

 55:     private static FuncVarPtg create(int numArgs, int functionIndex) {
 56:         FunctionMetadata fm = FunctionMetadataRegistry.getFunctionByIndex(functionIndex);
 57:         if(fm == null) {
 58:             // Happens only as a result of a call to FormulaParser.parse(), with a non-built-in function name
 59:             return new FuncVarPtg(functionIndex, Ptg.CLASS_VALUE, new byte[] {Ptg.CLASS_VALUE}, numArgs);
 60:         }
 61:         return new FuncVarPtg(functionIndex, fm.getReturnClassCode(), fm.getParameterClassCodes(), numArgs);
 62:     }
 63:

 65:         out.writeByte(sid + getPtgClass());
 66:         out.writeByte(getNumberOfOperands());
