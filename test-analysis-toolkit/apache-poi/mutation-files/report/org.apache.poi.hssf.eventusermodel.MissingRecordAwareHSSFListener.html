 33: /**
 34:  * <p>A HSSFListener which tracks rows and columns, and will
 35:  *  trigger your HSSFListener for all rows and cells,
 36:  *  even the ones that aren't actually stored in the file.</p>
 37:  * <p>This allows your code to have a more "Excel" like
 38:  *  view of the data in the file, and not have to worry
 39:  *  (as much) about if a particular row/cell is in the
 40:  *  file, or was skipped from being written as it was
 41:  *  blank.
 42:  */
 43: public final class MissingRecordAwareHSSFListener implements HSSFListener {
 44: 	private HSSFListener childListener;
 45: 
 46: 	// Need to have different counters for cell rows and
 47: 	//  row rows, as you sometimes get a RowRecord in the
 48: 	//  middle of some cells, and that'd break everything
 49: 	private int lastRowRow;
 50: 
 51: 	private int lastCellRow;
 52: 	private int lastCellColumn;
 53: 
 54: 	/**
 55: 	 * Constructs a new MissingRecordAwareHSSFListener, which
 56: 	 *  will fire processRecord on the supplied child
 57: 	 *  HSSFListener for all Records, and missing records.
 58: 	 * @param listener The HSSFListener to pass records on to
 59: 	 */
 60: 	public MissingRecordAwareHSSFListener(HSSFListener listener) {
 61: 		resetCounts();
 62: 		childListener = listener;
 63: 	}
 64: 
 65: 	public void processRecord(Record record) {
 66: 		int thisRow;
 67: 		int thisColumn;
 68: 		CellValueRecordInterface[] expandedRecords = null;
 69: 
 70: 		if (record instanceof CellValueRecordInterface) {
 71: 			CellValueRecordInterface valueRec = (CellValueRecordInterface) record;
 72: 			thisRow = valueRec.getRow();
 73: 			thisColumn = valueRec.getColumn();
 74: 		} else {
 75: 			thisRow = -1;
 76: 			thisColumn = -1;
 77: 
 78: 			switch (record.getSid()) {
 79: 				// the BOFRecord can represent either the beginning of a sheet or
 80: 				// the workbook
 81: 				case BOFRecord.sid:
 82: 					BOFRecord bof = (BOFRecord) record;
 83: 					if (bof.getType() == bof.TYPE_WORKBOOK || bof.getType() == bof.TYPE_WORKSHEET) {
 84: 						// Reset the row and column counts - new workbook / worksheet
 85: 						resetCounts();
 86: 					}
 87: 					break;
 88: 				case RowRecord.sid:
 89: 					RowRecord rowrec = (RowRecord) record;
 90: 					//System.out.println("Row " + rowrec.getRowNumber() + " found, first column at "
 91: 					//        + rowrec.getFirstCol() + " last column at " + rowrec.getLastCol());
 92: 
 93: 					// If there's a jump in rows, fire off missing row records
 94: 					if (lastRowRow + 1 < rowrec.getRowNumber()) {
 95: 						for (int i = (lastRowRow + 1); i < rowrec.getRowNumber(); i++) {
 96: 							MissingRowDummyRecord dr = new MissingRowDummyRecord(i);
 97: 							childListener.processRecord(dr);
 98: 						}
 99: 					}
100: 
101: 					// Record this as the last row we saw
102: 					lastRowRow = rowrec.getRowNumber();
103: 					break;
104: 
105: 				case SharedFormulaRecord.sid:
106: 					// SharedFormulaRecord occurs after the first FormulaRecord of the cell range.
107: 					// There are probably (but not always) more cell records after this
108: 					// - so don't fire off the LastCellOfRowDummyRecord yet
109: 					childListener.processRecord(record);
110: 					return;
111: 				case MulBlankRecord.sid:
112: 					// These appear in the middle of the cell records, to
113: 					//  specify that the next bunch are empty but styled
114: 					// Expand this out into multiple blank cells
115: 					MulBlankRecord mbr = (MulBlankRecord)record;
116: 					expandedRecords = RecordFactory.convertBlankRecords(mbr);
117: 					break;
118: 				case MulRKRecord.sid:
119: 					// This is multiple consecutive number cells in one record
120: 					// Exand this out into multiple regular number cells
121: 					MulRKRecord mrk = (MulRKRecord)record;
122: 					expandedRecords = RecordFactory.convertRKRecords(mrk);
123: 					break;
124: 				case NoteRecord.sid:
125: 					NoteRecord nrec = (NoteRecord) record;
126: 					thisRow = nrec.getRow();
127: 					thisColumn = nrec.getColumn();
128: 					break;
129: 			}
130: 		}
131: 
132: 		// First part of expanded record handling
133: 		if(expandedRecords != null && expandedRecords.length > 0) {
134: 			thisRow = expandedRecords[0].getRow();
135: 			thisColumn = expandedRecords[0].getColumn();
136: 		}
137: 
138: 		// If we're on cells, and this cell isn't in the same
139: 		//  row as the last one, then fire the
140: 		//  dummy end-of-row records
141: 		if(thisRow != lastCellRow && lastCellRow > -1) {
142: 			for(int i=lastCellRow; i<thisRow; i++) {
143: 				int cols = -1;
144: 				if(i == lastCellRow) {
145: 					cols = lastCellColumn;
146: 				}
147: 				childListener.processRecord(new LastCellOfRowDummyRecord(i, cols));
148: 			}
149: 		}
150: 
151: 		// If we've just finished with the cells, then fire the
152: 		// final dummy end-of-row record
153: 		if(lastCellRow != -1 && lastCellColumn != -1 && thisRow == -1) {
154: 			childListener.processRecord(new LastCellOfRowDummyRecord(lastCellRow, lastCellColumn));
155: 
156: 			lastCellRow = -1;
157: 			lastCellColumn = -1;
158: 		}
159: 
160: 		// If we've moved onto a new row, the ensure we re-set
161: 		//  the column counter
162: 		if(thisRow != lastCellRow) {
163: 			lastCellColumn = -1;
164: 		}
165: 
166: 		// If there's a gap in the cells, then fire
167: 		//  the dummy cell records
168: 		if(lastCellColumn != thisColumn-1) {
169: 			for(int i=lastCellColumn+1; i<thisColumn; i++) {
170: 				childListener.processRecord(new MissingCellDummyRecord(thisRow, i));
171: 			}
172: 		}
173: 
174: 		// Next part of expanded record handling
175: 		if(expandedRecords != null && expandedRecords.length > 0) {
176: 			thisColumn = expandedRecords[expandedRecords.length-1].getColumn();
177: 		}
178: 
179: 
180: 		// Update cell and row counts as needed
181: 		if(thisColumn != -1) {
182: 			lastCellColumn = thisColumn;
183: 			lastCellRow = thisRow;
184: 		}
185: 
186: 		// Pass along the record(s)
187: 		if(expandedRecords != null && expandedRecords.length > 0) {
188: 			for(CellValueRecordInterface r : expandedRecords) {
189: 				childListener.processRecord((Record)r);
190: 			}
191: 		} else {
192: 			childListener.processRecord(record);
193: 		}
194: 	}
195: 
196: 	private void resetCounts() {
197: 		lastRowRow = -1;
198: 		lastCellRow = -1;
199: 		lastCellColumn = -1;
200: 	}
