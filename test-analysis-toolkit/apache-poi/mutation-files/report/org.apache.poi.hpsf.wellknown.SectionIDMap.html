 22: /**
 23:  * <p>Maps section format IDs to {@link PropertyIDMap}s. It is
 24:  * initialized with two well-known section format IDs: those of the
 25:  * <tt>\005SummaryInformation</tt> stream and the
 26:  * <tt>\005DocumentSummaryInformation</tt> stream.</p>
 27:  *
 28:  * <p>If you have a section format ID you can use it as a key to query
 29:  * this map.  If you get a {@link PropertyIDMap} returned your section
 30:  * is well-known and you can query the {@link PropertyIDMap} for PID
 31:  * strings. If you get back <code>null</code> you are on your own.</p>
 32:  *
 33:  * <p>This {@link java.util.Map} expects the byte arrays of section format IDs
 34:  * as keys. A key maps to a {@link PropertyIDMap} describing the
 35:  * property IDs in sections with the specified section format ID.</p>
 36:  *
 37:  * @author Rainer Klute (klute@rainer-klute.de)
 38:  */

 85:     /**
 86:      * <p>Returns the singleton instance of the default {@link
 87:      * SectionIDMap}.</p>
 88:      *
 89:      * @return The instance value
 90:      */
 91:     public static SectionIDMap getInstance()
 92:     {
 93:         if (defaultMap == null)
 94:         {
 95:             final SectionIDMap m = new SectionIDMap();
 96:             m.put(SUMMARY_INFORMATION_ID,
 97:                   PropertyIDMap.getSummaryInformationProperties());
 98:             m.put(DOCUMENT_SUMMARY_INFORMATION_ID[0],
 99:                   PropertyIDMap.getDocumentSummaryInformationProperties());
100:             defaultMap = m;
101:         }
102:         return defaultMap;
103:     }

107:     /**
108:      * <p>Returns the property ID string that is associated with a
109:      * given property ID in a section format ID's namespace.</p>
110:      *
111:      * @param sectionFormatID Each section format ID has its own name
112:      * space of property ID strings and thus must be specified.
113:      * @param  pid The property ID
114:      * @return The well-known property ID string associated with the
115:      * property ID <var>pid</var> in the name space spanned by <var>
116:      * sectionFormatID</var> . If the <var>pid</var>
117:      * /<var>sectionFormatID </var> combination is not well-known, the
118:      * string "[undefined]" is returned.
119:      */
120:     public static String getPIDString(final byte[] sectionFormatID,
121:                                       final long pid)
122:     {
123:         final PropertyIDMap m = getInstance().get(sectionFormatID);
124:         if (m == null) {
125:             return UNDEFINED;
126:         }
127:         final String s = (String) m.get(pid);
128:         if (s == null)
129:             return UNDEFINED;
130:         return s;
131:     }
