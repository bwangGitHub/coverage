 35:     /**
 36:      * read a sub-record from the supplied stream
 37:      *
 38:      * @param in    the stream to read from
 39:      * @param cmoOt the objectType field of the containing CommonObjectDataSubRecord,
 40:      *   we need it to propagate to next sub-records as it defines what data follows
 41:      * @return the created sub-record
 42:      */
 43:     public static SubRecord createSubRecord(LittleEndianInput in, int cmoOt) {
 44: 		int sid = in.readUShort();
 45: 		int secondUShort = in.readUShort(); // Often (but not always) the datasize for the sub-record
 46: 
 47: 		switch (sid) {
 48: 			case CommonObjectDataSubRecord.sid:
 49: 				return new CommonObjectDataSubRecord(in, secondUShort);
 50: 			case EmbeddedObjectRefSubRecord.sid:
 51: 				return new EmbeddedObjectRefSubRecord(in, secondUShort);
 52: 			case GroupMarkerSubRecord.sid:
 53: 				return new GroupMarkerSubRecord(in, secondUShort);
 54: 			case EndSubRecord.sid:
 55: 				return new EndSubRecord(in, secondUShort);
 56: 			case NoteStructureSubRecord.sid:
 57: 				return new NoteStructureSubRecord(in, secondUShort);
 58: 			case LbsDataSubRecord.sid:
 59: 				return new LbsDataSubRecord(in, secondUShort, cmoOt);
 60:             case FtCblsSubRecord.sid:
 61:                 return new FtCblsSubRecord(in, secondUShort);
 62: 		}
 63: 		return new UnknownSubRecord(in, sid, secondUShort);
 64: 	}
 65: 
 66: 	/**
 67: 	 * @return the size of the data for this record (which is always 4 bytes less than the total
 68: 	 * record size).  Note however, that ushort encoded after the record sid is usually but not
 69: 	 * always the data size.
 70: 	 */
 71: 	protected abstract int getDataSize();
 72: 	public byte[] serialize() {
 73: 		int size = getDataSize() + 4;
 74: 		ByteArrayOutputStream baos = new ByteArrayOutputStream(size);
 75: 		serialize(new LittleEndianOutputStream(baos));
 76: 		if (baos.size() != size) {
 77: 			throw new RuntimeException("write size mismatch");
 78: 		}
 79: 		return baos.toByteArray();
 80: 	}
