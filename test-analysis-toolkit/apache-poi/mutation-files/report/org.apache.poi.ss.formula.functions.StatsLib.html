 38:     public static double avedev(double[] v) {
 39:         double r = 0;
 40:         double m = 0;
 41:         double s = 0;
 42:         for (int i=0, iSize=v.length; i<iSize; i++) {
 43:             s += v[i];
 44:         }
 45:         m = s / v.length;
 46:         s = 0;
 47:         for (int i=0, iSize=v.length; i<iSize; i++) {

 93:     public static double devsq(double[] v) {
 94:         double r = Double.NaN;
 95:         if (v!=null && v.length >= 1) {
 96:             double m = 0;
 97:             double s = 0;
 98:             int n = v.length;
 99:             for (int i=0; i<n; i++) {
100:                 s += v[i];
101:             }
102:             m = s / n;
103:             s = 0;
104:             for (int i=0; i<n; i++) {
105:                 s += (v[i]- m) * (v[i] - m);
106:             }
107: 
108:             r = (n == 1)
109:                     ? 0
110:                     : s;
111:         }
112:         return r;
113:     }
