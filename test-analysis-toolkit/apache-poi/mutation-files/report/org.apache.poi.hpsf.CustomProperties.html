 28: /**
 29:  * <p>Maintains the instances of {@link CustomProperty} that belong to a
 30:  * {@link DocumentSummaryInformation}. The class maintains the names of the
 31:  * custom properties in a dictionary. It implements the {@link Map} interface
 32:  * and by this provides a simplified view on custom properties: A property's
 33:  * name is the key that maps to a typed value. This implementation hides
 34:  * property IDs from the developer and regards the property names as keys to
 35:  * typed values.</p>
 36:  *
 37:  * <p>While this class provides a simple API to custom properties, it ignores
 38:  * the fact that not names, but IDs are the real keys to properties. Under the
 39:  * hood this class maintains a 1:1 relationship between IDs and names. Therefore
 40:  * you should not use this class to process property sets with several IDs
 41:  * mapping to the same name or with properties without a name: the result will
 42:  * contain only a subset of the original properties. If you really need to deal
 43:  * such property sets, use HPSF's low-level access methods.</p>
 44:  *
 45:  * <p>An application can call the {@link #isPure} method to check whether a
 46:  * property set parsed by {@link CustomProperties} is still pure (i.e.
 47:  * unmodified) or whether one or more properties have been dropped.</p>
 48:  *
 49:  * <p>This class is not thread-safe; concurrent access to instances of this
 50:  * class must be synchronized.</p>
 51:  *
 52:  * <p>While this class is roughly HashMap<Long,CustomProperty>, that's the
 53:  *  internal representation. To external calls, it should appear as
 54:  *  HashMap<String,Object> mapping between Names and Custom Property Values.</p>
 55:  *
 56:  * @author Rainer Klute <a
 57:  *         href="mailto:klute@rainer-klute.de">&lt;klute@rainer-klute.de&gt;</a>
 58:  */

 79:     /**
 80:      * <p>Puts a {@link CustomProperty} into this map. It is assumed that the
 81:      * {@link CustomProperty} already has a valid ID. Otherwise use
 82:      * {@link #put(CustomProperty)}.</p>
 83:      */
 84:     public CustomProperty put(final String name, final CustomProperty cp)
 85:     {
 86:         if (name == null)
 87:         {
 88:             /* Ignoring a property without a name. */
 89:             isPure = false;
 90:             return null;
 91:         }
 92:         if (!(name.equals(cp.getName())))
 93:             throw new IllegalArgumentException("Parameter \"name\" (" + name +
 94:                     ") and custom property's name (" + cp.getName() +
 95:                     ") do not match.");
 96: 
 97:         /* Register name and ID in the dictionary. Mapping in both directions is possible. If there is already a  */
 98:         final Long idKey = Long.valueOf(cp.getID());
 99:         final Long oldID = dictionaryNameToID.get(name);
100:         dictionaryIDToName.remove(oldID);
101:         dictionaryNameToID.put(name, idKey);
102:         dictionaryIDToName.put(idKey, name);
103: 
104:         /* Put the custom property into this map. */
105:         final CustomProperty oldCp = super.remove(oldID);
106:         super.put(idKey, cp);
107:         return oldCp;
108:     }

112:     /**
113:      * <p>Puts a {@link CustomProperty} that has not yet a valid ID into this
114:      * map. The method will allocate a suitable ID for the custom property:</p>
115:      *
116:      * <ul>
117:      *
118:      * <li><p>If there is already a property with the same name, take the ID
119:      * of that property.</p></li>
120:      *
121:      * <li><p>Otherwise find the highest ID and use its value plus one.</p></li>
122:      *
123:      * </ul>
124:      *
125:      * @param customProperty
126:      * @return If the was already a property with the same name, the
127:      * @throws ClassCastException
128:      */
129:     private Object put(final CustomProperty customProperty) throws ClassCastException
130:     {
131:         final String name = customProperty.getName();
132: 
133:         /* Check whether a property with this name is in the map already. */
134:         final Long oldId = (Long) dictionaryNameToID.get(name);
135:         if (oldId != null)
136:             customProperty.setID(oldId.longValue());
137:         else
138:         {
139:             long max = 1;
140:             for (final Iterator<Long> i = dictionaryIDToName.keySet().iterator(); i.hasNext();)
141:             {
142:                 final long id = i.next().longValue();
143:                 if (id > max)
144:                     max = id;
145:             }
146:             customProperty.setID(max + 1);
147:         }
148:         return this.put(name, customProperty);
149:     }

362:    /**
363:     * Checks against both the property, and its values. 
364:     */
365:    public boolean containsValue(Object value) {
366:       if(value instanceof CustomProperty) {
367:          return super.containsValue((CustomProperty)value);
368:       } else {
369:          for(CustomProperty cp : super.values()) {
370:             if(cp.getValue() == value) {
371:                return true;
372:             }
373:          }
374:       }
375:       return false;
376:    }
