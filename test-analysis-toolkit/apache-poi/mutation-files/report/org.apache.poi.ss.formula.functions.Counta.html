 28: /**
 29:  * Counts the number of cells that contain data within the list of arguments.
 30:  *
 31:  * Excel Syntax
 32:  * COUNTA(value1,value2,...)
 33:  * Value1, value2, ...   are 1 to 30 arguments representing the values or ranges to be counted.
 34:  *
 35:  * @author Josh Micich
 36:  */
 37: public final class Counta implements Function {
 38:     private final I_MatchPredicate _predicate;
 39: 
 40:     public Counta(){
 41:         _predicate = defaultPredicate;
 42:     }
 43: 
 44:     private Counta(I_MatchPredicate criteriaPredicate){
 45:         _predicate = criteriaPredicate;
 46:     }
 47: 
 48: 	public ValueEval evaluate(ValueEval[] args, int srcCellRow, int srcCellCol) {
 49: 		int nArgs = args.length;
 50: 		if (nArgs < 1) {
 51: 			// too few arguments
 52: 			return ErrorEval.VALUE_INVALID;
 53: 		}
 54: 
 55: 		if (nArgs > 30) {
 56: 			// too many arguments
 57: 			return ErrorEval.VALUE_INVALID;
 58: 		}
 59: 
 60: 		int temp = 0;
 61: 
 62: 		for(int i=0; i<nArgs; i++) {
 63: 			temp += CountUtils.countArg(args[i], _predicate);
 64: 
 65: 		}
 66: 		return new NumberEval(temp);
 67: 	}
 68: 
 69: 	private static final I_MatchPredicate defaultPredicate = new I_MatchPredicate() {
 70: 
 71: 		public boolean matches(ValueEval valueEval) {
 72: 			// Note - observed behavior of Excel:
 73: 			// Error values like #VALUE!, #REF!, #DIV/0!, #NAME? etc don't cause this COUNTA to return an error
 74: 			// in fact, they seem to get counted
 75: 
 76: 			if(valueEval == BlankEval.instance) {
 77: 				return false;
 78: 			}
 79: 			// Note - everything but BlankEval counts
 80: 			return true;
 81: 		}
 82: 	};
 83:     private static final I_MatchPredicate subtotalPredicate = new I_MatchAreaPredicate() {
 84:         public boolean matches(ValueEval valueEval) {
 85:             return defaultPredicate.matches(valueEval);
 86:         }
 87: 
 88:         /**
 89:          * don't count cells that are subtotals
 90:          */
 91:         public boolean matches(TwoDEval areEval, int rowIndex, int columnIndex) {
 92:             return !areEval.isSubTotal(rowIndex, columnIndex);
 93:         }
 94:     };
 95:
