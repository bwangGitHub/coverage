349: 	public void serialize(ContinuableRecordOutput out) {
350: 
351: 		int field_7_length_custom_menu = field_14_custom_menu_text.length();
352: 		int field_8_length_description_text = field_15_description_text.length();
353: 		int field_9_length_help_topic_text = field_16_help_topic_text.length();
354: 		int field_10_length_status_bar_text = field_17_status_bar_text.length();
355: 
356: 		// size defined below
357: 		out.writeShort(getOptionFlag());
358: 		out.writeByte(getKeyboardShortcut());
359: 		out.writeByte(getNameTextLength());
360: 		// Note - formula size is not immediately before encoded formula, and does not include any array constant data
361: 		out.writeShort(field_13_name_definition.getEncodedTokenSize());
362: 		out.writeShort(field_5_externSheetIndex_plus1);
363: 		out.writeShort(field_6_sheetNumber);
364: 		out.writeByte(field_7_length_custom_menu);
365: 		out.writeByte(field_8_length_description_text);
366: 		out.writeByte(field_9_length_help_topic_text);
367: 		out.writeByte(field_10_length_status_bar_text);
368: 		out.writeByte(field_11_nameIsMultibyte ? 1 : 0);
369: 
370: 		if (isBuiltInName()) {
371: 			//can send the builtin name directly in
372: 			out.writeByte(field_12_built_in_code);
373: 		} else {
374: 			String nameText = field_12_name_text;
375: 			if (field_11_nameIsMultibyte) {
376: 				StringUtil.putUnicodeLE(nameText, out);
377: 			} else {
378: 				StringUtil.putCompressedUnicode(nameText, out);
379: 			}
380: 		}
381: 		field_13_name_definition.serializeTokens(out);
382: 		field_13_name_definition.serializeArrayConstantData(out);
383: 
384: 		StringUtil.putCompressedUnicode( getCustomMenuText(), out);
385: 		StringUtil.putCompressedUnicode( getDescriptionText(), out);
386: 		StringUtil.putCompressedUnicode( getHelpTopicText(), out);
387: 		StringUtil.putCompressedUnicode( getStatusBarText(), out);
388: 	}
