149: 	 * Note - this method is expected to be called only when completed reading the current BIFF
150: 	 * record.
151: 	 * @throws LeftoverDataException if this method is called before reaching the end of the
152: 	 * current record.

154: 	public boolean hasNextRecord() throws LeftoverDataException {
155: 		if (_currentDataLength != -1 && _currentDataLength != _currentDataOffset) {
156: 			throw new LeftoverDataException(_currentSid, remaining());
157: 		}
158: 		if (_currentDataLength != DATA_LEN_NEEDS_TO_BE_READ) {
159: 			_nextSid = readNextSid();
160: 		}
161: 		return _nextSid != INVALID_SID_VALUE;
162: 	}

164: 	/**
165: 	 * @return the sid of the next record or {@link #INVALID_SID_VALUE} if at end of stream
166: 	 */
167: 	private int readNextSid() {
168: 		int nAvailable  = _bhi.available();
169: 		if (nAvailable < EOFRecord.ENCODED_SIZE) {
170: 			if (nAvailable > 0) {
171: 				// some scrap left over?
172: 				// ex45582-22397.xls has one extra byte after the last record
173: 				// Excel reads that file OK
174: 			}
175: 			return INVALID_SID_VALUE;
176: 		}
177: 		int result = _bhi.readRecordSID();
178: 		if (result == INVALID_SID_VALUE) {
179: 			throw new RecordFormatException("Found invalid sid (" + result + ")");
180: 		}
181: 		_currentDataLength = DATA_LEN_NEEDS_TO_BE_READ;
182: 		return result;
183: 	}

185: 	/** Moves to the next record in the stream.
186: 	 *
187: 	 * <i>Note: The auto continue flag is reset to true</i>
188: 	 */
189: 	public void nextRecord() throws RecordFormatException {
190: 		if (_nextSid == INVALID_SID_VALUE) {
191: 			throw new IllegalStateException("EOF - next record not available");
192: 		}
193: 		if (_currentDataLength != DATA_LEN_NEEDS_TO_BE_READ) {
194: 			throw new IllegalStateException("Cannot call nextRecord() without checking hasNextRecord() first");
195: 		}
196: 		_currentSid = _nextSid;
197: 		_currentDataOffset = 0;
198: 		_currentDataLength = _bhi.readDataSize();
199: 		if (_currentDataLength > MAX_RECORD_DATA_SIZE) {
200: 			throw new RecordFormatException("The content of an excel record cannot exceed "
201: 					+ MAX_RECORD_DATA_SIZE + " bytes");
202: 		}
203: 	}

205: 	private void checkRecordPosition(int requiredByteCount) {
206: 
207: 		int nAvailable = remaining();
208: 		if (nAvailable >= requiredByteCount) {
209: 			// all OK
210: 			return;
211: 		}
212: 		if (nAvailable == 0 && isContinueNext()) {
213: 			nextRecord();

318: 	private String readStringCommon(int requestedLength, boolean pIsCompressedEncoding) {
319: 		// Sanity check to detect garbage string lengths
320: 		if (requestedLength < 0 || requestedLength > 0x100000) { // 16 million chars?
321: 			throw new IllegalArgumentException("Bad requested string length (" + requestedLength + ")");
322: 		}
323: 		char[] buf = new char[requestedLength];
324: 		boolean isCompressedEncoding = pIsCompressedEncoding;
325: 		int curLen = 0;
326: 		while(true) {
327: 			int availableChars =isCompressedEncoding ?  remaining() : remaining() / LittleEndian.SHORT_SIZE;
328: 			if (requestedLength - curLen <= availableChars) {
329: 				// enough space in current record, so just read it out
330: 				while(curLen < requestedLength) {
331: 					char ch;
332: 					if (isCompressedEncoding) {
333: 						ch = (char)readUByte();
334: 					} else {
335: 						ch = (char)readShort();
336: 					}
337: 					buf[curLen] = ch;
338: 					curLen++;
339: 				}
340: 				return new String(buf);
341: 			}
342: 			// else string has been spilled into next continue record
343: 			// so read what's left of the current record
344: 			while(availableChars > 0) {
345: 				char ch;
346: 				if (isCompressedEncoding) {
347: 					ch = (char)readUByte();
348: 				} else {
349: 					ch = (char)readShort();
350: 				}
351: 				buf[curLen] = ch;
352: 				curLen++;
353: 				availableChars--;
354: 			}
355: 			if (!isContinueNext()) {
356: 				throw new RecordFormatException("Expected to find a ContinueRecord in order to read remaining "
357: 						+ (requestedLength-curLen) + " of " + requestedLength + " chars");
358: 			}
359: 			if(remaining() != 0) {
360: 				throw new RecordFormatException("Odd number of bytes(" + remaining() + ") left behind");
361: 			}
362: 			nextRecord();
363: 			// note - the compressed flag may change on the fly
364: 			byte compressFlag = readByte();
365: 			isCompressedEncoding = (compressFlag == 0);
366: 		}
367: 	}

406: 	 *
407: 	 * @return The number of bytes remaining in the current record

409: 	public int remaining() {
410: 		if (_currentDataLength == DATA_LEN_NEEDS_TO_BE_READ) {
411: 			// already read sid of next record. so current one is finished
412: 			return 0;
413: 		}
414: 		return _currentDataLength - _currentDataOffset;

417: 	/**
418: 	 *
419: 	 * @return <code>true</code> when a {@link ContinueRecord} is next.
420: 	 */
421: 	private boolean isContinueNext() {
422: 		if (_currentDataLength != DATA_LEN_NEEDS_TO_BE_READ && _currentDataOffset != _currentDataLength) {
423: 			throw new IllegalStateException("Should never be called before end of current record");
424: 		}
425: 		if (!hasNextRecord()) {
426: 			return false;
427: 		}
428: 		// At what point are records continued?
429: 		//  - Often from within the char data of long strings (caller is within readStringCommon()).
430: 		//  - From UnicodeString construction (many different points - call via checkRecordPosition)
431: 		//  - During TextObjectRecord construction (just before the text, perhaps within the text,
432: 		//    and before the formatting run data)
433: 		return _nextSid == ContinueRecord.sid;
434: 	}
