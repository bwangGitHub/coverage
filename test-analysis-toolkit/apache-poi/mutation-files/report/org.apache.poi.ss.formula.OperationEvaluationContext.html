 75: 	SheetRefEvaluator createExternSheetRefEvaluator(int externSheetIndex) {
 76: 		ExternalSheet externalSheet = _workbook.getExternalSheet(externSheetIndex);
 77: 		WorkbookEvaluator targetEvaluator;
 78: 		int otherSheetIndex;
 79: 		if (externalSheet == null) {
 80: 			// sheet is in same workbook
 81: 			otherSheetIndex = _workbook.convertFromExternSheetIndex(externSheetIndex);
 82: 			targetEvaluator = _bookEvaluator;
 83: 		} else {
 84: 			// look up sheet by name from external workbook
 85: 			String workbookName = externalSheet.getWorkbookName();
 86: 			try {
 87: 				targetEvaluator = _bookEvaluator.getOtherWorkbookEvaluator(workbookName);
 88: 			} catch (WorkbookNotFoundException e) {
 89: 				throw new RuntimeException(e.getMessage(), e);
 90: 			}
 91: 			otherSheetIndex = targetEvaluator.getSheetIndex(externalSheet.getSheetName());
 92: 			if (otherSheetIndex < 0) {
 93: 				throw new RuntimeException("Invalid sheet name '" + externalSheet.getSheetName()
 94: 						+ "' in bool '" + workbookName + "'.");
 95: 			}
 96: 		}
 97: 		return new SheetRefEvaluator(targetEvaluator, _tracker, otherSheetIndex);
 98: 	}

104: 		WorkbookEvaluator targetEvaluator;

106: 			targetEvaluator = _bookEvaluator;
107: 		} else {
108: 			if (sheetName == null) {
109: 				throw new IllegalArgumentException("sheetName must not be null if workbookName is provided");

117: 		int otherSheetIndex = sheetName == null ? _sheetIndex : targetEvaluator.getSheetIndex(sheetName);
118: 		if (otherSheetIndex < 0) {

130: 	/**
131: 	 * Resolves a cell or area reference dynamically.
132: 	 * @param workbookName the name of the workbook containing the reference.  If <code>null</code>
133: 	 * the current workbook is assumed.  Note - to evaluate formulas which use multiple workbooks,
134: 	 * a {@link CollaboratingWorkbooksEnvironment} must be set up.
135: 	 * @param sheetName the name of the sheet containing the reference.  May be <code>null</code>
136: 	 * (when <tt>workbookName</tt> is also null) in which case the current workbook and sheet is
137: 	 * assumed.
138: 	 * @param refStrPart1 the single cell reference or first part of the area reference.  Must not
139: 	 * be <code>null</code>.
140: 	 * @param refStrPart2 the second part of the area reference. For single cell references this
141: 	 * parameter must be <code>null</code>
142: 	 * @param isA1Style specifies the format for <tt>refStrPart1</tt> and <tt>refStrPart2</tt>.
143: 	 * Pass <code>true</code> for 'A1' style and <code>false</code> for 'R1C1' style.
144: 	 * TODO - currently POI only supports 'A1' reference style
145: 	 * @return a {@link RefEval} or {@link AreaEval}
146: 	 */
147: 	public ValueEval getDynamicReference(String workbookName, String sheetName, String refStrPart1,
148: 			String refStrPart2, boolean isA1Style) {
149: 		if (!isA1Style) {
150: 			throw new RuntimeException("R1C1 style not supported yet");
151: 		}
152: 		SheetRefEvaluator sre = createExternSheetRefEvaluator(workbookName, sheetName);
153: 		if (sre == null) {
154: 			return ErrorEval.REF_INVALID;
155: 		}
156: 		// ugly typecast - TODO - make spreadsheet version more easily accessible
157: 		SpreadsheetVersion ssVersion = ((FormulaParsingWorkbook)_workbook).getSpreadsheetVersion();
158: 
159: 		NameType part1refType = classifyCellReference(refStrPart1, ssVersion);
160: 		switch (part1refType) {
161: 			case BAD_CELL_OR_NAMED_RANGE:
162: 				return ErrorEval.REF_INVALID;
163: 			case NAMED_RANGE:
164:                 EvaluationName nm = ((FormulaParsingWorkbook)_workbook).getName(refStrPart1, _sheetIndex);
165:                 if(!nm.isRange()){
166:                     throw new RuntimeException("Specified name '" + refStrPart1 + "' is not a range as expected.");
167:                 }
168:                 return _bookEvaluator.evaluateNameFormula(nm.getNameDefinition(), this);
169: 		}
170: 		if (refStrPart2 == null) {
171: 			// no ':'
172: 			switch (part1refType) {
173: 				case COLUMN:
174: 				case ROW:
175: 					return ErrorEval.REF_INVALID;
176: 				case CELL:
177: 					CellReference cr = new CellReference(refStrPart1);
178: 					return new LazyRefEval(cr.getRow(), cr.getCol(), sre);
179: 			}
180: 			throw new IllegalStateException("Unexpected reference classification of '" + refStrPart1 + "'.");
181: 		}
182: 		NameType part2refType = classifyCellReference(refStrPart1, ssVersion);
183: 		switch (part2refType) {
184: 			case BAD_CELL_OR_NAMED_RANGE:
185: 				return ErrorEval.REF_INVALID;
186: 			case NAMED_RANGE:
187: 				throw new RuntimeException("Cannot evaluate '" + refStrPart1
188: 						+ "'. Indirect evaluation of defined names not supported yet");
189: 		}
190: 
191: 		if (part2refType != part1refType) {
192: 			// LHS and RHS of ':' must be compatible
193: 			return ErrorEval.REF_INVALID;
194: 		}
195: 		int firstRow, firstCol, lastRow, lastCol;
196: 		switch (part1refType) {
197: 			case COLUMN:
198: 				firstRow =0;
199: 				lastRow = ssVersion.getLastRowIndex();
200: 				firstCol = parseColRef(refStrPart1);
201: 				lastCol = parseColRef(refStrPart2);
202: 				break;
203: 			case ROW:
204: 				firstCol = 0;
205: 				lastCol = ssVersion.getLastColumnIndex();
206: 				firstRow = parseRowRef(refStrPart1);
207: 				lastRow = parseRowRef(refStrPart2);
208: 				break;
209: 			case CELL:
210: 				CellReference cr;
211: 				cr = new CellReference(refStrPart1);
212: 				firstRow = cr.getRow();
213: 				firstCol = cr.getCol();
214: 				cr = new CellReference(refStrPart2);
215: 				lastRow = cr.getRow();
216: 				lastCol = cr.getCol();
217: 				break;
218: 			default:
219: 				throw new IllegalStateException("Unexpected reference classification of '" + refStrPart1 + "'.");
220: 		}
221: 		return new LazyAreaEval(firstRow, firstCol, lastRow, lastCol, sre);
222: 	}
