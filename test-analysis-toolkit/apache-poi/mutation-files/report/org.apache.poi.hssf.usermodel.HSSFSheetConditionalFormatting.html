 45: 	 * A factory method allowing to create a conditional formatting rule
 46: 	 * with a cell comparison operator<p/>
 47: 	 * TODO - formulas containing cell references are currently not parsed properly
 48: 	 *
 49: 	 * @param comparisonOperation - a constant value from
 50: 	 *		 <tt>{@link org.apache.poi.hssf.record.CFRuleRecord.ComparisonOperator}</tt>: <p>
 51: 	 * <ul>
 52: 	 *		 <li>BETWEEN</li>
 53: 	 *		 <li>NOT_BETWEEN</li>
 54: 	 *		 <li>EQUAL</li>
 55: 	 *		 <li>NOT_EQUAL</li>
 56: 	 *		 <li>GT</li>
 57: 	 *		 <li>LT</li>
 58: 	 *		 <li>GE</li>
 59: 	 *		 <li>LE</li>
 60: 	 * </ul>
 61: 	 * </p>
 62: 	 * @param formula1 - formula for the valued, compared with the cell
 63: 	 * @param formula2 - second formula (only used with
 64: 	 * {@link org.apache.poi.hssf.record.CFRuleRecord.ComparisonOperator#BETWEEN}) and
 65: 	 * {@link org.apache.poi.hssf.record.CFRuleRecord.ComparisonOperator#NOT_BETWEEN} operations)
 66: 	 */
 67: 	public HSSFConditionalFormattingRule createConditionalFormattingRule(
 68: 			byte comparisonOperation,
 69: 			String formula1,
 70: 			String formula2) {
 71: 
 72: 		HSSFWorkbook wb = _sheet.getWorkbook();
 73: 		CFRuleRecord rr = CFRuleRecord.create(_sheet, comparisonOperation, formula1, formula2);
 74: 		return new HSSFConditionalFormattingRule(wb, rr);
 75: 	}

127: 	/**
128: 	 * Allows to add a new Conditional Formatting set to the sheet.
129: 	 *
130: 	 * @param regions - list of rectangular regions to apply conditional formatting rules
131: 	 * @param cfRules - set of up to three conditional formatting rules
132: 	 *
133: 	 * @return index of the newly created Conditional Formatting object
134: 	 */
135: 	public int addConditionalFormatting(CellRangeAddress[] regions, HSSFConditionalFormattingRule[] cfRules) {
136: 		if (regions == null) {
137: 			throw new IllegalArgumentException("regions must not be null");
138: 		}
139: 		for(CellRangeAddress range : regions) range.validate(SpreadsheetVersion.EXCEL97);
140: 
141: 		if (cfRules == null) {
142: 			throw new IllegalArgumentException("cfRules must not be null");
143: 		}
144: 		if (cfRules.length == 0) {
145: 			throw new IllegalArgumentException("cfRules must not be empty");
146: 		}
147: 		if (cfRules.length > 3) {
148: 			throw new IllegalArgumentException("Number of rules must not exceed 3");
149: 		}
150: 
151: 		CFRuleRecord[] rules = new CFRuleRecord[cfRules.length];
152: 		for (int i = 0; i != cfRules.length; i++) {
153: 			rules[i] = cfRules[i].getCfRuleRecord();
154: 		}
155: 		CFRecordsAggregate cfra = new CFRecordsAggregate(regions, rules);
156: 		return _conditionalFormattingTable.add(cfra);
157: 	}

159:     public int addConditionalFormatting(CellRangeAddress[] regions, ConditionalFormattingRule[] cfRules) {
160:         HSSFConditionalFormattingRule[] hfRules;
161:         if(cfRules instanceof HSSFConditionalFormattingRule[]) hfRules = (HSSFConditionalFormattingRule[])cfRules;
162:         else {
163:             hfRules = new HSSFConditionalFormattingRule[cfRules.length];
164:             System.arraycopy(cfRules, 0, hfRules, 0, hfRules.length);
165:         }
166:         return addConditionalFormatting(regions, hfRules);
167:     }
