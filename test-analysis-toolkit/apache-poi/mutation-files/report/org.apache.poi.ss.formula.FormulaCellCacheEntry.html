 28: /**
 29:  * Stores the cached result of a formula evaluation, along with the set of sensitive input cells
 30:  * 
 31:  * @author Josh Micich
 32:  */
 33: final class FormulaCellCacheEntry extends CellCacheEntry {
 34: 	
 35: 	/**
 36: 	 * Cells 'used' in the current evaluation of the formula corresponding to this cache entry
 37: 	 *
 38: 	 * If any of the following cells change, this cache entry needs to be cleared
 39: 	 */
 40: 	private CellCacheEntry[] _sensitiveInputCells;
 41: 
 42: 	private FormulaUsedBlankCellSet _usedBlankCellGroup;
 43: 
 44: 	public FormulaCellCacheEntry() {
 45: 		// leave fields un-set
 46: 	}
 47: 	
 48: 	public boolean isInputSensitive() {
 49: 		if (_sensitiveInputCells != null) {
 50: 			if (_sensitiveInputCells.length > 0 ) {
 51: 				return true;
 52: 			}
 53: 		}
 54: 		return _usedBlankCellGroup == null ? false : !_usedBlankCellGroup.isEmpty();
 55: 	}
 56: 
 57: 	public void setSensitiveInputCells(CellCacheEntry[] sensitiveInputCells) {
 58: 		// need to tell all cells that were previously used, but no longer are, 
 59: 		// that they are not consumed by this cell any more
 60: 		changeConsumingCells(sensitiveInputCells == null ? CellCacheEntry.EMPTY_ARRAY : sensitiveInputCells);
 61: 		_sensitiveInputCells = sensitiveInputCells;
 62: 	}
 63: 
 64: 	public void clearFormulaEntry() {
 65: 		CellCacheEntry[] usedCells = _sensitiveInputCells;
 66: 		if (usedCells != null) {
 67: 			for (int i = usedCells.length-1; i>=0; i--) {
 68: 				usedCells[i].clearConsumingCell(this);
 69: 			}
 70: 		}
 71: 		_sensitiveInputCells = null;
 72: 		clearValue();
 73: 	}
 74: 	
 75: 	private void changeConsumingCells(CellCacheEntry[] usedCells) {
 76: 
 77: 		CellCacheEntry[] prevUsedCells = _sensitiveInputCells;
 78: 		int nUsed = usedCells.length;
 79: 		for (int i = 0; i < nUsed; i++) {
 80: 			usedCells[i].addConsumingCell(this);
 81: 		}
 82: 		if (prevUsedCells == null) {
 83: 			return;
 84: 		}
 85: 		int nPrevUsed = prevUsedCells.length;
 86: 		if (nPrevUsed < 1) {
 87: 			return;
 88: 		}
 89: 		Set<CellCacheEntry> usedSet;
 90: 		if (nUsed < 1) {
 91: 			usedSet = Collections.emptySet();
 92: 		} else {
 93: 			usedSet = new HashSet<CellCacheEntry>(nUsed * 3 / 2);
 94: 			for (int i = 0; i < nUsed; i++) {
 95: 				usedSet.add(usedCells[i]);
 96: 			}
 97: 		}
 98: 		for (int i = 0; i < nPrevUsed; i++) {
 99: 			CellCacheEntry prevUsed = prevUsedCells[i];
100: 			if (!usedSet.contains(prevUsed)) {
101: 				// previously was used by cellLoc, but not anymore
102: 				prevUsed.clearConsumingCell(this);
103: 			}
104: 		}
105: 	}
106: 
107: 	public void updateFormulaResult(ValueEval result, CellCacheEntry[] sensitiveInputCells, FormulaUsedBlankCellSet usedBlankAreas) {
108: 		updateValue(result);
109: 		setSensitiveInputCells(sensitiveInputCells);
110: 		_usedBlankCellGroup = usedBlankAreas;
111: 	}
112: 
113: 	public void notifyUpdatedBlankCell(BookSheetKey bsk, int rowIndex, int columnIndex, IEvaluationListener evaluationListener) {
114: 		if (_usedBlankCellGroup != null) {
115: 			if (_usedBlankCellGroup.containsCell(bsk, rowIndex, columnIndex)) {
116: 				clearFormulaEntry();
117: 				recurseClearCachedFormulaResults(evaluationListener);
118: 			}
119: 		}
120: 	}
121: }
