 24: /**
 25:  * Title: Merged Cells Record (0x00E5)
 26:  * <br/>
 27:  * Description:  Optional record defining a square area of cells to "merged" into one cell. <br>
 28:  * @author Andrew C. Oliver (acoliver at apache dot org)
 29:  */
 30: public final class MergeCellsRecord extends StandardRecord {
 31:     public final static short sid = 0x00E5;
 32:     /** sometimes the regions array is shared with other MergedCellsRecords */ 
 33:     private CellRangeAddress[] _regions;
 34:     private final int _startIndex;
 35:     private final int _numberOfRegions;
 36: 
 37:     public MergeCellsRecord(CellRangeAddress[] regions, int startIndex, int numberOfRegions) {
 38: 		_regions = regions;
 39: 		_startIndex = startIndex;
 40: 		_numberOfRegions = numberOfRegions;
 41:     }
 42:     /**
 43:      * Constructs a MergedCellsRecord and sets its fields appropriately
 44:      * @param in the RecordInputstream to read the record from
 45:      */
 46:     public MergeCellsRecord(RecordInputStream in) {
 47:      	int nRegions = in.readUShort();
 48:     	CellRangeAddress[] cras = new CellRangeAddress[nRegions];
 49:     	for (int i = 0; i < nRegions; i++) {
 50: 			cras[i] = new CellRangeAddress(in);
 51: 		}
 52:     	_numberOfRegions = nRegions;
 53:     	_startIndex = 0;
 54:     	_regions = cras;
 55:     }
 56:     /**
 57:      * get the number of merged areas.  If this drops down to 0 you should just go
 58:      * ahead and delete the record.
 59:      * @return number of areas
 60:      */
 61:     public short getNumAreas() {
 62:         return (short)_numberOfRegions;
 63:     }
 64: 
 65:     /**
 66:      * @return MergedRegion at the given index representing the area that is Merged (r1,c1 - r2,c2)
 67:      */
 68:     public CellRangeAddress getAreaAt(int index) {
 69:         return _regions[_startIndex + index];
 70:     }
 71: 
 72:     protected int getDataSize() {
 73: 		return CellRangeAddressList.getEncodedSize(_numberOfRegions);
 74: 	}
 75: 
 76:     public short getSid() {
 77:         return sid;
 78:     }
 79: 
 80:     public void serialize(LittleEndianOutput out) {
 81:         int nItems = _numberOfRegions;
 82:         out.writeShort(nItems);
 83:         for (int i = 0; i < _numberOfRegions; i++) {
 84: 			_regions[_startIndex + i].serialize(out);
 85: 		}
 86:     }
 87: 
 88:     public String toString() {
 89:         StringBuffer retval = new StringBuffer();
 90: 
 91:         retval.append("[MERGEDCELLS]").append("\n");
 92:         retval.append("     .numregions =").append(getNumAreas()).append("\n");
 93:         for (int k = 0; k < _numberOfRegions; k++) {
 94:             CellRangeAddress r = _regions[_startIndex + k];
 95: 
 96:             retval.append("     .rowfrom =").append(r.getFirstRow()).append("\n");
 97:             retval.append("     .rowto   =").append(r.getLastRow()).append("\n");
 98:             retval.append("     .colfrom =").append(r.getFirstColumn()).append("\n");
 99:             retval.append("     .colto   =").append(r.getLastColumn()).append("\n");
100:         }
101:         retval.append("[MERGEDCELLS]").append("\n");
102:         return retval.toString();
103:     }
104: 
105:     public Object clone() {
106:     	int nRegions = _numberOfRegions;
107:     	CellRangeAddress[] clonedRegions = new CellRangeAddress[nRegions];
108: 		for (int i = 0; i < clonedRegions.length; i++) {
109: 			clonedRegions[i] = _regions[_startIndex + i].copy();
110: 		}
111:         return new MergeCellsRecord(clonedRegions, 0, nRegions);
112:     }
113: }
