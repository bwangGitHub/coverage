 33: /**
 34:  * Implementation for the MATCH() Excel function.<p/>
 35:  *
 36:  * <b>Syntax:</b><br/>
 37:  * <b>MATCH</b>(<b>lookup_value</b>, <b>lookup_array</b>, match_type)<p/>
 38:  *
 39:  * Returns a 1-based index specifying at what position in the <b>lookup_array</b> the specified
 40:  * <b>lookup_value</b> is found.<p/>
 41:  *
 42:  * Specific matching behaviour can be modified with the optional <b>match_type</b> parameter.
 43:  *
 44:  *    <table border="0" cellpadding="1" cellspacing="0" summary="match_type parameter description">
 45:  *      <tr><th>Value</th><th>Matching Behaviour</th></tr>
 46:  *      <tr><td>1</td><td>(default) find the largest value that is less than or equal to lookup_value.
 47:  *        The lookup_array must be in ascending <i>order</i>*.</td></tr>
 48:  *      <tr><td>0</td><td>find the first value that is exactly equal to lookup_value.
 49:  *        The lookup_array can be in any order.</td></tr>
 50:  *      <tr><td>-1</td><td>find the smallest value that is greater than or equal to lookup_value.
 51:  *        The lookup_array must be in descending <i>order</i>*.</td></tr>
 52:  *    </table>
 53:  *
 54:  * * Note regarding <i>order</i> - For the <b>match_type</b> cases that require the lookup_array to
 55:  *  be ordered, MATCH() can produce incorrect results if this requirement is not met.  Observed
 56:  *  behaviour in Excel is to return the lowest index value for which every item after that index
 57:  *  breaks the match rule.<br>
 58:  *  The (ascending) sort order expected by MATCH() is:<br/>
 59:  *  numbers (low to high), strings (A to Z), boolean (FALSE to TRUE)<br/>
 60:  *  MATCH() ignores all elements in the lookup_array with a different type to the lookup_value.
 61:  *  Type conversion of the lookup_array elements is never performed.
 62:  *
 63:  *
 64:  * @author Josh Micich
 65:  */

 74: 	public ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0, ValueEval arg1,
 75: 			ValueEval arg2) {
 76: 
 77: 		double match_type;
 78: 
 79: 		try {
 80: 			match_type = evaluateMatchTypeArg(arg2, srcRowIndex, srcColumnIndex);
 81: 		} catch (EvaluationException e) {
 82: 			// Excel/MATCH() seems to have slightly abnormal handling of errors with
 83: 			// the last parameter.  Errors do not propagate up.  Every error gets
 84: 			// translated into #REF!
 85: 			return ErrorEval.REF_INVALID;
 86: 		}
 87: 
 88: 		return eval(srcRowIndex, srcColumnIndex, arg0, arg1, match_type);
 89: 	}

 91: 	private static  ValueEval eval(int srcRowIndex, int srcColumnIndex, ValueEval arg0, ValueEval arg1,
 92: 			double match_type) {
 93: 		boolean matchExact = match_type == 0;
 94: 		// Note - Excel does not strictly require -1 and +1
 95: 		boolean findLargestLessThanOrEqual = match_type > 0;
 96: 
 97: 		try {
 98: 			ValueEval lookupValue = OperandResolver.getSingleValue(arg0, srcRowIndex, srcColumnIndex);
 99: 			ValueVector lookupRange = evaluateLookupRange(arg1);
100: 			int index = findIndexOfValue(lookupValue, lookupRange, matchExact, findLargestLessThanOrEqual);
101: 			return new NumberEval(index + 1); // +1 to convert to 1-based
102: 		} catch (EvaluationException e) {
103: 			return e.getErrorEval();
104: 		}
105: 	}

128: 	private static ValueVector evaluateLookupRange(ValueEval eval) throws EvaluationException {
129: 		if (eval instanceof RefEval) {
130: 			RefEval re = (RefEval) eval;
131: 			return new SingleValueVector(re.getInnerValueEval());
132: 		}
133: 		if (eval instanceof TwoDEval) {
134: 			ValueVector result = LookupUtils.createVector((TwoDEval)eval);
135: 			if (result == null) {
136: 				throw new EvaluationException(ErrorEval.NA);
137: 			}
138: 			return result;
139: 		}
140: 
141: 		// Error handling for lookup_range arg is also unusual
142: 		if(eval instanceof NumericValueEval) {
143: 			throw new EvaluationException(ErrorEval.NA);
144: 		}
145: 		if (eval instanceof StringEval) {
146: 			StringEval se = (StringEval) eval;
147: 			Double d = OperandResolver.parseDouble(se.getStringValue());
148: 			if(d == null) {
149: 				// plain string
150: 				throw new EvaluationException(ErrorEval.VALUE_INVALID);
151: 			}
152: 			// else looks like a number
153: 			throw new EvaluationException(ErrorEval.NA);
154: 		}
155: 		throw new RuntimeException("Unexpected eval type (" + eval.getClass().getName() + ")");
156: 	}

160: 	private static double evaluateMatchTypeArg(ValueEval arg, int srcCellRow, int srcCellCol)
161: 			throws EvaluationException {
162: 		ValueEval match_type = OperandResolver.getSingleValue(arg, srcCellRow, srcCellCol);
163: 
164: 		if(match_type instanceof ErrorEval) {
165: 			throw new EvaluationException((ErrorEval)match_type);
166: 		}
167: 		if(match_type instanceof NumericValueEval) {
168: 			NumericValueEval ne = (NumericValueEval) match_type;
169: 			return ne.getNumberValue();
170: 		}
171: 		if (match_type instanceof StringEval) {
172: 			StringEval se = (StringEval) match_type;
173: 			Double d = OperandResolver.parseDouble(se.getStringValue());
174: 			if(d == null) {
175: 				// plain string
176: 				throw new EvaluationException(ErrorEval.VALUE_INVALID);
177: 			}
178: 			// if the string parses as a number, it is OK
179: 			return d.doubleValue();
180: 		}
181: 		throw new RuntimeException("Unexpected match_type type (" + match_type.getClass().getName() + ")");
182: 	}

184: 	/**
185: 	 * @return zero based index
186: 	 */
187: 	private static int findIndexOfValue(ValueEval lookupValue, ValueVector lookupRange,
188: 			boolean matchExact, boolean findLargestLessThanOrEqual) throws EvaluationException {
189: 
190: 		LookupValueComparer lookupComparer = createLookupComparer(lookupValue, matchExact);
191: 
192: 		int size = lookupRange.getSize();
193: 		if(matchExact) {
194: 			for (int i = 0; i < size; i++) {
195: 				if(lookupComparer.compareTo(lookupRange.getItem(i)).isEqual()) {
196: 					return i;
197: 				}
198: 			}
199: 			throw new EvaluationException(ErrorEval.NA);
200: 		}
201: 
202: 		if(findLargestLessThanOrEqual) {
203: 			// Note - backward iteration
204: 			for (int i = size - 1; i>=0;  i--) {
205: 				CompareResult cmp = lookupComparer.compareTo(lookupRange.getItem(i));
206: 				if(cmp.isTypeMismatch()) {
207: 					continue;
208: 				}
209: 				if(!cmp.isLessThan()) {
210: 					return i;
211: 				}
212: 			}
213: 			throw new EvaluationException(ErrorEval.NA);
214: 		}
215: 
216: 		// else - find smallest greater than or equal to
217: 		// TODO - is binary search used for (match_type==+1) ?
218: 		for (int i = 0; i<size; i++) {
219: 			CompareResult cmp = lookupComparer.compareTo(lookupRange.getItem(i));
220: 			if(cmp.isEqual()) {
221: 				return i;
222: 			}
223: 			if(cmp.isGreaterThan()) {
224: 				if(i<1) {
225: 					throw new EvaluationException(ErrorEval.NA);
226: 				}
227: 				return i-1;
228: 			}
229: 		}
230: 
231: 		throw new EvaluationException(ErrorEval.NA);
232: 	}

234: 	private static LookupValueComparer createLookupComparer(ValueEval lookupValue, boolean matchExact) {
235: 		if (matchExact && lookupValue instanceof StringEval) {
236: 			String stringValue = ((StringEval) lookupValue).getStringValue();
237: 			if(isLookupValueWild(stringValue)) {
238: 				throw new RuntimeException("Wildcard lookup values '" + stringValue + "' not supported yet");
239: 			}
240: 
241: 		}
242: 		return LookupUtils.createLookupComparer(lookupValue);
243: 	}
244: 
245: 	private static boolean isLookupValueWild(String stringValue) {
246: 		if(stringValue.indexOf('?') >=0 || stringValue.indexOf('*') >=0) {
247: 			return true;
248: 		}
249: 		return false;
250: 	}
