 22: /**
 23:  * Calculates the internal rate of return.
 24:  *
 25:  * Syntax is IRR(values) or IRR(values,guess)
 26:  *
 27:  * @author Marcel May
 28:  * @author Yegor Kozlov
 29:  *
 30:  * @see <a href="http://en.wikipedia.org/wiki/Internal_rate_of_return#Numerical_solution">Wikipedia on IRR</a>
 31:  * @see <a href="http://office.microsoft.com/en-us/excel-help/irr-HP005209146.aspx">Excel IRR</a>
 32:  */
 33: public final class Irr implements Function {
 34: 
 35: 
 36:     public ValueEval evaluate(final ValueEval[] args, final int srcRowIndex, final int srcColumnIndex) {
 37:         if(args.length == 0 || args.length > 2) {
 38:             // Wrong number of arguments
 39:             return ErrorEval.VALUE_INVALID;
 40:         }
 41: 
 42:         try {
 43:             double[] values = AggregateFunction.ValueCollector.collectValues(args[0]);
 44:             double guess;
 45:             if(args.length == 2) {
 46:                 guess = NumericFunction.singleOperandEvaluate(args[1], srcRowIndex, srcColumnIndex);
 47:             } else {
 48:                 guess = 0.1d;
 49:             }
 50:             double result = irr(values, guess);
 51:             NumericFunction.checkValue(result);
 52:             return new NumberEval(result);
 53:         } catch (EvaluationException e){
 54:             return e.getErrorEval();
 55:         }
 56:     }
 57: 
 58:     /**
 59:      * Computes the internal rate of return using an estimated irr of 10 percent.
 60:      *
 61:      * @param income the income values.
 62:      * @return the irr.
 63:      */
 64:     public static double irr(double[] income) {
 65:         return irr(income, 0.1d);
 66:     }
 67: 
 68: 
 69:     /**
 70:      * Calculates IRR using the Newton-Raphson Method.
 71:      * <p>
 72:      * Starting with the guess, the method cycles through the calculation until the result
 73:      * is accurate within 0.00001 percent. If IRR can't find a result that works
 74:      * after 20 tries, the Double.NaN<> is returned.
 75:      * </p>
 76:      * <p>
 77:      *   The implementation is inspired by the NewtonSolver from the Apache Commons-Math library,
 78:      *   @see <a href="http://commons.apache.org">http://commons.apache.org</a>
 79:      * </p>
 80:      *
 81:      * @param values        the income values.
 82:      * @param guess         the initial guess of irr.
 83:      * @return the irr value. The method returns <code>Double.NaN</code>
 84:      *  if the maximum iteration count is exceeded
 85:      *
 86:      * @see <a href="http://en.wikipedia.org/wiki/Internal_rate_of_return#Numerical_solution">
 87:      *     http://en.wikipedia.org/wiki/Internal_rate_of_return#Numerical_solution</a>
 88:      * @see <a href="http://en.wikipedia.org/wiki/Newton%27s_method">
 89:      *     http://en.wikipedia.org/wiki/Newton%27s_method</a>
 90:      */
 91:     public static double irr(double[] values, double guess) {
 92:         int maxIterationCount = 20;
 93:         double absoluteAccuracy = 1E-7;
 94: 
 95:         double x0 = guess;
 96:         double x1;
 97: 
 98:         int i = 0;
 99:         while (i < maxIterationCount) {
100: 
101:             // the value of the function (NPV) and its derivate can be calculated in the same loop
102:             double fValue = 0;
103:             double fDerivative = 0;
104:             for (int k = 0; k < values.length; k++) {
105:                 fValue += values[k] / Math.pow(1.0 + x0, k);
106:                 fDerivative += -k * values[k] / Math.pow(1.0 + x0, k + 1);
107:             }
108: 
109:             // the essense of the Newton-Raphson Method
110:             x1 = x0 - fValue/fDerivative;
111: 
112:             if (Math.abs(x1 - x0) <= absoluteAccuracy) {
113:                 return x1;
114:             }
115: 
116:             x0 = x1;
117:             ++i;
118:         }
119:         // maximum number of iterations is exceeded
120:         return Double.NaN;
121:     }
122: }
