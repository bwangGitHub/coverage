 22: /**
 23:  * Manage operations dealing with bit-mapped fields.
 24:  *
 25:  * @author Marc Johnson (mjohnson at apache dot org)
 26:  * @author Andrew C. Oliver (acoliver at apache dot org)
 27:  */
 28: 
 29: public class BitField
 30: {
 31:     private final int _mask;
 32:     private final int _shift_count;
 33: 
 34:     /**
 35:      * Create a BitField instance
 36:      *
 37:      * @param mask the mask specifying which bits apply to this
 38:      *             BitField. Bits that are set in this mask are the
 39:      *             bits that this BitField operates on
 40:      */
 41: 
 42:     public BitField(final int mask)
 43:     {
 44:         _mask = mask;
 45:         int count       = 0;
 46:         int bit_pattern = mask;
 47: 
 48:         if (bit_pattern != 0)
 49:         {
 50:             while ((bit_pattern & 1) == 0)
 51:             {
 52:                 count++;
 53:                 bit_pattern >>= 1;
 54:             }
 55:         }
 56:         _shift_count = count;
 57:     }
 58: 
 59:     /**
 60:      * Obtain the value for the specified BitField, appropriately
 61:      * shifted right. Many users of a BitField will want to treat the
 62:      * specified bits as an int value, and will not want to be aware
 63:      * that the value is stored as a BitField (and so shifted left so
 64:      * many bits)
 65:      *
 66:      * @param holder the int data containing the bits we're interested
 67:      *               in
 68:      *
 69:      * @return the selected bits, shifted right appropriately
 70:      */
 71: 
 72:     public int getValue(final int holder)
 73:     {
 74:         return getRawValue(holder) >>> _shift_count;
 75:     }
 76: 
 77:     /**
 78:      * Obtain the value for the specified BitField, appropriately
 79:      * shifted right, as a short. Many users of a BitField will want
 80:      * to treat the specified bits as an int value, and will not want
 81:      * to be aware that the value is stored as a BitField (and so
 82:      * shifted left so many bits)
 83:      *
 84:      * @param holder the short data containing the bits we're
 85:      *               interested in
 86:      *
 87:      * @return the selected bits, shifted right appropriately
 88:      */
 89: 
 90:     public short getShortValue(final short holder)
 91:     {
 92:         return ( short ) getValue(holder);
 93:     }
 94: 
 95:     /**
 96:      * Obtain the value for the specified BitField, unshifted
 97:      *
 98:      * @param holder the int data containing the bits we're interested
 99:      *               in
100:      *
101:      * @return the selected bits
102:      */
103: 
104:     public int getRawValue(final int holder)
105:     {
106:         return (holder & _mask);
107:     }
108: 
109:     /**
110:      * Obtain the value for the specified BitField, unshifted
111:      *
112:      * @param holder the short data containing the bits we're
113:      *               interested in
114:      *
115:      * @return the selected bits
116:      */
117: 
118:     public short getShortRawValue(final short holder)
119:     {
120:         return ( short ) getRawValue(holder);
121:     }
122: 
123:     /**
124:      * Is the field set or not? This is most commonly used for a
125:      * single-bit field, which is often used to represent a boolean
126:      * value; the results of using it for a multi-bit field is to
127:      * determine whether *any* of its bits are set
128:      *
129:      * @param holder the int data containing the bits we're interested
130:      *               in
131:      *
132:      * @return true if any of the bits are set, else false
133:      */
134: 
135:     public boolean isSet(final int holder)
136:     {
137:         return (holder & _mask) != 0;
138:     }
139: 
140:     /**
141:      * Are all of the bits set or not? This is a stricter test than
142:      * isSet, in that all of the bits in a multi-bit set must be set
143:      * for this method to return true
144:      *
145:      * @param holder the int data containing the bits we're interested
146:      *               in
147:      *
148:      * @return true if all of the bits are set, else false
149:      */
150: 
151:     public boolean isAllSet(final int holder)
152:     {
153:         return (holder & _mask) == _mask;
154:     }
155: 
156:     /**
157:      * Replace the bits with new values.
158:      *
159:      * @param holder the int data containint the bits we're interested
160:      *               in
161:      * @param value the new value for the specified bits
162:      *
163:      * @return the value of holder with the bits from the value
164:      *         parameter replacing the old bits
165:      */
166: 
167:     public int setValue(final int holder, final int value)
168:     {
169:         return (holder & ~_mask) | ((value << _shift_count) & _mask);
170:     }
171: 
172:     /**
173:      * Replace the bits with new values.
174:      *
175:      * @param holder the short data containing the bits we're
176:      *               interested in
177:      * @param value the new value for the specified bits
178:      *
179:      * @return the value of holder with the bits from the value
180:      *         parameter replacing the old bits
181:      */
182: 
183:     public short setShortValue(final short holder, final short value)
184:     {
185:         return ( short ) setValue(holder, value);
186:     }
187: 
188:     /**
189:      * Clear the bits.
190:      *
191:      * @param holder the int data containing the bits we're interested
192:      *               in
193:      *
194:      * @return the value of holder with the specified bits cleared
195:      *         (set to 0)
196:      */
197: 
198:     public int clear(final int holder)
199:     {
200:         return holder & ~_mask;
201:     }
202: 
203:     /**
204:      * Clear the bits.
205:      *
206:      * @param holder the short data containing the bits we're
207:      *               interested in
208:      *
209:      * @return the value of holder with the specified bits cleared
210:      *         (set to 0)
211:      */
212: 
213:     public short clearShort(final short holder)
214:     {
215:         return ( short ) clear(holder);
216:     }
217: 
218:     /**
219:      * Clear the bits.
220:      *
221:      * @param holder the byte data containing the bits we're
222:      *               interested in
223:      *
224:      * @return the value of holder with the specified bits cleared
225:      *         (set to 0)
226:      */
227: 
228:     public byte clearByte(final byte holder)
229:     {
230:         return ( byte ) clear(holder);
231:     }
232: 
233:     /**
234:      * Set the bits.
235:      *
236:      * @param holder the int data containing the bits we're interested
237:      *               in
238:      *
239:      * @return the value of holder with the specified bits set to 1
240:      */
241: 
242:     public int set(final int holder)
243:     {
244:         return holder | _mask;
245:     }
246: 
247:     /**
248:      * Set the bits.
249:      *
250:      * @param holder the short data containing the bits we're
251:      *               interested in
252:      *
253:      * @return the value of holder with the specified bits set to 1
254:      */
255: 
256:     public short setShort(final short holder)
257:     {
258:         return ( short ) set(holder);
259:     }
260: 
261:     /**
262:      * Set the bits.
263:      *
264:      * @param holder the byte data containing the bits we're
265:      *               interested in
266:      *
267:      * @return the value of holder with the specified bits set to 1
268:      */
269: 
270:     public byte setByte(final byte holder)
271:     {
272:         return ( byte ) set(holder);
273:     }
274: 
275:     /**
276:      * Set a boolean BitField
277:      *
278:      * @param holder the int data containing the bits we're interested
279:      *               in
280:      * @param flag indicating whether to set or clear the bits
281:      *
282:      * @return the value of holder with the specified bits set or
283:      *         cleared
284:      */
285: 
286:     public int setBoolean(final int holder, final boolean flag)
287:     {
288:         return flag ? set(holder)
289:                     : clear(holder);
290:     }
291: 
292:     /**
293:      * Set a boolean BitField
294:      *
295:      * @param holder the short data containing the bits we're
296:      *               interested in
297:      * @param flag indicating whether to set or clear the bits
298:      *
299:      * @return the value of holder with the specified bits set or
300:      *         cleared
301:      */
302: 
303:     public short setShortBoolean(final short holder, final boolean flag)
304:     {
305:         return flag ? setShort(holder)
306:                     : clearShort(holder);
307:     }
308: 
309:     /**
310:      * Set a boolean BitField
311:      *
312:      * @param holder the byte data containing the bits we're
313:      *               interested in
314:      * @param flag indicating whether to set or clear the bits
315:      *
316:      * @return the value of holder with the specified bits set or
317:      *         cleared
318:      */
319: 
320:     public byte setByteBoolean(final byte holder, final boolean flag)
321:     {
322:         return flag ? setByte(holder)
323:                     : clearByte(holder);
324:     }
325: }   // end public class BitField
