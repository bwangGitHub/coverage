707:      *            the InputStream from which the int is to be read
708:      * @return the int (32-bit) value
709:      * @exception IOException
710:      *                will be propagated back to the caller
711:      * @exception BufferUnderrunException
712:      *                if the stream cannot provide enough bytes
713:      */
714:     public static int readInt( InputStream stream ) throws IOException,
715:             BufferUnderrunException
716:     {
717:         int ch1 = stream.read();
718:         int ch2 = stream.read();
719:         int ch3 = stream.read();
720:         int ch4 = stream.read();
721:         if ( ( ch1 | ch2 | ch3 | ch4 ) < 0 )
722:         {
723:             throw new BufferUnderrunException();
724:         }
725:         return ( ch4 << 24 ) + ( ch3 << 16 ) + ( ch2 << 8 ) + ( ch1 << 0 );
