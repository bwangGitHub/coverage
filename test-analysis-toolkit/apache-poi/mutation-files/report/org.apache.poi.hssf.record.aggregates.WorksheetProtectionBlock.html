 28: /**
 29:  * Groups the sheet protection records for a worksheet.
 30:  * <p/>
 31:  *
 32:  * See OOO excelfileformat.pdf sec 4.18.2 'Sheet Protection in a Workbook
 33:  * (BIFF5-BIFF8)'
 34:  *
 35:  * @author Josh Micich
 36:  */
 37: public final class WorksheetProtectionBlock extends RecordAggregate {
 38: 	// Every one of these component records is optional
 39: 	// (The whole WorksheetProtectionBlock may not be present)
 40: 	private ProtectRecord _protectRecord;
 41: 	private ObjectProtectRecord _objectProtectRecord;
 42: 	private ScenarioProtectRecord _scenarioProtectRecord;
 43: 	private PasswordRecord _passwordRecord;
 44: 
 45: 	/**
 46: 	 * Creates an empty WorksheetProtectionBlock
 47: 	 */
 48: 	public WorksheetProtectionBlock() {
 49: 		// all fields empty
 50: 	}
 51: 
 52: 	/**
 53: 	 * @return <code>true</code> if the specified Record sid is one belonging to
 54: 	 *         the 'Page Settings Block'.
 55: 	 */
 56: 	public static boolean isComponentRecord(int sid) {
 57: 		switch (sid) {
 58: 			case ProtectRecord.sid:
 59: 			case ObjectProtectRecord.sid:
 60: 			case ScenarioProtectRecord.sid:
 61: 			case PasswordRecord.sid:
 62: 				return true;
 63: 		}
 64: 		return false;
 65: 	}
 66: 
 67: 	private boolean readARecord(RecordStream rs) {
 68: 		switch (rs.peekNextSid()) {
 69: 			case ProtectRecord.sid:
 70: 				checkNotPresent(_protectRecord);
 71: 				_protectRecord = (ProtectRecord) rs.getNext();
 72: 				break;
 73: 			case ObjectProtectRecord.sid:
 74: 				checkNotPresent(_objectProtectRecord);
 75: 				_objectProtectRecord = (ObjectProtectRecord) rs.getNext();
 76: 				break;
 77: 			case ScenarioProtectRecord.sid:
 78: 				checkNotPresent(_scenarioProtectRecord);
 79: 				_scenarioProtectRecord = (ScenarioProtectRecord) rs.getNext();
 80: 				break;
 81: 			case PasswordRecord.sid:
 82: 				checkNotPresent(_passwordRecord);
 83: 				_passwordRecord = (PasswordRecord) rs.getNext();
 84: 				break;
 85: 			default:
 86: 				// all other record types are not part of the PageSettingsBlock
 87: 				return false;
 88: 		}
 89: 		return true;
 90: 	}
 91: 
 92: 	private void checkNotPresent(Record rec) {
 93: 		if (rec != null) {
 94: 			throw new RecordFormatException("Duplicate PageSettingsBlock record (sid=0x"
 95: 					+ Integer.toHexString(rec.getSid()) + ")");
 96: 		}
 97: 	}

 99: 	public void visitContainedRecords(RecordVisitor rv) {
100: 		// Replicates record order from Excel 2007, though this is not critical
101: 
102: 		visitIfPresent(_protectRecord, rv);
103: 		visitIfPresent(_objectProtectRecord, rv);
104: 		visitIfPresent(_scenarioProtectRecord, rv);
105: 		visitIfPresent(_passwordRecord, rv);
106: 	}
107: 
108: 	private static void visitIfPresent(Record r, RecordVisitor rv) {
109: 		if (r != null) {
110: 			rv.visitRecord(r);
111: 		}
112: 	}

122: 	/**
123: 	 * This method reads {@link WorksheetProtectionBlock} records from the supplied RecordStream
124: 	 * until the first non-WorksheetProtectionBlock record is encountered. As each record is read,
125: 	 * it is incorporated into this WorksheetProtectionBlock.
126: 	 * <p/>
127: 	 * As per the OOO documentation, the protection block records can be expected to be written
128: 	 * together (with no intervening records), but earlier versions of POI (prior to Jun 2009)
129: 	 * didn't do this.  Workbooks with sheet protection created by those earlier POI versions
130: 	 * seemed to be valid (Excel opens them OK). So PO allows continues to support reading of files
131: 	 * with non continuous worksheet protection blocks.
132: 	 *
133: 	 * <p/>
134: 	 * <b>Note</b> - when POI writes out this WorksheetProtectionBlock, the records will always be
135: 	 * written in one consolidated block (in the standard ordering) regardless of how scattered the
136: 	 * records were when they were originally read.
137: 	 */
138: 	public void addRecords(RecordStream rs) {
139: 		while (true) {
140: 			if (!readARecord(rs)) {
141: 				break;
142: 			}
143: 		}
144: 	}

162: 		if (_passwordRecord == null) {
163: 			_passwordRecord = createPassword();
164: 		}
165: 		return _passwordRecord;

168: 	/**
169: 	 * protect a spreadsheet with a password (not encrypted, just sets protect
170: 	 * flags and the password.
171: 	 *
172: 	 * @param password to set. Pass <code>null</code> to remove all protection
173: 	 * @param shouldProtectObjects are protected
174: 	 * @param shouldProtectScenarios are protected
175: 	 */
176: 	public void protectSheet(String password, boolean shouldProtectObjects,
177: 			boolean shouldProtectScenarios) {
178: 		if (password == null) {
179: 			_passwordRecord = null;
180: 			_protectRecord = null;
181: 			_objectProtectRecord = null;
182: 			_scenarioProtectRecord = null;
183: 			return;
184: 		}
185: 
186: 		ProtectRecord prec = getProtect();
187: 		PasswordRecord pass = getPassword();
188: 		prec.setProtect(true);
189: 		pass.setPassword(PasswordRecord.hashPassword(password));
190: 		if (_objectProtectRecord == null && shouldProtectObjects) {
191: 			ObjectProtectRecord rec = createObjectProtect();
192: 			rec.setProtect(true);
193: 			_objectProtectRecord = rec;
194: 		}
195: 		if (_scenarioProtectRecord == null && shouldProtectScenarios) {
196: 			ScenarioProtectRecord srec = createScenarioProtect();
197: 			srec.setProtect(true);
198: 			_scenarioProtectRecord = srec;
199: 		}
200: 	}
201:

203: 		return _protectRecord != null && _protectRecord.getProtect();
204: 	}

207: 		return _objectProtectRecord != null && _objectProtectRecord.getProtect();
208: 	}

210: 	public boolean isScenarioProtected() {
211: 		return _scenarioProtectRecord != null && _scenarioProtectRecord.getProtect();
212: 	}
