 44: 	 * Intersect this range with the specified range.
 45: 	 * 
 46: 	 * @param crB - the specified range
 47: 	 * @return code which reflects how the specified range is related to this range.<br/>
 48: 	 * Possible return codes are:	
 49: 	 * 		NO_INTERSECTION - the specified range is outside of this range;<br/> 
 50: 	 * 		OVERLAP - both ranges partially overlap;<br/>
 51: 	 * 		INSIDE - the specified range is inside of this one<br/>
 52: 	 * 		ENCLOSES - the specified range encloses (possibly exactly the same as) this range<br/>
 53: 	 */
 54: 	public static int intersect(CellRangeAddress crA, CellRangeAddress crB )
 55: 	{
 56: 		
 57: 		int firstRow = crB.getFirstRow();
 58: 		int lastRow  = crB.getLastRow();
 59: 		int firstCol = crB.getFirstColumn();
 60: 		int lastCol  = crB.getLastColumn();
 61: 		
 62: 		if
 63: 		( 
 64: 				gt(crA.getFirstRow(),lastRow) || 
 65: 				lt(crA.getLastRow(),firstRow) ||
 66: 				gt(crA.getFirstColumn(),lastCol) || 
 67: 				lt(crA.getLastColumn(),firstCol) 
 68: 		)
 69: 		{
 70: 			return NO_INTERSECTION;
 71: 		}
 72: 		else if( contains(crA, crB) )
 73: 		{
 74: 			return INSIDE;
 75: 		}
 76: 		else if( contains(crB, crA))
 77: 		{
 78: 			return ENCLOSES;
 79: 		}
 80: 		else
 81: 		{
 82: 			return OVERLAP;
 83: 		}

 89: 	 * 	<li>if a cell range is completely inside of another cell range, it gets removed from the list 
 90: 	 * 	<li>if two cells have a shared border, merge them into one bigger cell range
 91: 	 * @param cellRanges
 92: 	 * @return updated List of cell ranges

 94: 	public static CellRangeAddress[] mergeCellRanges(CellRangeAddress[] cellRanges) {
 95: 		if(cellRanges.length < 1) {
 96: 			return cellRanges;
 97: 		}
 98: 
 99:         List<CellRangeAddress> lst = new ArrayList<CellRangeAddress>();
100:         for(CellRangeAddress cr : cellRanges) lst.add(cr);

104: 	private static List mergeCellRanges(List cellRangeList)
105: 	{
106: 
107: 		while(cellRangeList.size() > 1)
108: 		{
109: 			boolean somethingGotMerged = false;
110: 			
111: 			for( int i=0; i<cellRangeList.size(); i++)
112: 			{
113: 				CellRangeAddress range1 = (CellRangeAddress)cellRangeList.get(i);
114: 				for( int j=i+1; j<cellRangeList.size(); j++)
115: 				{
116: 					CellRangeAddress range2 = (CellRangeAddress)cellRangeList.get(j);
117: 					
118: 					CellRangeAddress[] mergeResult = mergeRanges(range1, range2);
119: 					if(mergeResult == null) {
120: 						continue;
121: 					}
122: 					somethingGotMerged = true;
123: 					// overwrite range1 with first result 
124: 					cellRangeList.set(i, mergeResult[0]);
125: 					// remove range2
126: 					cellRangeList.remove(j--);
127: 					// add any extra results beyond the first
128: 					for(int k=1; k<mergeResult.length; k++) {
129: 						j++;
130: 						cellRangeList.add(j, mergeResult[k]);
131: 					}
132: 				}
133: 			}
134: 			if(!somethingGotMerged) {
135: 				break;
136: 			}
137: 		}
138: 		
139: 
140: 		return cellRangeList;
141: 	}

144: 	 * @return the new range(s) to replace the supplied ones.  <code>null</code> if no merge is possible
145: 	 */
146: 	private static CellRangeAddress[] mergeRanges(CellRangeAddress range1, CellRangeAddress range2) {
147: 		
148: 		int x = intersect(range1, range2);
149: 		switch(x)
150: 		{
151: 			case CellRangeUtil.NO_INTERSECTION: 
152: 				if(hasExactSharedBorder(range1, range2)) {
153: 					return new CellRangeAddress[] { createEnclosingCellRange(range1, range2), };
154: 				}
155: 				// else - No intersection and no shared border: do nothing 
156: 				return null;
157: 			case CellRangeUtil.OVERLAP:
158: 				return resolveRangeOverlap(range1, range2);
159: 			case CellRangeUtil.INSIDE:
160: 				// Remove range2, since it is completely inside of range1
161: 				return new CellRangeAddress[] { range1, };
162: 			case CellRangeUtil.ENCLOSES:
163: 				// range2 encloses range1, so replace it with the enclosing one
164: 				return new CellRangeAddress[] { range2, };
165: 		}
166: 		throw new RuntimeException("unexpected intersection result (" + x + ")");
167: 	}

171:

196: 	 * @param crB never a full row or full column range

200: 	private static CellRangeAddress[] sliceUp(CellRangeAddress crA, CellRangeAddress crB) {
201: 		
202: 		List temp = new ArrayList();
203: 		
204: 		// Chop up range horizontally and vertically
205: 		temp.add(crB);
206: 		if(!crA.isFullColumnRange()) {
207: 			temp = cutHorizontally(crA.getFirstRow(), temp);
208: 			temp = cutHorizontally(crA.getLastRow()+1, temp);
209: 		}
210: 		if(!crA.isFullRowRange()) {
211: 			temp = cutVertically(crA.getFirstColumn(), temp);
212: 			temp = cutVertically(crA.getLastColumn()+1, temp);
213: 		}
214: 		CellRangeAddress[] crParts = toArray(temp);
215: 
216: 		// form result array
217: 		temp.clear();
218: 		temp.add(crA);
219: 		
220: 		for (int i = 0; i < crParts.length; i++) {
221: 			CellRangeAddress crPart = crParts[i];
222: 			// only include parts that are not enclosed by this
223: 			if(intersect(crA, crPart) != ENCLOSES) {
224: 				temp.add(crPart);
225: 			}
226: 		}
227: 		return toArray(temp);
228: 	}

230: 	private static List cutHorizontally(int cutRow, List input) {
231: 		
232: 		List result = new ArrayList();
233: 		CellRangeAddress[] crs = toArray(input);
234: 		for (int i = 0; i < crs.length; i++) {
235: 			CellRangeAddress cr = crs[i];
236: 			if(cr.getFirstRow() < cutRow && cutRow < cr.getLastRow()) {
237: 				result.add(new CellRangeAddress(cr.getFirstRow(), cutRow, cr.getFirstColumn(), cr.getLastColumn()));
238: 				result.add(new CellRangeAddress(cutRow+1, cr.getLastRow(), cr.getFirstColumn(), cr.getLastColumn()));
239: 			} else {
240: 				result.add(cr);
241: 			}
242: 		}

276:    public static boolean contains(CellRangeAddress crA, CellRangeAddress crB)
277:    {
278: 		int firstRow = crB.getFirstRow();
279: 		int lastRow = crB.getLastRow();
280: 		int firstCol = crB.getFirstColumn();
281: 		int lastCol = crB.getLastColumn();
282: 		return le(crA.getFirstRow(), firstRow) && ge(crA.getLastRow(), lastRow)
283: 				&& le(crA.getFirstColumn(), firstCol) && ge(crA.getLastColumn(), lastCol);
284: 	}

286:    /**
287: 	* Check if the specified cell range has a shared border with the current range.
288: 	* 
289: 	* @return <code>true</code> if the ranges have a complete shared border (i.e.
290: 	* the two ranges together make a simple rectangular region.
291: 	*/
292:    	public static boolean hasExactSharedBorder(CellRangeAddress crA, CellRangeAddress crB) {
293: 		int oFirstRow = crB.getFirstRow();
294: 		int oLastRow  = crB.getLastRow();
295: 		int oFirstCol = crB.getFirstColumn();
296: 		int oLastCol  = crB.getLastColumn();
297: 		
298: 		if (crA.getFirstRow() > 0 && crA.getFirstRow()-1 == oLastRow || 
299: 			oFirstRow > 0 && oFirstRow-1 == crA.getLastRow()) {
300: 			// ranges have a horizontal border in common
301: 			// make sure columns are identical:
302: 			return crA.getFirstColumn() == oFirstCol && crA.getLastColumn() == oLastCol;
303: 		}
304: 
305: 		if (crA.getFirstColumn()>0 && crA.getFirstColumn() - 1 == oLastCol ||
306: 			oFirstCol>0 && crA.getLastColumn() == oFirstCol -1) {
307: 			// ranges have a vertical border in common
308: 			// make sure rows are identical:
309: 			return crA.getFirstRow() == oFirstRow && crA.getLastRow() == oLastRow;
310: 		}
311: 		return false;
312:    	}
