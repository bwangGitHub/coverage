 30: /**
 31:  * The base abstract record from which all escher records are defined.  Subclasses will need
 32:  * to define methods for serialization/deserialization and for determining the record size.
 33:  *
 34:  * @author Glen Stampoultzis
 35:  */
 36: public abstract class EscherRecord {
 37:     private static BitField fInstance = BitFieldFactory.getInstance(0xfff0);
 38:     private static BitField fVersion = BitFieldFactory.getInstance(0x000f);
 39: 
 40:     private short _options;
 41:     private short _recordId;
 42: 
 43:     /**
 44:      * Create a new instance
 45:      */
 46:     public EscherRecord() {
 47:         // fields uninitialised
 48:     }
 49: 
 50:     /**
 51:      * Delegates to fillFields(byte[], int, EscherRecordFactory)
 52:      *
 53:      * @see #fillFields(byte[], int, org.apache.poi.ddf.EscherRecordFactory)
 54:      */
 55:     protected int fillFields( byte[] data, EscherRecordFactory f )
 56:     {
 57:         return fillFields( data, 0, f );
 58:     }
 59: 
 60:     /**
 61:      * The contract of this method is to deserialize an escher record including
 62:      * it's children.
 63:      *
 64:      * @param data      The byte array containing the serialized escher
 65:      *                  records.
 66:      * @param offset    The offset into the byte array.
 67:      * @param recordFactory     A factory for creating new escher records.
 68:      * @return          The number of bytes written.
 69:      */
 70:     public abstract int fillFields( byte[] data, int offset, EscherRecordFactory recordFactory );
 71: 
 72:     /**
 73:      * Reads the 8 byte header information and populates the <code>options</code>
 74:      * and <code>recordId</code> records.
 75:      *
 76:      * @param data      the byte array to read from
 77:      * @param offset    the offset to start reading from
 78:      * @return          the number of bytes remaining in this record.  This
 79:      *                  may include the children if this is a container.
 80:      */
 81:     protected int readHeader( byte[] data, int offset )
 82:     {
 83:         _options = LittleEndian.getShort( data, offset );
 84:         _recordId = LittleEndian.getShort( data, offset + 2 );
 85:         int remainingBytes = LittleEndian.getInt( data, offset + 4 );
 86:         return remainingBytes;
 87:     }
 88: 
 89:     /**
 90:      * Read the options field from header and return instance part of it.
 91:      * @param data      the byte array to read from
 92:      * @param offset    the offset to start reading from
 93:      * @return          value of instance part of options field
 94:      */
 95:     protected static short readInstance( byte data[], int offset )
 96:     {
 97:         final short options = LittleEndian.getShort( data, offset );
 98:         return fInstance.getShortValue( options );
 99:     }
100: 
101:     /**
102:      * Determine whether this is a container record by inspecting the option
103:      * field.
104:      * @return  true is this is a container field.
105:      */
106:     public boolean isContainerRecord() {
107:         return getVersion() == (short)0x000f;
108:     }
109: 
110:     /**
111:      * <p
112:      * Note that <code>options</code> is an internal field. Use {@link #setInstance(short)} ()} and
113:      *             {@link #setVersion(short)} ()} to set the actual fields.
114:      * </p>
115:      * @return The options field for this record. All records have one.
116:      */
117:     @Internal
118:     public short getOptions()
119:     {
120:         return _options;
121:     }
122: 
123:     /**
124:      * Set the options this this record.  Container records should have the
125:      * last nibble set to 0xF.
126:      *
127:      * <p
128:      * Note that <code>options</code> is an internal field. Use {@link #getInstance()} and
129:      *             {@link #getVersion()} to access actual fields.
130:      * </p>
131:      */
132:     @Internal
133:     public void setOptions( short options ) {
134:         // call to handle correct/incorrect values
135:         setVersion( fVersion.getShortValue( options ) );
136:         setInstance( fInstance.getShortValue( options ) );
137:         _options = options;
138:     }
139: 
140:     /**
141:      * Serializes to a new byte array.  This is done by delegating to
142:      * serialize(int, byte[]);
143:      *
144:      * @return  the serialized record.
145:      * @see #serialize(int, byte[])
146:      */
147:     public byte[] serialize()
148:     {
149:         byte[] retval = new byte[getRecordSize()];
150: 
151:         serialize( 0, retval );
152:         return retval;
153:     }
154: 
155:     /**
156:      * Serializes to an existing byte array without serialization listener.
157:      * This is done by delegating to serialize(int, byte[], EscherSerializationListener).
158:      *
159:      * @param offset    the offset within the data byte array.
160:      * @param data      the data array to serialize to.
161:      * @return          The number of bytes written.
162:      *
163:      * @see #serialize(int, byte[], org.apache.poi.ddf.EscherSerializationListener)
164:      */
165:     public int serialize( int offset, byte[] data)
166:     {
167:         return serialize( offset, data, new NullEscherSerializationListener() );
168:     }
169: 
170:     /**
171:      * Serializes the record to an existing byte array.
172:      *
173:      * @param offset    the offset within the byte array
174:      * @param data      the data array to serialize to
175:      * @param listener  a listener for begin and end serialization events.  This
176:      *                  is useful because the serialization is
177:      *                  hierarchical/recursive and sometimes you need to be able
178:      *                  break into that.
179:      * @return the number of bytes written.
180:      */
181:     public abstract int serialize( int offset, byte[] data, EscherSerializationListener listener );
182: 
183:     /**
184:      * Subclasses should effeciently return the number of bytes required to
185:      * serialize the record.
186:      *
187:      * @return  number of bytes
188:      */
189:     abstract public int getRecordSize();
190: 
191:     /**
192:      * Return the current record id.
193:      *
194:      * @return  The 16 bit record id.
195:      */
196:     public short getRecordId() {
197:         return _recordId;
198:     }
199: 
200:     /**
201:      * Sets the record id for this record.
202:      */
203:     public void setRecordId( short recordId ) {
204:         _recordId = recordId;
205:     }
206: 
207:     /**
208:      * @return  Returns the children of this record.  By default this will
209:      *          be an empty list.  EscherCotainerRecord is the only record
210:      *          that may contain children.
211:      *
212:      * @see EscherContainerRecord
213:      */
214:     public List<EscherRecord> getChildRecords() { return Collections.emptyList(); }
215: 
216:     /**
217:      * Sets the child records for this record.  By default this will throw
218:      * an exception as only EscherContainerRecords may have children.
219:      *
220:      * @param childRecords  Not used in base implementation.
221:      */
222:     public void setChildRecords(List<EscherRecord> childRecords) {
223:         throw new UnsupportedOperationException("This record does not support child records.");
224:     }

227:      * Escher records may need to be clonable in the future.
228:      */
229:     public Object clone()

231:         throw new RuntimeException( "The class " + getClass().getName() + " needs to define a clone method" );
232:     }
233:

235:      * Returns the indexed child record.
236:      */
237:     public EscherRecord getChild( int index ) {

241:     /**
242:      * The display methods allows escher variables to print the record names

244:      *
245:      * @param w         The print writer to output to.
246:      * @param indent    The current indent level.
247:      */
248:     public void display(PrintWriter w, int indent)
249:     {
250:         for (int i = 0; i < indent * 4; i++) w.print(' ');

266:         return fInstance.getShortValue( _options );
