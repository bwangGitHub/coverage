488:     /**
489:      * <p>Returns the variant type name associated with a variant type
490:      * number.</p>
491:      *
492:      * @param variantType The variant type number
493:      * @return The variant type name or the string "unknown variant type"
494:      */
495:     public static String getVariantName(final long variantType)
496:     {
497:         final String name = (String) numberToName.get(Long.valueOf(variantType));
498:         return name != null ? name : "unknown variant type";
499:     }
500: 
501:     /**
502:      * <p>Returns a variant type's length.</p>
503:      *
504:      * @param variantType The variant type number
505:      * @return The length of the variant type's data in bytes. If the length is
506:      * variable, i.e. the length of a string, -1 is returned. If HPSF does not
507:      * know the length, -2 is returned. The latter usually indicates an
508:      * unsupported variant type.
509:      */
510:     public static int getVariantLength(final long variantType)
511:     {
512:         final Long key = Long.valueOf((int) variantType);
513:         final Long length = (Long) numberToLength.get(key);
514:         if (length == null)
515:             return -2;
516:         return length.intValue();
517:     }
