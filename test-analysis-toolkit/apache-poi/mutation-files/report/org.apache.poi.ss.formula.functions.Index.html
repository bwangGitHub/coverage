 30:  * Implementation for the Excel function INDEX
 31:  * <p>
 32:  *
 33:  * Syntax : <br/>
 34:  *  INDEX ( reference, row_num[, column_num [, area_num]])</br>
 35:  *  INDEX ( array, row_num[, column_num])
 36:  *    <table border="0" cellpadding="1" cellspacing="0" summary="Parameter descriptions">
 37:  *      <tr><th>reference</th><td>typically an area reference, possibly a union of areas</td></tr>
 38:  *      <tr><th>array</th><td>a literal array value (currently not supported)</td></tr>
 39:  *      <tr><th>row_num</th><td>selects the row within the array or area reference</td></tr>
 40:  *      <tr><th>column_num</th><td>selects column within the array or area reference. default is 1</td></tr>

 45:  * @author Josh Micich

 49: 	public ValueEval evaluate(int srcRowIndex, int srcColumnIndex, ValueEval arg0, ValueEval arg1) {
 50: 		TwoDEval reference = convertFirstArg(arg0);
 51: 
 52: 		int columnIx = 0;
 53: 		try {
 54: 			int rowIx = resolveIndexArg(arg1, srcRowIndex, srcColumnIndex);
 55: 
 56: 			if (!reference.isColumn()) {
 57: 				if (!reference.isRow()) {
 58: 					// always an error with 2-D area refs
 59: 					// Note - the type of error changes if the pRowArg is negative
 60: 					return ErrorEval.REF_INVALID;
 61: 				}
 62: 				// When the two-arg version of INDEX() has been invoked and the reference
 63: 				// is a single column ref, the row arg seems to get used as the column index
 64: 				columnIx = rowIx;
 65: 				rowIx = 0;
 66: 			}
 67: 
 68: 			return getValueFromArea(reference, rowIx, columnIx);
 69: 		} catch (EvaluationException e) {
 70: 			return e.getErrorEval();

 96: 		ValueEval firstArg = arg0;
 97: 		if (firstArg instanceof RefEval) {
 98: 			// convert to area ref for simpler code in getValueFromArea()
 99: 			return ((RefEval)firstArg).offset(0, 0, 0, 0);

123: 	private static ValueEval getValueFromArea(TwoDEval ae, int pRowIx, int pColumnIx)
124: 			throws EvaluationException {
125: 		assert pRowIx >= 0;
126: 		assert pColumnIx >= 0;
127: 
128: 		TwoDEval result = ae;
129: 
130: 		if (pRowIx != 0) {
131: 			// Slightly irregular logic for bounds checking errors
132: 			if (pRowIx > ae.getHeight()) {
133: 				// high bounds check fail gives #REF! if arg was explicitly passed
134: 				throw new EvaluationException(ErrorEval.REF_INVALID);
135: 			}
136: 			result = result.getRow(pRowIx-1);
137: 		}

142: 				// high bounds check fail gives #REF! if arg was explicitly passed
143: 				throw new EvaluationException(ErrorEval.REF_INVALID);
144: 			}
145: 			result = result.getColumn(pColumnIx-1);
146: 		}
147: 		return result;

153: 	 * @return the resolved 1-based index. Zero if the arg was missing or blank
154: 	 * @throws EvaluationException if the arg is an error value evaluates to a negative numeric value

156: 	private static int resolveIndexArg(ValueEval arg, int srcCellRow, int srcCellCol) throws EvaluationException {
157: 
158: 		ValueEval ev = OperandResolver.getSingleValue(arg, srcCellRow, srcCellCol);
159: 		if (ev == MissingArgEval.instance) {
160: 			return 0;
161: 		}
162: 		if (ev == BlankEval.instance) {
163: 			return 0;
164: 		}
165: 		int result = OperandResolver.coerceValueToInt(ev);
166: 		if (result < 0) {
167: 			throw new EvaluationException(ErrorEval.VALUE_INVALID);
168: 		}
169: 		return result;
