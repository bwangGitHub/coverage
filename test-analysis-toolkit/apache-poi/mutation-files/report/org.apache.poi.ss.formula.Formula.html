135: 	public static Formula create(Ptg[] ptgs) {
136: 		if (ptgs == null || ptgs.length < 1) {
137: 			return EMPTY;
138: 		}
139: 		int totalSize = Ptg.getEncodedSize(ptgs);
140: 		byte[] encodedData = new byte[totalSize];
141: 		Ptg.serializePtgs(ptgs, encodedData, 0);
142: 		int encodedTokenLen = Ptg.getEncodedSizeWithoutArrayData(ptgs);
143: 		return new Formula(encodedData, encodedTokenLen);
144: 	}

175: 		byte[] data = _byteEncoding;
176: 		if (data.length != 5) {
177: 			// tExp and tTbl are always 5 bytes long, and the only ptg in the formula
178: 			return null;
179: 		}
180: 		switch (data[0]) {
181: 			case ExpPtg.sid:
182: 				break;
183: 			case TblPtg.sid:

187: 		}
188: 		int firstRow = LittleEndian.getUShort(data, 1);
189: 		int firstColumn = LittleEndian.getUShort(data, 3);
190: 		return new CellReference(firstRow, firstColumn);
