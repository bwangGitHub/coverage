 84: 			Ptg[] formula1, Ptg[] formula2,

111: 		_errorTitle = readUnicodeString(in);
112: 		_promptText = readUnicodeString(in);
113: 		_errorText = readUnicodeString(in);
114: 
115: 		int field_size_first_formula = in.readUShort();
116: 		_not_used_1 = in.readShort();
117:

207: 	public String toString() {
208: 		StringBuffer sb = new StringBuffer();
209: 		sb.append("[DV]\n");
210: 		sb.append(" options=").append(Integer.toHexString(_option_flags));
211: 		sb.append(" title-prompt=").append(formatTextTitle(_promptTitle));
212: 		sb.append(" title-error=").append(formatTextTitle(_errorTitle));
213: 		sb.append(" text-prompt=").append(formatTextTitle(_promptText));
214: 		sb.append(" text-error=").append(formatTextTitle(_errorText));
215: 		sb.append("\n");
216: 		appendFormula(sb, "Formula 1:",  _formula1);
217: 		appendFormula(sb, "Formula 2:",  _formula2);
218: 		sb.append("Regions: ");
219: 		int nRegions = _regions.countRanges();
220: 		for(int i=0; i<nRegions; i++) {
221: 			if (i>0) {
222: 				sb.append(", ");
223: 			}
224: 			CellRangeAddress addr = _regions.getCellRangeAddress(i);
225: 			sb.append('(').append(addr.getFirstRow()).append(',').append(addr.getLastRow());
226: 			sb.append(',').append(addr.getFirstColumn()).append(',').append(addr.getLastColumn()).append(')');
227: 		}
228: 		sb.append("\n");
229: 		sb.append("[/DV]");
230: 
231: 		return sb.toString();
232: 	}

234: 	private static String formatTextTitle(UnicodeString us) {
235: 		String str = us.getString();
236: 		if (str.length() == 1 && str.charAt(0) == '\0') {
237: 			return "'\\0'";
238: 		}
239: 		return str;
240: 	}

242: 	private static void appendFormula(StringBuffer sb, String label, Formula f) {
243: 		sb.append(label);
244: 		
245: 		if (f == null) {
246: 			sb.append("<empty>\n");
247: 			return;
248: 		}
249: 		Ptg[] ptgs = f.getTokens();
250: 		sb.append('\n');
251: 		for (int i = 0; i < ptgs.length; i++) {
252: 			sb.append('\t').append(ptgs[i].toString()).append('\n');
253: 		}

275: 	/**
276: 	 * When entered via the UI, Excel translates empty string into "\0"
277: 	 * While it is possible to encode the title/text as empty string (Excel doesn't exactly crash),
278: 	 * the resulting tool-tip text / message box looks wrong.  It is best to do the same as the 
279: 	 * Excel UI and encode 'not present' as "\0". 
280: 	 */
281: 	private static UnicodeString resolveTitleText(String str) {
282: 		if (str == null || str.length() < 1) {
283: 			return NULL_TEXT_STRING;
284: 		}
285: 		return new UnicodeString(str);
286: 	}

292: 	private static void serializeUnicodeString(UnicodeString us, LittleEndianOutput out) {
293: 		StringUtil.writeUnicodeString(out, us.getString());

295: 	private static int getUnicodeStringSize(UnicodeString us) {
296: 		String str = us.getString();
297: 		return 3 + str.length() * (StringUtil.hasMultibyte(str) ? 2 : 1);
298: 	}
