305:      * returns the remainder resulting from operation:
306:      * n / d.
307:      * <br/> The result has the sign of the divisor.
308:      * <br/> Examples:
309:      * <ul>
310:      * <li>mod(3.4, 2) = 1.4</li>
311:      * <li>mod(-3.4, 2) = 0.6</li>
312:      * <li>mod(-3.4, -2) = -1.4</li>
313:      * <li>mod(3.4, -2) = -0.6</li>
314:      * </ul>
315:      * If d == 0, result is NaN
316:      * @param n
317:      * @param d
318:      */
319:     public static double mod(double n, double d) {
320:         double result = 0;
321: 
322:         if (d == 0) {
323:             result = Double.NaN;
324:         }
325:         else if (sign(n) == sign(d)) {
326:             result = n % d;
327:         }
328:         else {
329:             result = ((n % d) + d) % d;
330:         }
