 59:  * This class creates <tt>OperationEval</tt> instances to help evaluate <tt>OperationPtg</tt>
 60:  * formula tokens.

 62:  * @author Josh Micich

 96: 	private static void put(Map<OperationPtg, Function> m, OperationPtg ptgKey,
 97: 			Function instance) {
 98: 		// make sure ptg has single private constructor because map lookups assume singleton keys
 99: 		Constructor[] cc = ptgKey.getClass().getDeclaredConstructors();
100: 		if (cc.length > 1 || !Modifier.isPrivate(cc[0].getModifiers())) {
101: 			throw new RuntimeException("Failed to verify instance ("
102: 					+ ptgKey.getClass().getName() + ") is a singleton.");
103: 		}
104: 		m.put(ptgKey, instance);
105: 	}
106: 
107: 	/**
108: 	 * returns the OperationEval concrete impl instance corresponding
109: 	 * to the supplied operationPtg
110: 	 */
111: 	public static ValueEval evaluate(OperationPtg ptg, ValueEval[] args,
112: 			OperationEvaluationContext ec) {
113: 		if(ptg == null) {
114: 			throw new IllegalArgumentException("ptg must not be null");
115: 		}
116: 		Function result = _instancesByPtgClass.get(ptg);
117: 
118: 		if (result != null) {
119: 			return  result.evaluate(args, ec.getRowIndex(), (short) ec.getColumnIndex());
120: 		}
121: 
122: 		if (ptg instanceof AbstractFunctionPtg) {
123: 			AbstractFunctionPtg fptg = (AbstractFunctionPtg)ptg;
124: 			int functionIndex = fptg.getFunctionIndex();
125: 			switch (functionIndex) {
126: 				case FunctionMetadataRegistry.FUNCTION_INDEX_INDIRECT:
127: 					return Indirect.instance.evaluate(args, ec);
128: 				case FunctionMetadataRegistry.FUNCTION_INDEX_EXTERNAL:
129: 					return UserDefinedFunction.instance.evaluate(args, ec);
130: 			}
131: 
132: 			return FunctionEval.getBasicFunction(functionIndex).evaluate(args, ec.getRowIndex(), (short) ec.getColumnIndex());
133: 		}
134: 		throw new RuntimeException("Unexpected operation ptg class (" + ptg.getClass().getName() + ")");
135: 	}
