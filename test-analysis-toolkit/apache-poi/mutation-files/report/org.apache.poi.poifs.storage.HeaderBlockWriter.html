 30: /**
 31:  * The block containing the archive header
 32:  *
 33:  * @author Marc Johnson (mjohnson at apache dot org)
 34:  */
 35: public class HeaderBlockWriter implements HeaderBlockConstants, BlockWritable
 36: {
 37:    private final HeaderBlock _header_block;
 38: 
 39:     /**
 40:      * Create a single instance initialized with default values
 41:      */
 42:     public HeaderBlockWriter(POIFSBigBlockSize bigBlockSize)
 43:     {
 44:        _header_block = new HeaderBlock(bigBlockSize);
 45:     }
 46: 
 47:     /**
 48:      * Create a single instance initialized with the specified 
 49:      *  existing values
 50:      */
 51:     public HeaderBlockWriter(HeaderBlock headerBlock)
 52:     {
 53:        _header_block = headerBlock;
 54:     }
 55: 
 56:     /**
 57:      * Set BAT block parameters. Assumes that all BAT blocks are
 58:      * contiguous. Will construct XBAT blocks if necessary and return
 59:      * the array of newly constructed XBAT blocks.
 60:      *
 61:      * @param blockCount count of BAT blocks
 62:      * @param startBlock index of first BAT block
 63:      *
 64:      * @return array of XBAT blocks; may be zero length, will not be
 65:      *         null
 66:      */
 67: 
 68:     public BATBlock [] setBATBlocks(final int blockCount,
 69:                                     final int startBlock)
 70:     {
 71:         BATBlock[] rvalue;
 72:         POIFSBigBlockSize bigBlockSize = _header_block.getBigBlockSize();
 73: 
 74:         _header_block.setBATCount(blockCount);
 75: 
 76:         // Set the BAT locations
 77:         int limit  = Math.min(blockCount, _max_bats_in_header);
 78:         int[] bat_blocks = new int[limit];
 79:         for (int j = 0; j < limit; j++) {
 80:            bat_blocks[j] = startBlock + j;
 81:         }
 82:         _header_block.setBATArray(bat_blocks);
 83:         
 84:         // Now do the XBATs
 85:         if (blockCount > _max_bats_in_header)
 86:         {
 87:             int   excess_blocks      = blockCount - _max_bats_in_header;
 88:             int[] excess_block_array = new int[ excess_blocks ];
 89: 
 90:             for (int j = 0; j < excess_blocks; j++)
 91:             {
 92:                 excess_block_array[ j ] = startBlock + j
 93:                                           + _max_bats_in_header;
 94:             }
 95:             rvalue = BATBlock.createXBATBlocks(bigBlockSize, excess_block_array,
 96:                                                startBlock + blockCount);
 97:             _header_block.setXBATStart(startBlock + blockCount);
 98:         }
 99:         else
100:         {
101:             rvalue = BATBlock.createXBATBlocks(bigBlockSize, new int[ 0 ], 0);
102:             _header_block.setXBATStart(POIFSConstants.END_OF_CHAIN);
103:         }
104:         _header_block.setXBATCount(rvalue.length);
105:         return rvalue;
106:     }
107: 
108:     /**
109:      * Set start of Property Table
110:      *
111:      * @param startBlock the index of the first block of the Property
112:      *                   Table
113:      */
114:     public void setPropertyStart(final int startBlock)
115:     {
116:        _header_block.setPropertyStart(startBlock);
117:     }
118: 
119:     /**
120:      * Set start of small block allocation table
121:      *
122:      * @param startBlock the index of the first big block of the small
123:      *                   block allocation table
124:      */
125:     public void setSBATStart(final int startBlock)
126:     {
127:         _header_block.setSBATStart(startBlock);
128:     }
129: 
130:     /**
131:      * Set count of SBAT blocks
132:      *
133:      * @param count the number of SBAT blocks
134:      */
135:     public void setSBATBlockCount(final int count)
136:     {
137:        _header_block.setSBATBlockCount(count);
138:     }
139: 
140:     /**
141:      * For a given number of BAT blocks, calculate how many XBAT
142:      * blocks will be needed
143:      *
144:      * @param blockCount number of BAT blocks
145:      *
146:      * @return number of XBAT blocks needed
147:      */
148: 
149:     static int calculateXBATStorageRequirements(POIFSBigBlockSize bigBlockSize, final int blockCount)
150:     {
151:         return (blockCount > _max_bats_in_header)
152:                ? BATBlock.calculateXBATStorageRequirements(
153:                      bigBlockSize, blockCount - _max_bats_in_header)
154:                : 0;
155:     }
156: 
157:     /* ********** START extension of BigBlock ********** */
158: 
159:     /**
160:      * Write the block's data to an OutputStream
161:      *
162:      * @param stream the OutputStream to which the stored data should
163:      *               be written
164:      *
165:      * @exception IOException on problems writing to the specified
166:      *            stream
167:      */
168:     public void writeBlocks(final OutputStream stream)
169:         throws IOException
170:     {
171:         _header_block.writeData(stream);
172:     }
173:     
174:     /**
175:      * Write the block's data to an existing block
176:      *
177:      * @param block the ByteBuffer of the block to which the 
178:      *               stored data should be written
179:      *
180:      * @exception IOException on problems writing to the block
181:      */
182:     public void writeBlock(ByteBuffer block)
183:         throws IOException
184:     {
185:        ByteArrayOutputStream baos = new ByteArrayOutputStream(
186:              _header_block.getBigBlockSize().getBigBlockSize()
187:        );
188:        _header_block.writeData(baos);
189:        
190:        block.put(baos.toByteArray());
191:     }
192: 
193:     /* **********  END  extension of BigBlock ********** */
194: }   // end public class HeaderBlockWriter
