 43: 			ValueEval arg2) {
 44: 		double result;
 45: 		try {
 46: 			result = evaluate(evalArg(arg0, srcRowIndex, srcColumnIndex), evalArg(arg1, srcRowIndex, srcColumnIndex), evalArg(arg2, srcRowIndex, srcColumnIndex));
 47: 		} catch (EvaluationException e) {
 48: 			return e.getErrorEval();
 49: 		}
 50: 		return new NumberEval(result);
 51: 	}
 52: 	private static int evalArg(ValueEval arg, int srcRowIndex, int srcColumnIndex) throws EvaluationException {
 53: 		if (arg == MissingArgEval.instance) {
 54: 			return 0;
 55: 		}
 56: 		ValueEval ev = OperandResolver.getSingleValue(arg, srcRowIndex, srcColumnIndex);
 57: 		// Excel silently truncates double values to integers
 58: 		return OperandResolver.coerceValueToInt(ev);
 59: 	}
 60: 	/**
 61: 	 * Converts the supplied hours, minutes and seconds to an Excel time value.
 62: 	 *
 63: 	 *
 64: 	 * @param ds array of 3 doubles containing hours, minutes and seconds.
 65: 	 * Non-integer inputs are truncated to an integer before further calculation
 66: 	 * of the time value.
 67: 	 * @return An Excel representation of a time of day.
 68: 	 * If the time value represents more than a day, the days are removed from
 69: 	 * the result, leaving only the time of day component.
 70: 	 * @throws org.apache.poi.ss.formula.eval.EvaluationException
 71: 	 * If any of the arguments are greater than 32767 or the hours
 72: 	 * minutes and seconds when combined form a time value less than 0, the function
 73: 	 * evaluates to an error.
 74: 	 */
 75: 	private static double evaluate(int hours, int minutes, int seconds) throws EvaluationException {
 76: 
 77: 		if (hours > 32767 || minutes > 32767 || seconds > 32767) {
 78: 			throw new EvaluationException(ErrorEval.VALUE_INVALID);
 79: 		}
 80: 		int totalSeconds = hours * SECONDS_PER_HOUR + minutes * SECONDS_PER_MINUTE + seconds;
 81: 
 82: 		if (totalSeconds < 0) {
 83: 			throw new EvaluationException(ErrorEval.VALUE_INVALID);
 84: 		}
 85: 		return (totalSeconds % SECONDS_PER_DAY) / (double)SECONDS_PER_DAY;
 86: 	}
