 25: /**
 26:  * DConRef records specify a range in a workbook (internal or external) that serves as a data source
 27:  * for pivot tables or data consolidation.
 28:  *
 29:  * Represents a <code>DConRef</code> Structure
 30:  * <a href="http://msdn.microsoft.com/en-us/library/dd923854(office.12).aspx">[MS-XLS s.
 31:  * 2.4.86]</a>, and the contained <code>DConFile</code> structure
 32:  * <a href="http://msdn.microsoft.com/en-us/library/dd950157(office.12).aspx">
 33:  * [MS-XLS s. 2.5.69]</a>. This in turn contains a <code>XLUnicodeStringNoCch</code>
 34:  * <a href="http://msdn.microsoft.com/en-us/library/dd910585(office.12).aspx">
 35:  * [MS-XLS s. 2.5.296]</a>.
 36:  *
 37:  * <pre>
 38:  *         _______________________________
 39:  *        |          DConRef              |
 40:  *(bytes) +-+-+-+-+-+-+-+-+-+-+...+-+-+-+-+
 41:  *        |    ref    |cch|  stFile   | un|
 42:  *        +-+-+-+-+-+-+-+-+-+-+...+-+-+-+-+
 43:  *                              |
 44:  *                     _________|_____________________
 45:  *                    |DConFile / XLUnicodeStringNoCch|
 46:  *                    +-+-+-+-+-+-+-+-+-+-+-+...+-+-+-+
 47:  *             (bits) |h|   reserved  |      rgb      |
 48:  *                    +-+-+-+-+-+-+-+-+-+-+-+...+-+-+-+
 49:  * </pre>
 50:  * Where
 51:  * <ul>
 52:  * <li><code>DConFile.h = 0x00</code> if the characters in<code>rgb</code> are single byte, and
 53:  * <code>DConFile.h = 0x01</code> if they are double byte. <br/>
 54:  * If they are double byte, then<br/>
 55:  * <ul>
 56:  * <li> If it exists, the length of <code>DConRef.un = 2</code>. Otherwise it is 1.
 57:  * <li> The length of <code>DConFile.rgb = (2 * DConRef.cch)</code>. Otherwise it is equal to
 58:  * <code>DConRef.cch</code>.
 59:  * </ul>
 60:  * <li><code>DConRef.rgb</code> starts with <code>0x01</code> if it is an external reference,
 61:  * and with <code>0x02</code> if it is a self-reference.
 62:  * </ul>
 63:  *
 64:  * At the moment this class is read-only.
 65:  *
 66:  * @author Niklas Rehfeld
 67:  */
 68: public class DConRefRecord extends StandardRecord
 69: {
 70: 
 71:     /**
 72:      * The id of the record type,
 73:      * <code>sid = {@value}</code>
 74:      */
 75:     public static final short sid = 0x0051;
 76:     /**
 77:      * A RefU structure specifying the range of cells if this record is part of an SXTBL.
 78:      * <a href="http://msdn.microsoft.com/en-us/library/dd920420(office.12).aspx">
 79:      * [MS XLS s.2.5.211]</a>
 80:      */
 81:     private int firstRow, lastRow, firstCol, lastCol;
 82:     /**
 83:      * the number of chars in the link
 84:      */
 85:     private int charCount;
 86:     /**
 87:      * the type of characters (single or double byte)
 88:      */
 89:     private int charType;
 90:     /**
 91:      * The link's path string. This is the <code>rgb</code> field of a
 92:      * <code>XLUnicodeStringNoCch</code>. Therefore it will contain at least one leading special
 93:      * character (0x01 or 0x02) and probably other ones.<p/>
 94:      * @see <A href="http://msdn.microsoft.com/en-us/library/dd923491(office.12).aspx">
 95:      * DConFile [MS-XLS s. 2.5.77]</A> and
 96:      * <A href="http://msdn.microsoft.com/en-us/library/dd950157(office.12).aspx">
 97:      * VirtualPath [MS-XLS s. 2.5.69]</a>
 98:      * <p/>
 99:      */
100:     private byte[] path;
101:     /**
102:      * unused bits at the end, must be set to 0.
103:      */
104:     private byte[] _unused;
105: 
106:     /**
107:      * Read constructor.
108:      *
109:      * @param data byte array containing a DConRef Record, including the header.
110:      */
111:     public DConRefRecord(byte[] data)
112:     {
113:         int offset = 0;
114:         if (!(LittleEndian.getShort(data, offset) == DConRefRecord.sid))
115:             throw new RecordFormatException("incompatible sid.");
116:         offset += LittleEndian.SHORT_SIZE;
117: 
118:         //length = LittleEndian.getShort(data, offset);
119:         offset += LittleEndian.SHORT_SIZE;
120: 
121:         firstRow = LittleEndian.getUShort(data, offset);
122:         offset += LittleEndian.SHORT_SIZE;
123:         lastRow = LittleEndian.getUShort(data, offset);
124:         offset += LittleEndian.SHORT_SIZE;
125:         firstCol = LittleEndian.getUByte(data, offset);
126:         offset += LittleEndian.BYTE_SIZE;
127:         lastCol = LittleEndian.getUByte(data, offset);
128:         offset += LittleEndian.BYTE_SIZE;
129:         charCount = LittleEndian.getUShort(data, offset);
130:         offset += LittleEndian.SHORT_SIZE;
131:         if (charCount < 2)
132:             throw new org.apache.poi.hssf.record.RecordFormatException(
133:                     "Character count must be >= 2");
134: 
135:         charType = LittleEndian.getUByte(data, offset);
136:         offset += LittleEndian.BYTE_SIZE; //7 bits reserved + 1 bit type
137: 
138:         /*
139:          * bytelength is the length of the string in bytes, which depends on whether the string is
140:          * made of single- or double-byte chars. This is given by charType, which equals 0 if
141:          * single-byte, 1 if double-byte.
142:          */
143:         int byteLength = charCount * ((charType & 1) + 1);
144: 
145:         path = LittleEndian.getByteArray(data, offset, byteLength);
146:         offset += byteLength;
147: 
148:         /*
149:          * If it's a self reference, the last one or two bytes (depending on char type) are the
150:          * unused field. Not sure If i need to bother with this...
151:          */
152:         if (path[0] == 0x02)
153:             _unused = LittleEndian.getByteArray(data, offset, (charType + 1));
154: 
155:     }
156: 
157:     /**
158:      * Read Constructor.
159:      *
160:      * @param inStream RecordInputStream containing a DConRefRecord structure.
161:      */
162:     public DConRefRecord(RecordInputStream inStream)
163:     {
164:         if (inStream.getSid() != sid)
165:             throw new RecordFormatException("Wrong sid: " + inStream.getSid());
166: 
167:         firstRow = inStream.readUShort();
168:         lastRow = inStream.readUShort();
169:         firstCol = inStream.readUByte();
170:         lastCol = inStream.readUByte();
171: 
172:         charCount = inStream.readUShort();
173:         charType = inStream.readUByte() & 0x01; //first bit only.
174: 
175:         // byteLength depends on whether we are using single- or double-byte chars.
176:         int byteLength = charCount * (charType + 1);
177: 
178:         path = new byte[byteLength];
179:         inStream.readFully(path);
180: 
181:         if (path[0] == 0x02)
182:             _unused = inStream.readRemainder();
183: 
184:     }
185: 
186:     /*
187:      * assuming this wants the number of bytes returned by {@link serialize(LittleEndianOutput)},
188:      * that is, (length - 4).
189:      */
190:     @Override
191:     protected int getDataSize()
192:     {
193:         int sz = 9 + path.length;
194:         if (path[0] == 0x02)
195:             sz += _unused.length;
196:         return sz;
197:     }

199:     @Override
200:     protected void serialize(LittleEndianOutput out)
201:     {
202:         out.writeShort(firstRow);
203:         out.writeShort(lastRow);
204:         out.writeByte(firstCol);
205:         out.writeByte(lastCol);
206:         out.writeShort(charCount);
207:         out.writeByte(charType);
208:         out.write(path);
209:         if (path[0] == 0x02)
210:             out.write(_unused);
211:     }

254:         StringBuilder b = new StringBuilder();

270:     /**
271:      *
272:      * @return raw path byte array.
273:      */
274:     public byte[] getPath()
275:     {
276:         return ArrayUtil.copyOf(path, path.length);
277:     }
278: 
279:     /**
280:      * @return the link's path, with the special characters stripped/replaced. May be null.
281:      * @see MS-XLS 2.5.277 (VirtualPath)
282:      */
283:     public String getReadablePath()
284:     {
285:         if (path != null)
286:         {
287:             //all of the path strings start with either 0x02 or 0x01 followed by zero or
288:             //more of 0x01..0x08
289:             int offset = 1;
290:             while (path[offset] < 0x20 && offset < path.length)
291:             {
292:                 offset++;
293:             }
294:             String out = new String(ArrayUtil.copyOfRange(path, offset, path.length));
295:             //UNC paths have \u0003 chars as path separators.
296:             out = out.replaceAll("\u0003", "/");
297:             return out;
298:         }
299:         return null;
300:     }
301: 
302:     /**
303:      * Checks if the data source in this reference record is external to this sheet or internal.
304:      *
305:      * @return true iff this is an external reference.
306:      */
307:     public boolean isExternalRef()
308:     {
309:         if (path[0] == 0x01)
310:             return true;
311:         return false;
312:     }
