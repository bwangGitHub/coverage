 81:      * @see org.apache.poi.hssf.usermodel.HSSFSheet#createRow(int)
 82:      */
 83:     HSSFRow(HSSFWorkbook book, HSSFSheet sheet, RowRecord record) {
 84:         this.book = book;
 85:         this.sheet = sheet;
 86:         row = record;
 87:         setRowNum(record.getRowNumber());
 88:         
 89:         // Size the initial cell list such that a read only case won't waste
 90:         //  lots of memory, and a create/read followed by adding new cells can
 91:         //  add a bit without needing a resize
 92:         cells = new HSSFCell[record.getLastCol()+INITIAL_CAPACITY];

 94:         // Don't trust colIx boundaries as read by other apps

 97:         // subsequent calls to createCellFromRecord() will update the colIx boundaries properly
 98:     }

164:     private void removeCell(HSSFCell cell, boolean alsoRemoveRecords) {
165: 
166:         int column=cell.getColumnIndex();
167:         if(column < 0) {
168:             throw new RuntimeException("Negative cell indexes not allowed");
169:         }
170:         if(column >= cells.length || cell != cells[column]) {
171:             throw new RuntimeException("Specified cell is not from this row");
172:         }
173:         if(cell.isPartOfArrayFormulaGroup()){
174:             cell.notifyArrayFormulaChanging();
175:         }
176: 
177:         cells[column]=null;
178: 
179:         if(alsoRemoveRecords) {
180:             CellValueRecordInterface cval = cell.getCellValueRecord();
181:             sheet.getSheet().removeValueRecord(getRowNum(), cval);
182:         }
183:         if (cell.getColumnIndex()+1 == row.getLastCol()) {
184:             row.setLastCol(calculateNewLastCellPlusOne(row.getLastCol()));
185:         }
186:         if (cell.getColumnIndex() == row.getFirstCol()) {
187:             row.setFirstCol(calculateNewFirstCell(row.getFirstCol()));
188:         }
189:     }

277:     /**
278:      * Moves the supplied cell to a new column, which
279:      *  must not already have a cell there!
280:      * @param cell The cell to move
281:      * @param newColumn The new column number (0 based)
282:      */
283:     public void moveCell(HSSFCell cell, short newColumn) {
284:         // Ensure the destination is free
285:         if(cells.length > newColumn && cells[newColumn] != null) {
286:             throw new IllegalArgumentException("Asked to move cell to column " + newColumn + " but there's already a cell there");
287:         }
288: 
289:         // Check it's one of ours
290:         if(! cells[cell.getColumnIndex()].equals(cell)) {
291:             throw new IllegalArgumentException("Asked to move a cell, but it didn't belong to our row");
292:         }
293: 
294:         // Move the cell to the new position
295:         // (Don't remove the records though)
296:         removeCell(cell, false);
297:         cell.updateCellNum(newColumn);
298:         addCell(cell);
299:     }

301:     /**
302:      * used internally to add a cell.
303:      */
304:     private void addCell(HSSFCell cell) {
305: 
306:         int column=cell.getColumnIndex();
307:         // re-allocate cells array as required.
308:         if(column>=cells.length) {
309:             HSSFCell[] oldCells=cells;
310:             // New size based on the same logic as ArrayList
311:             int newSize=oldCells.length*3/2+1;
312:             if(newSize<column+1) {
313:                 newSize=column+INITIAL_CAPACITY;
314:             }
315:             cells=new HSSFCell[newSize];
316:             System.arraycopy(oldCells,0,cells,0,oldCells.length);
317:         }
318:         cells[column]=cell;
319: 
320:         // fix up firstCol and lastCol indexes
321:         if (row.isEmpty() || column < row.getFirstCol()) {
322:             row.setFirstCol((short)column);
323:         }
324: 
325:         if (row.isEmpty() || column >= row.getLastCol()) {
326:             row.setLastCol((short) (column+1)); // +1 -> for one past the last index
327:         }
328:     }

367:     /**
368:      * Get the hssfcell representing a given column (logical cell)
369:      *  0-based.  If you ask for a cell that is not defined, then
370:      *  your supplied policy says what to do
371:      *
372:      * @param cellnum  0 based column number
373:      * @param policy Policy on blank / missing cells
374:      * @return representing that column or null if undefined + policy allows.
375:      */
376:     public HSSFCell getCell(int cellnum, MissingCellPolicy policy) {
377:         HSSFCell cell = retrieveCell(cellnum);
378:         if(policy == RETURN_NULL_AND_BLANK) {
379:             return cell;
380:         }
381:         if(policy == RETURN_BLANK_AS_NULL) {
382:             if(cell == null) return cell;
383:             if(cell.getCellType() == HSSFCell.CELL_TYPE_BLANK) {
384:                 return null;
385:             }
386:             return cell;
387:         }
388:         if(policy == CREATE_NULL_AS_BLANK) {
389:             if(cell == null) {
390:                 return createCell(cellnum, HSSFCell.CELL_TYPE_BLANK);
391:             }
392:             return cell;
393:         }
394:         throw new IllegalArgumentException("Illegal policy " + policy + " (" + policy.id + ")");
395:     }

537:     /**
538:      * used internally to refresh the "last cell plus one" when the last cell is removed.
539:      * @return 0 when row contains no cells
540:      */
541:     private int calculateNewLastCellPlusOne(int lastcell) {
542:         int cellIx = lastcell - 1;
543:         HSSFCell r = retrieveCell(cellIx);
544: 
545:         while (r == null) {
546:             if (cellIx < 0) {
547:                 return 0;
548:             }
549:             r = retrieveCell(--cellIx);
550:         }
551:         return cellIx+1;
552:     }

554:     /**
555:      * used internally to refresh the "first cell" when the first cell is removed.
556:      * @return 0 when row contains no cells (also when first cell is occupied)
557:      */
558:     private int calculateNewFirstCell(int firstcell) {
559:         int cellIx = firstcell + 1;
560:         HSSFCell r = retrieveCell(cellIx);
561: 
562:         while (r == null) {
563:             if (cellIx <= cells.length) {
564:                 return 0;
565:             }
566:             r = retrieveCell(++cellIx);
567:         }
568:         return cellIx;
569:     }

624:     /**
625:      * An iterator over the (physical) cells in the row.
626:      */
627:     private class CellIterator implements Iterator<Cell> {
628:       int thisId=-1;
629:       int nextId=-1;
630: 
631:       public CellIterator()
632:       {
633:         findNext();
634:       }
635: 
636:       public boolean hasNext() {
637:         return nextId<cells.length;
638:       }
639: 
640:       public Cell next() {
641:           if (!hasNext())
642:               throw new NoSuchElementException("At last element");
643:         HSSFCell cell=cells[nextId];
644:         thisId=nextId;
645:         findNext();
646:         return cell;
647:       }
648: 
649:       public void remove() {
650:           if (thisId == -1)
651:               throw new IllegalStateException("remove() called before next()");
652:         cells[thisId]=null;
653:       }
654: 
655:       private void findNext()
656:       {
657:         int i=nextId+1;
