 56: 	public static double calculate(double pStartDateVal, double pEndDateVal, int basis) throws EvaluationException {
 57: 
 58: 		if (basis < 0 || basis >= 5) {
 59: 			// if basis is invalid the result is #NUM!
 60: 			throw new EvaluationException(ErrorEval.NUM_ERROR);
 61: 		}
 62: 
 63: 		// common logic for all bases
 64: 
 65: 		// truncate day values
 66: 		int startDateVal = (int) Math.floor(pStartDateVal);
 67: 		int endDateVal = (int) Math.floor(pEndDateVal);
 68: 		if (startDateVal == endDateVal) {
 69: 			// when dates are equal, result is zero 
 70: 			return 0;
 71: 		}
 72: 		// swap start and end if out of order
 73: 		if (startDateVal > endDateVal) {
 74: 			int temp = startDateVal;
 75: 			startDateVal = endDateVal;
 76: 			endDateVal = temp;
 77: 		}
 78: 
 79: 		switch (basis) {
 80: 			case 0: return basis0(startDateVal, endDateVal);
 81: 			case 1: return basis1(startDateVal, endDateVal);
 82: 			case 2: return basis2(startDateVal, endDateVal);
 83: 			case 3: return basis3(startDateVal, endDateVal);
 84: 			case 4: return basis4(startDateVal, endDateVal);
 85: 		}
 86: 		throw new IllegalStateException("cannot happen");
 87: 	}

 90: 	/**
 91: 	 * @param startDateVal assumed to be less than or equal to endDateVal
 92: 	 * @param endDateVal assumed to be greater than or equal to startDateVal
 93: 	 */
 94: 	public static double basis0(int startDateVal, int endDateVal) {
 95: 		SimpleDate startDate = createDate(startDateVal);
 96: 		SimpleDate endDate = createDate(endDateVal);
 97: 		int date1day = startDate.day;
 98: 		int date2day = endDate.day;
 99: 
100: 		// basis zero has funny adjustments to the day-of-month fields when at end-of-month 
101: 		if (date1day == LONG_MONTH_LEN && date2day == LONG_MONTH_LEN) {
102: 			date1day = SHORT_MONTH_LEN;
103: 			date2day = SHORT_MONTH_LEN;
104: 		} else if (date1day == LONG_MONTH_LEN) {
105: 			date1day = SHORT_MONTH_LEN;
106: 		} else if (date1day == SHORT_MONTH_LEN && date2day == LONG_MONTH_LEN) {
107: 			date2day = SHORT_MONTH_LEN;
108: 			// Note: If date2day==31, it STAYS 31 if date1day < 30.
109: 			// Special fixes for February:
110: 		} else if (startDate.month == 2 && isLastDayOfMonth(startDate)) {
111: 			// Note - these assignments deliberately set Feb 30 date.
112: 			date1day = SHORT_MONTH_LEN;
113: 			if (endDate.month == 2 && isLastDayOfMonth(endDate)) {
114: 				// only adjusted when first date is last day in Feb
115: 				date2day = SHORT_MONTH_LEN;
116: 			}
117: 		}
118: 		return calculateAdjusted(startDate, endDate, date1day, date2day);
119: 	}

121: 	 * @param startDateVal assumed to be less than or equal to endDateVal
122: 	 * @param endDateVal assumed to be greater than or equal to startDateVal
123: 	 */
124: 	public static double basis1(int startDateVal, int endDateVal) {
125: 		SimpleDate startDate = createDate(startDateVal);
126: 		SimpleDate endDate = createDate(endDateVal);
127: 		double yearLength;
128: 		if (isGreaterThanOneYear(startDate, endDate)) {
129: 			yearLength = averageYearLength(startDate.year, endDate.year);
130: 		} else if (shouldCountFeb29(startDate, endDate)) {
131: 			yearLength = DAYS_PER_LEAP_YEAR;
132: 		} else {
133: 			yearLength = DAYS_PER_NORMAL_YEAR;
134: 		}
135: 		return dateDiff(startDate.tsMilliseconds, endDate.tsMilliseconds) / yearLength;
136: 	}

175: 	private static double calculateAdjusted(SimpleDate startDate, SimpleDate endDate, int date1day,
176: 			int date2day) {
177: 		double dayCount 
178: 			= (endDate.year - startDate.year) * 360
179: 			+ (endDate.month - startDate.month) * SHORT_MONTH_LEN
180: 			+ (date2day - date1day) * 1;
181: 		return dayCount / 360;
182: 	}

184: 	private static boolean isLastDayOfMonth(SimpleDate date) {
185: 		if (date.day < SHORT_FEB_LEN) {
186: 			return false;
187: 		}
188: 		return date.day == getLastDayOfMonth(date);
189: 	}

191: 	private static int getLastDayOfMonth(SimpleDate date) {
192: 		switch (date.month) {
193: 			case 1:
194: 			case 3:
195: 			case 5:
196: 			case 7:
197: 			case 8:
198: 			case 10:
199: 			case 12:
200: 				return LONG_MONTH_LEN;
201: 			case 4:
202: 			case 6:
203: 			case 9:
204: 			case 11:
205: 				return SHORT_MONTH_LEN;
206: 		}
207: 		if (isLeapYear(date.year)) {
208: 			return LONG_FEB_LEN;
209: 		}
210: 		return SHORT_FEB_LEN;
211: 	}

213: 	/**
214: 	 * Assumes dates are no more than 1 year apart.
215: 	 * @return <code>true</code> if dates both within a leap year, or span a period including Feb 29
216: 	 */
217: 	private static boolean shouldCountFeb29(SimpleDate start, SimpleDate end) {
218: 		boolean startIsLeapYear = isLeapYear(start.year);
219: 		if (startIsLeapYear && start.year == end.year) {
220: 			// note - dates may not actually span Feb-29, but it gets counted anyway in this case
221: 			return true;
222: 		}
223: 
224: 		boolean endIsLeapYear = isLeapYear(end.year);
225: 		if (!startIsLeapYear && !endIsLeapYear) {
226: 			return false;
227: 		}
228: 		if (startIsLeapYear) {
229: 			switch (start.month) {
230: 				case SimpleDate.JANUARY:
231: 				case SimpleDate.FEBRUARY:
232: 					return true;
233: 			}
234: 			return false;
235: 		}
236: 		if (endIsLeapYear) {
237: 			switch (end.month) {
238: 				case SimpleDate.JANUARY:
239: 					return false;
240: 				case SimpleDate.FEBRUARY:
241: 					break;
242: 				default:
243: 					return true;
244: 			}
245: 			return end.day == LONG_FEB_LEN;
246: 		}
247: 		return false;
248: 	}

250: 	/**
251: 	 * @return the whole number of days between the two time-stamps.  Both time-stamps are
252: 	 * assumed to represent 12:00 midnight on the respective day.
253: 	 */
254: 	private static int dateDiff(long startDateMS, long endDateMS) {
255: 		long msDiff = endDateMS - startDateMS;
256: 
257: 		// some extra checks to make sure we don't hide some other bug with the rounding 
258: 		int remainderHours = (int) ((msDiff % MS_PER_DAY) / MS_PER_HOUR);
259: 		switch (remainderHours) {
260: 			case 0:  // normal case
261: 				break;
262: 			case 1:  // transition from normal time to daylight savings adjusted
263: 			case 23: // transition from daylight savings adjusted to normal time
264: 				// Unexpected since we are using UTC_TIME_ZONE 
265: 			default:
266: 				throw new RuntimeException("Unexpected date diff between " + startDateMS + " and " + endDateMS);
267: 
268: 		}
269: 		return (int) (0.5 + ((double)msDiff / MS_PER_DAY));
270: 	}

272: 	private static double averageYearLength(int startYear, int endYear) {
273: 		int dayCount = 0;
274: 		for (int i=startYear; i<=endYear; i++) {
275: 			dayCount += DAYS_PER_NORMAL_YEAR;
276: 			if (isLeapYear(i)) {
277: 				dayCount++;
278: 			}
279: 		}
280: 		double numberOfYears = endYear-startYear+1;
281: 		return dayCount / numberOfYears;
282: 	}
283: 
284: 	private static boolean isLeapYear(int i) {
285: 		// leap years are always divisible by 4
286: 		if (i % 4 != 0) {
287: 			return false;
288: 		}
289: 		// each 4th century is a leap year
290: 		if (i % 400 == 0) {
291: 			return true;
292: 		}
293: 		// all other centuries are *not* leap years
294: 		if (i % 100 == 0) {
295: 			return false;
296: 		}
297: 		return true;
298: 	}
299: 
300: 	private static boolean isGreaterThanOneYear(SimpleDate start, SimpleDate end) {
301: 		if (start.year == end.year) {
302: 			return false;
303: 		}
304: 		if (start.year + 1 != end.year) {
305: 			return true;
306: 		}
307: 
308: 		if (start.month > end.month) {
309: 			return false;
310: 		}
311: 		if (start.month < end.month) {
312: 			return true;
313: 		}
314: 
315: 		return start.day < end.day;
316: 	}

318: 	private static SimpleDate createDate(int dayCount) {
319: 		GregorianCalendar calendar = new GregorianCalendar(UTC_TIME_ZONE);
320: 		DateUtil.setCalendar(calendar, dayCount, 0, false);
321: 		return new SimpleDate(calendar);
322: 	}
