 36: /**
 37:  * <p>Adds writing support to the {@link PropertySet} class.</p>
 38:  *
 39:  * <p>Please be aware that this class' functionality will be merged into the
 40:  * {@link PropertySet} class at a later time, so the API will change.</p>
 41:  *
 42:  * @author Rainer Klute <a
 43:  * href="mailto:klute@rainer-klute.de">&lt;klute@rainer-klute.de&gt;</a>
 44:  */
 45: public class MutablePropertySet extends PropertySet

 47: 
 48:     /**
 49:      * <p>Constructs a <code>MutablePropertySet</code> instance. Its
 50:      * primary task is to initialize the immutable field with their proper
 51:      * values. It also sets fields that might change to reasonable defaults.</p>
 52:      */
 53:     public MutablePropertySet()
 54:     {
 55:         /* Initialize the "byteOrder" field. */
 56:         byteOrder = LittleEndian.getUShort(BYTE_ORDER_ASSERTION);
 57: 
 58:         /* Initialize the "format" field. */
 59:         format = LittleEndian.getUShort(FORMAT_ASSERTION);
 60: 
 61:         /* Initialize "osVersion" field as if the property has been created on
 62:          * a Win32 platform, whether this is the case or not. */
 63:         osVersion = (OS_WIN32 << 16) | 0x0A04;
 64: 
 65:         /* Initailize the "classID" field. */
 66:         classID = new ClassID();
 67: 
 68:         /* Initialize the sections. Since property set must have at least
 69:          * one section it is added right here. */
 70:         sections = new LinkedList<Section>();
 71:         sections.add(new MutableSection());
 72:     }
 73: 
 74: 
 75: 
 76:     /**
 77:      * <p>Constructs a <code>MutablePropertySet</code> by doing a deep copy of
 78:      * an existing <code>PropertySet</code>. All nested elements, i.e.
 79:      * <code>Section</code>s and <code>Property</code> instances, will be their
 80:      * mutable counterparts in the new <code>MutablePropertySet</code>.</p>
 81:      *
 82:      * @param ps The property set to copy
 83:      */
 84:     public MutablePropertySet(final PropertySet ps)
 85:     {
 86:         byteOrder = ps.getByteOrder();
 87:         format = ps.getFormat();
 88:         osVersion = ps.getOSVersion();
 89:         setClassID(ps.getClassID());
 90:         clearSections();
 91:         if (sections == null)
 92:             sections = new LinkedList();
 93:         for (final Iterator i = ps.getSections().iterator(); i.hasNext();)
 94:         {
 95:             final MutableSection s = new MutableSection((Section) (i.next()));
 96:             addSection(s);
 97:         }
 98:     }
 99:

102:     /**
103:      * <p>The length of the property set stream header.</p>
104:      */
105:     private final int OFFSET_HEADER =
106:         BYTE_ORDER_ASSERTION.length + /* Byte order    */
107:         FORMAT_ASSERTION.length +     /* Format        */
108:         LittleEndianConsts.INT_SIZE + /* OS version    */
109:         ClassID.LENGTH +              /* Class ID      */
110:         LittleEndianConsts.INT_SIZE;  /* Section count */

191:     /**
192:      * <p>Writes the property set to an output stream.</p>
193:      *
194:      * @param out the output stream to write the section to
195:      * @exception IOException if an error when writing to the output stream
196:      * occurs
197:      * @exception WritingNotSupportedException if HPSF does not yet support
198:      * writing a property's variant type.
199:      */
200:     public void write(final OutputStream out)
201:         throws WritingNotSupportedException, IOException
202:     {
203:         /* Write the number of sections in this property set stream. */
204:         final int nrSections = sections.size();
205:         int length = 0;
206: 
207:         /* Write the property set's header. */
208:         length += TypeWriter.writeToStream(out, (short) getByteOrder());
209:         length += TypeWriter.writeToStream(out, (short) getFormat());
210:         length += TypeWriter.writeToStream(out, getOSVersion());
211:         length += TypeWriter.writeToStream(out, getClassID());
212:         length += TypeWriter.writeToStream(out, nrSections);
213:         int offset = OFFSET_HEADER;
214: 
215:         /* Write the section list, i.e. the references to the sections. Each
216:          * entry in the section list consist of the section's class ID and the
217:          * section's offset relative to the beginning of the stream. */
218:         offset += nrSections * (ClassID.LENGTH + LittleEndian.INT_SIZE);
219:         final int sectionsBegin = offset;
220:         for (final ListIterator i = sections.listIterator(); i.hasNext();)
221:         {
222:             final MutableSection s = (MutableSection) i.next();
223:             final ClassID formatID = s.getFormatID();
224:             if (formatID == null)
225:                 throw new NoFormatIDException();
226:             length += TypeWriter.writeToStream(out, s.getFormatID());
227:             length += TypeWriter.writeUIntToStream(out, offset);
228:             try
229:             {
230:                 offset += s.getSize();
231:             }
232:             catch (HPSFRuntimeException ex)
233:             {
234:                 final Throwable cause = ex.getReason();
235:                 if (cause instanceof UnsupportedEncodingException) {
236:                     throw new IllegalPropertySetDataException(cause);
237:                 }
238:                 throw ex;
239:             }
240:         }
241: 
242:         /* Write the sections themselves. */
243:         offset = sectionsBegin;
244:         for (final ListIterator i = sections.listIterator(); i.hasNext();)
245:         {
246:             final MutableSection s = (MutableSection) i.next();
247:             offset += s.write(out);
248:         }
249:     }

253:     /**
254:      * <p>Returns the contents of this property set stream as an input stream.
255:      * The latter can be used for example to write the property set into a POIFS
256:      * document. The input stream represents a snapshot of the property set.
257:      * If the latter is modified while the input stream is still being
258:      * read, the modifications will not be reflected in the input stream but in
259:      * the {@link MutablePropertySet} only.</p>
260:      *
261:      * @return the contents of this property set stream
262:      *
263:      * @throws WritingNotSupportedException if HPSF does not yet support writing
264:      * of a property's variant type.
265:      * @throws IOException if an I/O exception occurs.
266:      */
267:     public InputStream toInputStream()
268:         throws IOException, WritingNotSupportedException
269:     {
270:         final ByteArrayOutputStream psStream = new ByteArrayOutputStream();
271:         write(psStream);
272:         psStream.close();
273:         final byte[] streamData = psStream.toByteArray();
274:         return new ByteArrayInputStream(streamData);
275:     }
276: 
277:     /**
278:      * <p>Writes a property set to a document in a POI filesystem directory.</p>
279:      *
280:      * @param dir The directory in the POI filesystem to write the document to.
281:      * @param name The document's name. If there is already a document with the
282:      * same name in the directory the latter will be overwritten.
283:      *
284:      * @throws WritingNotSupportedException
285:      * @throws IOException
286:      */
287:     public void write(final DirectoryEntry dir, final String name)
288:     throws WritingNotSupportedException, IOException
289:     {
290:         /* If there is already an entry with the same name, remove it. */
291:         try
292:         {
293:             final Entry e = dir.getEntry(name);
294:             e.delete();
295:         }
296:         catch (FileNotFoundException ex)
297:         {
298:             /* Entry not found, no need to remove it. */
299:         }
300:         /* Create the new entry. */
301:         dir.createDocument(name, toInputStream());
302:     }
