201: 	private static byte[] readFirst512(InputStream stream) throws IOException {
202:       // Grab the first 512 bytes
203:       // (For 4096 sized blocks, the remaining 3584 bytes are zero)
204:       byte[] data = new byte[512];
205:       int bsCount = IOUtils.readFully(stream, data);
206:       if(bsCount != 512) {
207:          throw alertShortRead(bsCount, 512);
208:       }
209:       return data;
210: 	}
