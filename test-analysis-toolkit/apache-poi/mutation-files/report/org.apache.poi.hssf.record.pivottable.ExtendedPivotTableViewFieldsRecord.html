 32: public final class ExtendedPivotTableViewFieldsRecord extends StandardRecord {
 33: 	public static final short sid = 0x0100;
 34: 
 35: 	/** the value of the subname length when the {@link #_subtotalName} is not present */
 36: 	private static final int STRING_NOT_PRESENT_LEN = 0xFFFF;
 37: 
 38: 	private int _grbit1;
 39: 	private int _grbit2;
 40: 	private int _citmShow;
 41: 	private int _isxdiSort;
 42: 	private int _isxdiShow;
 43: 	private int _reserved1;
 44: 	private int _reserved2;
 45: 	/** custom sub-total name */
 46: 	private String _subtotalName;
 47: 
 48: 	public ExtendedPivotTableViewFieldsRecord(RecordInputStream in) {
 49: 
 50: 		_grbit1 = in.readInt();
 51: 		_grbit2 = in.readUByte();
 52: 		_citmShow = in.readUByte();
 53: 		_isxdiSort = in.readUShort();
 54: 		_isxdiShow = in.readUShort();
 55: 		// This record seems to have different valid encodings
 56: 		switch (in.remaining()) {
 57: 			case 0:
 58: 				// as per "Microsoft Excel Developer's Kit" book
 59: 				// older version of SXVDEX - doesn't seem to have a sub-total name
 60: 				_reserved1 = 0;
 61: 				_reserved2 = 0;
 62: 				_subtotalName = null;
 63: 				return;
 64: 			case 10:
 65: 				// as per "MICROSOFT OFFICE EXCEL 97-2007 BINARY FILE FORMAT SPECIFICATION" pdf
 66: 				break;
 67: 			default:
 68: 				throw new RecordFormatException("Unexpected remaining size (" + in.remaining() + ")");
 69: 		}
 70: 		int cchSubName = in.readUShort();
 71: 		_reserved1 = in.readInt();
 72: 		_reserved2 = in.readInt();
 73: 		if (cchSubName != STRING_NOT_PRESENT_LEN) {
 74: 			_subtotalName = in.readUnicodeLEString(cchSubName);
 75: 		}
 76: 	}
 77: 
 78: 	@Override
 79: 	protected void serialize(LittleEndianOutput out) {
 80: 
 81: 		out.writeInt(_grbit1);
 82: 		out.writeByte(_grbit2);
 83: 		out.writeByte(_citmShow);
 84: 		out.writeShort(_isxdiSort);
 85: 		out.writeShort(_isxdiShow);
 86: 
 87: 		if (_subtotalName == null) {
 88: 			out.writeShort(STRING_NOT_PRESENT_LEN);
 89: 		} else {
 90: 			out.writeShort(_subtotalName.length());
 91: 		}
 92: 
 93: 		out.writeInt(_reserved1);
 94: 		out.writeInt(_reserved2);
 95: 		if (_subtotalName != null) {
 96: 			StringUtil.putUnicodeLE(_subtotalName, out);
 97: 		}
 98: 	}
 99:

101: 	protected int getDataSize() {
102: 		
103: 		return 4 + 1 + 1 + 2 + 2 + 2 +  4 + 4 +
104: 					(_subtotalName == null ? 0 : (2*_subtotalName.length())); // in unicode
105: 	}

108: 	public short getSid() {

110: 	}
