130: 			int rowIndex, int columnIndex, ValueEval value) {
131: 
132: 		Loc loc = new Loc(bookIndex, sheetIndex, rowIndex, columnIndex);
133: 		PlainValueCellCacheEntry result = _plainCellCache.get(loc);
134: 		if (result == null) {

151: 	private boolean areValuesEqual(ValueEval a, ValueEval b) {
152: 		if (a == null) {
153: 			return false;
154: 		}
155: 		Class<?> cls = a.getClass();
156: 		if (cls != b.getClass()) {
157: 			// value type is changing
158: 			return false;
159: 		}
160: 		if (a == BlankEval.instance) {
161: 			return b == a;
162: 		}
163: 		if (cls == NumberEval.class) {
164: 			return ((NumberEval)a).getNumberValue() == ((NumberEval)b).getNumberValue();
165: 		}
166: 		if (cls == StringEval.class) {
167: 			return ((StringEval)a).getStringValue().equals(((StringEval)b).getStringValue());
168: 		}
169: 		if (cls == BoolEval.class) {
170: 			return ((BoolEval)a).getBooleanValue() == ((BoolEval)b).getBooleanValue();
171: 		}
172: 		if (cls == ErrorEval.class) {
173: 			return ((ErrorEval)a).getErrorCode() == ((ErrorEval)b).getErrorCode();
174: 		}
175: 		throw new IllegalStateException("Unexpected value class (" + cls.getName() + ")");
176: 	}
