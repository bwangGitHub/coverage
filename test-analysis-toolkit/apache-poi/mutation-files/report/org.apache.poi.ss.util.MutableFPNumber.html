 27: 	 * The minimum value in 'Base-10 normalised form'.<br/>
 28: 	 * When {@link #_binaryExponent} == 46 this is the the minimum {@link #_frac} value
 29: 	 *  (10<sup>14</sup>-0.05) * 2^17

 31: 	 *  Values between (10<sup>14</sup>-0.05) and 10<sup>14</sup> will be represented as '1'
 32: 	 *  followed by 14 zeros.
 33: 	 *  Values less than (10<sup>14</sup>-0.05) will get shifted by one more power of 10
 34: 	 *
 35: 	 *  This frac value rounds to '1' followed by fourteen zeros with an incremented decimal exponent
 36: 	 */

 65: 	public void normalise64bit() {
 66: 		int oldBitLen = _significand.bitLength();
 67: 		int sc = oldBitLen - C_64;
 68: 		if (sc == 0) {
 69: 			return;
 70: 		}
 71: 		if (sc < 0) {
 72: 			throw new IllegalStateException("Not enough precision");
 73: 		}
 74: 		_binaryExponent += sc;
 75: 		if (sc > 32) {
 76: 			int highShift = (sc-1) & 0xFFFFE0;
 77: 			_significand = _significand.shiftRight(highShift);
 78: 			sc -= highShift;
 79: 			oldBitLen -= highShift;
 80: 		}
 81: 		if (sc < 1) {
 82: 			throw new IllegalStateException();
 83: 		}
 84: 		_significand = Rounder.round(_significand, sc);
 85: 		if (_significand.bitLength() > oldBitLen) {
 86: 			sc++;
 87: 			_binaryExponent++;
 88: 		}
 89: 		_significand = _significand.shiftRight(sc);
 90: 	}

119: 	private void mulShift(BigInteger multiplicand, int multiplierShift) {
120: 		_significand = _significand.multiply(multiplicand);
121: 		_binaryExponent += multiplierShift;
122: 		// check for too much precision
123: 		int sc = (_significand.bitLength() - MIN_PRECISION) & 0xFFFFFFE0;
124: 		// mask makes multiples of 32 which optimises BigInteger.shiftRight
125: 		if (sc > 0) {
126: 			// no need to round because we have at least 8 bits of extra precision
127: 			_significand = _significand.shiftRight(sc);
128: 			_binaryExponent += sc;
129: 		}

132: 	private static final class Rounder {
133: 		private static final BigInteger[] HALF_BITS;
134: 
135: 		static {
136: 			BigInteger[] bis = new BigInteger[33];
137: 			long acc=1;
138: 			for (int i = 1; i < bis.length; i++) {
139: 				bis[i] = BigInteger.valueOf(acc);
140: 				acc <<=1;
141: 			}
142: 			HALF_BITS = bis;
143: 		}
144: 		/**
145: 		 * @param nBits number of bits to shift right
146: 		 */
147: 		public static BigInteger round(BigInteger bi, int nBits) {
148: 			if (nBits < 1) {
149: 				return bi;
150: 			}
151: 			return bi.add(HALF_BITS[nBits]);
