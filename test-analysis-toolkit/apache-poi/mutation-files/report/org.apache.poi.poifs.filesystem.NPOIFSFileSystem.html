219:     /**
220:      * Create a POIFSFileSystem from an <tt>InputStream</tt>.  Normally the stream is read until
221:      * EOF.  The stream is always closed.<p/>
222:      *
223:      * Some streams are usable after reaching EOF (typically those that return <code>true</code>
224:      * for <tt>markSupported()</tt>).  In the unlikely case that the caller has such a stream
225:      * <i>and</i> needs to use it after this constructor completes, a work around is to wrap the
226:      * stream in order to trap the <tt>close()</tt> call.  A convenience method (
227:      * <tt>createNonClosingInputStream()</tt>) has been provided for this purpose:
228:      * <pre>
229:      * InputStream wrappedStream = POIFSFileSystem.createNonClosingInputStream(is);
230:      * HSSFWorkbook wb = new HSSFWorkbook(wrappedStream);
231:      * is.reset();
232:      * doSomethingElse(is);
233:      * </pre>
234:      * Note also the special case of <tt>ByteArrayInputStream</tt> for which the <tt>close()</tt>
235:      * method does nothing.
236:      * <pre>
237:      * ByteArrayInputStream bais = ...
238:      * HSSFWorkbook wb = new HSSFWorkbook(bais); // calls bais.close() !
239:      * bais.reset(); // no problem
240:      * doSomethingElse(bais);
241:      * </pre>
242:      *
243:      * @param stream the InputStream from which to read the data
244:      *
245:      * @exception IOException on errors reading, or on invalid data
246:      */
247: 
248:     public NPOIFSFileSystem(InputStream stream)
249:         throws IOException
250:     {
251:         this(false);
252:         
253:         ReadableByteChannel channel = null;
254:         boolean success = false;
255:         
256:         try {
257:            // Turn our InputStream into something NIO based
258:            channel = Channels.newChannel(stream);
259:            
260:            // Get the header
261:            ByteBuffer headerBuffer = ByteBuffer.allocate(POIFSConstants.SMALLER_BIG_BLOCK_SIZE);
262:            IOUtils.readFully(channel, headerBuffer);
263:            
264:            // Have the header processed
265:            _header = new HeaderBlock(headerBuffer);
266:            
267:            // Sanity check the block count
268:            BlockAllocationTableReader.sanityCheckBlockCount(_header.getBATCount());
269:    
270:            // We need to buffer the whole file into memory when
271:            //  working with an InputStream.
272:            // The max possible size is when each BAT block entry is used
273:            int maxSize = BATBlock.calculateMaximumSize(_header); 
274:            ByteBuffer data = ByteBuffer.allocate(maxSize);
275:            // Copy in the header
276:            headerBuffer.position(0);
277:            data.put(headerBuffer);
278:            data.position(headerBuffer.capacity());
279:            // Now read the rest of the stream
280:            IOUtils.readFully(channel, data);
281:            success = true;
282:            
283:            // Turn it into a DataSource
284:            _data = new ByteArrayBackedDataSource(data.array(), data.position());
285:         } finally {
286:            // As per the constructor contract, always close the stream
287:            if(channel != null)
288:               channel.close();
289:            closeInputStream(stream, success);
290:         }
291:         
292:         // Now process the various entries
293:         readCoreContents();
294:     }
295:     /**
296:      * @param stream the stream to be closed
297:      * @param success <code>false</code> if an exception is currently being thrown in the calling method
298:      */
299:     private void closeInputStream(InputStream stream, boolean success) {
300:         try {
301:             stream.close();
302:         } catch (IOException e) {
303:             if(success) {
304:                 throw new RuntimeException(e);
305:             }
306:             // else not success? Try block did not complete normally
307:             // just print stack trace and leave original ex to be thrown
308:             e.printStackTrace();
309:         }
310:     }

312:     /**
313:      * Checks that the supplied InputStream (which MUST
314:      *  support mark and reset, or be a PushbackInputStream)
315:      *  has a POIFS (OLE2) header at the start of it.
316:      * If your InputStream does not support mark / reset,
317:      *  then wrap it in a PushBackInputStream, then be
318:      *  sure to always use that, and not the original!
319:      * @param inp An InputStream which supports either mark/reset, or is a PushbackInputStream
320:      */
321:     public static boolean hasPOIFSHeader(InputStream inp) throws IOException {
322:         // We want to peek at the first 8 bytes
323:         inp.mark(8);
324: 
325:         byte[] header = new byte[8];
326:         IOUtils.readFully(inp, header);
327:         LongField signature = new LongField(HeaderBlockConstants._signature_offset, header);
328: 
329:         // Wind back those 8 bytes
330:         if(inp instanceof PushbackInputStream) {
331:             PushbackInputStream pin = (PushbackInputStream)inp;
332:             pin.unread(header);
333:         } else {
334:             inp.reset();
335:         }
336: 
337:         // Did it match the signature?
338:         return (signature.get() == HeaderBlockConstants._signature);
339:     }

341:     /**
342:      * Read and process the PropertiesTable and the
343:      *  FAT / XFAT blocks, so that we're ready to
344:      *  work with the file
345:      */
346:     private void readCoreContents() throws IOException {
347:        // Grab the block size
348:        bigBlockSize = _header.getBigBlockSize();
349:        
350:        // Each block should only ever be used by one of the
351:        //  FAT, XFAT or Property Table. Ensure it does
352:        ChainLoopDetector loopDetector = getChainLoopDetector();
353:        
354:        // Read the FAT blocks
355:        for(int fatAt : _header.getBATArray()) {
356:           readBAT(fatAt, loopDetector);
357:        }
358:        
359:        // Now read the XFAT blocks, and the FATs within them
360:        BATBlock xfat; 
361:        int nextAt = _header.getXBATIndex();
362:        for(int i=0; i<_header.getXBATCount(); i++) {
363:           loopDetector.claim(nextAt);
364:           ByteBuffer fatData = getBlockAt(nextAt);
365:           xfat = BATBlock.createBATBlock(bigBlockSize, fatData);
366:           xfat.setOurBlockIndex(nextAt);
367:           nextAt = xfat.getValueAt(bigBlockSize.getXBATEntriesPerBlock());
368:           _xbat_blocks.add(xfat);
369:           
370:           for(int j=0; j<bigBlockSize.getXBATEntriesPerBlock(); j++) {
371:              int fatAt = xfat.getValueAt(j);
372:              if(fatAt == POIFSConstants.UNUSED_BLOCK || fatAt == POIFSConstants.END_OF_CHAIN) break;
373:              readBAT(fatAt, loopDetector);
374:           }
375:        }
376:        
377:        // We're now able to load steams
378:        // Use this to read in the properties
379:        _property_table = new NPropertyTable(_header, this);
380:        
381:        // Finally read the Small Stream FAT (SBAT) blocks
382:        BATBlock sfat;
383:        List<BATBlock> sbats = new ArrayList<BATBlock>();
384:        _mini_store     = new NPOIFSMiniStore(this, _property_table.getRoot(), sbats, _header);
385:        nextAt = _header.getSBATStart();
386:        for(int i=0; i<_header.getSBATCount(); i++) {
387:           loopDetector.claim(nextAt);
388:           ByteBuffer fatData = getBlockAt(nextAt);
389:           sfat = BATBlock.createBATBlock(bigBlockSize, fatData);
390:           sfat.setOurBlockIndex(nextAt);
391:           sbats.add(sfat);
392:           nextAt = getNextBlock(nextAt);  
393:        }
394:     }
395:     private void readBAT(int batAt, ChainLoopDetector loopDetector) throws IOException {
396:        loopDetector.claim(batAt);
397:        ByteBuffer fatData = getBlockAt(batAt);
398:        BATBlock bat = BATBlock.createBATBlock(bigBlockSize, fatData);
399:        bat.setOurBlockIndex(batAt);
400:        _bat_blocks.add(bat);
401:     }
402:     private BATBlock createBAT(int offset, boolean isBAT) throws IOException {
403:        // Create a new BATBlock
404:        BATBlock newBAT = BATBlock.createEmptyBATBlock(bigBlockSize, !isBAT);
405:        newBAT.setOurBlockIndex(offset);
406:        // Ensure there's a spot in the file for it
407:        ByteBuffer buffer = ByteBuffer.allocate(bigBlockSize.getBigBlockSize());
408:        int writeTo = (1+offset) * bigBlockSize.getBigBlockSize(); // Header isn't in BATs
409:        _data.write(buffer, writeTo);
410:        // All done
411:        return newBAT;
412:     }

469:     /**
470:      * Finds a free block, and returns its offset.
471:      * This method will extend the file if needed, and if doing
472:      *  so, allocate new FAT blocks to address the extra space.
473:      */
474:     protected int getFreeBlock() throws IOException {
475:        // First up, do we have any spare ones?
476:        int offset = 0;
477:        for(int i=0; i<_bat_blocks.size(); i++) {
478:           int numSectors = bigBlockSize.getBATEntriesPerBlock();
479: 
480:           // Check this one
481:           BATBlock bat = _bat_blocks.get(i);
482:           if(bat.hasFreeSectors()) {
483:              // Claim one of them and return it
484:              for(int j=0; j<numSectors; j++) {
485:                 int batValue = bat.getValueAt(j);
486:                 if(batValue == POIFSConstants.UNUSED_BLOCK) {
487:                    // Bingo
488:                    return offset + j;
489:                 }
490:              }
491:           }
492:           
493:           // Move onto the next BAT
494:           offset += numSectors;
495:        }
496:        
497:        // If we get here, then there aren't any free sectors
498:        //  in any of the BATs, so we need another BAT
499:        BATBlock bat = createBAT(offset, true);
500:        bat.setValueAt(0, POIFSConstants.FAT_SECTOR_BLOCK);
501:        _bat_blocks.add(bat);
502:        
503:        // Now store a reference to the BAT in the required place 
504:        if(_header.getBATCount() >= 109) {
505:           // Needs to come from an XBAT
506:           BATBlock xbat = null;
507:           for(BATBlock x : _xbat_blocks) {
508:              if(x.hasFreeSectors()) {
509:                 xbat = x;
510:                 break;
511:              }
512:           }
513:           if(xbat == null) {
514:              // Oh joy, we need a new XBAT too...
515:              xbat = createBAT(offset+1, false);
516:              xbat.setValueAt(0, offset);
517:              bat.setValueAt(1, POIFSConstants.DIFAT_SECTOR_BLOCK);
518:              
519:              // Will go one place higher as XBAT added in
520:              offset++;
521:              
522:              // Chain it
523:              if(_xbat_blocks.size() == 0) {
524:                 _header.setXBATStart(offset);
525:              } else {
526:                 _xbat_blocks.get(_xbat_blocks.size()-1).setValueAt(
527:                       bigBlockSize.getXBATEntriesPerBlock(), offset
528:                 );
529:              }
530:              _xbat_blocks.add(xbat);
531:              _header.setXBATCount(_xbat_blocks.size());
532:           }
533:           // Allocate us in the XBAT
534:           for(int i=0; i<bigBlockSize.getXBATEntriesPerBlock(); i++) {
535:              if(xbat.getValueAt(i) == POIFSConstants.UNUSED_BLOCK) {
536:                 xbat.setValueAt(i, offset);
537:              }
538:           }
539:        } else {
540:           // Store us in the header
541:           int[] newBATs = new int[_header.getBATCount()+1];
542:           System.arraycopy(_header.getBATArray(), 0, newBATs, 0, newBATs.length-1);
543:           newBATs[newBATs.length-1] = offset;
544:           _header.setBATArray(newBATs);
545:        }
546:        _header.setBATCount(_bat_blocks.size());
547:        
548:        // The current offset stores us, but the next one is free
549:        return offset+1;
550:     }
