200:     /**
201:      * used internally -- given a cell value record, figure out its type
202:      */
203:     private static int determineType(CellValueRecordInterface cval) {
204:         if (cval instanceof FormulaRecordAggregate) {
205:             return HSSFCell.CELL_TYPE_FORMULA;
206:         }
207:         // all others are plain BIFF records
208:         Record record = ( Record ) cval;
209:         switch (record.getSid()) {
210: 
211:             case NumberRecord.sid :   return HSSFCell.CELL_TYPE_NUMERIC;
212:             case BlankRecord.sid :    return HSSFCell.CELL_TYPE_BLANK;
213:             case LabelSSTRecord.sid : return HSSFCell.CELL_TYPE_STRING;
214:             case BoolErrRecord.sid :
215:                 BoolErrRecord boolErrRecord = ( BoolErrRecord ) record;
216: 
217:                 return boolErrRecord.isBoolean()
218:                          ? HSSFCell.CELL_TYPE_BOOLEAN
219:                          : HSSFCell.CELL_TYPE_ERROR;
220:         }
221:         throw new RuntimeException("Bad cell value rec (" + cval.getClass().getName() + ")");
222:     }

328:                 if (cellType != _cellType)
329:                 {
