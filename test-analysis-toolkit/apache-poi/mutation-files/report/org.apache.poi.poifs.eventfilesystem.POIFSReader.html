 38: /**
 39:  * An event-driven reader for POIFS file systems. Users of this class
 40:  * first create an instance of it, then use the registerListener
 41:  * methods to register POIFSReaderListener instances for specific
 42:  * documents. Once all the listeners have been registered, the read()
 43:  * method is called, which results in the listeners being notified as
 44:  * their documents are read.
 45:  *
 46:  * @author Marc Johnson (mjohnson at apache dot org)
 47:  */
 48: 
 49: public class POIFSReader
 50: {
 51:     private POIFSReaderRegistry registry;
 52:     private boolean             registryClosed;
 53: 
 54:     /**
 55:      * Create a POIFSReader
 56:      */
 57: 
 58:     public POIFSReader()
 59:     {
 60:         registry       = new POIFSReaderRegistry();
 61:         registryClosed = false;
 62:     }
 63: 
 64:     /**
 65:      * Read from an InputStream and process the documents we get
 66:      *
 67:      * @param stream the InputStream from which to read the data
 68:      *
 69:      * @exception IOException on errors reading, or on invalid data
 70:      */
 71: 
 72:     public void read(final InputStream stream)
 73:         throws IOException
 74:     {
 75:         registryClosed = true;
 76: 
 77:         // read the header block from the stream
 78:         HeaderBlock header_block = new HeaderBlock(stream);
 79: 
 80:         // read the rest of the stream into blocks
 81:         RawDataBlockList data_blocks = new RawDataBlockList(stream, header_block.getBigBlockSize());
 82: 
 83:         // set up the block allocation table (necessary for the
 84:         // data_blocks to be manageable
 85:         new BlockAllocationTableReader(header_block.getBigBlockSize(),
 86:                                        header_block.getBATCount(),
 87:                                        header_block.getBATArray(),
 88:                                        header_block.getXBATCount(),
 89:                                        header_block.getXBATIndex(),
 90:                                        data_blocks);
 91: 
 92:         // get property table from the document
 93:         PropertyTable properties =
 94:             new PropertyTable(header_block, data_blocks);
 95: 
 96:         // process documents
 97:         processProperties(SmallBlockTableReader
 98:             .getSmallDocumentBlocks(
 99:                   header_block.getBigBlockSize(),
100:                   data_blocks, properties.getRoot(), 
101:                   header_block.getSBATStart()), 
102:                   data_blocks, properties.getRoot()
103:                         .getChildren(), new POIFSDocumentPath());
104:     }
105: 
106:     /**
107:      * Register a POIFSReaderListener for all documents
108:      *
109:      * @param listener the listener to be registered
110:      *
111:      * @exception NullPointerException if listener is null
112:      * @exception IllegalStateException if read() has already been
113:      *                                  called
114:      */
115: 
116:     public void registerListener(final POIFSReaderListener listener)
117:     {
118:         if (listener == null)
119:         {
120:             throw new NullPointerException();
121:         }
122:         if (registryClosed)
123:         {
124:             throw new IllegalStateException();
125:         }
126:         registry.registerListener(listener);
127:     }
128: 
129:     /**
130:      * Register a POIFSReaderListener for a document in the root
131:      * directory
132:      *
133:      * @param listener the listener to be registered
134:      * @param name the document name
135:      *
136:      * @exception NullPointerException if listener is null or name is
137:      *                                 null or empty
138:      * @exception IllegalStateException if read() has already been
139:      *                                  called
140:      */
141: 
142:     public void registerListener(final POIFSReaderListener listener,
143:                                  final String name)
144:     {
145:         registerListener(listener, null, name);
146:     }
147: 
148:     /**
149:      * Register a POIFSReaderListener for a document in the specified
150:      * directory
151:      *
152:      * @param listener the listener to be registered
153:      * @param path the document path; if null, the root directory is
154:      *             assumed
155:      * @param name the document name
156:      *
157:      * @exception NullPointerException if listener is null or name is
158:      *                                 null or empty
159:      * @exception IllegalStateException if read() has already been
160:      *                                  called
161:      */
162: 
163:     public void registerListener(final POIFSReaderListener listener,
164:                                  final POIFSDocumentPath path,
165:                                  final String name)
166:     {
167:         if ((listener == null) || (name == null) || (name.length() == 0))
168:         {
169:             throw new NullPointerException();
170:         }
171:         if (registryClosed)
172:         {
173:             throw new IllegalStateException();
174:         }
175:         registry.registerListener(listener,
176:                                   (path == null) ? new POIFSDocumentPath()
177:                                                  : path, name);
178:     }

180:     /**
181:      * read in files
182:      *
183:      * @param args names of the files
184:      *
185:      * @exception IOException
186:      */
187: 
188:     public static void main(String args[])
189:         throws IOException
190:     {
191:         if (args.length == 0)
192:         {
193:             System.err
194:                 .println("at least one argument required: input filename(s)");
195:             System.exit(1);
196:         }
197: 
198:         // register for all
199:         for (int j = 0; j < args.length; j++)
200:         {
201:             POIFSReader         reader   = new POIFSReader();
202:             POIFSReaderListener listener = new SampleListener();
203: 
204:             reader.registerListener(listener);
205:             System.out.println("reading " + args[ j ]);
206:             FileInputStream istream = new FileInputStream(args[ j ]);
207: 
208:             reader.read(istream);
209:             istream.close();
210:         }
211:     }

213:     private void processProperties(final BlockList small_blocks,
214:                                    final BlockList big_blocks,
215:                                    final Iterator properties,
216:                                    final POIFSDocumentPath path)
217:         throws IOException
218:     {
219:         while (properties.hasNext())
220:         {
221:             Property property = ( Property ) properties.next();
222:             String   name     = property.getName();
223: 
224:             if (property.isDirectory())
225:             {
226:                 POIFSDocumentPath new_path = new POIFSDocumentPath(path,
227:                                                  new String[]
228:                 {
229:                     name
230:                 });
231: 
232:                 processProperties(
233:                     small_blocks, big_blocks,
234:                     (( DirectoryProperty ) property).getChildren(), new_path);
235:             }
236:             else
237:             {
238:                 int      startBlock = property.getStartBlock();
239:                 Iterator listeners  = registry.getListeners(path, name);
240: 
241:                 if (listeners.hasNext())
242:                 {
243:                     int           size     = property.getSize();
244:                     POIFSDocument document = null;
245: 
246:                     if (property.shouldUseSmallBlocks())
247:                     {
248:                         document =
249:                             new POIFSDocument(name, small_blocks
250:                                 .fetchBlocks(startBlock, -1), size);
251:                     }
252:                     else
253:                     {
254:                         document =
255:                             new POIFSDocument(name, big_blocks
256:                                 .fetchBlocks(startBlock, -1), size);
257:                     }
258:                     while (listeners.hasNext())
259:                     {
260:                         POIFSReaderListener listener =
261:                             ( POIFSReaderListener ) listeners.next();
262: 
263:                         listener.processPOIFSReaderEvent(
264:                             new POIFSReaderEvent(
265:                                 new DocumentInputStream(document), path,
266:                                 name));
267:                     }
268:                 }
269:                 else
270:                 {
271: 
272:                     // consume the document's data and discard it
273:                     if (property.shouldUseSmallBlocks())
274:                     {
275:                         small_blocks.fetchBlocks(startBlock, -1);
276:                     }
277:                     else
278:                     {
279:                         big_blocks.fetchBlocks(startBlock, -1);
280:                     }
281:                 }
282:             }
283:         }
284:     }

287:         implements POIFSReaderListener
288:     {
289: 
290:         /**
291:          * Constructor SampleListener
292:          */
293: 
294:         SampleListener()

298:         /**
299:          * Method processPOIFSReaderEvent
300:          *
301:          * @param event
302:          */
303: 
304:         public void processPOIFSReaderEvent(final POIFSReaderEvent event)
305:         {
306:             DocumentInputStream istream = event.getStream();
307:             POIFSDocumentPath   path    = event.getPath();
308:             String              name    = event.getName();
309: 
310:             try
311:             {
312:                 byte[] data = new byte[ istream.available() ];
313: 
314:                 istream.read(data);
315:                 int pathLength = path.length();
316: 
317:                 for (int k = 0; k < pathLength; k++)
318:                 {
319:                     System.out.print("/" + path.getComponent(k));
320:                 }
321:                 System.out.println("/" + name + ": " + data.length
322:                                    + " bytes read");
323:             }
324:             catch (IOException ignored)
325:             {
326:             }
327:         }
328:     }   // end private class SampleListener
329: }       // end public class POIFSReader
