 24: /**
 25:  * This class provides a wrapper over an OutputStream so that Document
 26:  * writers can't accidently go over their size limits
 27:  *
 28:  * @author Marc Johnson (mjohnson at apache dot org)
 29:  */
 30: 
 31: public final class DocumentOutputStream extends OutputStream {
 32:     private final OutputStream _stream;
 33:     private final int          _limit;
 34:     private int          _written;
 35: 
 36:     /**
 37:      * Create a DocumentOutputStream
 38:      *
 39:      * @param stream the OutputStream to which the data is actually
 40:      *               read
 41:      * @param limit the maximum number of bytes that can be written
 42:      */
 43:     DocumentOutputStream(OutputStream stream, int limit) {
 44:         _stream  = stream;
 45:         _limit   = limit;
 46:         _written = 0;
 47:     }
 48: 
 49:     /**
 50:      * Writes the specified byte to this output stream. The general
 51:      * contract for write is that one byte is written to the output
 52:      * stream. The byte to be written is the eight low-order bits of
 53:      * the argument b. The 24 high-order bits of b are ignored.
 54:      *
 55:      * @param b the byte.
 56:      * @exception IOException if an I/O error occurs. In particular,
 57:      *                        an IOException may be thrown if the
 58:      *                        output stream has been closed, or if the
 59:      *                        writer tries to write too much data.
 60:      */
 61:     public void write(int b)
 62:         throws IOException
 63:     {
 64:         limitCheck(1);
 65:         _stream.write(b);
 66:     }
 67: 
 68:     /**
 69:      * Writes b.length bytes from the specified byte array
 70:      * to this output stream.
 71:      *
 72:      * @param b the data.
 73:      * @exception IOException if an I/O error occurs.
 74:      */
 75:     public void write(byte b[])
 76:         throws IOException
 77:     {
 78:         write(b, 0, b.length);
 79:     }

 81:     /**
 82:      * Writes len bytes from the specified byte array starting at
 83:      * offset off to this output stream.  The general contract for
 84:      * write(b, off, len) is that some of the bytes in the array b are
 85:      * written to the output stream in order; element b[off] is the
 86:      * first byte written and b[off+len-1] is the last byte written by
 87:      * this operation.<p>
 88:      * If b is null, a NullPointerException is thrown.<p>
 89:      * If off is negative, or len is negative, or off+len is greater
 90:      * than the length of the array b, then an
 91:      * IndexOutOfBoundsException is thrown.
 92:      *
 93:      * @param b the data.
 94:      * @param off the start offset in the data.
 95:      * @param len the number of bytes to write.
 96:      * @exception IOException if an I/O error occurs. In particular,
 97:      *                        an IOException</code> is thrown if the
 98:      *                        output stream is closed or if the writer
 99:      *                        tries to write too many bytes.
100:      */
101:     public void write(byte b[], int off, int len)
102:         throws IOException
103:     {
104:         limitCheck(len);
105:         _stream.write(b, off, len);
106:     }
107: 
108:     /**
109:      * Flushes this output stream and forces any buffered output bytes

111:      *
112:      * @exception IOException if an I/O error occurs.
113:      */
114:     public void flush()
115:         throws IOException
116:     {
117:         _stream.flush();
118:     }

123:      * that it closes the output stream. A closed stream cannot
124:      * perform output operations and cannot be reopened.
125:      *
126:      * @exception IOException if an I/O error occurs.
127:      */
128:     public void close() {
129: 
130:         // ignore this call
131:     }
132: 
133:     /**
134:      * write the rest of the document's data (fill in at the end)
135:      *
136:      * @param totalLimit the actual number of bytes the corresponding
137:      *                   document must fill
138:      * @param fill the byte to fill remaining space with
139:      *
140:      * @exception IOException on I/O error
141:      */
142:     void writeFiller(int totalLimit, byte fill)
143:         throws IOException
144:     {
145:         if (totalLimit > _written)
146:         {
147:             byte[] filler = new byte[ totalLimit - _written ];
148: 
149:             Arrays.fill(filler, fill);
150:             _stream.write(filler);
151:         }
152:     }
153: 
154:     private void limitCheck(int toBeWritten)
155:         throws IOException
156:     {
157:         if ((_written + toBeWritten) > _limit)
158:         {
159:             throw new IOException("tried to write too much data");
160:         }
161:         _written += toBeWritten;
162:     }
