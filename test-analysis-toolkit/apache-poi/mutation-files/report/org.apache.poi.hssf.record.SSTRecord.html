 81:      * Add a string.
 82:      *
 83:      * @param string string to be added
 84:      *
 85:      * @return the index of that string in the table
 86:      */
 87:     public int addString(UnicodeString string)
 88:     {
 89:         field_1_num_strings++;
 90:         UnicodeString ucs = ( string == null ) ? EMPTY_STRING
 91:                 : string;
 92:         int rval;
 93:         int index = field_3_strings.getIndex(ucs);
 94: 
 95:         if ( index != -1 ) {
 96:             rval = index;
 97:         } else {
 98:             // This is a new string -- we didn't see it among the
 99:             // strings we've already collected
100:             rval = field_3_strings.size();
101:             field_2_num_unique_strings++;
102:             SSTDeserializer.addToStringTable( field_3_strings, ucs );
103:         }
104:         return rval;
105:     }

164:     /**
165:      * Fill the fields from the data
166:      * <P>
167:      * The data consists of sets of string data. This string data is
168:      * arranged as follows:
169:      * <P>
170:      * <CODE><pre>
171:      * short  string_length;   // length of string data
172:      * byte   string_flag;     // flag specifying special string
173:      *                         // handling
174:      * short  run_count;       // optional count of formatting runs
175:      * int    extend_length;   // optional extension length
176:      * char[] string_data;     // string data, can be byte[] or
177:      *                         // short[] (length of array is
178:      *                         // string_length)
179:      * int[]  formatting_runs; // optional formatting runs (length of
180:      *                         // array is run_count)
181:      * byte[] extension;       // optional extension (length of array
182:      *                         // is extend_length)
183:      * </pre></CODE>
184:      * <P>
185:      * The string_flag is bit mapped as follows:
186:      * <P>
187:      * <TABLE>
188:      *   <TR>
189:      *      <TH>Bit number</TH>
190:      *      <TH>Meaning if 0</TH>
191:      *      <TH>Meaning if 1</TH>
192:      *   <TR>
193:      *   <TR>
194:      *      <TD>0</TD>
195:      *      <TD>string_data is byte[]</TD>
196:      *      <TD>string_data is short[]</TH>
197:      *   <TR>
198:      *   <TR>
199:      *      <TD>1</TD>
200:      *      <TD>Should always be 0</TD>
201:      *      <TD>string_flag is defective</TH>
202:      *   <TR>
203:      *   <TR>
204:      *      <TD>2</TD>
205:      *      <TD>extension is not included</TD>
206:      *      <TD>extension is included</TH>
207:      *   <TR>
208:      *   <TR>
209:      *      <TD>3</TD>
210:      *      <TD>formatting run data is not included</TD>
211:      *      <TD>formatting run data is included</TH>
212:      *   <TR>
213:      *   <TR>
214:      *      <TD>4</TD>
215:      *      <TD>Should always be 0</TD>
216:      *      <TD>string_flag is defective</TH>
217:      *   <TR>
218:      *   <TR>
219:      *      <TD>5</TD>
220:      *      <TD>Should always be 0</TD>
221:      *      <TD>string_flag is defective</TH>
222:      *   <TR>
223:      *   <TR>
224:      *      <TD>6</TD>
225:      *      <TD>Should always be 0</TD>
226:      *      <TD>string_flag is defective</TH>
227:      *   <TR>
228:      *   <TR>
229:      *      <TD>7</TD>
230:      *      <TD>Should always be 0</TD>
231:      *      <TD>string_flag is defective</TH>
232:      *   <TR>
233:      * </TABLE>
234:      * <P>
235:      * We can handle eating the overhead associated with bits 2 or 3
236:      * (or both) being set, but we have no idea what to do with the
237:      * associated data. The UnicodeString class can handle the byte[]
238:      * vs short[] nature of the actual string data
239:      *
240:      * @param in the RecordInputstream to read the record from
241:      */
242:     public SSTRecord(RecordInputStream in) {
243:         // this method is ALWAYS called after construction -- using
244:         // the nontrivial constructor, of course -- so this is where
245:         // we initialize our fields
246:         field_1_num_strings = in.readInt();
247:         field_2_num_unique_strings = in.readInt();
248:         field_3_strings = new IntMapper<UnicodeString>();
249:         deserializer = new SSTDeserializer(field_3_strings);
250:         deserializer.manufactureStrings( field_2_num_unique_strings, in );
251:     }

306:     }
