 91:     private DirectoryNode(final DirectoryProperty property,
 92:                           final DirectoryNode parent,
 93:                           final POIFSFileSystem ofilesystem,
 94:                           final NPOIFSFileSystem nfilesystem)
 95:     {
 96:         super(property, parent);
 97:         this._ofilesystem = ofilesystem;
 98:         this._nfilesystem = nfilesystem;
 99: 
100:         if (parent == null)
101:         {
102:             _path = new POIFSDocumentPath();
103:         }
104:         else
105:         {
106:             _path = new POIFSDocumentPath(parent._path, new String[]
107:             {
108:                 property.getName()
109:             });
110:         }
111:         _byname     = new HashMap<String, Entry>();
112:         _entries    = new ArrayList<Entry>();
113:         Iterator<Property> iter = property.getChildren();
114: 
115:         while (iter.hasNext())
116:         {
117:             Property child     = iter.next();
118:             Entry    childNode = null;
119: 
120:             if (child.isDirectory())
121:             {
122:                 DirectoryProperty childDir = (DirectoryProperty) child;
123:                 if(_ofilesystem != null) {
124:                    childNode = new DirectoryNode(childDir, _ofilesystem, this);
125:                 } else {
126:                    childNode = new DirectoryNode(childDir, _nfilesystem, this);
127:                 }
128:             }
129:             else
130:             {
131:                 childNode = new DocumentNode((DocumentProperty) child, this);
132:             }
133:             _entries.add(childNode);
134:             _byname.put(childNode.getName(), childNode);
135:         }
136:     }
