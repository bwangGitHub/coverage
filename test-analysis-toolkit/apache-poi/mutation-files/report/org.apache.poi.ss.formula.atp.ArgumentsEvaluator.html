 31: /**
 32:  * Evaluator for formula arguments.
 33:  * 
 34:  * @author jfaenomoto@gmail.com
 35:  */
 36: final class ArgumentsEvaluator {
 37: 
 38:     public static final ArgumentsEvaluator instance = new ArgumentsEvaluator();
 39: 
 40:     private ArgumentsEvaluator() {
 41:         // enforces singleton
 42:     }
 43: 
 44:     /**
 45:      * Evaluate a generic {@link ValueEval} argument to a double value that represents a date in POI.
 46:      * 
 47:      * @param arg {@link ValueEval} an argument.
 48:      * @param srcCellRow number cell row.
 49:      * @param srcCellCol number cell column.
 50:      * @return a double representing a date in POI.
 51:      * @throws EvaluationException exception upon argument evaluation.
 52:      */
 53:     public double evaluateDateArg(ValueEval arg, int srcCellRow, int srcCellCol) throws EvaluationException {
 54:         ValueEval ve = OperandResolver.getSingleValue(arg, srcCellRow, (short) srcCellCol);
 55: 
 56:         if (ve instanceof StringEval) {
 57:             String strVal = ((StringEval) ve).getStringValue();
 58:             Double dVal = OperandResolver.parseDouble(strVal);
 59:             if (dVal != null) {
 60:                 return dVal.doubleValue();
 61:             }
 62:             Calendar date = DateParser.parseDate(strVal);
 63:             return DateUtil.getExcelDate(date, false);
 64:         }
 65:         return OperandResolver.coerceValueToDouble(ve);
 66:     }
 67: 
 68:     /**
 69:      * Evaluate a generic {@link ValueEval} argument to an array of double values that represents dates in POI.
 70:      * 
 71:      * @param arg {@link ValueEval} an argument.
 72:      * @param srcCellRow number cell row.
 73:      * @param srcCellCol number cell column.
 74:      * @return an array of doubles representing dates in POI.
 75:      * @throws EvaluationException exception upon argument evaluation.
 76:      */
 77:     public double[] evaluateDatesArg(ValueEval arg, int srcCellRow, int srcCellCol) throws EvaluationException {
 78:         if (arg == null) {
 79:             return new double[0];
 80:         }
 81: 
 82:         if (arg instanceof StringEval) {
 83:             return new double[]{ evaluateDateArg(arg, srcCellRow, srcCellCol) };
 84:         } else if (arg instanceof AreaEvalBase) {
 85:             List<Double> valuesList = new ArrayList<Double>();
 86:             AreaEvalBase area = (AreaEvalBase) arg;
 87:             for (int i = area.getFirstRow(); i <= area.getLastRow(); i++) {
 88:                 for (int j = area.getFirstColumn(); j <= area.getLastColumn(); j++) {
 89:                     valuesList.add(evaluateDateArg(area.getValue(i, j), i, j));
 90:                 }
 91:             }
 92:             double[] values = new double[valuesList.size()];
 93:             for (int i = 0; i < valuesList.size(); i++) {
 94:                 values[i] = valuesList.get(i).doubleValue();
 95:             }
 96:             return values;
 97:         }
 98:         return new double[]{ OperandResolver.coerceValueToDouble(arg) };
 99:     }
100: 
101:     /**
102:      * Evaluate a generic {@link ValueEval} argument to a double value.
103:      * 
104:      * @param arg {@link ValueEval} an argument.
105:      * @param srcCellRow number cell row.
106:      * @param srcCellCol number cell column.
107:      * @return a double value.
108:      * @throws EvaluationException exception upon argument evaluation.
109:      */
110:     public double evaluateNumberArg(ValueEval arg, int srcCellRow, int srcCellCol) throws EvaluationException {
111:         if (arg == null) {
112:             return 0f;
113:         }
114: 
115:         return OperandResolver.coerceValueToDouble(arg);
116:     }
117: }
