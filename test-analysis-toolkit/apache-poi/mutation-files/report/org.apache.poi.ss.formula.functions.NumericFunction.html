399:         /**
400:          * This checks is x = 0 and the mean = 0.
401:          * Excel currently returns the value 1 where as the
402:          * maths common implementation will error.
403:          * @param x  The number.
404:          * @param mean The mean.
405:          * @return If a default value should be returned.
406:          */
407:         private boolean isDefaultResult(double x, double mean) {
408: 
409:             if ( x == 0 && mean == 0 ) {
410:                 return true;
411:             }
412:             return false;
413:         }
414: 
415:         private boolean checkArgument(double aDouble) throws EvaluationException {
416: 
417:             NumericFunction.checkValue(aDouble);
418: 
419:             // make sure that the number is positive
420:             if (aDouble < 0) {
421:                 throw new EvaluationException(ErrorEval.NUM_ERROR);
422:             }
423:             
424:             return true;
425:         }

427:         private double probability(int k, double lambda) {
428:             return Math.pow(lambda, k) * Math.exp(-lambda) / factorial(k);
429:         }
430: 
431:         private double cumulativeProbability(int x, double lambda) {
432:             double result = 0;
433:             for(int k = 0; k <= x; k++){
434:                 result += probability(k, lambda);
435:             }
436:             return result;
437:         }

441:                            1l,                  1l,                   2l,

444:                       362880l,            3628800l,            39916800l,
