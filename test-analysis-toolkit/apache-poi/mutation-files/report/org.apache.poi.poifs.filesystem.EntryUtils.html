 32:     /**
 33:      * Copies an Entry into a target POIFS directory, recursively
 34:      */
 35:     @Internal
 36:     public static void copyNodeRecursively( Entry entry, DirectoryEntry target )
 37:             throws IOException
 38:     {
 39:         // System.err.println("copyNodeRecursively called with "+entry.getName()+
 40:         // ","+target.getName());
 41:         DirectoryEntry newTarget = null;
 42:         if ( entry.isDirectoryEntry() )
 43:         {
 44:             newTarget = target.createDirectory( entry.getName() );
 45:             Iterator<Entry> entries = ( (DirectoryEntry) entry ).getEntries();
 46: 
 47:             while ( entries.hasNext() )
 48:             {
 49:                 copyNodeRecursively( entries.next(), newTarget );
 50:             }
 51:         }
 52:         else
 53:         {
 54:             DocumentEntry dentry = (DocumentEntry) entry;
 55:             DocumentInputStream dstream = new DocumentInputStream( dentry );
 56:             target.createDocument( dentry.getName(), dstream );
 57:             dstream.close();
 58:         }
 59:     }

 61:     /**
 62:      * Copies all the nodes from one POIFS Directory to another
 63:      * 
 64:      * @param sourceRoot
 65:      *            is the source Directory to copy from
 66:      * @param targetRoot
 67:      *            is the target Directory to copy to
 68:      */
 69:     public static void copyNodes(DirectoryEntry sourceRoot,
 70:             DirectoryEntry targetRoot) throws IOException
 71:     {
 72:         for (Entry entry : sourceRoot) {
 73:             copyNodeRecursively( entry, targetRoot );

112:                 copyNodeRecursively( entry, targetRoot );
113:             }

150:     /**
151:      * Checks to see if the two Directories hold the same contents.
152:      * For this to be true, they must have entries with the same names,
153:      *  no entries in one but not the other, and the size+contents
154:      *  of each entry must match, and they must share names.
155:      * To exclude certain parts of the Directory from being checked,
156:      *  use a {@link FilteringDirectoryNode}
157:      */
158:     public static boolean areDirectoriesIdentical(DirectoryEntry dirA, DirectoryEntry dirB) {
159:        // First, check names
160:        if (! dirA.getName().equals(dirB.getName())) {
161:           return false;
162:        }
163:        
164:        // Next up, check they have the same number of children
165:        if (dirA.getEntryCount() != dirB.getEntryCount()) {
166:           return false;
167:        }
168:        
169:        // Next, check entries and their types/sizes
170:        Map<String,Integer> aSizes = new HashMap<String, Integer>();
171:        final int isDirectory = -12345; 
172:        for (Entry a : dirA) {
173:           String aName = a.getName();
174:           if (a.isDirectoryEntry()) {
175:              aSizes.put(aName, isDirectory);
176:           } else {
177:              aSizes.put(aName, ((DocumentNode)a).getSize());
178:           }
179:        }
180:        for (Entry b : dirB) {
181:           String bName = b.getName();
182:           if (! aSizes.containsKey(bName)) {
183:              // In B but not A
184:              return false;
185:           }
186:           
187:           int size;
188:           if (b.isDirectoryEntry()) {
189:              size = isDirectory;
190:           } else {
191:              size = ((DocumentNode)b).getSize();
192:           }
193:           if (size != aSizes.get(bName)) {
194:              // Either the wrong type, or they're different sizes
195:              return false;
196:           }
197:           
198:           // Track it as checked
199:           aSizes.remove(bName);
200:        }
201:        if (!aSizes.isEmpty()) {
202:           // Nodes were in A but not B
203:           return false;
204:        }
205:        
206:        // If that passed, check entry contents
207:        for (Entry a : dirA) {
208:           try {
209:              Entry b = dirB.getEntry(a.getName());
210:              boolean match;
211:              if (a.isDirectoryEntry()) {
212:                 match = areDirectoriesIdentical(
213:                       (DirectoryEntry)a, (DirectoryEntry)b);
214:              } else {
215:                 match = areDocumentsIdentical(
216:                       (DocumentEntry)a, (DocumentEntry)b);
217:              }
218:              if (!match) return false;
219:           } catch(FileNotFoundException e) {
220:              // Shouldn't really happen...
221:              return false;
222:           } catch(IOException e) {
223:              // Something's messed up with one document, not a match
224:              return false;
225:           }
226:        }
227:        
228:        // If we get here, they match!
229:        return true;
230:     }

232:     /**
233:      * Checks to see if two Documents have the same name
234:      *  and the same contents. (Their parent directories are
235:      *  not checked)
236:      */
237:     public static boolean areDocumentsIdentical(DocumentEntry docA, DocumentEntry docB) throws IOException {
238:        if (! docA.getName().equals(docB.getName())) {
239:           // Names don't match, not the same
240:           return false;
241:        }
242:        if (docA.getSize() != docB.getSize()) {
243:           // Wrong sizes, can't have the same contents
244:           return false;
245:        }
246: 
247:        boolean matches = true;
248:        DocumentInputStream inpA = null, inpB = null;
249:        try {
250:           inpA = new DocumentInputStream(docA);
251:           inpB = new DocumentInputStream(docB);
252:           
253:           int readA, readB;
254:           do {
255:              readA = inpA.read();
256:              readB = inpB.read();
257:              if (readA != readB) {
258:                 matches = false;
259:                 break;
260:              }
261:           } while(readA != -1 && readB != -1);
262:        } finally {
263:           if (inpA != null) inpA.close();
264:           if (inpB != null) inpB.close();
265:        }
266:        
267:        return matches;
268:     }
