 24:  * Title:        DBCell Record (0x00D7)<p/>
 25:  * Description:  Used by Excel and other MS apps to quickly find rows in the sheets.<P>

 30: public final class DBCellRecord extends StandardRecord {
 31:     public final static short sid = 0x00D7;
 32:     public final static int BLOCK_SIZE = 32;
 33:     
 34:     public static final class Builder {
 35:         private short[] _cellOffsets;
 36:         private int _nCellOffsets;
 37:         public Builder() {
 38:         	_cellOffsets = new short[4];
 39: 		}
 40: 
 41:         public void addCellOffset(int cellRefOffset) {
 42:             if (_cellOffsets.length <= _nCellOffsets) {
 43:                 short[] temp = new short[_nCellOffsets * 2];
 44:                 System.arraycopy(_cellOffsets, 0, temp, 0, _nCellOffsets);
 45:                 _cellOffsets = temp;
 46:             }
 47:             _cellOffsets[_nCellOffsets] = (short) cellRefOffset;
 48:             _nCellOffsets++;
 49:         }
 50: 
 51:         public DBCellRecord build(int rowOffset) {
 52:             short[] cellOffsets = new short[_nCellOffsets];
 53:             System.arraycopy(_cellOffsets, 0, cellOffsets, 0, _nCellOffsets);
 54:             return new DBCellRecord(rowOffset, cellOffsets);
 55:         }
 56:     }
 57:     /**
 58:      * offset from the start of this DBCellRecord to the start of the first cell in
 59:      * the next DBCell block.
 60:      */
 61:     private final int     field_1_row_offset;
 62:     private final short[] field_2_cell_offsets;
 63: 
 64:     DBCellRecord(int rowOffset, short[]cellOffsets) {
 65:         field_1_row_offset = rowOffset;
 66:         field_2_cell_offsets = cellOffsets;
 67:     }
 68: 
 69:     public DBCellRecord(RecordInputStream in) {
 70:         field_1_row_offset   = in.readUShort();
 71:         int size = in.remaining();        
 72:         field_2_cell_offsets = new short[ size / 2 ];
 73: 
 74:         for (int i=0;i<field_2_cell_offsets.length;i++)
 75:         {
 76:             field_2_cell_offsets[ i ] = in.readShort();
 77:         }
 78:     }
 79: 
 80: 
 81:     public String toString() {
 82:         StringBuffer buffer = new StringBuffer();
 83: 
 84:         buffer.append("[DBCELL]\n");
 85:         buffer.append("    .rowoffset = ").append(HexDump.intToHex(field_1_row_offset)).append("\n");
 86:         for (int k = 0; k < field_2_cell_offsets.length; k++) {
 87:             buffer.append("    .cell_").append(k).append(" = ")
 88:                 .append(HexDump.shortToHex(field_2_cell_offsets[ k ])).append("\n");
 89:         }
 90:         buffer.append("[/DBCELL]\n");
 91:         return buffer.toString();
 92:     }
 93: 
 94:     public void serialize(LittleEndianOutput out) {
 95:         out.writeInt(field_1_row_offset);
 96:         for (int k = 0; k < field_2_cell_offsets.length; k++) {
 97:             out.writeShort(field_2_cell_offsets[ k ]);
 98:         }
 99:     }
100:     protected int getDataSize() {
101:         return 4 + field_2_cell_offsets.length * 2;
102:     }
103:     
104:     /**
105:      *  @return the size of the group of <tt>DBCellRecord</tt>s needed to encode
106:      *  the specified number of blocks and rows
107:      */
108:     public static int calculateSizeOfRecords(int nBlocks, int nRows) {
109:         // One DBCell per block.
110:         // 8 bytes per DBCell (non variable section)
111:         // 2 bytes per row reference
112:         return nBlocks * 8 + nRows * 2;
113:     }
114: 
115:     public short getSid() {

117:     }
