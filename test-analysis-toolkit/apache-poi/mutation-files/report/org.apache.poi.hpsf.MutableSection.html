336:     /**
337:      * <p>Calculates the section's size. It is the sum of the lengths of the
338:      * section's header (8), the properties list (16 times the number of
339:      * properties) and the properties themselves.</p>
340:      *

342:      * @throws WritingNotSupportedException
343:      * @throws IOException

348:         write(out);

358:     /**
359:      * <p>Writes this section into an output stream.</p>
360:      *
361:      * <p>Internally this is done by writing into three byte array output
362:      * streams: one for the properties, one for the property list and one for
363:      * the section as such. The two former are appended to the latter when they
364:      * have received all their data.</p>
365:      *
366:      * @param out The stream to write into.
367:      *
368:      * @return The number of bytes written, i.e. the section's size.
369:      * @exception IOException if an I/O error occurs
370:      * @exception WritingNotSupportedException if HPSF does not yet support
371:      * writing a property's variant type.
372:      */
373:     public int write(final OutputStream out)
374:         throws WritingNotSupportedException, IOException
375:     {
376:         /* Check whether we have already generated the bytes making out the
377:          * section. */
378:         if (!dirty && sectionBytes != null)
379:         {
380:             out.write(sectionBytes);
381:             return sectionBytes.length;
382:         }
383: 
384:         /* The properties are written to this stream. */
385:         final ByteArrayOutputStream propertyStream =
386:             new ByteArrayOutputStream();
387: 
388:         /* The property list is established here. After each property that has
389:          * been written to "propertyStream", a property list entry is written to
390:          * "propertyListStream". */
391:         final ByteArrayOutputStream propertyListStream =
392:             new ByteArrayOutputStream();
393: 
394:         /* Maintain the current position in the list. */
395:         int position = 0;
396: 
397:         /* Increase the position variable by the size of the property list so
398:          * that it points behind the property list and to the beginning of the
399:          * properties themselves. */
400:         position += 2 * LittleEndian.INT_SIZE +
401:                     getPropertyCount() * 2 * LittleEndian.INT_SIZE;
402: 
403:         /* Writing the section's dictionary it tricky. If there is a dictionary
404:          * (property 0) the codepage property (property 1) must be set, too. */
405:         int codepage = -1;
406:         if (getProperty(PropertyIDMap.PID_DICTIONARY) != null)
407:         {
408:             final Object p1 = getProperty(PropertyIDMap.PID_CODEPAGE);
409:             if (p1 != null)
410:             {
411:                 if (!(p1 instanceof Integer))
412:                     throw new IllegalPropertySetDataException
413:                         ("The codepage property (ID = 1) must be an " +
414:                          "Integer object.");
415:             }
416:             else
417:                 /* Warning: The codepage property is not set although a
418:                  * dictionary is present. In order to cope with this problem we
419:                  * add the codepage property and set it to Unicode. */
420:                 setProperty(PropertyIDMap.PID_CODEPAGE, Variant.VT_I2,
421:                             Integer.valueOf(Constants.CP_UNICODE));
422:             codepage = getCodepage();
423:         }
424: 
425:         /* Sort the property list by their property IDs: */
426:         Collections.sort(preprops, new Comparator<Property>()
427:             {
428:                 public int compare(final Property p1, final Property p2)
429:                 {
430:                     if (p1.getID() < p2.getID())
431:                         return -1;
432:                     else if (p1.getID() == p2.getID())
433:                         return 0;
434:                     else
435:                         return 1;
436:                 }
437:             });
438: 
439:         /* Write the properties and the property list into their respective
440:          * streams: */
441:         for (final ListIterator<Property> i = preprops.listIterator(); i.hasNext();)
442:         {
443:             final MutableProperty p = (MutableProperty) i.next();
444:             final long id = p.getID();
445: 
446:             /* Write the property list entry. */
447:             TypeWriter.writeUIntToStream(propertyListStream, p.getID());
448:             TypeWriter.writeUIntToStream(propertyListStream, position);
449: 
450:             /* If the property ID is not equal 0 we write the property and all
451:              * is fine. However, if it equals 0 we have to write the section's
452:              * dictionary which has an implicit type only and an explicit
453:              * value. */
454:             if (id != 0)
455:                 /* Write the property and update the position to the next
456:                  * property. */
457:                 position += p.write(propertyStream, getCodepage());
458:             else
459:             {
460:                 if (codepage == -1)
461:                     throw new IllegalPropertySetDataException
462:                         ("Codepage (property 1) is undefined.");
463:                 position += writeDictionary(propertyStream, dictionary,
464:                                             codepage);
465:             }
466:         }
467:         propertyStream.close();
468:         propertyListStream.close();
469: 
470:         /* Write the section: */
471:         byte[] pb1 = propertyListStream.toByteArray();
472:         byte[] pb2 = propertyStream.toByteArray();
473: 
474:         /* Write the section's length: */
475:         TypeWriter.writeToStream(out, LittleEndian.INT_SIZE * 2 +
476:                                       pb1.length + pb2.length);
477: 
478:         /* Write the section's number of properties: */
479:         TypeWriter.writeToStream(out, getPropertyCount());
480: 
481:         /* Write the property list: */
482:         out.write(pb1);
483: 
484:         /* Write the properties: */
485:         out.write(pb2);
486: 
487:         int streamLength = LittleEndian.INT_SIZE * 2 + pb1.length + pb2.length;
488:         return streamLength;
489:     }

493:     /**
494:      * <p>Writes the section's dictionary.</p>
495:      *
496:      * @param out The output stream to write to.
497:      * @param dictionary The dictionary.
498:      * @param codepage The codepage to be used to write the dictionary items.
499:      * @return The number of bytes written
500:      * @exception IOException if an I/O exception occurs.
501:      */
502:     private static int writeDictionary(final OutputStream out,
503:                                        final Map<Long,String> dictionary, final int codepage)
504:         throws IOException
505:     {
506:         int length = TypeWriter.writeUIntToStream(out, dictionary.size());
507:         for (final Iterator<Long> i = dictionary.keySet().iterator(); i.hasNext();)
508:         {
509:             final Long key = i.next();
510:             final String value = dictionary.get(key);
511: 
512:             if (codepage == Constants.CP_UNICODE)
513:             {
514:                 /* Write the dictionary item in Unicode. */
515:                 int sLength = value.length() + 1;
516:                 if (sLength % 2 == 1)
517:                     sLength++;
518:                 length += TypeWriter.writeUIntToStream(out, key.longValue());
519:                 length += TypeWriter.writeUIntToStream(out, sLength);
520:                 final byte[] ca =
521:                     value.getBytes(VariantSupport.codepageToEncoding(codepage));
522:                 for (int j = 2; j < ca.length; j += 2)
523:                 {
524:                     out.write(ca[j+1]);
525:                     out.write(ca[j]);
526:                     length += 2;
527:                 }
528:                 sLength -= value.length();
529:                 while (sLength > 0)
530:                 {
531:                     out.write(0x00);
532:                     out.write(0x00);
533:                     length += 2;
534:                     sLength--;
535:                 }
536:             }
537:             else
538:             {
539:                 /* Write the dictionary item in another codepage than
540:                  * Unicode. */
541:                 length += TypeWriter.writeUIntToStream(out, key.longValue());
542:                 length += TypeWriter.writeUIntToStream(out, value.length() + 1);
543:                 final byte[] ba =
544:                     value.getBytes(VariantSupport.codepageToEncoding(codepage));
545:                 for (int j = 0; j < ba.length; j++)
546:                 {
547:                     out.write(ba[j]);
548:                     length++;
549:                 }
550:                 out.write(0x00);
551:                 length++;
552:             }
553:         }
554:         return length;
555:     }

602:     /**
603:      * <p>Sets the section's dictionary. All keys in the dictionary must be
604:      * {@link java.lang.Long} instances, all values must be
605:      * {@link java.lang.String}s. This method overwrites the properties with IDs
606:      * 0 and 1 since they are reserved for the dictionary and the dictionary's
607:      * codepage. Setting these properties explicitly might have surprising
608:      * effects. An application should never do this but always use this
609:      * method.</p>
610:      *
611:      * @param dictionary The dictionary
612:      *
613:      * @exception IllegalPropertySetDataException if the dictionary's key and
614:      * value types are not correct.
615:      *
616:      * @see Section#getDictionary()
617:      */
618:     public void setDictionary(final Map<Long,String> dictionary)
619:         throws IllegalPropertySetDataException
620:     {
621:         if (dictionary != null)
622:         {
623:             this.dictionary = dictionary;
624: 
625:             /* Set the dictionary property (ID 0). Please note that the second
626:              * parameter in the method call below is unused because dictionaries
627:              * don't have a type. */
628:             setProperty(PropertyIDMap.PID_DICTIONARY, -1, dictionary);
629: 
630:             /* If the codepage property (ID 1) for the strings (keys and
631:              * values) used in the dictionary is not yet defined, set it to
632:              * Unicode. */
633:             final Integer codepage =
634:                 (Integer) getProperty(PropertyIDMap.PID_CODEPAGE);
635:             if (codepage == null)
636:                 setProperty(PropertyIDMap.PID_CODEPAGE, Variant.VT_I2,
637:                             Integer.valueOf(Constants.CP_UNICODE));
638:         }
639:         else
640:             /* Setting the dictionary to null means to remove property 0.
641:              * However, it does not mean to remove property 1 (codepage). */
642:             removeProperty(PropertyIDMap.PID_DICTIONARY);
643:     }
