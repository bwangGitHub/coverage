 28:     // convenient access to namespace
 29:     private static final ErrorConstants EC = null;

 31:     /** <b>#NULL!</b>  - Intersection of two cell ranges is empty */
 32:     public static final ErrorEval NULL_INTERSECTION = new ErrorEval(EC.ERROR_NULL);
 33:     /** <b>#DIV/0!</b> - Division by zero */
 34:     public static final ErrorEval DIV_ZERO = new ErrorEval(EC.ERROR_DIV_0);
 35:     /** <b>#VALUE!</b> - Wrong type of operand */
 36:     public static final ErrorEval VALUE_INVALID = new ErrorEval(EC.ERROR_VALUE);
 37:     /** <b>#REF!</b> - Illegal or deleted cell reference */
 38:     public static final ErrorEval REF_INVALID = new ErrorEval(EC.ERROR_REF);
 39:     /** <b>#NAME?</b> - Wrong function or range name */
 40:     public static final ErrorEval NAME_INVALID = new ErrorEval(EC.ERROR_NAME);
 41:     /** <b>#NUM!</b> - Value range overflow */
 42:     public static final ErrorEval NUM_ERROR = new ErrorEval(EC.ERROR_NUM);
 43:     /** <b>#N/A</b> - Argument or function not available */
 44:     public static final ErrorEval NA = new ErrorEval(EC.ERROR_NA);
 45: 
 46: 
 47:     // POI internal error codes
 48:     private static final int CIRCULAR_REF_ERROR_CODE = 0xFFFFFFC4;
 49:     private static final int FUNCTION_NOT_IMPLEMENTED_CODE = 0xFFFFFFE2;
 50: 
 51:     // Note - Excel does not seem to represent this condition with an error code
 52:     public static final ErrorEval CIRCULAR_REF_ERROR = new ErrorEval(CIRCULAR_REF_ERROR_CODE);
 53: 
 54: 
 55:     /**
 56:      * Translates an Excel internal error code into the corresponding POI ErrorEval instance
 57:      * @param errorCode
 58:      */
 59:     public static ErrorEval valueOf(int errorCode) {
 60:         switch(errorCode) {
 61:             case ErrorConstants.ERROR_NULL:  return NULL_INTERSECTION;
 62:             case ErrorConstants.ERROR_DIV_0: return DIV_ZERO;
 63:             case ErrorConstants.ERROR_VALUE: return VALUE_INVALID;
 64:             case ErrorConstants.ERROR_REF:   return REF_INVALID;
 65:             case ErrorConstants.ERROR_NAME:  return NAME_INVALID;
 66:             case ErrorConstants.ERROR_NUM:   return NUM_ERROR;
 67:             case ErrorConstants.ERROR_NA:    return NA;
 68:             // non-std errors (conditions modeled as errors by POI)
 69:             case CIRCULAR_REF_ERROR_CODE:        return CIRCULAR_REF_ERROR;
 70:         }
 71:         throw new RuntimeException("Unexpected error code (" + errorCode + ")");
 72:     }
 73: 
 74:     /**
 75:      * Converts error codes to text.  Handles non-standard error codes OK.  
 76:      * For debug/test purposes (and for formatting error messages).
 77:      * @return the String representation of the specified Excel error code.
 78:      */
 79:     public static String getText(int errorCode) {
 80:         if(ErrorConstants.isValidCode(errorCode)) {
 81:             return ErrorConstants.getText(errorCode);
 82:         }
 83:         // It is desirable to make these (arbitrary) strings look clearly different from any other
 84:         // value expression that might appear in a formula.  In addition these error strings should
 85:         // look unlike the standard Excel errors.  Hence tilde ('~') was used.
 86:         switch(errorCode) {
 87:             case CIRCULAR_REF_ERROR_CODE: return "~CIRCULAR~REF~";
 88:             case FUNCTION_NOT_IMPLEMENTED_CODE: return "~FUNCTION~NOT~IMPLEMENTED~";
 89:         }
 90:         return "~non~std~err(" + errorCode + ")~";
 91:     }
 92: 
 93:     private int _errorCode;
 94:     /**
 95:      * @param errorCode an 8-bit value
 96:      */
 97:     private ErrorEval(int errorCode) {
 98:         _errorCode = errorCode;
 99:     }
100: 
101:     public int getErrorCode() {
102:         return _errorCode;
103:     }

109:         return sb.toString();
110:     }
