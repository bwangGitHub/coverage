 23: /**
 24:  * <tt>RecordAggregate</tt>s are groups of of BIFF <tt>Record</tt>s that are typically stored
 25:  * together and/or updated together.  Workbook / Sheet records are typically stored in a sequential
 26:  * list, which does not provide much structure to coordinate updates.
 27:  *
 28:  * @author Josh Micich
 29:  */
 30: public abstract class RecordAggregate extends RecordBase {
 31: 
 32: 	/**
 33: 	 * Visit each of the atomic BIFF records contained in this {@link RecordAggregate} in the order
 34: 	 * that they should be written to file.  Implementors may or may not return the actual
 35: 	 * {@link Record}s being used to manage POI's internal implementation.  Callers should not
 36: 	 * assume either way, and therefore only attempt to modify those {@link Record}s after cloning
 37: 	 */
 38: 	public abstract void visitContainedRecords(RecordVisitor rv);
 39: 
 40: 	public final int serialize(int offset, byte[] data) {
 41: 		SerializingRecordVisitor srv = new SerializingRecordVisitor(data, offset);
 42: 		visitContainedRecords(srv);
 43: 		return srv.countBytesWritten();
 44: 	}
 45: 	public int getRecordSize() {
 46: 		RecordSizingVisitor rsv = new RecordSizingVisitor();
 47: 		visitContainedRecords(rsv);
 48: 		return rsv.getTotalSize();
 49: 	}

 60: 
 61: 		private final byte[] _data;
 62: 		private final int _startOffset;
 63: 		private int _countBytesWritten;
 64: 
 65: 		public SerializingRecordVisitor(byte[] data, int startOffset) {
 66: 			_data = data;
