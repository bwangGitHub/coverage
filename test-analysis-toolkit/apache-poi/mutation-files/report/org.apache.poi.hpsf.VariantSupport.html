 90:     protected static void writeUnsupportedTypeMessage
 91:         (final UnsupportedVariantTypeException ex)
 92:     {
 93:         if (isLogUnsupportedTypes())
 94:         {
 95:             if (unsupportedMessage == null)
 96:                 unsupportedMessage = new LinkedList();
 97:             Long vt = Long.valueOf(ex.getVariantType());
 98:             if (!unsupportedMessage.contains(vt))
 99:             {
100:                 System.err.println(ex.getMessage());
101:                 unsupportedMessage.add(vt);
102:             }
103:         }

389:     /**
390:      * <p>Writes a variant value to an output stream. This method ensures that
391:      * always a multiple of 4 bytes is written.</p>
392:      *
393:      * <p>If the codepage is UTF-16, which is encouraged, strings
394:      * <strong>must</strong> always be written as {@link Variant#VT_LPWSTR}
395:      * strings, not as {@link Variant#VT_LPSTR} strings. This method ensure this
396:      * by converting strings appropriately, if needed.</p>
397:      *
398:      * @param out The stream to write the value to.
399:      * @param type The variant's type.
400:      * @param value The variant's value.
401:      * @param codepage The codepage to use to write non-wide strings
402:      * @return The number of entities that have been written. In many cases an
403:      * "entity" is a byte but this is not always the case.
404:      * @exception IOException if an I/O exceptions occurs
405:      * @exception WritingNotSupportedException if a property is to be written
406:      * who's variant type HPSF does not yet support
407:      */
408:     public static int write(final OutputStream out, final long type,
409:                             final Object value, final int codepage)
410:         throws IOException, WritingNotSupportedException
411:     {
412:         int length = 0;
413:         switch ((int) type)
414:         {
415:             case Variant.VT_BOOL:
416:             {
417:                 if ( ( (Boolean) value ).booleanValue() )
418:                 {
419:                     out.write( 0xff );
420:                     out.write( 0xff );
421:                 }
422:                 else
423:                 {
424:                     out.write( 0x00 );
425:                     out.write( 0x00 );
426:                 }
427:                 length += 2;
428:                 break;
429:             }
430:             case Variant.VT_LPSTR:
431:             {
432:                 CodePageString codePageString = new CodePageString( (String) value,
433:                         codepage );
434:                 length += codePageString.write( out );
435:                 break;
436:             }
437:             case Variant.VT_LPWSTR:
438:             {
439:                 final int nrOfChars = ( (String) value ).length() + 1;
440:                 length += TypeWriter.writeUIntToStream( out, nrOfChars );
441:                 char[] s = ( (String) value ).toCharArray();
442:                 for ( int i = 0; i < s.length; i++ )
443:                 {
444:                     final int high = ( ( s[i] & 0x0000ff00 ) >> 8 );
445:                     final int low = ( s[i] & 0x000000ff );
446:                     final byte highb = (byte) high;
447:                     final byte lowb = (byte) low;
448:                     out.write( lowb );
449:                     out.write( highb );
450:                     length += 2;
451:                 }
452:                 // NullTerminator
453:                 out.write( 0x00 );
454:                 out.write( 0x00 );
455:                 length += 2;
456:                 break;
457:             }
458:             case Variant.VT_CF:
459:             {
460:                 final byte[] b = (byte[]) value;
461:                 out.write(b);
462:                 length = b.length;
463:                 break;
464:             }
465:             case Variant.VT_EMPTY:
466:             {
467:                 length += TypeWriter.writeUIntToStream( out, Variant.VT_EMPTY );
468:                 break;
469:             }
470:             case Variant.VT_I2:
471:             {
472:                 length += TypeWriter.writeToStream( out,
473:                         ( (Integer) value ).shortValue() );
474:                 break;
475:             }
476:             case Variant.VT_I4:
477:             {
478:                 if (!(value instanceof Integer))
479:                 {
480:                     throw new ClassCastException("Could not cast an object to "
481:                             + Integer.class.toString() + ": "
482:                             + value.getClass().toString() + ", "
483:                             + value.toString());
484:                 }
485:                 length += TypeWriter.writeToStream(out,
486:                           ((Integer) value).intValue());
487:                 break;
488:             }
489:             case Variant.VT_I8:
490:             {
491:                 length += TypeWriter.writeToStream(out, ((Long) value).longValue());
492:                 break;
493:             }
494:             case Variant.VT_R8:
495:             {
496:                 length += TypeWriter.writeToStream(out,
497:                           ((Double) value).doubleValue());
498:                 break;
499:             }
500:             case Variant.VT_FILETIME:
501:             {
502:                 long filetime = Util.dateToFileTime((Date) value);
503:                 int high = (int) ((filetime >> 32) & 0x00000000FFFFFFFFL);
504:                 int low = (int) (filetime & 0x00000000FFFFFFFFL);
505:                 Filetime filetimeValue = new Filetime( low, high);
506:                 length += filetimeValue.write( out );
507:                 break;
508:             }
509:             default:
510:             {
511:                 /* The variant type is not supported yet. However, if the value
512:                  * is a byte array we can write it nevertheless. */
513:                 if (value instanceof byte[])
514:                 {
515:                     final byte[] b = (byte[]) value;
516:                     out.write(b);
517:                     length = b.length;
518:                     writeUnsupportedTypeMessage
519:                         (new WritingNotSupportedException(type, value));
520:                 }
521:                 else
522:                     throw new WritingNotSupportedException(type, value);
523:                 break;
524:             }
525:         }
526: 
527:         /* pad values to 4-bytes */
528:         while ( ( length & 0x3 ) != 0 )
529:         {
530:             out.write( 0x00 );
531:             length++;
532:         }
533: 
534:         return length;
535:     }
