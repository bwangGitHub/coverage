 32: /**
 33:  * This class implements storage for writing the small blocks used by
 34:  * small documents.
 35:  *
 36:  * @author Marc Johnson (mjohnson at apache dot org)
 37:  */
 38: 
 39: public class SmallBlockTableWriter
 40:     implements BlockWritable, BATManaged
 41: {
 42:     private BlockAllocationTableWriter _sbat;
 43:     private List                       _small_blocks;
 44:     private int                        _big_block_count;
 45:     private RootProperty               _root;
 46: 
 47:     /**
 48:      * Creates new SmallBlockTable
 49:      *
 50:      * @param documents a List of POIFSDocument instances
 51:      * @param root the Filesystem's root property
 52:      */
 53: 
 54:     public SmallBlockTableWriter(final POIFSBigBlockSize bigBlockSize,
 55:                                  final List documents,
 56:                                  final RootProperty root)
 57:     {
 58:         _sbat         = new BlockAllocationTableWriter(bigBlockSize);
 59:         _small_blocks = new ArrayList();
 60:         _root         = root;
 61:         Iterator iter = documents.iterator();
 62: 
 63:         while (iter.hasNext())
 64:         {
 65:             POIFSDocument   doc    = ( POIFSDocument ) iter.next();
 66:             BlockWritable[] blocks = doc.getSmallBlocks();
 67: 
 68:             if (blocks.length != 0)
 69:             {
 70:                 doc.setStartBlock(_sbat.allocateSpace(blocks.length));
 71:                 for (int j = 0; j < blocks.length; j++)
 72:                 {
 73:                     _small_blocks.add(blocks[ j ]);
 74:                 }
 75:             } else {
 76:             	doc.setStartBlock(POIFSConstants.END_OF_CHAIN);
 77:             }
 78:         }
 79:         _sbat.simpleCreateBlocks();
 80:         _root.setSize(_small_blocks.size());
 81:         _big_block_count = SmallDocumentBlock.fill(bigBlockSize,_small_blocks);
 82:     }
 83: 
 84:     /**
 85:      * Get the number of SBAT blocks
 86:      *
 87:      * @return number of SBAT big blocks
 88:      */
 89:     
 90:     public int getSBATBlockCount()
 91:     {
 92: 	return (_big_block_count + 15) / 16;
 93:     }
 94: 
 95:     /**
 96:      * Get the SBAT
 97:      *
 98:      * @return the Small Block Allocation Table
 99:      */
100: 
101:     public BlockAllocationTableWriter getSBAT()
102:     {
103:         return _sbat;
104:     }

109:      * Return the number of BigBlock's this instance uses
110:      *
111:      * @return count of BigBlock instances
112:      */
113: 
114:     public int countBlocks()
115:     {
116:         return _big_block_count;
117:     }
118: 
119:     /**

121:      *
122:      * @param start_block
123:      */
124: 
125:     public void setStartBlock(int start_block)
126:     {
127:         _root.setStartBlock(start_block);
128:     }
129: 
130:     /* **********  END  implementation of BATManaged ********** */
131:     /* ********** START implementation of BlockWritable ********** */
132: 
133:     /**
134:      * Write the storage to an OutputStream
135:      *
136:      * @param stream the OutputStream to which the stored data should
137:      *               be written
138:      *
139:      * @exception IOException on problems writing to the specified
140:      *            stream
141:      */
142: 
143:     public void writeBlocks(final OutputStream stream)
144:         throws IOException
145:     {
146:         Iterator iter = _small_blocks.iterator();
147: 
148:         while (iter.hasNext())
149:         {
150:             (( BlockWritable ) iter.next()).writeBlocks(stream);
151:         }
152:     }
