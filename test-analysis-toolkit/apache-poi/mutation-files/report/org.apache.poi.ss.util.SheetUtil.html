 36:     /**
 37:      * Excel measures columns in units of 1/256th of a character width
 38:      * but the docs say nothing about what particular character is used.
 39:      * '0' looks to be a good choice.
 40:      */
 41:     private static final char defaultChar = '0';

 43:     /**
 44:      * This is the multiple that the font height is scaled by when determining the
 45:      * boundary of rotated text.

 47:     private static final double fontHeightMultiple = 2.0;

 50:      *  Dummy formula evaluator that does nothing.
 51:      *  YK: The only reason of having this class is that
 52:      *  {@link org.apache.poi.ss.usermodel.DataFormatter#formatCellValue(org.apache.poi.ss.usermodel.Cell)}
 53:      *  returns formula string for formula cells. Dummy evaluator makes it to format the cached formula result.

 77:     /**
 78:      * Compute width of a single cell
 79:      *
 80:      * @param cell the cell whose width is to be calculated
 81:      * @param defaultCharWidth the width of a single character
 82:      * @param formatter formatter used to prepare the text to be measured
 83:      * @param useMergedCells    whether to use merged cells
 84:      * @return  the width in pixels
 85:      */
 86:     public static double getCellWidth(Cell cell, int defaultCharWidth, DataFormatter formatter, boolean useMergedCells) {
 87: 
 88:         Sheet sheet = cell.getSheet();
 89:         Workbook wb = sheet.getWorkbook();
 90:         Row row = cell.getRow();
 91:         int column = cell.getColumnIndex();
 92: 
 93:         int colspan = 1;
 94:         for (int i = 0 ; i < sheet.getNumMergedRegions(); i++) {
 95:             CellRangeAddress region = sheet.getMergedRegion(i);
 96:             if (containsCell(region, row.getRowNum(), column)) {
 97:                 if (!useMergedCells) {
 98:                     // If we're not using merged cells, skip this one and move on to the next.
 99:                     return -1;
100:                 }
101:                 cell = row.getCell(region.getFirstColumn());
102:                 colspan = 1 + region.getLastColumn() - region.getFirstColumn();
103:             }
104:         }
105: 
106:         CellStyle style = cell.getCellStyle();
107:         int cellType = cell.getCellType();
108: 
109:         // for formula cells we compute the cell width for the cached formula result
110:         if(cellType == Cell.CELL_TYPE_FORMULA) cellType = cell.getCachedFormulaResultType();
111: 
112:         Font font = wb.getFontAt(style.getFontIndex());
113: 
114:         AttributedString str;
115:         TextLayout layout;
116: 
117:         double width = -1;
118:         if (cellType == Cell.CELL_TYPE_STRING) {
119:             RichTextString rt = cell.getRichStringCellValue();
120:             String[] lines = rt.getString().split("\\n");
121:             for (int i = 0; i < lines.length; i++) {
122:                 String txt = lines[i] + defaultChar;
123: 
124:                 str = new AttributedString(txt);
125:                 copyAttributes(font, str, 0, txt.length());
126: 
127:                 if (rt.numFormattingRuns() > 0) {
128:                     // TODO: support rich text fragments
129:                 }
130: 
131:                 layout = new TextLayout(str.getIterator(), fontRenderContext);
132:                 if(style.getRotation() != 0){
133:                     /*
134:                      * Transform the text using a scale so that it's height is increased by a multiple of the leading,
135:                      * and then rotate the text before computing the bounds. The scale results in some whitespace around
136:                      * the unrotated top and bottom of the text that normally wouldn't be present if unscaled, but
137:                      * is added by the standard Excel autosize.
138:                      */
139:                     AffineTransform trans = new AffineTransform();
140:                     trans.concatenate(AffineTransform.getRotateInstance(style.getRotation()*2.0*Math.PI/360.0));
141:                     trans.concatenate(
142:                     AffineTransform.getScaleInstance(1, fontHeightMultiple)
143:                     );
144:                     width = Math.max(width, ((layout.getOutline(trans).getBounds().getWidth() / colspan) / defaultCharWidth) + cell.getCellStyle().getIndention());
145:                 } else {
146:                     width = Math.max(width, ((layout.getBounds().getWidth() / colspan) / defaultCharWidth) + cell.getCellStyle().getIndention());
147:                 }
148:             }
149:         } else {
150:             String sval = null;
151:             if (cellType == Cell.CELL_TYPE_NUMERIC) {
152:                 // Try to get it formatted to look the same as excel
153:                 try {
154:                     sval = formatter.formatCellValue(cell, dummyEvaluator);
155:                 } catch (Exception e) {
156:                     sval = String.valueOf(cell.getNumericCellValue());
157:                 }
158:             } else if (cellType == Cell.CELL_TYPE_BOOLEAN) {
159:                 sval = String.valueOf(cell.getBooleanCellValue()).toUpperCase();
160:             }
161:             if(sval != null) {
162:                 String txt = sval + defaultChar;
163:                 str = new AttributedString(txt);
164:                 copyAttributes(font, str, 0, txt.length());
165: 
166:                 layout = new TextLayout(str.getIterator(), fontRenderContext);
167:                 if(style.getRotation() != 0){
168:                     /*
169:                      * Transform the text using a scale so that it's height is increased by a multiple of the leading,
170:                      * and then rotate the text before computing the bounds. The scale results in some whitespace around
171:                      * the unrotated top and bottom of the text that normally wouldn't be present if unscaled, but
172:                      * is added by the standard Excel autosize.
173:                      */
174:                     AffineTransform trans = new AffineTransform();
175:                     trans.concatenate(AffineTransform.getRotateInstance(style.getRotation()*2.0*Math.PI/360.0));
176:                     trans.concatenate(
177:                     AffineTransform.getScaleInstance(1, fontHeightMultiple)
178:                     );
179:                     width = Math.max(width, ((layout.getOutline(trans).getBounds().getWidth() / colspan) / defaultCharWidth) + cell.getCellStyle().getIndention());
180:                 } else {
181:                     width = Math.max(width, ((layout.getBounds().getWidth() / colspan) / defaultCharWidth) + cell.getCellStyle().getIndention());
182:                 }
183:             }
184:         }
185:         return width;
186:     }

188:     /**
189:      * Compute width of a column and return the result
190:      *
191:      * @param sheet the sheet to calculate
192:      * @param column    0-based index of the column
193:      * @param useMergedCells    whether to use merged cells
194:      * @return  the width in pixels
195:      */
196:     public static double getColumnWidth(Sheet sheet, int column, boolean useMergedCells){
197:         AttributedString str;
198:         TextLayout layout;
199: 
200:         Workbook wb = sheet.getWorkbook();
201:         DataFormatter formatter = new DataFormatter();
202:         Font defaultFont = wb.getFontAt((short) 0);
203: 
204:         str = new AttributedString(String.valueOf(defaultChar));
205:         copyAttributes(defaultFont, str, 0, 1);
206:         layout = new TextLayout(str.getIterator(), fontRenderContext);
207:         int defaultCharWidth = (int)layout.getAdvance();
208: 
209:         double width = -1;
210:         for (Row row : sheet) {
211:             Cell cell = row.getCell(column);
212: 
213:             if (cell == null) {
214:                 continue;
215:             }
216: 
217:             double cellWidth = getCellWidth(cell, defaultCharWidth, formatter, useMergedCells);
218:             width = Math.max(width, cellWidth);
219:         }
220:         return width;
221:     }

223:     /**
224:      * Compute width of a column based on a subset of the rows and return the result
225:      *
226:      * @param sheet the sheet to calculate
227:      * @param column    0-based index of the column
228:      * @param useMergedCells    whether to use merged cells
229:      * @param firstRow  0-based index of the first row to consider (inclusive)
230:      * @param lastRow   0-based index of the last row to consider (inclusive)
231:      * @return  the width in pixels
232:      */
233:     public static double getColumnWidth(Sheet sheet, int column, boolean useMergedCells, int firstRow, int lastRow){
234:         AttributedString str;
235:         TextLayout layout;
236: 
237:         Workbook wb = sheet.getWorkbook();
238:         DataFormatter formatter = new DataFormatter();
239:         Font defaultFont = wb.getFontAt((short) 0);
240: 
241:         str = new AttributedString(String.valueOf(defaultChar));
242:         copyAttributes(defaultFont, str, 0, 1);
243:         layout = new TextLayout(str.getIterator(), fontRenderContext);
244:         int defaultCharWidth = (int)layout.getAdvance();
245: 
246:         double width = -1;
247:         for (int rowIdx = firstRow; rowIdx <= lastRow; ++rowIdx) {
248:             Row row = sheet.getRow(rowIdx);
249:             if( row != null ) {
250: 
251:                 Cell cell = row.getCell(column);
252: 
253:                 if (cell == null) {
254:                     continue;
255:                 }
256: 
257:                 double cellWidth = getCellWidth(cell, defaultCharWidth, formatter, useMergedCells);
258:                 width = Math.max(width, cellWidth);
259:             }
260:         }
261:         return width;
262:     }

264:     /**
265:      * Copy text attributes from the supplied Font to Java2D AttributedString
266:      */
267:     private static void copyAttributes(Font font, AttributedString str, int startIdx, int endIdx) {
268:         str.addAttribute(TextAttribute.FAMILY, font.getFontName(), startIdx, endIdx);
269:         str.addAttribute(TextAttribute.SIZE, (float)font.getFontHeightInPoints());
270:         if (font.getBoldweight() == Font.BOLDWEIGHT_BOLD) str.addAttribute(TextAttribute.WEIGHT, TextAttribute.WEIGHT_BOLD, startIdx, endIdx);
271:         if (font.getItalic() ) str.addAttribute(TextAttribute.POSTURE, TextAttribute.POSTURE_OBLIQUE, startIdx, endIdx);
272:         if (font.getUnderline() == Font.U_SINGLE ) str.addAttribute(TextAttribute.UNDERLINE, TextAttribute.UNDERLINE_ON, startIdx, endIdx);
273:     }

275:     public static boolean containsCell(CellRangeAddress cr, int rowIx, int colIx) {
276:         if (cr.getFirstRow() <= rowIx && cr.getLastRow() >= rowIx
277:                 && cr.getFirstColumn() <= colIx && cr.getLastColumn() >= colIx)
278:         {
279:             return true;
280:         }
281:         return false;
282:     }
