 23:  * Title: Sheet Tab Index Array Record (0x013D)<p/>

 30: public final class TabIdRecord extends StandardRecord {

 32:     private static final short[] EMPTY_SHORT_ARRAY = { };
 33:

 38:     }
 39: 
 40:     public TabIdRecord(RecordInputStream in) {
 41:         int nTabs = in.remaining() / 2;
 42:         _tabids = new short[nTabs];
 43:         for (int i = 0; i < _tabids.length; i++) {
 44:             _tabids[i] = in.readShort();
 45:         }
 46:     }
 47:

 50:      * @param array of tab id's {0,1,2}
 51:      */
 52:     public void setTabIdArray(short[] array) {
 53:         _tabids = array;
 54:     }
 55:

 57:         StringBuffer buffer = new StringBuffer();
 58: 
 59:         buffer.append("[TABID]\n");
 60:         buffer.append("    .elements        = ").append(_tabids.length).append("\n");
 61:         for (int i = 0; i < _tabids.length; i++) {
 62:             buffer.append("    .element_").append(i).append(" = ").append(_tabids[i]).append("\n");
 63:         }

 65:         return buffer.toString();
 66:     }

 72:             out.writeShort(tabids[i]);

 74:     }
