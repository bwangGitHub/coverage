 25: /**
 26:  * Title: Protect Record (0x0012) <p/>
 27:  * Description:  defines whether a sheet or workbook is protected (HSSF DOES NOT SUPPORT ENCRYPTION)<p/>
 28:  * HSSF now supports the simple "protected" sheets (where they are not encrypted and open office et al
 29:  * ignore the password record entirely).
 30:  * REFERENCE:  PG 373 Microsoft Excel 97 Developer's Kit (ISBN: 1-57231-498-2)<p/>
 31:  * @author Andrew C. Oliver (acoliver at apache dot org)
 32:  */
 33: public final class ProtectRecord extends StandardRecord {
 34:     public final static short sid = 0x0012;
 35: 
 36:     private static final BitField protectFlag = BitFieldFactory.getInstance(0x0001);
 37: 
 38:     private int _options;
 39: 
 40:     private ProtectRecord(int options) {
 41:         _options = options;
 42:     }
 43: 
 44:     public ProtectRecord(boolean isProtected) {
 45:         this(0);
 46:         setProtect(isProtected);
 47:     }
 48: 
 49:     public ProtectRecord(RecordInputStream in) {
 50:         this(in.readShort());
 51:     }
 52: 
 53:     /**
 54:      * set whether the sheet is protected or not
 55:      * @param protect whether to protect the sheet or not
 56:      */
 57:     public void setProtect(boolean protect) {
 58:         _options = protectFlag.setBoolean(_options, protect);
 59:     }
 60: 
 61:     /**
 62:      * get whether the sheet is protected or not
 63:      * @return whether to protect the sheet or not
 64:      */
 65:     public boolean getProtect() {
 66:         return protectFlag.isSet(_options);
 67:     }
 68: 
 69:     public String toString() {
 70:         StringBuffer buffer = new StringBuffer();
 71: 
 72:         buffer.append("[PROTECT]\n");
 73:         buffer.append("    .options = ").append(HexDump.shortToHex(_options)).append("\n");
 74:         buffer.append("[/PROTECT]\n");
 75:         return buffer.toString();
 76:     }
 77: 
 78:     public void serialize(LittleEndianOutput out) {
 79:         out.writeShort(_options);
 80:     }
 81: 
 82:     protected int getDataSize() {
 83:         return 2;
 84:     }
 85: 
 86:     public short getSid() {
 87:         return sid;
 88:     }
 89: 
 90:     public Object clone() {
 91:         return new ProtectRecord(_options);
 92:     }
 93: }
