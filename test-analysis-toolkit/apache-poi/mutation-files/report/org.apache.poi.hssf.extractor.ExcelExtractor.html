288: 		StringBuffer text = new StringBuffer();
289: 
290: 		// We don't care about the difference between
291: 		//  null (missing) and blank cells
292: 		_wb.setMissingCellPolicy(HSSFRow.RETURN_BLANK_AS_NULL);
293: 
294: 		// Process each sheet in turn
295: 		for(int i=0;i<_wb.getNumberOfSheets();i++) {
296: 			HSSFSheet sheet = _wb.getSheetAt(i);
297: 			if(sheet == null) { continue; }
298: 
299: 			if(_includeSheetNames) {
300: 				String name = _wb.getSheetName(i);
301: 				if(name != null) {
302: 					text.append(name);
303: 					text.append("\n");
304: 				}
305: 			}
306: 
307: 			// Header text, if there is any
308: 			if(_includeHeadersFooters) {
309: 				text.append(_extractHeaderFooter(sheet.getHeader()));
310: 			}
311: 
312: 			int firstRow = sheet.getFirstRowNum();
313: 			int lastRow = sheet.getLastRowNum();
314: 			for(int j=firstRow;j<=lastRow;j++) {
315: 				HSSFRow row = sheet.getRow(j);
316: 				if(row == null) { continue; }
317: 
318: 				// Check each cell in turn
319: 				int firstCell = row.getFirstCellNum();
320: 				int lastCell = row.getLastCellNum();
321: 				if(_includeBlankCells) {
322: 					firstCell = 0;
323: 				}
324: 
325: 				for(int k=firstCell;k<lastCell;k++) {
326: 					HSSFCell cell = row.getCell(k);
327: 					boolean outputContents = true;
328: 
329: 					if(cell == null) {
330: 						// Only output if requested
331: 						outputContents = _includeBlankCells;
332: 					} else {
333: 						switch(cell.getCellType()) {
334: 							case HSSFCell.CELL_TYPE_STRING:
335: 								text.append(cell.getRichStringCellValue().getString());
336: 								break;
337: 							case HSSFCell.CELL_TYPE_NUMERIC:
338: 								text.append(
339: 								      _formatter.formatCellValue(cell)
340: 								);
341: 								break;
342: 							case HSSFCell.CELL_TYPE_BOOLEAN:
343: 								text.append(cell.getBooleanCellValue());
344: 								break;
345: 							case HSSFCell.CELL_TYPE_ERROR:
346: 								text.append(ErrorEval.getText(cell.getErrorCellValue()));
347: 								break;
348: 							case HSSFCell.CELL_TYPE_FORMULA:
349: 								if(!_shouldEvaluateFormulas) {
350: 									text.append(cell.getCellFormula());
351: 								} else {
352: 									switch(cell.getCachedFormulaResultType()) {
353: 										case HSSFCell.CELL_TYPE_STRING:
354: 											HSSFRichTextString str = cell.getRichStringCellValue();
355: 											if(str != null && str.length() > 0) {
356: 												text.append(str.toString());
357: 											}
358: 											break;
359: 										case HSSFCell.CELL_TYPE_NUMERIC:
360: 										   HSSFCellStyle style = cell.getCellStyle();
361: 										   if(style == null) {
362: 										      text.append( cell.getNumericCellValue() );
363: 										   } else {
364: 	                                 text.append(
365: 	                                       _formatter.formatRawCellContents(
366: 	                                             cell.getNumericCellValue(),
367: 	                                             style.getDataFormat(),
368: 	                                             style.getDataFormatString()
369: 	                                       )
370: 	                                 );
371: 										   }
372: 											break;
373: 										case HSSFCell.CELL_TYPE_BOOLEAN:
374: 											text.append(cell.getBooleanCellValue());
375: 											break;
376: 										case HSSFCell.CELL_TYPE_ERROR:
377: 											text.append(ErrorEval.getText(cell.getErrorCellValue()));
378: 											break;
379: 
380: 									}
381: 								}
382: 								break;
383: 							default:
384: 								throw new RuntimeException("Unexpected cell type (" + cell.getCellType() + ")");
385: 						}
386: 
387: 						// Output the comment, if requested and exists
388: 						HSSFComment comment = cell.getCellComment();
389: 						if(_includeCellComments && comment != null) {
390: 							// Replace any newlines with spaces, otherwise it
391: 							//  breaks the output
392: 							String commentText = comment.getString().getString().replace('\n', ' ');
393: 							text.append(" Comment by "+comment.getAuthor()+": "+commentText);
394: 						}
395: 					}
396: 
397: 					// Output a tab if we're not on the last cell
398: 					if(outputContents && k < (lastCell-1)) {
399: 						text.append("\t");
400: 					}
401: 				}
402: 
403: 				// Finish off the row
404: 				text.append("\n");
405: 			}
406: 
407: 			// Finally Footer text, if there is any
408: 			if(_includeHeadersFooters) {
409: 				text.append(_extractHeaderFooter(sheet.getFooter()));
410: 			}
411: 		}
412: 
413: 		return text.toString();
414: 	}
