195: 	 * Distinguishes formula expressions from simple value expressions.  This logic is only 
196: 	 * required by a few factory methods in this class that create data validation constraints
197: 	 * from more or less the same parameters that would have been entered in the Excel UI.  The
198: 	 * data validation dialog box uses the convention that formulas begin with '='.  Other methods
199: 	 * in this class follow the POI convention (formulas and values are distinct), so the '=' 
200: 	 * convention is not used there.
201: 	 *  
202: 	 * @param textExpr a formula or value expression
203: 	 * @return all text after '=' if textExpr begins with '='. Otherwise <code>null</code> if textExpr does not begin with '='
204: 	 */
205: 	private static String getFormulaFromTextExpression(String textExpr) {
206: 		if (textExpr == null) {
207: 			return null;
208: 		}
209: 		if (textExpr.length() < 1) {
210: 			throw new IllegalArgumentException("Empty string is not a valid formula/value expression");
211: 		}
212: 		if (textExpr.charAt(0) == '=') {
213: 			return textExpr.substring(1);
214: 		}
215: 		return null;
