 26: /**
 27:  * Excel can get cranky if you give it files containing too
 28:  *  many (especially duplicate) objects, and this class can
 29:  *  help to avoid those.
 30:  * In general, it's much better to make sure you don't 
 31:  *  duplicate the objects in your code, as this is likely
 32:  *  to be much faster than creating lots and lots of
 33:  *  excel objects+records, only to optimise them down to
 34:  *  many fewer at a later stage.
 35:  * However, sometimes this is too hard / tricky to do, which
 36:  *  is where the use of this class comes in.
 37:  */
 38: public class HSSFOptimiser {
 39: 	/**
 40: 	 * Goes through the Workbook, optimising the fonts by
 41: 	 *  removing duplicate ones.
 42: 	 * For now, only works on fonts used in {@link HSSFCellStyle}
 43: 	 *  and {@link HSSFRichTextString}. Any other font uses
 44: 	 *  (eg charts, pictures) may well end up broken!
 45: 	 * This can be a slow operation, especially if you have
 46: 	 *  lots of cells, cell styles or rich text strings
 47: 	 * @param workbook The workbook in which to optimise the fonts
 48: 	 */
 49: 	public static void optimiseFonts(HSSFWorkbook workbook) {
 50: 		// Where each font has ended up, and if we need to
 51: 		//  delete the record for it. Start off with no change
 52: 		short[] newPos = 
 53: 			new short[workbook.getWorkbook().getNumberOfFontRecords()+1];
 54: 		boolean[] zapRecords = new boolean[newPos.length];
 55: 		for(int i=0; i<newPos.length; i++) {
 56: 			newPos[i] = (short)i;
 57: 			zapRecords[i] = false;
 58: 		}
 59: 		
 60: 		// Get each font record, so we can do deletes
 61: 		//  without getting confused
 62: 		FontRecord[] frecs = new FontRecord[newPos.length]; 
 63: 		for(int i=0; i<newPos.length; i++) {
 64: 			// There is no 4!
 65: 			if(i == 4) continue;
 66: 			
 67: 			frecs[i] = workbook.getWorkbook().getFontRecordAt(i);
 68: 		}
 69: 		
 70: 		// Loop over each font, seeing if it is the same
 71: 		//  as an earlier one. If it is, point users of the
 72: 		//  later duplicate copy to the earlier one, and 
 73: 		//  mark the later one as needing deleting
 74: 		// Note - don't change built in fonts (those before 5)
 75: 		for(int i=5; i<newPos.length; i++) {
 76: 			// Check this one for being a duplicate
 77: 			//  of an earlier one
 78: 			int earlierDuplicate = -1;
 79: 			for(int j=0; j<i && earlierDuplicate == -1; j++) {
 80: 				if(j == 4) continue;
 81: 				
 82: 				FontRecord frCheck = workbook.getWorkbook().getFontRecordAt(j);
 83: 				if(frCheck.sameProperties(frecs[i])) {
 84: 					earlierDuplicate = j;
 85: 				}
 86: 			}
 87: 			
 88: 			// If we got a duplicate, mark it as such
 89: 			if(earlierDuplicate != -1) {
 90: 				newPos[i] = (short)earlierDuplicate;
 91: 				zapRecords[i] = true;
 92: 			}
 93: 		}
 94: 		
 95: 		// Update the new positions based on
 96: 		//  deletes that have occurred between
 97: 		//  the start and them
 98: 		// Only need to worry about user fonts
 99: 		for(int i=5; i<newPos.length; i++) {
100: 			// Find the number deleted to that
101: 			//  point, and adjust
102: 			short preDeletePos = newPos[i];
103: 			short newPosition = preDeletePos;
104: 			for(int j=0; j<preDeletePos; j++) {
105: 				if(zapRecords[j]) newPosition--;
106: 			}
107: 			
108: 			// Update the new position
109: 			newPos[i] = newPosition;
110: 		}
111: 		
112: 		// Zap the un-needed user font records
113: 		for(int i=5; i<newPos.length; i++) {
114: 			if(zapRecords[i]) {
115: 				workbook.getWorkbook().removeFontRecord(
116: 						frecs[i]
117: 				);
118: 			}
119: 		}
120: 		
121: 		// Tell HSSFWorkbook that it needs to
122: 		//  re-start its HSSFFontCache
123: 		workbook.resetFontCache();
124: 		
125: 		// Update the cell styles to point at the 
126: 		//  new locations of the fonts
127: 		for(int i=0; i<workbook.getWorkbook().getNumExFormats(); i++) {
128: 			ExtendedFormatRecord xfr = workbook.getWorkbook().getExFormatAt(i);
129: 			xfr.setFontIndex(
130: 					newPos[ xfr.getFontIndex() ]
131: 			);
132: 		}
133: 		
134: 		// Update the rich text strings to point at
135: 		//  the new locations of the fonts
136: 		// Remember that one underlying unicode string
137: 		//  may be shared by multiple RichTextStrings!
138: 		HashSet doneUnicodeStrings = new HashSet();
139: 		for(int sheetNum=0; sheetNum<workbook.getNumberOfSheets(); sheetNum++) {
140: 			HSSFSheet s = workbook.getSheetAt(sheetNum);
141: 			Iterator rIt = s.rowIterator();
142: 			while(rIt.hasNext()) {
143: 				HSSFRow row = (HSSFRow)rIt.next();
144: 				Iterator cIt = row.cellIterator();
145: 				while(cIt.hasNext()) {
146: 					HSSFCell cell = (HSSFCell)cIt.next();
147: 					if(cell.getCellType() == HSSFCell.CELL_TYPE_STRING) {
148: 						HSSFRichTextString rtr = cell.getRichStringCellValue();
149: 						UnicodeString u = rtr.getRawUnicodeString();
150: 						
151: 						// Have we done this string already?
152: 						if(! doneUnicodeStrings.contains(u)) {
153: 							// Update for each new position
154: 							for(short i=5; i<newPos.length; i++) {
155: 								if(i != newPos[i]) {
156: 									u.swapFontUse(i, newPos[i]);
157: 								}
158: 							}
159: 							
160: 							// Mark as done
161: 							doneUnicodeStrings.add(u);
162: 						}
163: 					}
164: 				}
165: 			}
166: 		}
167: 	}

239: 				);
240: 			}
241: 		}

243: 		// Finally, update the cells to point at
244: 		//  their new extended format records
245: 		for(int sheetNum=0; sheetNum<workbook.getNumberOfSheets(); sheetNum++) {
