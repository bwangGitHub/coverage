278: 	public static Record [] createRecord(RecordInputStream in) {
279: 
280: 		Record record = createSingleRecord(in);
281: 		if (record instanceof DBCellRecord) {
282: 			// Not needed by POI.  Regenerated from scratch by POI when spreadsheet is written
283: 			return new Record[] { null, };
284: 		}

286: 			return new Record[] { convertToNumberRecord((RKRecord) record), };
287: 		}

289: 			return convertRKRecords((MulRKRecord)record);
290: 		}

292: 	}

371: 	/**
372: 	 * gets the record constructors and sticks them in the map by SID
373: 	 * @return map of SIDs to short,short,byte[] constructors for Record classes
374: 	 * most of org.apache.poi.hssf.record.*
375: 	 */
376: 	private static Map<Integer, I_RecordCreator> recordsToMap(Class<? extends Record> [] records) {
377: 		Map<Integer, I_RecordCreator> result = new HashMap<Integer, I_RecordCreator>();
378: 		Set<Class<?>> uniqueRecClasses = new HashSet<Class<?>>(records.length * 3 / 2);
379: 
380: 		for (int i = 0; i < records.length; i++) {
381: 
382: 			Class<? extends Record> recClass = records[ i ];
383: 			if(!Record.class.isAssignableFrom(recClass)) {
384: 				throw new RuntimeException("Invalid record sub-class (" + recClass.getName() + ")");
385: 			}
386: 			if(Modifier.isAbstract(recClass.getModifiers())) {
387: 				throw new RuntimeException("Invalid record class (" + recClass.getName() + ") - must not be abstract");
388: 			}
389: 			if(!uniqueRecClasses.add(recClass)) {
390: 				throw new RuntimeException("duplicate record class (" + recClass.getName() + ")");
391: 			}
392: 
393: 			int sid;
394: 			try {
395: 				sid = recClass.getField("sid").getShort(null);
396: 			} catch (Exception illegalArgumentException) {
397: 				throw new RecordFormatException(
398: 					"Unable to determine record types");
399: 			}
400: 			Integer key = Integer.valueOf(sid);
401: 			if (result.containsKey(key)) {
402: 				Class<?> prevClass = result.get(key).getRecordClass();
403: 				throw new RuntimeException("duplicate record sid 0x" + Integer.toHexString(sid).toUpperCase()
404: 						+ " for classes (" + recClass.getName() + ") and (" + prevClass.getName() + ")");
405: 			}
406: 			result.put(key, getRecordCreator(recClass));
407: 		}
408: //		result.put(Integer.valueOf(0x0406), result.get(Integer.valueOf(0x06)));
409: 		return result;
410: 	}

412: 	private static I_RecordCreator getRecordCreator(Class<? extends Record> recClass) {
413: 		try {
414: 			Constructor<? extends Record> constructor;
415: 			constructor = recClass.getConstructor(CONSTRUCTOR_ARGS);
416: 			return new ReflectionConstructorRecordCreator(constructor);
417: 		} catch (NoSuchMethodException e) {
418: 			// fall through and look for other construction methods
419: 		}
420: 		try {
421: 			Method m = recClass.getDeclaredMethod("create", CONSTRUCTOR_ARGS);
422: 			return new ReflectionMethodRecordCreator(m);
423: 		} catch (NoSuchMethodException e) {
424: 			throw new RuntimeException("Failed to find constructor or create method for (" + recClass.getName() + ").");
425: 		}
426: 	}

429: 	 *
430: 	 * @param in the InputStream from which the records will be obtained
431: 	 *

433: 	 *

437:
