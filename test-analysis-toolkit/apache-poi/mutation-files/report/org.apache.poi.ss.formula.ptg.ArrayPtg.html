 26: /**
 27:  * ArrayPtg - handles arrays
 28:  *
 29:  * The ArrayPtg is a little weird, the size of the Ptg when parsing initially only
 30:  * includes the Ptg sid and the reserved bytes. The next Ptg in the expression then follows.
 31:  * It is only after the "size" of all the Ptgs is met, that the ArrayPtg data is actually
 32:  * held after this. So Ptg.createParsedExpression keeps track of the number of
 33:  * ArrayPtg elements and need to parse the data upto the FORMULA record size.
 34:  *
 35:  * @author Jason Height (jheight at chariot dot net dot au)
 36:  */
 37: public final class ArrayPtg extends Ptg {
 38: 	public static final byte sid  = 0x20;
 39: 
 40: 	private static final int RESERVED_FIELD_LEN = 7;
 41: 	/**
 42: 	 * The size of the plain tArray token written within the standard formula tokens
 43: 	 * (not including the data which comes after all formula tokens)
 44: 	 */
 45: 	public static final int PLAIN_TOKEN_SIZE = 1+RESERVED_FIELD_LEN;
 46: 
 47: 	// 7 bytes of data (stored as an int, short and byte here)
 48: 	private final int _reserved0Int;
 49: 	private final int _reserved1Short;
 50: 	private final int _reserved2Byte;
 51: 
 52: 	// data from these fields comes after the Ptg data of all tokens in current formula
 53: 	private final int  _nColumns;
 54: 	private final int _nRows;
 55: 	private final Object[] _arrayValues;
 56: 
 57: 	ArrayPtg(int reserved0, int reserved1, int reserved2, int nColumns, int nRows, Object[] arrayValues) {
 58: 		_reserved0Int = reserved0;
 59: 		_reserved1Short = reserved1;
 60: 		_reserved2Byte = reserved2;
 61: 		_nColumns = nColumns;
 62: 		_nRows = nRows;
 63: 		_arrayValues = arrayValues;
 64: 	}
 65: 	/**
 66: 	 * @param values2d array values arranged in rows
 67: 	 */
 68: 	public ArrayPtg(Object[][] values2d) {
 69: 		int nColumns = values2d[0].length;
 70: 		int nRows = values2d.length;
 71: 		// convert 2-d to 1-d array (row by row according to getValueIndex())
 72: 		_nColumns = (short) nColumns;
 73: 		_nRows = (short) nRows;
 74: 
 75: 		Object[] vv = new Object[_nColumns * _nRows];
 76: 		for (int r=0; r<nRows; r++) {
 77: 			Object[] rowData = values2d[r];
 78: 			for (int c=0; c<nColumns; c++) {
 79: 				vv[getValueIndex(c, r)] = rowData[c];
 80: 			}
 81: 		}
 82: 
 83: 		_arrayValues = vv;
 84: 		_reserved0Int = 0;
 85: 		_reserved1Short = 0;
 86: 		_reserved2Byte = 0;
 87: 	}
 88: 	/**
 89: 	 * @return 2-d array (inner index is rowIx, outer index is colIx)
 90: 	 */
 91: 	public Object[][] getTokenArrayValues() {
 92: 		if (_arrayValues == null) {
 93: 			throw new IllegalStateException("array values not read yet");
 94: 		}
 95: 		Object[][] result = new Object[_nRows][_nColumns];
 96: 		for (int r = 0; r < _nRows; r++) {
 97: 			Object[] rowData = result[r];
 98: 			for (int c = 0; c < _nColumns; c++) {
 99: 				rowData[c] = _arrayValues[getValueIndex(c, r)];
100: 			}
101: 		}
102: 		return result;
103: 	}
104: 
105: 	public boolean isBaseToken() {
106: 		return false;
107: 	}
108: 
109: 	public String toString() {
110: 		StringBuffer sb = new StringBuffer("[ArrayPtg]\n");
111: 
112: 		sb.append("nRows = ").append(getRowCount()).append("\n");
113: 		sb.append("nCols = ").append(getColumnCount()).append("\n");
114: 		if (_arrayValues == null) {
115: 			sb.append("  #values#uninitialised#\n");
116: 		} else {
117: 			sb.append("  ").append(toFormulaString());
118: 		}
119: 		return sb.toString();
120: 	}
121: 
122: 	/**
123: 	 * Note - (2D) array elements are stored row by row
124: 	 * @return the index into the internal 1D array for the specified column and row
125: 	 */
126: 	/* package */ int getValueIndex(int colIx, int rowIx) {
127: 		if(colIx < 0 || colIx >= _nColumns) {
128: 			throw new IllegalArgumentException("Specified colIx (" + colIx
129: 					+ ") is outside the allowed range (0.." + (_nColumns-1) + ")");
130: 		}
131: 		if(rowIx < 0 || rowIx >= _nRows) {
132: 			throw new IllegalArgumentException("Specified rowIx (" + rowIx
133: 					+ ") is outside the allowed range (0.." + (_nRows-1) + ")");
134: 		}
135: 		return rowIx * _nColumns + colIx;
136: 	}

138: 	public void write(LittleEndianOutput out) {
139: 		out.writeByte(sid + getPtgClass());
140: 		out.writeInt(_reserved0Int);
141: 		out.writeShort(_reserved1Short);
142: 		out.writeByte(_reserved2Byte);
143: 	}

145: 	public int writeTokenValueBytes(LittleEndianOutput out) {
146: 
147: 		out.writeByte(_nColumns-1);
148: 		out.writeShort(_nRows-1);
149: 		ConstantValueParser.encode(out, _arrayValues);
150: 		return 3 + ConstantValueParser.getEncodedSize(_arrayValues);
151: 	}

154: 		return _nRows;
155: 	}
156: 
157: 	public int getColumnCount() {
158: 		return _nColumns;

161: 	/** This size includes the size of the array Ptg plus the Array Ptg Token value size*/
162: 	public int getSize() {
163: 		return PLAIN_TOKEN_SIZE
164: 			// data written after the all tokens:
165: 			+ 1 + 2 // column, row
166: 			+ ConstantValueParser.getEncodedSize(_arrayValues);
167: 	}

169: 	public String toFormulaString() {
170: 		StringBuffer b = new StringBuffer();
171: 		b.append("{");
172: 	  	for (int y=0;y<getRowCount();y++) {
173: 			if (y > 0) {
174: 				b.append(";");
175: 			}
176: 			for (int x=0;x<getColumnCount();x++) {
177: 			  	if (x > 0) {
178: 					b.append(",");
179: 				}
180: 		  		Object o = _arrayValues[getValueIndex(x, y)];
181: 		  		b.append(getConstantText(o));
182: 		  	}
183: 		  }
184: 		b.append("}");
185: 		return b.toString();
186: 	}

188: 	private static String getConstantText(Object o) {
189: 
190: 		if (o == null) {
191: 			throw new RuntimeException("Array item cannot be null");
192: 		}
193: 		if (o instanceof String) {
194: 			return "\"" + (String)o + "\"";
195: 		}
196: 		if (o instanceof Double) {
197: 			return NumberToTextConverter.toText(((Double)o).doubleValue());
198: 		}
199: 		if (o instanceof Boolean) {
200: 			return ((Boolean)o).booleanValue() ? "TRUE" : "FALSE";
201: 		}
202: 		if (o instanceof ErrorConstant) {
203: 			return ((ErrorConstant)o).getText();
204: 		}
205: 		throw new IllegalArgumentException("Unexpected constant class (" + o.getClass().getName() + ")");
206: 	}

212: 	/**
213: 	 * Represents the initial plain tArray token (without the constant data that trails the whole
214: 	 * formula).  Objects of this class are only temporary and cannot be used as {@link Ptg}s.
215: 	 * These temporary objects get converted to {@link ArrayPtg} by the
216: 	 * {@link #finishReading(LittleEndianInput)} method.
217: 	 */
218: 	static final class Initial extends Ptg {
219: 		private final int _reserved0;
220: 		private final int _reserved1;
221: 		private final int _reserved2;
222: 
223: 		public Initial(LittleEndianInput in) {
224: 			_reserved0 = in.readInt();
225: 			_reserved1 = in.readUShort();
226: 			_reserved2 = in.readUByte();
227: 		}
228: 		private static RuntimeException invalid() {
229: 			throw new IllegalStateException("This object is a partially initialised tArray, and cannot be used as a Ptg");
230: 		}
231: 		public byte getDefaultOperandClass() {
232: 			throw invalid();
233: 		}
234: 		public int getSize() {
235: 			return PLAIN_TOKEN_SIZE;
236: 		}
237: 		public boolean isBaseToken() {
238: 			return false;
239: 		}
240: 		public String toFormulaString() {
241: 			throw invalid();
242: 		}
243: 		public void write(LittleEndianOutput out) {
244: 			throw invalid();
245: 		}
246: 		/**
247: 		 * Read in the actual token (array) values. This occurs
248: 		 * AFTER the last Ptg in the expression.
249: 		 * See page 304-305 of Excel97-2007BinaryFileFormat(xls)Specification.pdf
250: 		 */
251: 		public ArrayPtg finishReading(LittleEndianInput in) {
252: 			int nColumns = in.readUByte();
253: 			short nRows = in.readShort();
254: 			//The token_1_columns and token_2_rows do not follow the documentation.
255: 			//The number of physical rows and columns is actually +1 of these values.
256: 			//Which is not explicitly documented.
257: 			nColumns++;
258: 			nRows++;
259: 
260: 			int totalCount = nRows * nColumns;
261: 			Object[] arrayValues = ConstantValueParser.parse(in, totalCount);
262: 
263: 			ArrayPtg result = new ArrayPtg(_reserved0, _reserved1, _reserved2, nColumns, nRows, arrayValues);
264: 			result.setClass(getPtgClass());
265: 			return result;
266: 		}
267: 	}
