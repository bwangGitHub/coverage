223:     /**
224:      * Does this HSSFPatriarch contain a chart?
225:      * (Technically a reference to a chart, since they
226:      *  get stored in a different block of records)
227:      * FIXME - detect chart in all cases (only seems
228:      *  to work on some charts so far)
229:      */
230:     public boolean containsChart() {
231:         // TODO - support charts properly in usermodel
232: 
233:         // We're looking for a EscherOptRecord
234:         EscherOptRecord optRecord = (EscherOptRecord)
235:             _boundAggregate.findFirstWithId(EscherOptRecord.RECORD_ID);
236:         if(optRecord == null) {
237:             // No opt record, can't have chart
238:             return false;
239:         }
240: 
241:         for(Iterator<EscherProperty> it = optRecord.getEscherProperties().iterator(); it.hasNext();) {
242:             EscherProperty prop = it.next();
243:             if(prop.getPropertyNumber() == 896 && prop.isComplex()) {
244:                 EscherComplexProperty cp = (EscherComplexProperty)prop;
245:                 String str = StringUtil.getFromUnicodeLE(cp.getComplexData());
246: 
247:                 if(str.equals("Chart 1\0")) {
248:                     return true;
249:                 }
250:             }
251:         }
252: 
253:         return false;
254:     }
