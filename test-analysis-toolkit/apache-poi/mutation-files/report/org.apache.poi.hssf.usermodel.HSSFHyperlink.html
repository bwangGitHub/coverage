 59:     /**
 60:      * Construct a new hyperlink
 61:      *
 62:      * @param type the type of hyperlink to create
 63:      */
 64:     public HSSFHyperlink( int type )
 65:     {
 66:         this.link_type = type;
 67:         record = new HyperlinkRecord();
 68:         switch(type){
 69:             case LINK_URL:
 70:             case LINK_EMAIL:
 71:                 record.newUrlLink();
 72:                 break;
 73:             case LINK_FILE:
 74:                 record.newFileLink();
 75:                 break;
 76:             case LINK_DOCUMENT:
 77:                 record.newDocumentLink();
 78:                 break;
 79:         }
 80:     }
 81: 
 82:     /**
 83:      * Initialize the hyperlink by a <code>HyperlinkRecord</code> record
 84:      *
 85:      * @param record
 86:      */
 87:     protected HSSFHyperlink( HyperlinkRecord record )
 88:     {
 89:         this.record = record;
 90:         
 91:         // Figure out the type
 92:         if(record.isFileLink()) {
 93:            link_type = LINK_FILE;
 94:         } else if(record.isDocumentLink()) {
 95:            link_type = LINK_DOCUMENT;
 96:         } else {
 97:            if(record.getAddress() != null &&
 98:                  record.getAddress().startsWith("mailto:")) {
 99:               link_type = LINK_EMAIL;
100:            } else {
101:               link_type = LINK_URL;
102:            }
103:         }
104:     }
