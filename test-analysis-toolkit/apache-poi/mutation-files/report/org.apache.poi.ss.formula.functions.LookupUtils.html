125: 	/**
126: 	 * Enumeration to support <b>4</b> valued comparison results.<p/>
127: 	 * Excel lookup functions have complex behaviour in the case where the lookup array has mixed
128: 	 * types, and/or is unordered.  Contrary to suggestions in some Excel documentation, there
129: 	 * does not appear to be a universal ordering across types.  The binary search algorithm used
130: 	 * changes behaviour when the evaluated 'mid' value has a different type to the lookup value.<p/>
131: 	 *
132: 	 * A simple int might have done the same job, but there is risk in confusion with the well
133: 	 * known <tt>Comparable.compareTo()</tt> and <tt>Comparator.compare()</tt> which both use
134: 	 * a ubiquitous 3 value result encoding.
135: 	 */
136: 	public static final class CompareResult {
137: 		private final boolean _isTypeMismatch;
138: 		private final boolean _isLessThan;
139: 		private final boolean _isEqual;
140: 		private final boolean _isGreaterThan;
141: 
142: 		private CompareResult(boolean isTypeMismatch, int simpleCompareResult) {
143: 			if(isTypeMismatch) {
144: 				_isTypeMismatch = true;
145: 				_isLessThan = false;
146: 				_isEqual = false;
147: 				_isGreaterThan = false;
148: 			} else {
149: 				_isTypeMismatch = false;
150: 				_isLessThan = simpleCompareResult < 0;
151: 				_isEqual = simpleCompareResult == 0;
152: 				_isGreaterThan = simpleCompareResult > 0;
153: 			}
154: 		}
155: 		public static final CompareResult TYPE_MISMATCH = new CompareResult(true, 0);
156: 		public static final CompareResult LESS_THAN = new CompareResult(false, -1);
157: 		public static final CompareResult EQUAL = new CompareResult(false, 0);
158: 		public static final CompareResult GREATER_THAN = new CompareResult(false, +1);
159: 
160: 		public static final CompareResult valueOf(int simpleCompareResult) {
161: 			if(simpleCompareResult < 0) {
162: 				return LESS_THAN;
163: 			}
164: 			if(simpleCompareResult > 0) {
165: 				return GREATER_THAN;
166: 			}
167: 			return EQUAL;
168: 		}
169: 
170: 		public boolean isTypeMismatch() {
171: 			return _isTypeMismatch;
172: 		}
173: 		public boolean isLessThan() {
174: 			return _isLessThan;
175: 		}
176: 		public boolean isEqual() {
177: 			return _isEqual;
178: 		}
179: 		public boolean isGreaterThan() {
180: 			return _isGreaterThan;
181: 		}
182: 		public String toString() {
183: 			StringBuffer sb = new StringBuffer(64);
184: 			sb.append(getClass().getName()).append(" [");
185: 			sb.append(formatAsString());
186: 			sb.append("]");
187: 			return sb.toString();
188: 		}
189: 
190: 		private String formatAsString() {
191: 			if(_isTypeMismatch) {
192: 				return "TYPE_MISMATCH";
193: 			}
194: 			if(_isLessThan) {
195: 				return "LESS_THAN";
196: 			}
197: 			if(_isEqual) {
198: 				return "EQUAL";
199: 			}
200: 			if(_isGreaterThan) {
201: 				return "GREATER_THAN";
202: 			}
203: 			// toString must be reliable
204: 			return "??error??";
205: 		}

300: 	/**
301: 	 * Processes the third argument to VLOOKUP, or HLOOKUP (<b>col_index_num</b>
302: 	 * or <b>row_index_num</b> respectively).<br>
303: 	 * Sample behaviour:
304: 	 *    <table border="0" cellpadding="1" cellspacing="2" summary="Sample behaviour">
305: 	 *      <tr><th>Input&nbsp;&nbsp;&nbsp;Return</th><th>Value&nbsp;&nbsp;</th><th>Thrown Error</th></tr>
306: 	 *      <tr><td>5</td><td>4</td><td>&nbsp;</td></tr>
307: 	 *      <tr><td>2.9</td><td>2</td><td>&nbsp;</td></tr>
308: 	 *      <tr><td>"5"</td><td>4</td><td>&nbsp;</td></tr>
309: 	 *      <tr><td>"2.18e1"</td><td>21</td><td>&nbsp;</td></tr>
310: 	 *      <tr><td>"-$2"</td><td>-3</td><td>*</td></tr>
311: 	 *      <tr><td>FALSE</td><td>-1</td><td>*</td></tr>
312: 	 *      <tr><td>TRUE</td><td>0</td><td>&nbsp;</td></tr>
313: 	 *      <tr><td>"TRUE"</td><td>&nbsp;</td><td>#REF!</td></tr>
314: 	 *      <tr><td>"abc"</td><td>&nbsp;</td><td>#REF!</td></tr>
315: 	 *      <tr><td>""</td><td>&nbsp;</td><td>#REF!</td></tr>
316: 	 *      <tr><td>&lt;blank&gt;</td><td>&nbsp;</td><td>#VALUE!</td></tr>
317: 	 *    </table><br/>
318: 	 *
319: 	 * Note - out of range errors (result index too high) are handled by the caller.
320: 	 * @return column or row index as a zero-based value, never negative.
321: 	 * @throws EvaluationException when the specified arg cannot be coerced to a non-negative integer
322: 	 */
323: 	public static int resolveRowOrColIndexArg(ValueEval rowColIndexArg, int srcCellRow, int srcCellCol) throws EvaluationException {
324: 		if(rowColIndexArg == null) {
325: 			throw new IllegalArgumentException("argument must not be null");
326: 		}
327: 
328: 		ValueEval veRowColIndexArg;
329: 		try {
330: 			veRowColIndexArg = OperandResolver.getSingleValue(rowColIndexArg, srcCellRow, (short)srcCellCol);
331: 		} catch (EvaluationException e) {
332: 			// All errors get translated to #REF!
333: 			throw EvaluationException.invalidRef();
334: 		}
335: 		int oneBasedIndex;
336: 		if(veRowColIndexArg instanceof StringEval) {
337: 			StringEval se = (StringEval) veRowColIndexArg;
338: 			String strVal = se.getStringValue();
339: 			Double dVal = OperandResolver.parseDouble(strVal);
340: 			if(dVal == null) {
341: 				// String does not resolve to a number. Raise #REF! error.
342: 				throw EvaluationException.invalidRef();
343: 				// This includes text booleans "TRUE" and "FALSE".  They are not valid.
344: 			}
345: 			// else - numeric value parses OK
346: 		}
347: 		// actual BoolEval values get interpreted as FALSE->0 and TRUE->1
348: 		oneBasedIndex = OperandResolver.coerceValueToInt(veRowColIndexArg);
349: 		if (oneBasedIndex < 1) {
350: 			// note this is asymmetric with the errors when the index is too large (#REF!)
351: 			throw EvaluationException.invalidValue();
352: 		}
353: 		return oneBasedIndex - 1; // convert to zero based
354: 	}

378: 	/**
379: 	 * Resolves the last (optional) parameter (<b>range_lookup</b>) to the VLOOKUP and HLOOKUP functions.
380: 	 * @param rangeLookupArg must not be <code>null</code>
381: 	 */
382: 	public static boolean resolveRangeLookupArg(ValueEval rangeLookupArg, int srcCellRow, int srcCellCol) throws EvaluationException {
383: 
384: 		ValueEval valEval = OperandResolver.getSingleValue(rangeLookupArg, srcCellRow, srcCellCol);
385: 		if(valEval instanceof BlankEval) {
386: 			// Tricky:
387: 			// fourth arg supplied but evaluates to blank
388: 			// this does not get the default value
389: 			return false;
390: 		}
391: 		if(valEval instanceof BoolEval) {
392: 			// Happy day flow
393: 			BoolEval boolEval = (BoolEval) valEval;
394: 			return boolEval.getBooleanValue();
395: 		}
396: 
397: 		if (valEval instanceof StringEval) {
398: 			String stringValue = ((StringEval) valEval).getStringValue();
399: 			if(stringValue.length() < 1) {
400: 				// More trickiness:
401: 				// Empty string is not the same as BlankEval.  It causes #VALUE! error
402: 				throw EvaluationException.invalidValue();
403: 			}
404: 			// TODO move parseBoolean to OperandResolver
405: 			Boolean b = Countif.parseBoolean(stringValue);
406: 			if(b != null) {
407: 				// string converted to boolean OK
408: 				return b.booleanValue();
409: 			}
410: 			// Even more trickiness:
411: 			// Note - even if the StringEval represents a number value (for example "1"),
412: 			// Excel does not resolve it to a boolean.
413: 			throw EvaluationException.invalidValue();
414: 			// This is in contrast to the code below,, where NumberEvals values (for
415: 			// example 0.01) *do* resolve to equivalent boolean values.
416: 		}
417: 		if (valEval instanceof NumericValueEval) {
418: 			NumericValueEval nve = (NumericValueEval) valEval;
419: 			// zero is FALSE, everything else is TRUE
420: 			return 0.0 != nve.getNumberValue();
421: 		}
422: 		throw new RuntimeException("Unexpected eval type (" + valEval.getClass().getName() + ")");
423: 	}

442: 	 * @param lookupValue the value to be found in column or row vector
443: 	 * @param vector the values to be searched. For VLOOKUP this is the first column of the
444: 	 * 	tableArray. For HLOOKUP this is the first row of the tableArray.

461: 	 * Encapsulates some standard binary search functionality so the unusual Excel behaviour can
462: 	 * be clearly distinguished.
463: 	 */
464: 	private static final class BinarySearchIndexes {
465: 
466: 		private int _lowIx;
467: 		private int _highIx;
468: 
469: 		public BinarySearchIndexes(int highIx) {
470: 			_lowIx = -1;
471: 			_highIx = highIx;
472: 		}
473: 
474: 		/**
475: 		 * @return -1 if the search range is empty
476: 		 */
477: 		public int getMidIx() {
478: 			int ixDiff = _highIx - _lowIx;
479: 			if(ixDiff < 2) {
480: 				return -1;
481: 			}
482: 			return _lowIx + (ixDiff / 2);
483: 		}
484: 
485: 		public int getLowIx() {
486: 			return _lowIx;
487: 		}
488: 		public int getHighIx() {
489: 			return _highIx;
490: 		}
491: 		public void narrowSearch(int midIx, boolean isLessThan) {
492: 			if(isLessThan) {
493: 				_highIx = midIx;
494: 			} else {
495: 				_lowIx = midIx;
496: 			}
497: 		}

499: 	/**
500: 	 * Excel has funny behaviour when the some elements in the search vector are the wrong type.
501: 	 *
502: 	 */
503: 	private static int performBinarySearch(ValueVector vector, LookupValueComparer lookupComparer) {
504: 		// both low and high indexes point to values assumed too low and too high.
505: 		BinarySearchIndexes bsi = new BinarySearchIndexes(vector.getSize());
506: 
507: 		while(true) {
508: 			int midIx = bsi.getMidIx();
509: 
510: 			if(midIx < 0) {
511: 				return bsi.getLowIx();
512: 			}
513: 			CompareResult cr = lookupComparer.compareTo(vector.getItem(midIx));
514: 			if(cr.isTypeMismatch()) {
515: 				int newMidIx = handleMidValueTypeMismatch(lookupComparer, vector, bsi, midIx);
516: 				if(newMidIx < 0) {
517: 					continue;
518: 				}
519: 				midIx = newMidIx;
520: 				cr = lookupComparer.compareTo(vector.getItem(midIx));
521: 			}
522: 			if(cr.isEqual()) {
523: 				return findLastIndexInRunOfEqualValues(lookupComparer, vector, midIx, bsi.getHighIx());
524: 			}
525: 			bsi.narrowSearch(midIx, cr.isLessThan());
526: 		}
527: 	}
528: 	/**
529: 	 * Excel seems to handle mismatched types initially by just stepping 'mid' ix forward to the
530: 	 * first compatible value.
531: 	 * @param midIx 'mid' index (value which has the wrong type)
532: 	 * @return usually -1, signifying that the BinarySearchIndex has been narrowed to the new mid
533: 	 * index.  Zero or greater signifies that an exact match for the lookup value was found
534: 	 */
535: 	private static int handleMidValueTypeMismatch(LookupValueComparer lookupComparer, ValueVector vector,
536: 			BinarySearchIndexes bsi, int midIx) {
537: 		int newMid = midIx;
538: 		int highIx = bsi.getHighIx();
539: 
540: 		while(true) {
541: 			newMid++;
542: 			if(newMid == highIx) {
543: 				// every element from midIx to highIx was the wrong type
544: 				// move highIx down to the low end of the mid values
545: 				bsi.narrowSearch(midIx, true);
546: 				return -1;
547: 			}
548: 			CompareResult cr = lookupComparer.compareTo(vector.getItem(newMid));
549: 			if(cr.isLessThan() && newMid == highIx-1) {
550: 				// move highIx down to the low end of the mid values
551: 				bsi.narrowSearch(midIx, true);
552: 				return -1;
553: 				// but only when "newMid == highIx-1"? slightly weird.
554: 				// It would seem more efficient to always do this.
555: 			}
556: 			if(cr.isTypeMismatch()) {
557: 				// keep stepping over values until the right type is found
558: 				continue;
559: 			}
560: 			if(cr.isEqual()) {
561: 				return newMid;
