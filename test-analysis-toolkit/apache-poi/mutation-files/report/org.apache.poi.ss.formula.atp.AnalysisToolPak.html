187:      *
188:      * @return an array of supported functions

191:     public static Collection<String> getSupportedFunctionNames(){
192:         AnalysisToolPak inst = (AnalysisToolPak)instance;
193:         Collection<String> lst = new TreeSet<String>();
194:         for(String name : inst._functionsByName.keySet()){
