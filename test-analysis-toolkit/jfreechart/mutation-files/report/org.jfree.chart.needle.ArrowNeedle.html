 66:     
 67:     /** 
 68:      * A flag controlling whether or not there is an arrow at the top of the 
 69:      * needle. 
 70:      */
 71:     private boolean isArrowAtTop = true;
 72: 
 73:     /**
 74:      * Constructs a new arrow needle.
 75:      *
 76:      * @param isArrowAtTop  a flag that controls whether or not there is an 
 77:      *                      arrow at the top of the needle.
 78:      */
 79:     public ArrowNeedle(boolean isArrowAtTop) {
 80:         this.isArrowAtTop = isArrowAtTop;
 81:     }
 82: 
 83:     /**
 84:      * Draws the needle.
 85:      *
 86:      * @param g2  the graphics device.
 87:      * @param plotArea  the plot area.
 88:      * @param rotate  the rotation point.
 89:      * @param angle  the angle.
 90:      */
 91:     protected void drawNeedle(Graphics2D g2, Rectangle2D plotArea, 
 92:                               Point2D rotate, double angle) {
 93: 
 94:         Line2D shape = new Line2D.Float();
 95:         Shape d = null;
 96: 
 97:         float x = (float) (plotArea.getMinX() +  (plotArea.getWidth() / 2));
 98:         float minY = (float) plotArea.getMinY();
 99:         float maxY = (float) plotArea.getMaxY();
100:         shape.setLine(x, minY, x, maxY);
101: 
102:         GeneralPath shape1 = new GeneralPath();
103:         if (this.isArrowAtTop) {
104:             shape1.moveTo(x, minY);
105:             minY += 4 * getSize();
106:         }
107:         else {
108:             shape1.moveTo(x, maxY);
109:             minY = maxY - 4 * getSize();
110:         }
111:         shape1.lineTo(x + getSize(), minY);
112:         shape1.lineTo(x - getSize(), minY);
113:         shape1.closePath();
114: 
115:         if ((rotate != null) && (angle != 0)) {
116:             getTransform().setToRotation(angle, rotate.getX(), rotate.getY());
117:             d = getTransform().createTransformedShape(shape);
118:         }
119:         else {
120:             d = shape;
121:         }
122:         defaultDisplay(g2, d);
123: 
124:         if ((rotate != null) && (angle != 0)) {
125:             d = getTransform().createTransformedShape(shape1);
126:         }
127:         else {
128:             d = shape1;
129:         }
130:         defaultDisplay(g2, d);
131: 
132:     }
133:

138:      * 
139:      * @return A boolean.
140:      */
141:     public boolean equals(Object obj) {
142:         if (obj == this) {
143:             return true;
144:         }
145:         if (!(obj instanceof ArrowNeedle)) {
146:             return false;

149:             return false;

152:         if (this.isArrowAtTop != that.isArrowAtTop) {
153:             return false;
154:         }
155:         return true;
156:     }
